377   @Test
378   public void testMultiThreadInsertIdenticalFullJson() throws InterruptedException {
379     Fragment fragment = fragmentService.get(jsonKey);
380     fragment.setKey(null);
381     HolyTriplet triplet = new HolyTriplet(JSON_DATASET_KEY, "fake", "fake", "fake", null);
382     Set<UniqueIdentifier> ids = Sets.newHashSet();
383     ids.add(triplet);
384 
385     int threadCount = 100;
386     ExecutorService tp = Executors.newFixedThreadPool(threadCount);
387     for (int i=0; i < threadCount; i++) {
388       tp.submit(new FragmentInserter(fragment, ids));
389     }
390     tp.shutdown();
391     tp.awaitTermination(1, TimeUnit.MINUTES);
392 
393     FragmentCreationResult got = fragmentService.insert(fragment, ids);
394     assertFalse(got.isKeyCreated());
395     assertEquals(3, got.getFragment().getKey().intValue());
396   }

398   private class FragmentInserter implements Runnable {
399     private final Fragment fragment;
400     private final Set<UniqueIdentifier> ids;
401 
402     private FragmentInserter(Fragment fragment, Set<UniqueIdentifier> ids) {
403       this.fragment = fragment;
404       this.ids = ids;
405     }
406 
407     @Override
408     public void run() {
409       try {
410         Thread.sleep(new Random().nextInt(5));
411       } catch (InterruptedException e) {
412         // TODO: Handle exception
413       }
414       FragmentCreationResult result = fragmentService.insert(fragment, ids);
415       if (result.isKeyCreated()) {
416         System.out.println(new Date().getTime() + " " + Thread.currentThread().getName() + " Created id [" + result.getFragment().getKey() + "]");
417       } else {
418         System.out.println(new Date().getTime() + " " + Thread.currentThread().getName() + " Reusing existing id [" + result.getFragment().getKey() + "]");
419       }
420 
421     }
