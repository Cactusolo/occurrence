110   private void testContention(KeyPersistenceService kps) {
111     String scope = UUID.randomUUID().toString();
112 
113     // the constants A,B,C,D,E are meant to all refer to the same occurrence, used to introduce contention
114 
115     // generate 1M sets of uniqueStrings
116     List<Set<String>> testSets = Lists.newArrayList();
117     int testSize = 1000000;
118     int abc = 0;
119     int bcd = 0;
120     int cde = 0;
121     int non = 0;
122     for (int i = 0; i < testSize; i++) {
123       Set<String> uniqueStrings = Sets.newHashSet();
124       if (i % 7 == 0) {
125         cde++;
126         uniqueStrings.add(C);
127         uniqueStrings.add(D);
128         uniqueStrings.add(E);
129       } else if (i % 5 == 0) {
130         bcd++;
131         uniqueStrings.add(B);
132         uniqueStrings.add(C);
133         uniqueStrings.add(D);
134       } else if (i % 3 == 0) {
135         abc++;
136         uniqueStrings.add(A);
137         uniqueStrings.add(B);
138         uniqueStrings.add(C);
139       } else {
140         non++;
141         uniqueStrings.add(String.valueOf(i));
142       }
143       testSets.add(uniqueStrings);
144     }
145 
146     // 1M produces abc [228571] bcd [171428] cde [142858] non [457143] so expect 457144 ids to be generated
147     //    System.out.println(
148     //      "got abc [" + abc + "] bcd [" + bcd + "] cde [" + cde + "] non [" + non + "] sum [" + (abc + bcd + cde + non)
149     //      + "]");
150 
151     // call generateKey for each of them, using X threads to do it concurrently
152     int threadCount = 100;
153     int submitCount = 0;
154     ExecutorService threadPool = Executors.newFixedThreadPool(threadCount);
155     Stopwatch stopwatch = new Stopwatch().start();
156     for (Set<String> uniques : testSets) {
157       threadPool.execute(new KeyGenRunnable(kps, uniques, scope, ++submitCount));
158     }
159     threadPool.shutdown();
160     try {
161       threadPool.awaitTermination(30, TimeUnit.MINUTES);
162     } catch (InterruptedException e) {
163       // oh well
164     }
165     stopwatch.stop();
166     System.out.println("1M keygen with [" + threadCount + "] threads took [" + stopwatch + "]");
167 
168     // expect Y new ids (test that nextId would = Y + 1)
169     Set<String> finalStrings = Sets.newHashSet();
170     finalStrings.add(UUID.randomUUID().toString());
171     KeyLookupResult nextResult = kps.generateKey(finalStrings, scope);
172     assertEquals(457145, nextResult.getKey());
173   }
