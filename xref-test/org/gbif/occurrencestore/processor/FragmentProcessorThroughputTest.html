84    public void testNoContention(int threadCount) throws InterruptedException {
85      // test generating ids as fast as possible in the ideal case of no waiting for contention (all ids are globally
86      // unique)
87      int genPerThread = 10000;
88      List<Thread> threads = Lists.newArrayList();
89      for (int i = 0; i < threadCount; i++) {
90        Thread thread = new Thread(new FragmentPersister(fragmentProcessor, UUID.randomUUID(), genPerThread));
91        thread.start();
92        threads.add(thread);
93      }
94  
95      Thread rateReporter = new Thread(new RateReporter(threadCount));
96      rateReporter.start();
97  
98      for (Thread thread : threads) {
99        thread.join();
100     }
101 
102     rateReporter.interrupt();
103     rateReporter.join();
104   }

106   private static class RateReporter implements Runnable {
107 
108     private final int threadCount;
109 
110     private RateReporter(int threadCount) {
111       this.threadCount = threadCount;
112     }
113 
114     @Override
115     public void run() {
116       int periods = 0;
117       int runningAvg = 0;
118       int buildAverageAfter = 15;
119       int lastCount = 0;
120       boolean interrupted = false;
121       while (!interrupted) {
122         try {
123           TimeUnit.SECONDS.sleep(1);
124         } catch (InterruptedException e) {
125           interrupted = true;
126         }
127         int generated = fragsPersisted.intValue() - lastCount;
128         if (periods > buildAverageAfter) {
129           if (runningAvg == 0) {
130             runningAvg = generated;
131           } else {
132             int netPeriods = periods - buildAverageAfter;
133             runningAvg = (netPeriods * runningAvg + generated) / (netPeriods + 1);
134           }
135           System.out.println("Frags persisted at [" + generated + " frags/s] for running avg of [" + runningAvg
136                              + " frags/s] and per thread [" + (runningAvg / threadCount)
137                              + " frags/sec] with frag process time of [" + (threadCount * 1000 / runningAvg)
138                              + " ms/frag]");
139         } else {
140           System.out.println("Stats in [" + (buildAverageAfter - periods) + "] seconds.");
141         }
142         periods++;
143         lastCount = fragsPersisted.intValue();
144       }
145     }
146   }
