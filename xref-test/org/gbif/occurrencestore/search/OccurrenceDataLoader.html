193   /**
194    * Reads a CSV file and produces occurrence records for each line.
195    * Each occurrence object is processed by the list of processors.
196    * 
197    * @param fileName CSV file
198    * @param processors list of processors(predicates) that consume occurrence objects
199    */
200   public static void processOccurrences(String fileName, Predicate<Occurrence>... processors) {
201     ICsvBeanReader reader = null;
202     int line = 1;
203     try {
204       reader =
205         new CsvBeanReader(new FileReader(new File(Resources.getResource(fileName).toURI())),
206           CsvPreference.STANDARD_PREFERENCE);
207       reader.getHeader(true);
208       Occurrence occurrence;
209       while ((occurrence = reader.read(Occurrence.class, HEADER, CELL_PROCESSORS)) != null) {
210         for (Predicate<Occurrence> predicate : processors) {
211           predicate.apply(occurrence);
212         }
213         line++;
214       }
215     } catch (Exception e) {
216       LOG.error(String.format("Error parsing occurrence object from file %d", line), e);
217     } finally {
218       Closeables.closeQuietly(reader);
219     }
220   }
