39  /**
40   * Utility class that stores an Occurrence record into a Solr index.
41   */
42  public class SolrOccurrenceWriter implements Predicate<Occurrence> {
43  
44  
45    // Joins coordinates with ','
46    private static final Joiner COORD_JOINER = Joiner.on(',').useForNull("");
47  
48    // Allowed latitude range
49    private static final Range<Double> LAT_RANGE = Range.closed(-90.0, 90.0);
50  
51    // Allowed longitude range
52    private static final Range<Double> LNG_RANGE = Range.closed(-180.0, 180.0);
53  
54    // Basis of record converter
55    private static final BasisOfRecordConverter BOR_CONVERTER = new BasisOfRecordConverter();
56  
57    // SolrServer that stores the occurrence records
58    private final SolrServer solrServer;
59  
60    /**
61     * Default constructor.
62     */
63    public SolrOccurrenceWriter(SolrServer solrServer) {
64      this.solrServer = solrServer;
65    }
66  
67  
68    /**
69     * Validate if the latitude and longitude values are not null and have correct values: latitude:[-90.0,90.0] and
70     * longitude[-180.0,180.0].
71     */
72    private static boolean isValidCoordinate(Double latitude, Double longitude) {
73      if (latitude != null && longitude != null) {
74        return LAT_RANGE.contains(latitude) && LNG_RANGE.contains(longitude);
75      }
76      return false;
77    }
78  
79  
80    /**
81     * Processes the occurrence object.
82     */
83    @Override
84    public boolean apply(Occurrence input) {
85      try {
86        solrServer.add(buildOccSolrDocument(input));
87      } catch (Exception e) {
88        return false;
89      }
90      return true;
91    }
92  
93    /**
94     * Populates the Solr document using the occurrence object.
95     */
96    private SolrInputDocument buildOccSolrDocument(Occurrence occurrence) {
97      SolrInputDocument doc = new SolrInputDocument();
98      final Double latitude = occurrence.getLatitude();
99      final Double longitude = occurrence.getLongitude();
100     // TODO: geospatial issue has changed a lot
101 //    final Integer geospatialIssue = occurrence.getGeospatialIssue();
102 
103     doc.setField(KEY.getFieldName(), occurrence.getKey());
104     doc.setField(YEAR.getFieldName(), occurrence.getYear());
105     doc.setField(MONTH.getFieldName(), occurrence.getMonth());
106     doc.setField(BASIS_OF_RECORD.getFieldName(), BOR_CONVERTER.fromEnum(occurrence.getBasisOfRecord()));
107     doc.setField(CATALOG_NUMBER.getFieldName(), occurrence.getField(DwcTerm.catalogNumber));
108     doc.setField(COLLECTOR_NAME.getFieldName(), occurrence.getField(DwcTerm.recordedBy));
109     doc.setField(COUNTRY.getFieldName(), occurrence.getCountry() == null ? null : occurrence.getCountry()
110       .getIso2LetterCode());
111     doc.setField(PUBLISHING_COUNTRY.getFieldName(), occurrence.getPublishingCountry() == null ? null : occurrence
112       .getPublishingCountry().getIso2LetterCode());
113     doc.setField(DATASET_KEY.getFieldName(), occurrence.getDatasetKey().toString());
114     Set<Integer> taxonKey = buildTaxonKey(occurrence);
115     if (!taxonKey.isEmpty()) {
116       doc.setField(TAXON_KEY.getFieldName(), taxonKey);
117     }
118     doc.setField(ALTITUDE.getFieldName(), occurrence.getAltitude());
119     doc.setField(DEPTH.getFieldName(), occurrence.getDepth());
120     doc.setField(INSTITUTION_CODE.getFieldName(), occurrence.getField(DwcTerm.institutionCode));
121     doc.setField(COLLECTION_CODE.getFieldName(), occurrence.getField(DwcTerm.collectionCode));
122 //    doc.setField(GEOSPATIAL_ISSUE.getFieldName(), geospatialIssue != null && geospatialIssue > 0);
123     doc.setField(GEOREFERENCED.getFieldName(), latitude != null && longitude != null);
124     doc.setField(DATE.getFieldName(),
125       occurrence.getEventDate() != null ? toDateQueryFormat(occurrence.getEventDate()) : null);
126     if (isValidCoordinate(latitude, longitude)) {
127       doc.setField(COORDINATE.getFieldName(), COORD_JOINER.join(latitude, longitude));
128     } else {
129       doc.setField(COORDINATE.getFieldName(), null);
130     }
131     return doc;
132   }

134   /**
135    * Return a set of integer that contains the taxon key values.
136    */
137   private Set<Integer> buildTaxonKey(Occurrence occurrence) {
138 
139     Set<Integer> taxonKey = new HashSet<Integer>();
140 
141     if (occurrence.getTaxonKey() != null) {
142       taxonKey.add(occurrence.getTaxonKey());
143     }
144 
145     if (occurrence.getKingdomKey() != null) {
146       taxonKey.add(occurrence.getKingdomKey());
147     }
148 
149     if (occurrence.getPhylumKey() != null) {
150       taxonKey.add(occurrence.getPhylumKey());
151     }
152 
153     if (occurrence.getClassKey() != null) {
154       taxonKey.add(occurrence.getClassKey());
155     }
156 
157     if (occurrence.getOrderKey() != null) {
158       taxonKey.add(occurrence.getOrderKey());
159     }
160 
161     if (occurrence.getFamilyKey() != null) {
162       taxonKey.add(occurrence.getFamilyKey());
163     }
164 
165     if (occurrence.getGenusKey() != null) {
166       taxonKey.add(occurrence.getGenusKey());
167     }
168 
169     if (occurrence.getSpeciesKey() != null) {
170       taxonKey.add(occurrence.getSpeciesKey());
171     }
172 
173     return taxonKey;
174   }
