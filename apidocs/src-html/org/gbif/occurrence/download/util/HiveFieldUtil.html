117  /**
118   * This map contains the hive names.
119   */
120  private static final Map<FieldName, String> HIVE_MAP = new ImmutableMap.Builder<FieldName, String>()
121    .put(FieldName.HOST_COUNTRY, "host_country").putAll(Maps.transformEntries(NAME_MAP,
122      new Maps.EntryTransformer<FieldName, Term, String>() {
123
124        public String transformEntry(@NotNull FieldName key, @NotNull Term value) {
125          switch (key) {
126            case ID:
127              return "id";
128            case I_ORDER:
129              return "order_";
130            case I_CLASS:
131              return "class_";
132            case I_LATITUDE:
133              return "latitude";
134            case I_LONGITUDE:
135              return "longitude";
136            case AUTHOR:
137              return "scientific_name_author";
138            case CONTINENT_OCEAN:
139              return "continent_ocean"; // see dev.gbif.org/issues/browse/OCC-197
140            case OWNING_ORG_KEY:
141              return "owning_organization";
142            case I_GEOSPATIAL_ISSUE:
143              return "geospatial_issue";
144            default:
145              return CaseFormat.UPPER_CAMEL.to(CaseFormat.LOWER_UNDERSCORE, value.simpleName())
146                .replace("_i_d", "_id");
147          }
148        }
149      })).build();
150
151  /**
152   * Private default constructor.
153   */
154  private HiveFieldUtil() {
155    throw new UnsupportedOperationException("Can't initialize class");
156  }
157
158  /**
159   * @return the column name used in the hive occurrence table.
160   */
161  public static String getHiveField(FieldName field) {
162    return HIVE_MAP.get(field);
163  }
164
165  /**
166   * @return the concept term enum for an hbase field
167   */
168  public static Term getTerm(FieldName field) {
169    return NAME_MAP.get(field);
170  }
