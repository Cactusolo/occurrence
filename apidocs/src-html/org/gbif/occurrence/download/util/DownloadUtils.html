001package org.gbif.occurrence.download.util;
002
003/**
004 * Shared download utilities.
005 */
006public class DownloadUtils {
007
008  private static final String OOZIE_SUFFIX = "-oozie-oozi-W";
009  public static final String DOWNLOAD_ID_PLACEHOLDER = "_DOWNLOAD_ID_";
010
011  public static final String DELIMETERS_MATCH =
012    "\\t|\\n|\\r|(?:(?>\\u000D\\u000A)|[\\u000A\\u000B\\u000C\\u000D\\u0085\\u2028\\u2029])";
013
014  public static final String ISO_8601_FORMAT = "yyyy-MM-dd'T'HH:mm'Z'";
015
016  /**
017   * Private default constructor.
018   */
019  private DownloadUtils() {
020    throw new UnsupportedOperationException("Can't initialize class");
021  }
022
023  /**
024   * Adds a path to a base url and makes sure the path / separator only exists once.
025   * 
026   * @param base the base url to add to with or without trailing slash
027   * @param path the path to be added without a prefix /
028   * @return a string with the form base/path
029   */
030  public static String concatUrlPaths(String base, String path) {
031    StringBuilder sb = new StringBuilder();
032    sb.append(base);
033    if (!base.endsWith("/")) {
034      sb.append('/');
035    }
036    sb.append(path);
037    return sb.toString();
038  }
039
040  /**
041   * Creates a URL pointing to the download file.
042   */
043  public static String downloadLink(String baseUrl, String downloadId) {
044    return DownloadUtils.concatUrlPaths(baseUrl, "occurrence/download/request/" + downloadId + ".zip");
045  }
046
047  public static String downloadToWorkflowId(String downloadId) {
048    return downloadId + OOZIE_SUFFIX;
049  }
050
051  public static String workflowToDownloadId(String workflowId) {
052    if (workflowId.contains(OOZIE_SUFFIX)) {
053      return workflowId.replace(OOZIE_SUFFIX, "");
054    }
055    throw new IllegalArgumentException("WorkflowId given in unknown format: " + workflowId);
056  }
057
058}
