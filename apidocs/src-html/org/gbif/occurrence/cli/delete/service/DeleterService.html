001package org.gbif.occurrence.cli.delete.service;
002
003import org.gbif.common.messaging.DefaultMessagePublisher;
004import org.gbif.common.messaging.MessageListener;
005import org.gbif.occurrencestore.deleter.OccurrenceDeletionService;
006import org.gbif.occurrencestore.deleter.messaging.DeleteOccurrenceListener;
007import org.gbif.occurrencestore.persistence.OccurrenceKeyPersistenceServiceImpl;
008import org.gbif.occurrencestore.persistence.OccurrencePersistenceServiceImpl;
009import org.gbif.occurrencestore.persistence.VerbatimOccurrencePersistenceServiceImpl;
010import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
011import org.gbif.occurrencestore.persistence.api.OccurrencePersistenceService;
012import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
013import org.gbif.occurrencestore.persistence.keygen.HBaseLockingKeyService;
014import org.gbif.occurrencestore.persistence.keygen.KeyPersistenceService;
015
016import com.google.common.util.concurrent.AbstractIdleService;
017import org.apache.hadoop.hbase.HBaseConfiguration;
018import org.apache.hadoop.hbase.client.HTablePool;
019
020public class DeleterService extends AbstractIdleService {
021
022  private final DeleterConfiguration config;
023  private MessageListener listener;
024  private HTablePool tablePool;
025
026  public DeleterService(DeleterConfiguration config) {
027    this.config = config;
028  }
029
030  @Override
031  protected void startUp() throws Exception {
032    config.ganglia.start();
033
034    tablePool = new HTablePool(HBaseConfiguration.create(), config.hbasePoolSize);
035
036    KeyPersistenceService<Integer> keyService =
037      new HBaseLockingKeyService(config.lookupTable, config.counterTable, config.occTable, tablePool);
038    OccurrenceKeyPersistenceService occurrenceKeyService = new OccurrenceKeyPersistenceServiceImpl(keyService);
039    OccurrencePersistenceService occurrenceService = new OccurrencePersistenceServiceImpl(config.occTable, tablePool);
040    VerbatimOccurrencePersistenceService verbatimService =
041      new VerbatimOccurrencePersistenceServiceImpl(config.occTable, tablePool);
042    OccurrenceDeletionService occurrenceDeletionService =
043      new OccurrenceDeletionService(occurrenceService, occurrenceKeyService, verbatimService);
044
045    listener = new MessageListener(config.messaging.getConnectionParameters());
046    listener.listen(config.queueName, 1, new DeleteOccurrenceListener(occurrenceDeletionService,
047      new DefaultMessagePublisher(config.messaging.getConnectionParameters())));
048  }
049
050  @Override
051  protected void shutDown() throws Exception {
052    if (listener != null) {
053      listener.close();
054    }
055    if (tablePool != null) {
056      tablePool.close();
057    }
058  }
059}
