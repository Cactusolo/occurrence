043/**
044 * Utility class that stores an Occurrence record into a Solr index.
045 */
046public class SolrOccurrenceWriter {
047
048
049  // Joins coordinates with ','
050  private static final Joiner COORD_JOINER = Joiner.on(',').useForNull("");
051
052  // Allowed latitude range
053  private static final Range<Double> LAT_RANGE = Range.closed(-90.0, 90.0);
054
055  // Allowed longitude range
056  private static final Range<Double> LNG_RANGE = Range.closed(-180.0, 180.0);
057
058  // Basis of record converter
059  private static final BasisOfRecordConverter BOR_CONVERTER = new BasisOfRecordConverter();
060
061  // SolrServer that stores the occurrence records
062  private final SolrServer solrServer;
063
064  private final int commitWithinMs;
065
066  /**
067   * Default constructor.
068   */
069  public SolrOccurrenceWriter(SolrServer solrServer, int commitWithinMs) {
070    this.solrServer = solrServer;
071    this.commitWithinMs = commitWithinMs;
072  }
073
074
075  /**
076   * Validate if the latitude and longitude values are not null and have correct values: latitude:[-90.0,90.0] and
077   * longitude[-180.0,180.0].
078   */
079  private static boolean isValidCoordinate(Double latitude, Double longitude) {
080    if (latitude != null && longitude != null) {
081      return LAT_RANGE.contains(latitude) && LNG_RANGE.contains(longitude);
082    }
083    return false;
084  }
085
086
087  public void delete(Occurrence input) throws IOException, SolrServerException {
088    solrServer.deleteById(input.getKey().toString(), commitWithinMs);
089  }
090
091  /**
092   * Processes the occurrence object.
093   */
094  public void update(Occurrence input) throws IOException, SolrServerException {
095    solrServer.add(buildOccSolrDocument(input), commitWithinMs);
096  }
097
098  /**
099   * Populates the Solr document using the occurrence object.
100   */
101  private SolrInputDocument buildOccSolrDocument(Occurrence occurrence) {
102    SolrInputDocument doc = new SolrInputDocument();
103    final Double latitude = occurrence.getLatitude();
104    final Double longitude = occurrence.getLongitude();
105    // TODO: geospatial issue has changed
106//    final Integer geospatialIssue = occurrence.getGeospatialIssue();
107
108    doc.setField(KEY.getFieldName(), occurrence.getKey());
109    doc.setField(YEAR.getFieldName(), occurrence.getYear());
110    doc.setField(MONTH.getFieldName(), occurrence.getMonth());
111    doc.setField(BASIS_OF_RECORD.getFieldName(), BOR_CONVERTER.fromEnum(occurrence.getBasisOfRecord()));
112    doc.setField(CATALOG_NUMBER.getFieldName(), occurrence.getField(DwcTerm.catalogNumber));
113    doc.setField(COLLECTOR_NAME.getFieldName(), occurrence.getField(DwcTerm.recordedBy));
114    doc.setField(COUNTRY.getFieldName(), occurrence.getCountry() == null ? null : occurrence.getCountry()
115      .getIso2LetterCode());
116    doc.setField(PUBLISHING_COUNTRY.getFieldName(), occurrence.getPublishingCountry() == null ? null : occurrence
117      .getPublishingCountry().getIso2LetterCode());
118    doc.setField(DATASET_KEY.getFieldName(), occurrence.getDatasetKey().toString());
119    Set<Integer> taxonKey = buildTaxonKey(occurrence);
120    if (!taxonKey.isEmpty()) {
121      doc.setField(TAXON_KEY.getFieldName(), taxonKey);
122    } else {
123      doc.setField(TAXON_KEY.getFieldName(), null);
124    }
125    doc.setField(ALTITUDE.getFieldName(), occurrence.getAltitude());
126    doc.setField(DEPTH.getFieldName(), occurrence.getDepth());
127    doc.setField(INSTITUTION_CODE.getFieldName(), occurrence.getField(DwcTerm.institutionCode));
128    doc.setField(COLLECTION_CODE.getFieldName(), occurrence.getField(DwcTerm.collectionCode));
129//    doc.setField(GEOSPATIAL_ISSUE.getFieldName(), geospatialIssue != null && geospatialIssue > 0);
130    doc.setField(LATITUDE.getFieldName(), latitude);
131    doc.setField(LONGITUDE.getFieldName(), longitude);
132    doc.setField(GEOREFERENCED.getFieldName(), latitude != null && longitude != null);
133    doc.setField(DATE.getFieldName(),
134      occurrence.getEventDate() != null ? toDateQueryFormat(occurrence.getEventDate()) : null);
135    doc.setField(MODIFIED.getFieldName(),
136      occurrence.getModified() != null ? toDateQueryFormat(occurrence.getModified()) : null);
137    if (isValidCoordinate(latitude, longitude)) {
138      doc.setField(COORDINATE.getFieldName(), COORD_JOINER.join(latitude, longitude));
139    } else {
140      doc.setField(COORDINATE.getFieldName(), null);
141    }
142
143    return doc;
144  }
145
146
147  /**
148   * Return a set of integer that contains the taxon key values.
149   */
150  private Set<Integer> buildTaxonKey(Occurrence occurrence) {
151
152    Set<Integer> taxonKey = new HashSet<Integer>();
153
154    if (occurrence.getTaxonKey() != null) {
155      taxonKey.add(occurrence.getTaxonKey());
156    }
157
158    if (occurrence.getKingdomKey() != null) {
159      taxonKey.add(occurrence.getKingdomKey());
160    }
161
162    if (occurrence.getPhylumKey() != null) {
163      taxonKey.add(occurrence.getPhylumKey());
164    }
165
166    if (occurrence.getClassKey() != null) {
167      taxonKey.add(occurrence.getClassKey());
168    }
169
170    if (occurrence.getOrderKey() != null) {
171      taxonKey.add(occurrence.getOrderKey());
172    }
173
174    if (occurrence.getFamilyKey() != null) {
175      taxonKey.add(occurrence.getFamilyKey());
176    }
177
178    if (occurrence.getGenusKey() != null) {
179      taxonKey.add(occurrence.getGenusKey());
180    }
181
182    if (occurrence.getSpeciesKey() != null) {
183      taxonKey.add(occurrence.getSpeciesKey());
184    }
185
186    return taxonKey;
187  }
