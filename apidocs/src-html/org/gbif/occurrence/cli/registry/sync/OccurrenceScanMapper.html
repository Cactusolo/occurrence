045/**
046 * A mapreduce Mapper that synchronizes occurrences with the registry. It checks for changed owning organizations,
047 * changed owning organization country, and dataset deletions. For organization changes the new values are written
048 * to the occurrence HBase table via occurrence persistence, and then an OccurrenceMutatedMessage is sent. For dataset
049 * deletions a DeleteDatasetMessage is sent.
050 */
051public class OccurrenceScanMapper extends TableMapper<ImmutableBytesWritable, NullWritable> {
052
053  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceScanMapper.class);
054
055  private static final Set<UUID> DEAD_DATASETS = Sets.newHashSet();
056  private static final Set<UUID> UNCHANGED_DATASETS = Sets.newHashSet();
057  private static final Map<UUID, Organization> DATASET_TO_OWNING_ORG = Maps.newHashMap();
058
059  private static final DatasetService DATASET_SERVICE;
060  private static final OrganizationService ORG_SERVICE;
061  private static final OccurrencePersistenceService OCCURRENCE_PERSISTENCE_SERVICE;
062  private static final MessagePublisher MESSAGE_PUBLISHER;
063
064  private int numRecords = 0;
065
066  static {
067    Properties props = SyncCommon.loadProperties();
068    ClientConfig cc = new DefaultClientConfig();
069    cc.getClasses().add(JacksonJsonProvider.class);
070    cc.getClasses().add(RegistryObjectMapperContextResolver.class);
071    cc.getFeatures().put(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES.toString(), false);
072    Client httpClient = ApacheHttpClient.create(cc);
073    WebResource regResource = httpClient.resource(props.getProperty(SyncCommon.REG_WS_PROPS_KEY));
074    DATASET_SERVICE = new DatasetWsClient(regResource, null);
075    ORG_SERVICE = new OrganizationWsClient(regResource, null);
076
077    Injector injector =
078      Guice.createInjector(new PostalServiceModule("sync", props), new OccurrencePersistenceModule(props));
079    OCCURRENCE_PERSISTENCE_SERVICE = injector.getInstance(OccurrencePersistenceService.class);
080    MESSAGE_PUBLISHER = injector.getInstance(MessagePublisher.class);
081  }
082
083  @Override
084  public void map(ImmutableBytesWritable row, Result values, Context context) throws IOException {
085    UUID datasetKey = UUID.fromString(Bytes.toString(values.getValue(SyncCommon.OCC_CF, SyncCommon.DK_COL)));
086    if (DEAD_DATASETS.contains(datasetKey) || UNCHANGED_DATASETS.contains(datasetKey)) {
087      return;
088    }
089
090    Dataset dataset = DATASET_SERVICE.get(datasetKey);
091    if (dataset.getDeleted() != null) {
092      DEAD_DATASETS.add(datasetKey);
093      try {
094        LOG.info("Sending delete dataset message for dataset [{}]", datasetKey);
095        MESSAGE_PUBLISHER
096          .send(new DeleteDatasetOccurrencesMessage(datasetKey, OccurrenceDeletionReason.DATASET_MANUAL));
097      } catch (IOException e) {
098        LOG.warn("Failed to send update message", e);
099      }
100      return;
101    }
102
103    // dataset exists, now compare with values we have on the occurrence
104    Organization owningOrg;
105
106    boolean needsUpdate;
107    if (DATASET_TO_OWNING_ORG.containsKey(datasetKey)) {
108      // seen it before, no need to do comparisons - record needs updating
109      owningOrg = DATASET_TO_OWNING_ORG.get(datasetKey);
110      needsUpdate = true;
111    } else {
112      UUID newOwningOrgKey = dataset.getOwningOrganizationKey();
113      owningOrg = ORG_SERVICE.get(newOwningOrgKey);
114      String rawOwningOrgKey = Bytes.toString(values.getValue(SyncCommon.OCC_CF, SyncCommon.OOK_COL));
115      UUID owningOrgKey = rawOwningOrgKey == null ? null : UUID.fromString(rawOwningOrgKey);
116      String rawHostCountry = Bytes.toString(values.getValue(SyncCommon.OCC_CF, SyncCommon.HC_COL));
117      Country hostCountry = rawHostCountry == null ? null : Country.fromIsoCode(rawHostCountry);
118      Country newHostCountry = owningOrg.getCountry();
119      if (newOwningOrgKey.equals(owningOrgKey) && newHostCountry == hostCountry) {
120        needsUpdate = false;
121        UNCHANGED_DATASETS.add(datasetKey);
122      } else {
123        needsUpdate = true;
124        DATASET_TO_OWNING_ORG.put(datasetKey, owningOrg);
125      }
126    }
127
128    if (needsUpdate) {
129      Occurrence origOcc = OCCURRENCE_PERSISTENCE_SERVICE.get(Bytes.toInt(row.get()));
130      // we have no clone or other easy copy method
131      Occurrence updatedOcc = OCCURRENCE_PERSISTENCE_SERVICE.get(Bytes.toInt(row.get()));
132      updatedOcc.setPublishingOrgKey(owningOrg.getKey());
133      updatedOcc.setPublishingCountry(owningOrg.getCountry());
134      OCCURRENCE_PERSISTENCE_SERVICE.update(updatedOcc);
135
136      int crawlId = Bytes.toInt(values.getValue(SyncCommon.OCC_CF, SyncCommon.CI_COL));
137      OccurrenceMutatedMessage msg =
138        OccurrenceMutatedMessage.buildUpdateMessage(datasetKey, origOcc, updatedOcc, crawlId);
139
140      try {
141        LOG.info("Sending update for key [{}], owning org changed from [{}] to [{}] and host country from [{}] to [{}]",
142          datasetKey, origOcc.getPublishingOrgKey(), updatedOcc.getPublishingOrgKey(), origOcc.getPublishingCountry(),
143          updatedOcc.getPublishingCountry());
144        MESSAGE_PUBLISHER.send(msg);
145      } catch (IOException e) {
146        LOG.warn("Failed to send update message", e);
147      }
148    }
149    numRecords++;
150    if (numRecords % 10000 == 0) {
151      context.setStatus("mapper processed " + numRecords + " records so far");
152    }
153  }
154}
