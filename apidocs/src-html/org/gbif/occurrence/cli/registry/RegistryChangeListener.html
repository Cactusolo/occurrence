037/**
038 * Listens for any registry changes of interest to occurrences: namely organization and dataset updates or deletions.
039 */
040public class RegistryChangeListener extends AbstractMessageCallback<RegistryChangeMessage> {
041
042  private static final Logger LOG = LoggerFactory.getLogger(RegistryChangeListener.class);
043  private static final int PAGING_LIMIT = 20;
044
045  private final MessagePublisher messagePublisher;
046  private final OrganizationService orgService;
047
048  public RegistryChangeListener(MessagePublisher messagePublisher, OrganizationService orgService) {
049    this.messagePublisher = messagePublisher;
050    this.orgService = orgService;
051  }
052
053  @Override
054  public void handleMessage(RegistryChangeMessage message) {
055    LOG.info("Handling registry [{}] msg for class [{}]", message.getChangeType(), message.getObjectClass());
056    Class<?> clazz = message.getObjectClass();
057    if ("Dataset".equals(clazz.getSimpleName())) {
058      handleDataset(message.getChangeType(), (Dataset) message.getOldObject(), (Dataset) message.getNewObject());
059    } else if ("Organization".equals(clazz.getSimpleName())) {
060      handleOrganization(message.getChangeType(), (Organization) message.getOldObject(),
061        (Organization) message.getNewObject());
062    }
063  }
064
065  private void handleDataset(RegistryChangeMessage.ChangeType changeType, Dataset oldDataset, Dataset newDataset) {
066    switch (changeType) {
067      case DELETED:
068        LOG.info("Sending delete for dataset [{}]", oldDataset.getKey());
069        try {
070          messagePublisher
071            .send(new DeleteDatasetOccurrencesMessage(oldDataset.getKey(), OccurrenceDeletionReason.DATASET_MANUAL));
072        } catch (IOException e) {
073          LOG.warn("Could not send delete dataset message for key [{}]", oldDataset.getKey(), e);
074        }
075        break;
076      case UPDATED:
077        if (oldDataset.getOwningOrganizationKey().equals(newDataset.getOwningOrganizationKey())) {
078          LOG.debug("Owning orgs match for updated dataset [{}] - taking no action", newDataset.getKey());
079        } else {
080          LOG.info("Starting m/r sync for changed owning org on dataset [{}]", newDataset.getKey());
081          try {
082            runMrSync(newDataset.getKey());
083          } catch (Exception e) {
084            LOG.warn("Failed to run RegistrySync m/r for dataset [{}]", newDataset.getKey(), e);
085          }
086        }
087        break;
088      case CREATED:
089        LOG.info("Sending crawl for new dataset [{}]", newDataset.getKey());
090        try {
091          messagePublisher.send(new StartCrawlMessage(newDataset.getKey()));
092        } catch (IOException e) {
093          LOG.warn("Could not send start crawl message for dataset key [{}]", newDataset.getKey(), e);
094        }
095        break;
096    }
097  }
098
099  private void handleOrganization(RegistryChangeMessage.ChangeType changeType, Organization oldOrg,
100    Organization newOrg) {
101    switch (changeType) {
102      case DELETED:
103        break;
104      case UPDATED:
105        if (!oldOrg.isEndorsementApproved() && newOrg.isEndorsementApproved()) {
106          LOG.info("Starting crawl of all datasets for newly endorsed org [{}]", newOrg.getKey());
107          DatasetVisitor visitor = new DatasetVisitor() {
108            @Override
109            public void visit(UUID datasetKey) {
110              try {
111                messagePublisher.send(new StartCrawlMessage(datasetKey));
112              } catch (IOException e) {
113                LOG.warn("Could not send start crawl message for newly endorsed dataset key [{}]", datasetKey, e);
114              }
115            }
116          };
117          visitOwnedDatasets(newOrg.getKey(), visitor);
118        } else if (oldOrg.getCountry() != newOrg.getCountry() && newOrg.isEndorsementApproved()) {
119          if (newOrg.getNumOwnedDatasets() > 0) {
120            LOG.info("Starting m/r sync for all datasets of org [{}] because it has changed country from [{}] to [{}]",
121              newOrg.getKey(), oldOrg.getCountry(), newOrg.getCountry());
122            DatasetVisitor visitor = new DatasetVisitor() {
123              @Override
124              public void visit(UUID datasetKey) {
125                runMrSync(datasetKey);
126              }
127            };
128            visitOwnedDatasets(newOrg.getKey(), visitor);
129          }
130        }
131        break;
132      case CREATED:
133        break;
134    }
135  }
136
137  private static void runMrSync(@Nullable UUID datasetKey) {
138    Configuration conf = HBaseConfiguration.create();
139    conf.set("hbase.client.scanner.timeout.period", "600000");
140    conf.set("hbase.regionserver.lease.period", "600000");
141    conf.set("hbase.rpc.timeout", "600000");
142
143    Scan scan = new Scan();
144    scan.addColumn(SyncCommon.OCC_CF, SyncCommon.DK_COL);
145    scan.addColumn(SyncCommon.OCC_CF, SyncCommon.HC_COL);
146    scan.addColumn(SyncCommon.OCC_CF, SyncCommon.OOK_COL);
147    scan.addColumn(SyncCommon.OCC_CF, SyncCommon.CI_COL);
148    scan.setCaching(200);
149    String rawDatasetKey = null;
150    if (datasetKey != null) {
151      rawDatasetKey = datasetKey.toString();
152      scan.setFilter(
153        new SingleColumnValueFilter(SyncCommon.OCC_CF, SyncCommon.DK_COL, CompareFilter.CompareOp.EQUAL,
154          Bytes.toBytes(rawDatasetKey)));
155    }
156
157    Properties props = SyncCommon.loadProperties();
158    String targetTable = props.getProperty(SyncCommon.OCC_TABLE_PROPS_KEY);
159
160    String jobTitle = "Registry-Occurrence Sync on table " + targetTable;
161    if (rawDatasetKey != null) {
162      jobTitle = jobTitle + " for dataset " + rawDatasetKey;
163    }
164
165    try {
166      Job job = new Job(conf, jobTitle);
167      job.setJarByClass(OccurrenceScanMapper.class);
168      job.setOutputFormatClass(NullOutputFormat.class);
169      job.setNumReduceTasks(0);
170      job.getConfiguration().set("mapred.map.tasks.speculative.execution", "false");
171      job.getConfiguration().set("mapred.reduce.tasks.speculative.execution", "false");
172
173
174      if (targetTable == null) {
175        LOG.error("Sync m/r not properly configured (occ table not set) - aborting");
176      } else {
177        TableMapReduceUtil
178          .initTableMapperJob(targetTable, scan, OccurrenceScanMapper.class, ImmutableBytesWritable.class,
179            NullWritable.class, job);
180        job.waitForCompletion(true);
181      }
182    } catch (Exception e) {
183      LOG.error("Could not start m/r job, aborting", e);
184    }
185
186    LOG.info("Finished running m/r sync for dataset [{}]", datasetKey);
187  }
188
189  private void visitOwnedDatasets(UUID orgKey, DatasetVisitor visitor) {
190    int datasetCount = 0;
191    boolean endOfRecords = false;
192    int offset = 0;
193    do {
194      Pageable page = new PagingRequest(offset, PAGING_LIMIT);
195      PagingResponse<Dataset> datasets = orgService.ownedDatasets(orgKey, page);
196
197      for (Dataset dataset : datasets.getResults()) {
198        visitor.visit(dataset.getKey());
199      }
200      datasetCount += datasets.getResults().size();
201      offset += PAGING_LIMIT;
202
203      if (datasets.isEndOfRecords()) {
204        endOfRecords = datasets.isEndOfRecords();
205      }
206    } while (!endOfRecords);
207    LOG.info("Visited [{}] datasets owned by org [{}]", datasetCount, orgKey);
208  }
209
210  private interface DatasetVisitor {
211
212    void visit(UUID datasetKey);
213  }
