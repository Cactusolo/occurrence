001package org.gbif.occurrence.cli.registry.service;
002
003import org.gbif.api.service.registry.OrganizationService;
004import org.gbif.common.messaging.DefaultMessagePublisher;
005import org.gbif.common.messaging.DefaultMessageRegistry;
006import org.gbif.common.messaging.MessageListener;
007import org.gbif.occurrence.cli.registry.RegistryChangeListener;
008import org.gbif.registry.ws.client.guice.RegistryWsClientModule;
009import org.gbif.ws.client.guice.AnonymousAuthModule;
010
011import java.util.Properties;
012
013import com.google.common.util.concurrent.AbstractIdleService;
014import com.google.inject.Guice;
015import com.google.inject.Injector;
016import org.codehaus.jackson.map.DeserializationConfig;
017import org.codehaus.jackson.map.ObjectMapper;
018
019public class RegistryChangeService extends AbstractIdleService {
020
021  private final RegistryChangeConfiguration configuration;
022  private MessageListener listener;
023
024  public RegistryChangeService(RegistryChangeConfiguration configuration) {
025    this.configuration = configuration;
026  }
027
028  @Override
029  protected void startUp() throws Exception {
030    // Create Registry WS Client
031    Properties properties = new Properties();
032    properties.setProperty("registry.ws.url", configuration.registryWsUrl);
033
034    Injector injector = Guice.createInjector(new RegistryWsClientModule(properties), new AnonymousAuthModule());
035    OrganizationService orgClient = injector.getInstance(OrganizationService.class);
036
037    // we have to create our own object mapper in order to set FAIL_ON_UNKNOWN, without which we can't deser reg objects
038    ObjectMapper objectMapper = new ObjectMapper();
039    objectMapper.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
040    listener = new MessageListener(configuration.messaging.getConnectionParameters(), new DefaultMessageRegistry(),
041      objectMapper);
042    listener.listen(configuration.registryChangeQueueName, 1,
043      new RegistryChangeListener(new DefaultMessagePublisher(configuration.messaging.getConnectionParameters()),
044        orgClient));
045  }
046
047  @Override
048  protected void shutDown() throws Exception {
049    if (listener != null) {
050      listener.close();
051    }
052  }
053}
