001package org.gbif.occurrence.cli.dataset.service;
002
003import org.gbif.common.messaging.DefaultMessagePublisher;
004import org.gbif.common.messaging.MessageListener;
005import org.gbif.occurrence.cli.dataset.DeleteDatasetListener;
006import org.gbif.occurrencestore.persistence.OccurrenceKeyPersistenceServiceImpl;
007import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
008import org.gbif.occurrencestore.persistence.keygen.HBaseLockingKeyService;
009import org.gbif.occurrencestore.persistence.keygen.KeyPersistenceService;
010
011import com.google.common.util.concurrent.AbstractIdleService;
012import org.apache.hadoop.hbase.HBaseConfiguration;
013import org.apache.hadoop.hbase.client.HTablePool;
014
015/**
016 * Handles messages to parse a dataset (from xml/json to verbatim) and to interpret a dataset (from verbatim to
017 * interpreted). Eventually should also handle deletions (which live in the delete package for now).
018 */
019public class DatasetMutationService extends AbstractIdleService {
020
021  private final DatasetMutationConfiguration config;
022  private MessageListener listener;
023  private HTablePool tablePool;
024
025  public DatasetMutationService(DatasetMutationConfiguration config) {
026    this.config = config;
027  }
028
029  @Override
030  protected void startUp() throws Exception {
031    config.ganglia.start();
032
033    tablePool = new HTablePool(HBaseConfiguration.create(), config.hbasePoolSize);
034
035    KeyPersistenceService<Integer> keyService =
036      new HBaseLockingKeyService(config.lookupTable, config.counterTable, config.occTable, tablePool);
037
038    OccurrenceKeyPersistenceService occurrenceService =
039      new OccurrenceKeyPersistenceServiceImpl(keyService);
040
041    // TODO: add listeners for parsing and interpreting
042//    OccurrencePersistenceService occurrencePersistenceService =
043//      new OccurrencePersistenceServiceImpl(config.occTable, tablePool);
044//    FragmentPersistenceService fragmentService =
045//      new FragmentPersistenceServiceImpl(config.occTable, tablePool, occurrenceService);
046//    VerbatimOccurrencePersistenceService verbatimService =
047//      new VerbatimOccurrencePersistenceServiceImpl(config.occTable, tablePool);
048
049    listener = new MessageListener(config.messaging.getConnectionParameters());
050    listener.listen(config.deleteDatasetQueueName, 1, new DeleteDatasetListener(occurrenceService,
051      new DefaultMessagePublisher(config.messaging.getConnectionParameters())));
052  }
053
054  @Override
055  protected void shutDown() throws Exception {
056    if (listener != null) {
057      listener.close();
058    }
059    if (tablePool != null) {
060      tablePool.close();
061    }
062  }
063}
