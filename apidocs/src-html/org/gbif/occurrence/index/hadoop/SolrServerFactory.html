053/**
054 * Instantiate a record writer that will build a Solr index.
055 * A zip file containing the solr config and additional libraries is expected to
056 * be passed via the distributed cache. The incoming written records are
057 * converted via the specified document converter, and written to the index in
058 * batches. When the job is done, the close copies the index to the destination
059 * output file system. <h2>Configuration Parameters</h2>
060 * <ul>
061 * <li>solr.record.writer.batch.size - the number of documents in a batch that is sent to the indexer.</li>
062 * <li>mapred.task.id - To build the unique temporary index directory file name.</li>
063 * <li>solr.output.format.setup - {@link SolrOutputFormat.SETUP_OK} The path to the configuration zip file.</li>
064 * <li> {@link SolrOutputFormat.zipName} - The file name of the configuration file.</li>
065 * <li>solr.document.converter.class - {@link SolrDocumentConverter.CONVERTER_NAME_KEY} the class used to convert the
066 * {@link #write} key, values into a {@link SolrInputDocument}. Set via {@link SolrDocumentConverter}.
067 * </ul>
068 */
069public class SolrServerFactory {

114  public SolrServerFactory(TaskAttemptContext context) {
115    conf = context.getConfiguration();
116
117    // setLogLevel("org.apache.solr.core", "WARN");
118    // setLogLevel("org.apache.solr.update", "WARN");
119
120    heartBeater = new HeartBeater(context);
121    try {
122      heartBeater.needHeartBeat();
123      /** The actual file in hdfs that holds the configuration. */
124
125      final String configuredSolrConfigPath = conf.get(SolrOutputFormat.SETUP_OK);
126      if (configuredSolrConfigPath == null) {
127        throw new IllegalStateException(String.format(
128          "The job did not pass %s", SolrOutputFormat.SETUP_OK));
129      }
130      outputZipFile = SolrOutputFormat.isOutputZipFormat(conf);
131
132      this.fs = FileSystem.get(conf);
133      perm = new Path(FileOutputFormat.getOutputPath(context), getOutFileName(context, "part"));
134
135      // Make a task unique name that contains the actual index output name to
136      // make debugging simpler
137      // Note: if using JVM reuse, the sequence number will not be reset for a
138      // new task using the jvm
139
140      temp = conf.getLocalPath("mapred.local.dir", "solr_" +
141        conf.get("mapred.task.id") + '.' + sequence.incrementAndGet());
142
143      if (outputZipFile && !perm.getName().endsWith(".zip")) {
144        perm = perm.suffix(".zip");
145      }
146      fs.delete(perm, true); // delete old, if any
147      Path local = fs.startLocalOutput(perm, temp);
148
149      solrHome = findSolrConfig(conf);
150
151      // }
152      // Verify that the solr home has a conf and lib directory
153      if (solrHome == null) {
154        throw new IOException("Unable to find solr home setting");
155      }
156
157      // Setup a solr instance that we can batch writes to
158      LOG.info("SolrHome: " + solrHome.toUri());
159      String dataDir = new File(local.toString(), "data").toString();
160      // copy the schema to the conf dir
161      File confDir = new File(local.toString(), "conf");
162      confDir.mkdirs();
163      File srcSchemaFile = new File(solrHome.toString(), "conf/schema.parsing");
164      assert srcSchemaFile.exists();
165      FileUtils.copyFile(srcSchemaFile, new File(confDir, "schema.parsing"));
166      Properties props = new Properties();
167      props.setProperty("solr.data.dir", dataDir);
168      props.setProperty("solr.home", solrHome.toString());
169      SolrResourceLoader loader = new SolrResourceLoader(solrHome.toString(),
170        null, props);
171      LOG
172        .info(String
173          .format(
174            "Constructed instance information solr.home %s (%s), instance dir %s, conf dir %s, writing index to temporary directory %s, with permdir %s",
175            solrHome, solrHome.toUri(), loader.getInstanceDir(), loader
176              .getConfigDir(), dataDir, perm));
177      CoreContainer container = new CoreContainer(loader);
178      CoreDescriptor descr = new CoreDescriptor(container, "core1", solrHome
179        .toString());
180      descr.setDataDir(dataDir);
181      descr.setCoreProperties(props);
182      core = container.create(descr);
183      container.register(core, false);
184      solr = new EmbeddedSolrServer(container, "core1");
185    } catch (Exception e) {
186      throw new IllegalStateException(String.format(
187        "Failed to initialize record writer for %s, %s", context.getJobName(), conf
188          .get("mapred.task.id")), e);
189    } finally {
190      heartBeater.cancelHeartBeat();
191    }
192  }

194  /**
195   * Write a file to a zip output stream, removing leading path name components
196   * from the actual file name when creating the zip file entry.
197   * The entry placed in the zip file is <code>baseName</code>/ <code>relativePath</code>, where
198   * <code>relativePath</code> is constructed
199   * by removing a leading <code>root</code> from the path for <code>itemToZip</code>.
200   * If <code>itemToZip</code> is an empty directory, it is ignored. If <code>itemToZip</code> is a directory, the
201   * contents of the directory are
202   * added recursively.
203   *
204   * @param zos The zip output stream
205   * @param baseName The base name to use for the file name entry in the zip
206   *        file
207   * @param root The path to remove from <code>itemToZip</code> to make a
208   *        relative path name
209   * @param itemToZip The path to the file to be added to the zip file
210   * @return the number of entries added
211   * @throws IOException
212   */
213  static public int zipDirectory(final Configuration conf,
214    final ZipOutputStream zos, final String baseName, final String root,
215    final Path itemToZip) throws IOException {
216    LOG
217      .info(String
218        .format("zipDirectory: %s %s %s", baseName, root, itemToZip));
219    LocalFileSystem localFs = FileSystem.getLocal(conf);
220    int count = 0;
221
222    final FileStatus itemStatus = localFs.getFileStatus(itemToZip);
223    if (itemStatus.isDir()) {
224      final FileStatus[] statai = localFs.listStatus(itemToZip);
225
226      // Add a directory entry to the zip file
227      final String zipDirName = relativePathForZipEntry(itemToZip.toUri()
228        .getPath(), baseName, root);
229      final ZipEntry dirZipEntry = new ZipEntry(zipDirName
230        + Path.SEPARATOR_CHAR);
231      LOG.info(String.format("Adding directory %s to zip", zipDirName));
232      zos.putNextEntry(dirZipEntry);
233      zos.closeEntry();
234      count++;
235
236      if (statai == null || statai.length == 0) {
237        LOG.info(String.format("Skipping empty directory %s", itemToZip));
238        return count;
239      }
240      for (FileStatus status : statai) {
241        count += zipDirectory(conf, zos, baseName, root, status.getPath());
242      }
243      LOG.info(String.format("Wrote %d entries for directory %s", count,
244        itemToZip));
245      return count;
246    }
247
248    final String inZipPath = relativePathForZipEntry(itemToZip.toUri()
249      .getPath(), baseName, root);
250
251    if (inZipPath.length() == 0) {
252      LOG.warn(String.format("Skipping empty zip file path for %s (%s %s)",
253        itemToZip, root, baseName));
254      return 0;
255    }
256
257    // Take empty files in case the place holder is needed
258    FSDataInputStream in = null;
259    try {
260      in = localFs.open(itemToZip);
261      final ZipEntry ze = new ZipEntry(inZipPath);
262      ze.setTime(itemStatus.getModificationTime());
263      // Comments confuse looking at the zip file
264      // ze.setComment(itemToZip.toString());
265      zos.putNextEntry(ze);
266
267      IOUtils.copyBytes(in, zos, conf, false);
268      zos.closeEntry();
269      LOG.info(String.format("Wrote %d entries for file %s", count, itemToZip));
270      return 1;
271    } finally {
272      in.close();
273    }
274
275  }
276
277  static String relativePathForZipEntry(final String rawPath,
278    final String baseName, final String root) {
279    String relativePath = rawPath.replaceFirst(Pattern.quote(root.toString()),
280      "");
281    LOG.info(String.format("RawPath %s, baseName %s, root %s, first %s",
282      rawPath, baseName, root, relativePath));
283
284    if (relativePath.startsWith(Path.SEPARATOR)) {
285      relativePath = relativePath.substring(1);
286    }
287    LOG.info(String.format(
288      "RawPath %s, baseName %s, root %s, post leading slash %s", rawPath,
289      baseName, root, relativePath));
290    if (relativePath.isEmpty()) {
291      LOG.warn(String.format(
292        "No data after root (%s) removal from raw path %s", root, rawPath));
293      return baseName;
294    }
295    // Construct the path that will be written to the zip file, including
296    // removing any leading '/' characters
297    String inZipPath = baseName + Path.SEPARATOR_CHAR + relativePath;
298
299    LOG.info(String.format("RawPath %s, baseName %s, root %s, inZip 1 %s",
300      rawPath, baseName, root, inZipPath));
301    if (inZipPath.startsWith(Path.SEPARATOR)) {
302      inZipPath = inZipPath.substring(1);
303    }
304    LOG.info(String.format("RawPath %s, baseName %s, root %s, inZip 2 %s",
305      rawPath, baseName, root, inZipPath));
306
307    return inZipPath;
308
309  }

318  private Path findSolrConfig(Configuration conf) throws IOException {
319    Path solrHome = null;
320    Path[] localArchives = DistributedCache.getLocalCacheArchives(conf);
321    if (localArchives.length == 0) {
322      throw new IOException(String.format(
323        "No local cache archives, where is %s:%s", SolrOutputFormat
324          .getSetupOk(), SolrOutputFormat.getZipName(conf)));
325    }
326    for (Path unpackedDir : localArchives) {
327      // Only logged if debugging
328      if (LOG.isDebugEnabled()) {
329        LOG.debug(String.format("Examining unpack directory %s for %s",
330          unpackedDir, SolrOutputFormat.getZipName(conf)));
331
332        ProcessBuilder lsCmd = new ProcessBuilder(new String[] {"/bin/ls",
333          "-lR", unpackedDir.toString()});
334        lsCmd.redirectErrorStream();
335        Process ls = lsCmd.start();
336        try {
337          byte[] buf = new byte[16 * 1024];
338          InputStream all = ls.getInputStream();
339          int count;
340          while ((count = all.read(buf)) > 0) {
341            System.err.write(buf, 0, count);
342          }
343        } catch (IOException ignore) {
344        }
345        System.err.format("Exit value is %d%n", ls.exitValue());
346      }
347      if (unpackedDir.getName().equals(SolrOutputFormat.getZipName(conf))) {
348
349        solrHome = unpackedDir;
350        break;
351      }
352    }
353    return solrHome;
354  }

356  private String getOutFileName(TaskAttemptContext context, String prefix) {
357    TaskID taskId = context.getTaskAttemptID().getTaskID();
358    int partition = taskId.getId();
359    NumberFormat nf = NumberFormat.getInstance();
360    nf.setMinimumIntegerDigits(5);
361    nf.setGroupingUsed(false);
362    StringBuilder result = new StringBuilder();
363    result.append(prefix);
364    result.append("-");
365    result.append(nf.format(partition));
366    return result.toString();
367  }
