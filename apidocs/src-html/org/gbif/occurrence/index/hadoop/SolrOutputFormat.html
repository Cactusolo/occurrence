149  public static void setupSolrHomeCache(File solrHome, Configuration jobConf)
150    throws IOException {
151    if (solrHome == null || !(solrHome.exists() && solrHome.isDirectory())) {
152      throw new IOException("Invalid solr.home: " + solrHome);
153    }
154    File tmpZip = File.createTempFile("solr", "zip");
155    createZip(solrHome, tmpZip);
156    // Make a reasonably unique name for the zip file in the distributed cache
157    // to avoid collisions if multiple jobs are running.
158    String hdfsZipName = UUID.randomUUID().toString() + '.'
159      + ZIP_FILE_BASE_NAME;
160    jobConf.set(ZIP_NAME, hdfsZipName);
161
162    Path zipPath = new Path("/tmp", getZipName(jobConf));
163    FileSystem fs = FileSystem.get(jobConf);
164    fs.copyFromLocalFile(new Path(tmpZip.toString()), zipPath);
165    final URI baseZipUrl = fs.getUri().resolve(
166      zipPath.toString() + '#' + getZipName(jobConf));
167
168    DistributedCache.addCacheArchive(baseZipUrl, jobConf);
169    LOG.info("Set Solr cache: "
170      + Arrays.asList(DistributedCache.getCacheArchives(jobConf)));
171    // Actually send the path for the configuration zip file
172    jobConf.set(SETUP_OK, zipPath.toString());
173  }
174
175  private static void createZip(File dir, File out) throws IOException {
176    HashSet<File> files = new HashSet<File>();
177    // take only conf/ and lib/
178    for (String allowedDirectory : SolrRecordWriter
179      .getAllowedConfigDirectories()) {
180      File configDir = new File(dir, allowedDirectory);
181      boolean configDirExists;
182      /** If the directory does not exist, and is required, bail out */
183      if (!(configDirExists = configDir.exists())
184        && SolrRecordWriter.isRequiredConfigDirectory(allowedDirectory)) {
185        throw new IOException(String.format(
186          "required configuration directory %s is not present in %s",
187          allowedDirectory, dir));
188      }
189      if (!configDirExists) {
190        continue;
191      }
192      listFiles(configDir, files); // Store the files in the existing, allowed
193                                   // directory configDir, in the list of files
194                                   // to store in the zip file
195    }
196
197    out.delete();
198    int subst = dir.toString().length();
199    ZipOutputStream zos = new ZipOutputStream(new FileOutputStream(out));
200    byte[] buf = new byte[1024];
201    for (File f : files) {
202      ZipEntry ze = new ZipEntry(f.toString().substring(subst));
203      zos.putNextEntry(ze);
204      InputStream is = new FileInputStream(f);
205      int cnt;
206      while ((cnt = is.read(buf)) >= 0) {
207        zos.write(buf, 0, cnt);
208      }
209      is.close();
210      zos.flush();
211      zos.closeEntry();
212    }
213    zos.close();
214  }
215
216  private static void listFiles(File dir, Set<File> files) throws IOException {
217    File[] list = dir.listFiles();
218    for (File f : list) {
219      if (f.isFile()) {
220        files.add(f);
221      } else {
222        listFiles(f, files);
223      }
224    }
225  }

232    }
