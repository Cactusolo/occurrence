062/**
063 * Instantiate a record writer that will build a Solr index.
064 * A zip file containing the solr config and additional libraries is expected to
065 * be passed via the distributed cache. The incoming written records are
066 * converted via the specified document converter, and written to the index in
067 * batches. When the job is done, the close copies the index to the destination
068 * output file system. <h2>Configuration Parameters</h2>
069 * <ul>
070 * <li>solr.record.writer.batch.size - the number of documents in a batch that is sent to the indexer.</li>
071 * <li>mapred.task.id - To build the unique temporary index directory file name.</li>
072 * <li>solr.output.format.setup - {@link SolrOutputFormat.SETUP_OK} The path to the configuration zip file.</li>
073 * <li> {@link SolrOutputFormat.zipName} - The file name of the configuration file.</li>
074 * <li>solr.document.converter.class - {@link SolrDocumentConverter.CONVERTER_NAME_KEY} the class used to convert the
075 * {@link #write} key, values into a {@link SolrInputDocument}. Set via {@link SolrDocumentConverter}.
076 * </ul>
077 */
078public class SolrRecordWriter<K, V> extends RecordWriter<K, V> {

129  @SuppressWarnings("unchecked")
130  public SolrRecordWriter(TaskAttemptContext context) {
131    conf = context.getConfiguration();
132    batchSize = SolrOutputFormat.getBatchSize(conf);
133
134    heartBeater = new HeartBeater(context);
135    try {
136      heartBeater.needHeartBeat();
137      /** The actual file in hdfs that holds the configuration. */
138
139      final String configuredSolrConfigPath = conf.get(SolrOutputFormat.SETUP_OK);
140      if (configuredSolrConfigPath == null) {
141        throw new IllegalStateException(String.format(
142          "The job did not pass %s", SolrOutputFormat.SETUP_OK));
143      }
144      outputZipFile = SolrOutputFormat.isOutputZipFormat(conf);
145
146      this.fs = FileSystem.get(conf);
147      perm = new Path(FileOutputFormat.getOutputPath(context), getOutFileName(context, "part"));
148
149      // Make a task unique name that contains the actual index output name to
150      // make debugging simpler
151      // Note: if using JVM reuse, the sequence number will not be reset for a
152      // new task using the jvm
153
154      temp = conf.getLocalPath("mapred.local.dir", "solr_" +
155        conf.get("mapred.task.id") + '.' + sequence.incrementAndGet());
156
157      if (outputZipFile && !perm.getName().endsWith(".zip")) {
158        perm = perm.suffix(".zip");
159      }
160      fs.delete(perm, true); // delete old, if any
161      Path local = fs.startLocalOutput(perm, temp);
162
163      solrHome = findSolrConfig(conf);
164
165      // }
166      // Verify that the solr home has a conf and lib directory
167      if (solrHome == null) {
168        throw new IOException("Unable to find solr home setting");
169      }
170
171      // Setup a solr instance that we can batch writes to
172      LOG.info("SolrHome: " + solrHome.toUri());
173      String dataDir = new File(local.toString(), "data").toString();
174      // copy the schema to the conf dir
175      File confDir = new File(local.toString(), "conf");
176      confDir.mkdirs();
177      File srcSchemaFile = new File(solrHome.toString(), "conf/schema.parsing");
178      assert srcSchemaFile.exists();
179      FileUtils.copyFile(srcSchemaFile, new File(confDir, "schema.parsing"));
180      Properties props = new Properties();
181      props.setProperty("solr.data.dir", dataDir);
182      props.setProperty("solr.home", solrHome.toString());
183      SolrResourceLoader loader = new SolrResourceLoader(solrHome.toString(),
184        null, props);
185      LOG
186        .info(String
187          .format(
188            "Constructed instance information solr.home %s (%s), instance dir %s, conf dir %s, writing index to temporary directory %s, with permdir %s",
189            solrHome, solrHome.toUri(), loader.getInstanceDir(), loader
190              .getConfigDir(), dataDir, perm));
191      CoreContainer container = new CoreContainer(loader);
192      CoreDescriptor descr = new CoreDescriptor(container, "core1", solrHome
193        .toString());
194      descr.setDataDir(dataDir);
195      descr.setCoreProperties(props);
196      core = container.create(descr);
197      container.register(core, false);
198      solr = new EmbeddedSolrServer(container, "core1");
199      batchWriter = new BatchWriter(solr, batchSize,
200        context.getTaskAttemptID().getTaskID(),
201        SolrOutputFormat.getSolrWriterThreadCount(conf),
202        SolrOutputFormat.getSolrWriterQueueSize(conf));
203
204      // instantiate the converter
205      String className = SolrDocumentConverter.getSolrDocumentConverter(conf);
206      Class<? extends SolrDocumentConverter> cls = (Class<? extends SolrDocumentConverter>) Class
207        .forName(className);
208      converter = ReflectionUtils.newInstance(cls, conf);
209    } catch (Exception e) {
210      throw new IllegalStateException(String.format(
211        "Failed to initialize record writer for %s, %s", context.getJobName(), conf
212          .get("mapred.task.id")), e);
213    } finally {
214      heartBeater.cancelHeartBeat();
215    }
216  }

253  /**
254   * Write a file to a zip output stream, removing leading path name components
255   * from the actual file name when creating the zip file entry.
256   * The entry placed in the zip file is <code>baseName</code>/ <code>relativePath</code>, where
257   * <code>relativePath</code> is constructed
258   * by removing a leading <code>root</code> from the path for <code>itemToZip</code>.
259   * If <code>itemToZip</code> is an empty directory, it is ignored. If <code>itemToZip</code> is a directory, the
260   * contents of the directory are
261   * added recursively.
262   *
263   * @param zos The zip output stream
264   * @param baseName The base name to use for the file name entry in the zip
265   *        file
266   * @param root The path to remove from <code>itemToZip</code> to make a
267   *        relative path name
268   * @param itemToZip The path to the file to be added to the zip file
269   * @return the number of entries added
270   * @throws IOException
271   */
272  static public int zipDirectory(final Configuration conf,
273    final ZipOutputStream zos, final String baseName, final String root,
274    final Path itemToZip) throws IOException {
275    LOG
276      .info(String
277        .format("zipDirectory: %s %s %s", baseName, root, itemToZip));
278    LocalFileSystem localFs = FileSystem.getLocal(conf);
279    int count = 0;
280
281    final FileStatus itemStatus = localFs.getFileStatus(itemToZip);
282    if (itemStatus.isDir()) {
283      final FileStatus[] statai = localFs.listStatus(itemToZip);
284
285      // Add a directory entry to the zip file
286      final String zipDirName = relativePathForZipEntry(itemToZip.toUri()
287        .getPath(), baseName, root);
288      final ZipEntry dirZipEntry = new ZipEntry(zipDirName
289        + Path.SEPARATOR_CHAR);
290      LOG.info(String.format("Adding directory %s to zip", zipDirName));
291      zos.putNextEntry(dirZipEntry);
292      zos.closeEntry();
293      count++;
294
295      if (statai == null || statai.length == 0) {
296        LOG.info(String.format("Skipping empty directory %s", itemToZip));
297        return count;
298      }
299      for (FileStatus status : statai) {
300        count += zipDirectory(conf, zos, baseName, root, status.getPath());
301      }
302      LOG.info(String.format("Wrote %d entries for directory %s", count,
303        itemToZip));
304      return count;
305    }
306
307    final String inZipPath = relativePathForZipEntry(itemToZip.toUri()
308      .getPath(), baseName, root);
309
310    if (inZipPath.length() == 0) {
311      LOG.warn(String.format("Skipping empty zip file path for %s (%s %s)",
312        itemToZip, root, baseName));
313      return 0;
314    }
315
316    // Take empty files in case the place holder is needed
317    FSDataInputStream in = null;
318    try {
319      in = localFs.open(itemToZip);
320      final ZipEntry ze = new ZipEntry(inZipPath);
321      ze.setTime(itemStatus.getModificationTime());
322      // Comments confuse looking at the zip file
323      // ze.setComment(itemToZip.toString());
324      zos.putNextEntry(ze);
325
326      IOUtils.copyBytes(in, zos, conf, false);
327      zos.closeEntry();
328      LOG.info(String.format("Wrote %d entries for file %s", count, itemToZip));
329      return 1;
330    } finally {
331      in.close();
332    }
333
334  }

336  static String relativePathForZipEntry(final String rawPath,
337    final String baseName, final String root) {
338    String relativePath = rawPath.replaceFirst(Pattern.quote(root.toString()),
339      "");
340    LOG.info(String.format("RawPath %s, baseName %s, root %s, first %s",
341      rawPath, baseName, root, relativePath));
342
343    if (relativePath.startsWith(Path.SEPARATOR)) {
344      relativePath = relativePath.substring(1);
345    }
346    LOG.info(String.format(
347      "RawPath %s, baseName %s, root %s, post leading slash %s", rawPath,
348      baseName, root, relativePath));
349    if (relativePath.isEmpty()) {
350      LOG.warn(String.format(
351        "No data after root (%s) removal from raw path %s", root, rawPath));
352      return baseName;
353    }
354    // Construct the path that will be written to the zip file, including
355    // removing any leading '/' characters
356    String inZipPath = baseName + Path.SEPARATOR_CHAR + relativePath;
357
358    LOG.info(String.format("RawPath %s, baseName %s, root %s, inZip 1 %s",
359      rawPath, baseName, root, inZipPath));
360    if (inZipPath.startsWith(Path.SEPARATOR)) {
361      inZipPath = inZipPath.substring(1);
362    }
363    LOG.info(String.format("RawPath %s, baseName %s, root %s, inZip 2 %s",
364      rawPath, baseName, root, inZipPath));
365
366    return inZipPath;
367
368  }

370  @Override
371  public void close(TaskAttemptContext context) throws IOException, InterruptedException {
372    if (context != null) {
373      heartBeater.setProgress(context);
374    }
375    try {
376      heartBeater.needHeartBeat();
377      batchWriter.close(context, core);
378      if (outputZipFile) {
379        context.setStatus("Writing Zip");
380        packZipFile(); // Written to the perm location
381      } else {
382        context.setStatus("Copying Index");
383        fs.completeLocalOutput(perm, temp); // copy to dfs
384      }
385    } catch (Exception e) {
386      if (e instanceof IOException) {
387        throw (IOException) e;
388      }
389      throw new IOException(e);
390    } finally {
391      heartBeater.cancelHeartBeat();
392      File tempFile = new File(temp.toString());
393      if (tempFile.exists()) {
394        FileUtils.forceDelete(new File(temp.toString()));
395      }
396    }
397
398    context.setStatus("Done");
399  }

401  /**
402   * Write a record. This method accumulates records in to a batch, and when {@link #batchSize} items are present
403   * flushes it to the indexer. The writes
404   * can take a substantial amount of time, depending on {@link #batchSize}. If
405   * there is heavy disk contention the writes may take more than the 600 second
406   * default timeout.
407   */
408  @Override
409  public void write(K key, V value) throws IOException {
410    if (isClosing()) {
411      throw new IOException("Index is already closing");
412    }
413    heartBeater.needHeartBeat();
414    try {
415      try {
416        Collection<SolrInputDocument> docs = converter.convert(key, value);
417        if (docs.size() > batchSize) {
418          ArrayList<SolrInputDocument> oneBatch = new ArrayList<SolrInputDocument>(
419            batchSize);
420          Iterator<SolrInputDocument> iterator = docs.iterator();
421          // Send the documents to the actual writer in batchSize chunks
422          for (int inBatch = 0; iterator.hasNext(); inBatch++) {
423            /** Flush the batch if it is the full size. */
424            if (inBatch == batchSize) {
425              batchWriter.queueBatch(oneBatch);
426              oneBatch.clear();
427              inBatch = 0;
428            }
429            oneBatch.add(iterator.next());
430          }
431          if (!oneBatch.isEmpty()) {
432            batchWriter.queueBatch(oneBatch);
433          }
434        } else {
435          batchWriter.queueBatch(docs);
436        }
437      } catch (SolrServerException e) {
438        throw new IOException(e);
439      }
440    } finally {
441      heartBeater.cancelHeartBeat();
442    }
443
444  }

454  private Path findSolrConfig(Configuration conf) throws IOException {
455    Path solrHome = null;
456    Path[] localArchives = DistributedCache.getLocalCacheArchives(conf);
457    if (localArchives.length == 0) {
458      throw new IOException(String.format(
459        "No local cache archives, where is %s:%s", SolrOutputFormat
460          .getSetupOk(), SolrOutputFormat.getZipName(conf)));
461    }
462    for (Path unpackedDir : localArchives) {
463      // Only logged if debugging
464      if (LOG.isDebugEnabled()) {
465        LOG.debug(String.format("Examining unpack directory %s for %s",
466          unpackedDir, SolrOutputFormat.getZipName(conf)));
467
468        ProcessBuilder lsCmd = new ProcessBuilder(new String[] {"/bin/ls",
469          "-lR", unpackedDir.toString()});
470        lsCmd.redirectErrorStream();
471        Process ls = lsCmd.start();
472        try {
473          byte[] buf = new byte[16 * 1024];
474          InputStream all = ls.getInputStream();
475          int count;
476          while ((count = all.read(buf)) > 0) {
477            System.err.write(buf, 0, count);
478          }
479        } catch (IOException ignore) {
480        }
481        System.err.format("Exit value is %d%n", ls.exitValue());
482      }
483      if (unpackedDir.getName().equals(SolrOutputFormat.getZipName(conf))) {
484
485        solrHome = unpackedDir;
486        break;
487      }
488    }
489    return solrHome;
490  }

492  private String getOutFileName(TaskAttemptContext context, String prefix) {
493    TaskID taskId = context.getTaskAttemptID().getTaskID();
494    int partition = taskId.getId();
495    NumberFormat nf = NumberFormat.getInstance();
496    nf.setMinimumIntegerDigits(5);
497    nf.setGroupingUsed(false);
498    StringBuilder result = new StringBuilder();
499    result.append(prefix);
500    result.append("-");
501    result.append(nf.format(partition));
502    return result.toString();
503  }

505  private void packZipFile() throws IOException {
506    FSDataOutputStream out = null;
507    ZipOutputStream zos = null;
508    int zipCount = 0;
509    LOG.info("Packing zip file for " + perm);
510    try {
511      out = fs.create(perm, false);
512      zos = new ZipOutputStream(out);
513
514      String name = perm.getName().replaceAll(".zip$", "");
515      LOG.info("adding index directory" + temp);
516      zipCount = zipDirectory(conf, zos, name, temp.toString(), temp);
517      /**
518       * for (String configDir : allowedConfigDirectories) {
519       * if (!isRequiredConfigDirectory(configDir)) {
520       * continue;
521       * }
522       * final Path confPath = new Path(solrHome, configDir);
523       * LOG.info("adding configdirectory" + confPath);
524       * zipCount += zipDirectory(conf, zos, name, solrHome.toString(), confPath);
525       * }
526       **/
527    } catch (Throwable ohFoo) {
528      LOG.error("packZipFile exception", ohFoo);
529      if (ohFoo instanceof RuntimeException) {
530        throw (RuntimeException) ohFoo;
531      }
532      if (ohFoo instanceof IOException) {
533        throw (IOException) ohFoo;
534      }
535      throw new IOException(ohFoo);
536
537    } finally {
538      if (zos != null) {
539        if (zipCount == 0) { // If no entries were written, only close out, as
540                             // the zip will throw an error
541          LOG.error("No entries written to zip file " + perm);
542          fs.delete(perm, false);
543          // out.close();
544        } else {
545          LOG.info(String.format("Wrote %d items to %s for %s", zipCount, perm,
546            temp));
547          zos.close();
548        }
549      }
550    }
551  }
