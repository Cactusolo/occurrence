001package org.gbif.occurrence.index.hadoop;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.io.IOException;
006
007import org.apache.solr.client.solrj.SolrServer;
008import org.apache.solr.client.solrj.embedded.EmbeddedSolrServer;
009import org.apache.solr.core.CoreContainer;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013
014public class EmbeddedSolrServerBuilder {
015
016  private static final Logger LOG = LoggerFactory.getLogger(EmbeddedSolrServerBuilder.class);
017
018  /**
019   * Builds a SolrServer using a home directory and data directory.
020   * 
021   * @param solrHome
022   * @return
023   */
024  public static EmbeddedSolrServer build(String solrHome) {
025    try {
026      File solrHomeFile = new File(solrHome);
027      // File solrDataDir = new File(dataDir);
028      if (!solrHomeFile.exists()) {
029        throw new FileNotFoundException(solrHome);
030      }
031      String solrHomePath = solrHomeFile.getAbsolutePath();
032      LOG.info(String.format("Constructed instance information solr.home %s", solrHomePath));
033      CoreContainer container = new CoreContainer(solrHomePath);
034      container.load();
035      System.out.println(container.getAllCoreNames());
036      return new EmbeddedSolrServer(container, "");
037    } catch (IOException e) {
038      throw new IllegalStateException(e);
039    }
040  }
041
042  public static void main(String[] args) {
043    SolrServer server = EmbeddedSolrServerBuilder.build("solr/");
044    System.out.println(server.toString());
045  }
046}
