001package org.gbif.occurrence.index.hadoop;
002
003import java.io.IOException;
004
005import org.apache.hadoop.io.LongWritable;
006import org.apache.hadoop.io.Text;
007import org.apache.hadoop.mapreduce.Mapper;
008import org.apache.solr.client.solrj.SolrServerException;
009import org.slf4j.Logger;
010import org.slf4j.LoggerFactory;
011
012
013public class IndexerMapper extends Mapper<LongWritable, Text, LongWritable, Text> {
014
015  private SolrServerFactory solrServerFactory;
016  private OccurrenceDocConverter occurrenceDocConverter;
017  private static final Logger LOG = LoggerFactory.getLogger(IndexerMapper.class);
018
019  /*
020   * (non-Javadoc)
021   * @see org.apache.hadoop.mapreduce.Mapper#cleanup(org.apache.hadoop.mapreduce.Mapper.Context)
022   */
023  @Override
024  protected void cleanup(Context context) throws IOException, InterruptedException {
025    super.cleanup(context);
026  }
027
028  /*
029   * (non-Javadoc)
030   * @see org.apache.hadoop.mapreduce.Mapper#map(java.lang.Object, java.lang.Object,
031   * org.apache.hadoop.mapreduce.Mapper.Context)
032   */
033  @Override
034  protected void map(LongWritable key, Text value, Context context) throws IOException, InterruptedException {
035    while (context.nextKeyValue()) {
036      try {
037        solrServerFactory.getSolr().add(
038          occurrenceDocConverter.convert(context.getCurrentKey(), context.getCurrentValue()));
039      } catch (SolrServerException e) {
040        LOG.error("Error adding document", e);
041      }
042    }
043  }
044
045  /*
046   * (non-Javadoc)
047   * @see org.apache.hadoop.mapreduce.Mapper#setup(org.apache.hadoop.mapreduce.Mapper.Context)
048   */
049  @Override
050  protected void setup(Context context) throws IOException, InterruptedException {
051    super.setup(context);
052    solrServerFactory = new SolrServerFactory(context);
053    occurrenceDocConverter = new OccurrenceDocConverter();
054  }
055}
