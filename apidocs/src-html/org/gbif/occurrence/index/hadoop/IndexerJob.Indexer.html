001package org.gbif.occurrence.index.hadoop;
002
003
004/**
005 * Mapreduce job to create a lucene index, this class implements the MapRunnable interfaces
006 * in order to avoid hadoop kill the tasks that are waiting for the lucene optimization
007 */
008
009import java.io.File;
010import java.io.IOException;
011import java.util.Random;
012
013import org.apache.hadoop.conf.Configuration;
014import org.apache.hadoop.fs.FileStatus;
015import org.apache.hadoop.fs.FileSystem;
016import org.apache.hadoop.fs.FileUtil;
017import org.apache.hadoop.fs.Path;
018import org.apache.hadoop.io.LongWritable;
019import org.apache.hadoop.io.Text;
020import org.apache.hadoop.mapred.JobClient;
021import org.apache.hadoop.mapred.JobConf;
022import org.apache.hadoop.mapred.MapRunnable;
023import org.apache.hadoop.mapred.OutputCollector;
024import org.apache.hadoop.mapred.RecordReader;
025import org.apache.hadoop.mapred.Reporter;
026import org.apache.hadoop.mapred.SequenceFileInputFormat;
027import org.apache.hadoop.mapred.TextOutputFormat;
028import org.apache.lucene.analysis.Analyzer;
029import org.apache.lucene.analysis.standard.StandardAnalyzer;
030import org.apache.lucene.document.Document;
031import org.apache.lucene.document.Field;
032import org.apache.lucene.document.StringField;
033import org.apache.lucene.index.IndexWriter;
034import org.apache.lucene.index.IndexWriterConfig;
035import org.apache.lucene.store.Directory;
036import org.apache.lucene.store.FSDirectory;
037import org.apache.lucene.util.Version;
038
039/**
040 * Implementation of a indexer as hadoop map reduce job.
041 */
042public class IndexerJob {
043
044  public static class Indexer implements MapRunnable<LongWritable, Text, Text, Text> {
045
046    private JobConf _conf;
047
048    public static final String INDEXER_ANALYZER = "indexer.analyzer";
049
050
051    public static Analyzer getAnalyzer(Configuration conf) {
052      Analyzer analyzer = null;
053      try {
054        analyzer = (Analyzer) Class.forName(
055          conf.get(INDEXER_ANALYZER, "org.apache.lucene.analysis.standard.StandardAnalyzer"))
056          .newInstance();
057      } catch (InstantiationException e1) {
058        e1.printStackTrace();
059      } catch (IllegalAccessException e1) {
060        e1.printStackTrace();
061      } catch (ClassNotFoundException e1) {
062        e1.printStackTrace();
063      }
064      return analyzer;
065    }
066
067
068    public void configure(JobConf conf) {
069      _conf = conf;
070
071    }
072
073    public void run(RecordReader<LongWritable, Text> reader, OutputCollector<Text, Text> output, final Reporter report)
074      throws IOException {
075      LongWritable key = reader.createKey();
076      Text value = reader.createValue();
077
078      String tmp = _conf.get("hadoop.tmp.dir");
079      long millis = System.currentTimeMillis();
080      String shardName = "" + millis + "-" + new Random().nextInt();
081      File file = new File(tmp, shardName);
082      report.progress();
083      Analyzer analyzer = getAnalyzer(_conf);
084      Directory directory = FSDirectory.open(file);
085      IndexWriterConfig idxConfig = new IndexWriterConfig(Version.LUCENE_41, new StandardAnalyzer(Version.LUCENE_41));
086      IndexWriter indexWriter = new IndexWriter(directory, idxConfig);
087      report.setStatus("Adding documents...");
088      while (reader.next(key, value)) {
089        report.progress();
090        Document doc = new Document();
091        String text = "" + value.toString();
092        Field contentField = new Field("content", text, StringField.TYPE_STORED);
093        doc.add(contentField);
094        indexWriter.addDocument(doc);
095      }
096
097      report.setStatus("Done adding documents.");
098      Thread t = new Thread() {
099
100        public boolean stop = false;
101
102        @Override
103        public void run() {
104          while (!stop) {
105            // Makes sure hadoop is not killing the task in case the
106            // optimization
107            // takes longer than the task timeout.
108            report.progress();
109            try {
110              sleep(10000);
111            } catch (InterruptedException e) {
112              // don't need to do anything.
113              stop = true;
114            }
115          }
116        }
117      };
118      t.start();
119      report.setStatus("Optimizing index...");
120      report.setStatus("Done optimizing!");
121      report.setStatus("Closing index...");
122      indexWriter.close();
123      report.setStatus("Closing done!");
124      FileSystem fileSystem = FileSystem.get(_conf);
125
126      report.setStatus("Starting copy to final destination...");
127      Path destination = new Path(_conf.get("finalDestination"));
128      fileSystem.copyFromLocalFile(new Path(file.getAbsolutePath()), destination);
129      report.setStatus("Copy to final destination done!");
130      report.setStatus("Deleting tmp files...");
131      FileUtil.fullyDelete(file);
132      report.setStatus("Delteing tmp files done!");
133      t.interrupt();
134    }
135  }
136
137  public static void main(String[] args) throws IOException {
138
139    if (args.length != 3) {
140      String usage = "IndexerJob <in text file/dir> <index dir> <numOfShards>";
141      System.out.println(usage);
142      System.exit(1);
143    }
144
145    IndexerJob indexerJob = new IndexerJob();
146    String input = args[0];
147    String output = args[1];
148    int numOfShards = Integer.parseInt(args[2]);
149    indexerJob.startIndexer(input, output, numOfShards);
150
151  }
152
153  public void startIndexer(String path, String finalDestination, int numOfShards) throws IOException {
154    // create job conf with class pointing into job jar.
155    JobConf jobConf = new JobConf(IndexerJob.class);
156    jobConf.setJobName("indexer");
157    jobConf.setMapRunnerClass(Indexer.class);
158    // alternative use a text file and a TextInputFormat
159    jobConf.setInputFormat(SequenceFileInputFormat.class);
160
161    Path input = new Path(path);
162    SequenceFileInputFormat.setInputPaths(jobConf, input);
163    // the output path is set just is because is required
164    TextOutputFormat.setOutputPath(jobConf, new Path(finalDestination));
165    // lucene indexes folder
166    jobConf.set("finalDestination", finalDestination);
167    // important to switch spec exec off to avoid duplicated documents
168    jobConf.setSpeculativeExecution(false);
169
170    // The num of map tasks is equal to the num of input splits.
171    // The num of input splits by default is equal to the num of hdf blocks
172    // for the input file(s). To get the right num of shards we need to
173    // calculate the best input split size.
174
175    FileSystem fs = FileSystem.get(input.toUri(), jobConf);
176    FileStatus[] status = fs.globStatus(input);
177    long size = 0;
178    for (FileStatus fileStatus : status) {
179      size += fileStatus.getLen();
180    }
181    long optimalSplisize = size / numOfShards;
182    jobConf.set("mapred.min.split.size", "" + optimalSplisize);
183
184    // give more mem to lucene tasks.
185    jobConf.set("mapred.child.java.opts", "-Xmx2G");
186    jobConf.setNumMapTasks(1);
187    jobConf.setNumReduceTasks(0);
188    JobClient.runJob(jobConf);
189  }
190}
