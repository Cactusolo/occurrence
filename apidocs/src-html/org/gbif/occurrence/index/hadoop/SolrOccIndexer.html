001package org.gbif.occurrence.index.hadoop;
002
003
004import java.io.File;
005import java.io.IOException;
006
007import org.apache.hadoop.conf.Configuration;
008import org.apache.hadoop.conf.Configured;
009import org.apache.hadoop.fs.Path;
010import org.apache.hadoop.io.LongWritable;
011import org.apache.hadoop.io.Text;
012import org.apache.hadoop.mapreduce.Job;
013import org.apache.hadoop.mapreduce.lib.input.FileInputFormat;
014import org.apache.hadoop.mapreduce.lib.output.FileOutputFormat;
015import org.apache.hadoop.util.Tool;
016import org.apache.hadoop.util.ToolRunner;
017
018public class SolrOccIndexer extends Configured implements Tool {
019
020  public static void main(String[] args) throws Exception {
021    Configuration conf = new Configuration();
022    int res = ToolRunner.run(conf, new SolrOccIndexer(), args);
023    System.exit(res);
024  }
025
026  public int run(String[] args) throws Exception {
027    if (args.length < 2) {
028      System.err
029        .println("Usage: SolrOccIndexer <outputDir> -solr <solrHome> <inputDir> [<inputDir2> ...] [-shards NNN] [-compress_output]");
030      System.err.println("\tinputDir\tinput directory(-ies) containing occurrence files");
031      System.err.println("\toutputDir\toutput directory containing Solr indexes.");
032      System.err.println("\tsolr <solrHome>\tlocal directory containing Solr conf/ and lib/");
033      System.err.println("\tshards NNN\tset the number of output shards to NNN");
034      System.err.println("\t\t(default: the default number of reduce tasks)");
035      System.err.println("\tcompress_output\tto compress the output of the reducer tasks (create .zip file)");
036      return -1;
037    }
038    Job job = new Job(getConf());
039    job.setJarByClass(SolrOccIndexer.class);
040
041    int shards = -1;
042    boolean compressOutput = false;
043    String solrHome = null;
044    Path out = new Path(args[0]);
045    for (int i = 1; i < args.length; i++) {
046      if (args[i] == null) {
047        continue;
048      }
049      if (args[i].equals("-shards")) {
050        shards = Integer.parseInt(args[++i]);
051      } else if (args[i].equals("-compress_output")) {
052        compressOutput = true;
053      } else if (args[i].equals("-solr")) {
054        solrHome = args[++i];
055        continue;
056      } else {
057        Path p = new Path(args[i]);
058        FileInputFormat.addInputPath(job, p);
059      }
060    }
061    if (solrHome == null || !new File(solrHome).exists()) {
062      throw new IOException("You must specify a valid solr.home directory!");
063    }
064    job.setMapperClass(IdentityMapper.class);
065    job.setReducerClass(SolrReducer.class);
066    job.setOutputFormatClass(SolrOutputFormat.class);
067    SolrOutputFormat.setupSolrHomeCache(new File(solrHome), job.getConfiguration());
068    if (shards > 0) {
069      job.setNumReduceTasks(shards);
070    }
071    job.setOutputKeyClass(LongWritable.class);
072    job.setOutputValueClass(Text.class);
073    SolrDocumentConverter.setSolrDocumentConverter(OccurrenceDocConverter.class, job.getConfiguration());
074
075    FileOutputFormat.setOutputPath(job, out);
076
077    SolrOutputFormat.setOutputZipFormat(compressOutput, job.getConfiguration());
078
079
080    return job.waitForCompletion(true) ? 0 : -1;
081  }
082}
