030/**
031 * Class that creates a csv file using an occurrence hbase table.
032 */
033public class CSVFileBuilder {
034
035  private static final Logger LOG = LoggerFactory.getLogger(CSVFileBuilder.class);
036
037  private static final String KEY_SOURCE_TABLE = "occurrence-index.sourceTable";
038  private static final String[] STR_ARR_TYPE = new String[0];
039  private final Properties props;
040
041
042  /**
043   * Default constructor.
044   * 
045   * @param props configuration properties
046   */
047  public CSVFileBuilder(Properties props) {
048    this.props = props;
049  }

051  /**
052   * Utility to build an API Occurrence from an HBase row.
053   * 
054   * @return A complete occurrence, or null
055   */
056  public static List<String> buildOccurrenceLine(@Nullable Result row) {
057    List<String> result = Lists.newArrayList();
058    if (row == null || row.isEmpty()) {
059      return null;
060    } else {
061      result.add(objectToString(Integer.toString(Bytes.toInt(row.getRow()))));
062      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_ALTITUDE)));
063      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_BASIS_OF_RECORD)));
064      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.CATALOG_NUMBER)));
065      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_CLASS_ID)));
066      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_CLASS)));
067      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.COLLECTION_CODE)));
068      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.DATA_PROVIDER_ID)));
069      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.DATA_RESOURCE_ID)));
070      result.add(objectToString(OccurrenceResultReader.getUuid(row, FieldName.DATASET_KEY)));
071      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_DEPTH)));
072      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.DWC_OCCURRENCE_ID)));
073      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_FAMILY)));
074      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_FAMILY_ID)));
075      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_GENUS)));
076      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_GENUS_ID)));
077      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_GEOSPATIAL_ISSUE)));
078      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.INSTITUTION_CODE)));
079      result.add(OccurrenceResultReader.getString(row, FieldName.I_ISO_COUNTRY_CODE));
080      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_KINGDOM)));
081      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_KINGDOM_ID)));
082      result.add(objectToString(OccurrenceResultReader.getDouble(row, FieldName.I_LATITUDE)));
083      result.add(objectToString(OccurrenceResultReader.getDouble(row, FieldName.I_LONGITUDE)));
084      result.add(objectToString(OccurrenceResultReader.getDate(row, FieldName.I_MODIFIED)));
085      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_MONTH)));
086      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_NUB_ID)));
087      result.add(objectToString(OccurrenceResultReader.getDate(row, FieldName.I_OCCURRENCE_DATE)));
088      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_ORDER)));
089      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_ORDER_ID)));
090      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_OTHER_ISSUE)));
091      result.add(objectToString(OccurrenceResultReader.getUuid(row, FieldName.OWNING_ORG_KEY)));
092      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_PHYLUM)));
093      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_PHYLUM_ID)));
094      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.RESOURCE_ACCESS_POINT_ID)));
095      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_SCIENTIFIC_NAME)));
096      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.I_SPECIES)));
097      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_SPECIES_ID)));
098      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_TAXONOMIC_ISSUE)));
099      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.UNIT_QUALIFIER)));
100      result.add(objectToString(OccurrenceResultReader.getInteger(row, FieldName.I_YEAR)));
101      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.LOCALITY)));
102      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.COUNTY)));
103      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.STATE_PROVINCE)));
104      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.CONTINENT_OCEAN)));
105      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.COLLECTOR_NAME)));
106      result.add(objectToString(OccurrenceResultReader.getString(row, FieldName.IDENTIFIER_NAME)));
107      result.add(objectToString(OccurrenceResultReader.getDate(row, FieldName.IDENTIFICATION_DATE)));
108      return result;
109    }
110  }

112  /**
113   * Main method.
114   * Requires 3 parameters: <hbaseInputTable> <solrHomeDir> <solrDataDir>.
115   * hbaseInputTable: name of the hbase table.
116   */
117  public static void main(String[] args) throws IOException {
118    if (args.length < 1) {
119      System.err
120        .println("Usage: SequentialOccurrenceIndexBuilder <hbaseInputTable>");
121      return;
122    }
123    Properties p = new Properties();
124    p.setProperty(KEY_SOURCE_TABLE, args[0]);
125    CSVFileBuilder builder = new CSVFileBuilder(p);
126    builder.build();
127  }
128
129
130  private static String objectToString(Object o) {
131    return o == null ? null : o.toString();
132  }
133
134  /**
135   * Builds the Occurrence Solr index.
136   */
137  private void build() throws IOException {
138    HTable table = null;
139    ResultScanner scanner = null;
140    CSVWriter writer = null;
141    int numDocs = 0;
142
143    // The doc instance is re-used
144
145    LOG.info("Initiating indexing job");
146    try {
147      Configuration conf = HBaseConfiguration.create();
148      table = new HTable(conf, props.getProperty(KEY_SOURCE_TABLE));
149      Scan scan = buildOccurrenceScan();
150      scan.setCaching(200);
151      scanner = table.getScanner(scan);
152      Iterator<Result> iter = scanner.iterator();
153      writer = new CSVWriter(new FileWriter("occurrences.csv"));
154      while (iter.hasNext() && numDocs < 10000) {
155        writer.writeNext(buildOccurrenceLine(iter.next()).toArray(STR_ARR_TYPE));
156        numDocs++;
157      }
158    } catch (Exception ex) {
159      LOG.error("A general error has occurred", ex);
160    } finally {
161      Closeables.closeQuietly(scanner);
162      Closeables.closeQuietly(table);
163      Closeables.closeQuietly(writer);
164      LOG.info("Indexing job has finished, # of documents indexed: {}", numDocs);
165    }
166  }
