001package org.gbif.occurrence.index.hbase;
002
003import org.gbif.occurrence.index.hadoop.EmbeddedSolrServerBuilder;
004
005import java.io.IOException;
006import java.util.Iterator;
007import java.util.Properties;
008
009import com.google.common.io.Closer;
010import org.apache.hadoop.conf.Configuration;
011import org.apache.hadoop.hbase.HBaseConfiguration;
012import org.apache.hadoop.hbase.client.HTable;
013import org.apache.hadoop.hbase.client.Result;
014import org.apache.hadoop.hbase.client.ResultScanner;
015import org.apache.hadoop.hbase.client.Scan;
016import org.apache.solr.client.solrj.SolrServer;
017import org.apache.solr.client.solrj.SolrServerException;
018import org.apache.solr.client.solrj.embedded.EmbeddedSolrServer;
019import org.apache.solr.common.SolrInputDocument;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022
023import static org.gbif.occurrence.index.hbase.IndexingUtils.addDocumentQuietly;
024import static org.gbif.occurrence.index.hbase.IndexingUtils.buildOccSolrDocument;
025import static org.gbif.occurrence.index.hbase.IndexingUtils.buildOccurrenceScan;
026
027
028/**
029 * Class that creates a Solr Index using reading sequentially a hbase table.
030 * The {@link SolrServer} instance is an embedded instance by default.
031 */
032public class SequentialOccurrenceIndexBuilder {
033
034  private static final Logger LOG = LoggerFactory.getLogger(SequentialOccurrenceIndexBuilder.class);
035
036  private static final String KEY_SOURCE_TABLE = "occurrence-index.sourceTable";
037  private static final String SOLR_HOME = "solr.home";
038  private static final String NR_OF_RECORDS = "nr_of_records";
039  private final Properties props;
040
041  /**
042   * Default constructor.
043   * 
044   * @param props configuration properties
045   */
046  public SequentialOccurrenceIndexBuilder(Properties props) {
047    this.props = props;
048  }
049
050  /**
051   * Main method.
052   * Requires 3 parameters: <hbaseInputTable> <solrHomeDir> <solrDataDir>.
053   * hbaseInputTable: name of the hbase table.
054   */
055  public static void main(String[] args) throws IOException {
056    if (args.length < 2) {
057      System.err
058        .println("Usage: SequentialOccurrenceIndexBuilder <hbaseInputTable> <solrHomeDir> <solrDataDir>");
059      return;
060    }
061    Properties p = new Properties();
062    p.setProperty(KEY_SOURCE_TABLE, args[0]);
063    p.setProperty(SOLR_HOME, args[1]);
064    if (args[2] != null) {
065      p.setProperty(NR_OF_RECORDS, args[2]);
066    }
067    SequentialOccurrenceIndexBuilder builder = new SequentialOccurrenceIndexBuilder(p);
068    builder.build();
069  }
070
071
072  /**
073   * Builds the Occurrence Solr index.
074   */
075  private void build() throws IOException {
076    HTable table = null;
077    ResultScanner scanner = null;
078    Closer closer = Closer.create();
079    SolrServer solrServer = EmbeddedSolrServerBuilder.build(props.getProperty(SOLR_HOME));
080    long nrOfRecords = -1;
081    if (props.containsKey(NR_OF_RECORDS)) {
082      nrOfRecords = Long.parseLong(props.getProperty(NR_OF_RECORDS));
083    }
084    int numDocs = 0;
085    // The doc instance is re-used
086    SolrInputDocument doc = new SolrInputDocument();
087    LOG.info("Initiating indexing job");
088    try {
089      Configuration conf = HBaseConfiguration.create();
090      table = new HTable(conf, props.getProperty(KEY_SOURCE_TABLE));
091      closer.register(table);
092      Scan scan = buildOccurrenceScan();
093      scan.setCaching(1200);
094      scanner = table.getScanner(scan);
095      closer.register(scanner);
096
097      Iterator<Result> iter = scanner.iterator();
098      while (iter.hasNext() && (nrOfRecords == -1 || numDocs < nrOfRecords)) {
099        buildOccSolrDocument(iter.next(), doc);
100        addDocumentQuietly(solrServer, doc);
101        numDocs++;
102      }
103    } catch (Exception ex) {
104      LOG.error("A general error has occurred", ex);
105      LOG.error("Last occurrence record visited", doc.getFieldValue("key"));
106    } finally {
107      closer.close();
108      LOG.info("Commiting {} documents", numDocs);
109      commitQuietly(solrServer);
110      LOG.info("Commit has finished");
111      shutdownSolrServer(solrServer);
112      LOG.info("Indexing job has finished, # of documents indexed: {}", numDocs);
113    }
114  }
115
116  /**
117   * Commits changes to the solr server.
118   * Exceptions {@link SolrServerException} and {@link IOException} are swallowed.
119   */
120  private void commitQuietly(SolrServer solrServer) {
121    try {
122      solrServer.commit();
123    } catch (SolrServerException e) {
124      LOG.error("Solr error commiting on Solr server", e);
125    } catch (IOException e) {
126      LOG.error("I/O error commiting on Solr server", e);
127    }
128  }
129
130  /**
131   * Shutdowns the Solr server.
132   * This methods works only if server is an {@link EmbeddedSolrServer}.
133   */
134  private void shutdownSolrServer(SolrServer solrServer) {
135    if (solrServer instanceof EmbeddedSolrServer) {
136      ((EmbeddedSolrServer) solrServer).shutdown();
137    }
138  }
139}
