070  /**
071   * Adds the solr document to instance managed by the solr server.
072   * Exceptions {@link SolrServerException} and {@link IOException} are swallowed.
073   */
074  public static void addDocumentQuietly(SolrServer solrServer, SolrInputDocument doc) {
075    try {
076      Long startTime = System.nanoTime();
077      solrServer.add(doc);
078      Long endTime = System.nanoTime();
079      Long totalTime = TimeUnit.NANOSECONDS.toSeconds(endTime - startTime);
080      if (TIMER_THRESHOLD <= totalTime) {
081        LOG
082          .info("Solr is delaying more than {} adding a document, the time adding a document was: {} ",
083            TIMER_THRESHOLD,
084            totalTime);
085      }
086    } catch (Exception e) {
087      LOG.error("Error adding document", e);
088    }
089  }

091  /**
092   * Populates the Solr document using the result row parameter.
093   */
094  public static void buildOccSolrDocument(Result row, SolrInputDocument doc) {
095    final Double latitude = OccurrenceResultReader.getDouble(row, FieldName.I_LATITUDE);
096    final Double longitude = OccurrenceResultReader.getDouble(row, FieldName.I_LONGITUDE);
097    final Integer geospatialIssue = OccurrenceResultReader.getInteger(row, FieldName.I_GEOSPATIAL_ISSUE);
098    final Date occurrenceDate = OccurrenceResultReader.getDate(row, FieldName.I_OCCURRENCE_DATE);
099    final Date modified = OccurrenceResultReader.getDate(row, FieldName.I_MODIFIED);
100
101    doc.setField(KEY.getFieldName(), OccurrenceResultReader.getKey(row));
102    doc.setField(YEAR.getFieldName(), OccurrenceResultReader.getInteger(row, FieldName.I_YEAR));
103    doc.setField(MONTH.getFieldName(), OccurrenceResultReader.getInteger(row, FieldName.I_MONTH));
104    doc.setField(BASIS_OF_RECORD.getFieldName(), OccurrenceResultReader.getInteger(row, FieldName.I_BASIS_OF_RECORD));
105    doc.setField(CATALOG_NUMBER.getFieldName(), OccurrenceResultReader.getString(row, FieldName.CATALOG_NUMBER));
106    doc.setField(COLLECTOR_NAME.getFieldName(), OccurrenceResultReader.getString(row, FieldName.COLLECTOR_NAME));
107    doc.setField(COUNTRY.getFieldName(), OccurrenceResultReader.getString(row, FieldName.I_ISO_COUNTRY_CODE));
108    doc.setField(PUBLISHING_COUNTRY.getFieldName(), OccurrenceResultReader.getString(row, FieldName.HOST_COUNTRY));
109    doc.setField(DATASET_KEY.getFieldName(), OccurrenceResultReader.getString(row, FieldName.DATASET_KEY));
110
111    Set<Integer> taxonKey = buildTaxonKey(row);
112    if (!taxonKey.isEmpty()) {
113      doc.setField(TAXON_KEY.getFieldName(), taxonKey);
114    } else {
115      doc.setField(TAXON_KEY.getFieldName(), null);
116    }
117    doc.setField(ALTITUDE.getFieldName(), OccurrenceResultReader.getInteger(row, FieldName.I_ALTITUDE));
118    doc.setField(DEPTH.getFieldName(), OccurrenceResultReader.getInteger(row, FieldName.I_DEPTH));
119    doc.setField(INSTITUTION_CODE.getFieldName(), OccurrenceResultReader.getString(row, FieldName.INSTITUTION_CODE));
120    doc.setField(COLLECTION_CODE.getFieldName(), OccurrenceResultReader.getString(row, FieldName.COLLECTION_CODE));
121    doc.setField(GEOSPATIAL_ISSUE.getFieldName(), geospatialIssue != null && geospatialIssue > 0);
122    doc.setField(GEOREFERENCED.getFieldName(), latitude != null && longitude != null);
123    doc.setField(LATITUDE.getFieldName(), latitude);
124    doc.setField(LONGITUDE.getFieldName(), longitude);
125    doc.setField(DATE.getFieldName(), occurrenceDate != null ? toDateQueryFormat(occurrenceDate) : null);
126    doc.setField(MODIFIED.getFieldName(), modified != null ? toDateQueryFormat(modified) : null);
127    if (isValidCoordinate(latitude, longitude)) {
128      doc.setField(COORDINATE.getFieldName(), COORD_JOINER.join(latitude, longitude));
129    } else {
130      doc.setField(COORDINATE.getFieldName(), null);
131    }
132  }

202  /**
203   * Creates a scan instance using the list of fields.
204   * 
205   * @param fieldNames list of {@link FieldName}
206   * @return a Scan containing the field names
207   */
208  public static Scan buildScan(FieldName... fieldNames) {
209    Scan scan = new Scan();
210    for (FieldName fieldName : fieldNames) {
211      scan.addColumn(Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnFamilyName()),
212        Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnName()));
213    }
214    return scan;
215  }

217  /**
218   * Converts a {@link OccurrenceIndexDocument} into {@link SolrInputDocument}.
219   */
220  public static SolrInputDocument toSolrInputDocument(OccurrenceIndexDocument occurrenceIndexDocument) {
221    SolrInputDocument doc = new SolrInputDocument();
222    final Double latitude = occurrenceIndexDocument.getLatitude();
223    final Double longitude = occurrenceIndexDocument.getLongitude();
224
225    doc.setField(KEY.getFieldName(), occurrenceIndexDocument.getKey());
226    doc.setField(YEAR.getFieldName(), occurrenceIndexDocument.getYear());
227    doc.setField(MONTH.getFieldName(), occurrenceIndexDocument.getMonth());
228    doc.setField(CATALOG_NUMBER.getFieldName(), occurrenceIndexDocument.getCatalogNumber());
229    doc.setField(COLLECTOR_NAME.getFieldName(), occurrenceIndexDocument.getCollectorName());
230    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getNubKey());
231    doc.setField(DATASET_KEY.getFieldName(), occurrenceIndexDocument.getDatasetKey());
232    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getKingdomKey());
233    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getPhylumKey());
234    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getClassKey());
235    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getOrderKey());
236    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getFamilyKey());
237    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getGenusKey());
238    doc.setField(TAXON_KEY.getFieldName(), occurrenceIndexDocument.getSpeciesKey());
239    doc.setField(COUNTRY.getFieldName(), occurrenceIndexDocument.getIsoCountryCode());
240    doc.setField(PUBLISHING_COUNTRY.getFieldName(), occurrenceIndexDocument.getHostCountry());
241    doc.setField(BASIS_OF_RECORD.getFieldName(), occurrenceIndexDocument.getBasisOfRecord());
242    if (latitude != null || longitude != null) {
243      doc.setField(COORDINATE.getFieldName(), COORD_JOINER.join(latitude, longitude));
244    }
245    return doc;
246  }

248  /**
249   * Reads the taxon key values to create a Set of integers with those values.
250   */
251  private static Set<Integer> buildTaxonKey(Result row) {
252    Set<Integer> taxonKey = new HashSet<Integer>();
253
254    Integer taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_NUB_ID);
255    if (taxaKey != null) {
256      taxonKey.add(taxaKey);
257    }
258
259    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_KINGDOM_ID);
260    if (taxaKey != null) {
261      taxonKey.add(taxaKey);
262    }
263
264    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_PHYLUM_ID);
265    if (taxaKey != null) {
266      taxonKey.add(taxaKey);
267    }
268
269    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_CLASS_ID);
270    if (taxaKey != null) {
271      taxonKey.add(taxaKey);
272    }
273
274    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_ORDER_ID);
275    if (taxaKey != null) {
276      taxonKey.add(taxaKey);
277    }
278
279    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_FAMILY_ID);
280    if (taxaKey != null) {
281      taxonKey.add(taxaKey);
282    }
283
284    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_GENUS_ID);
285    if (taxaKey != null) {
286      taxonKey.add(taxaKey);
287    }
288
289    taxaKey = OccurrenceResultReader.getInteger(row, FieldName.I_SPECIES_ID);
290    if (taxaKey != null) {
291      taxonKey.add(taxaKey);
292    }
293
294    return taxonKey;
295  }

297  /**
298   * Validate if the latitude and longitude values are not null and have correct values: latitude:[-90.0,90.0] and
299   * longitude[-180.0,180.0].
300   */
301  private static boolean isValidCoordinate(Double latitude, Double longitude) {
302    if (latitude != null && longitude != null) {
303      return LAT_RANGE.contains(latitude) && LNG_RANGE.contains(longitude);
304    }
305    return false;
306  }
