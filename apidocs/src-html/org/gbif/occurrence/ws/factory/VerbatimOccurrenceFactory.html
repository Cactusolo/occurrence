004package org.gbif.occurrence.ws.factory;
005
006import org.gbif.api.model.occurrence.VerbatimOccurrence;
007import org.gbif.dwc.terms.DwcTerm;
008import org.gbif.dwc.terms.Term;
009
010import java.util.List;
011import javax.annotation.Nullable;
012
013import com.google.common.base.Joiner;
014import com.google.common.collect.ImmutableMap;
015import com.google.common.collect.ImmutableMap.Builder;
016import com.google.common.collect.Lists;
017
018/**
019 * This is responsible for transforming the verbatim occurrence persistence object
020 * into the API representation which is Map based. All map entries are keyed using
021 * the DwC simple term names. Transposition of terms that don't map is implemented for
022 * <ul>
023 * <li>MinAltitude, MaxAltitude, Altitude Precision -> verbatimElevation</li>
024 * <li>MinDepth, MaxDepth, Depth Precision -> verbatimDepth</li>
025 * <li>yearIdentified, monthIdentified, dayIdentified, dateIdentified -> dateIdentified</li>
026 * </ul>
027 */
028public class VerbatimOccurrenceFactory {
029
030  public static VerbatimOccurrence build(org.gbif.occurrencestore.persistence.api.VerbatimOccurrence from) {
031    VerbatimOccurrence vo = new VerbatimOccurrence();
032    vo.setKey(from.getKey());
033    Builder<Term, String> b = ImmutableMap.builder();
034    putIfNotNull(b, DwcTerm.scientificName, from.getScientificName());
035    putIfNotNull(b, DwcTerm.scientificNameAuthorship, from.getAuthor());
036    putIfNotNull(b, DwcTerm.taxonRank, from.getRank());
037    putIfNotNull(b, DwcTerm.kingdom, from.getKingdom());
038    putIfNotNull(b, DwcTerm.phylum, from.getPhylum());
039    putIfNotNull(b, DwcTerm.class_, from.getKlass());
040    putIfNotNull(b, DwcTerm.order, from.getOrder());
041    putIfNotNull(b, DwcTerm.family, from.getFamily());
042    putIfNotNull(b, DwcTerm.genus, from.getGenus());
043    putIfNotNull(b, DwcTerm.specificEpithet, from.getSpecies());
044    putIfNotNull(b, DwcTerm.infraspecificEpithet, from.getSubspecies());
045    putIfNotNull(b, DwcTerm.verbatimLatitude, from.getLatitude());
046    putIfNotNull(b, DwcTerm.verbatimLongitude, from.getLongitude());
047    // based on the belief that we most commonly mapped uncertaintity in the OLD portal
048    putIfNotNull(b, DwcTerm.coordinateUncertaintyInMeters, from.getLatLongPrecision());
049    putIfNotNull(b, DwcTerm.verbatimElevation,
050      minMaxPrecision(from.getMinAltitude(), from.getMaxAltitude(), from.getAltitudePrecision()));
051    putIfNotNull(b, DwcTerm.verbatimDepth, minMaxPrecision(from.getMinDepth(), from.getMaxDepth(), from.getDepthPrecision()));
052    // DwC has expanded continentOrOcean into 2 terms, so they may be collapsed
053    putIfNotNull(b, DwcTerm.continent, from.getContinentOrOcean(), "(originally supplied in continent OR ocean)");
054    putIfNotNull(b, DwcTerm.country, from.getCountry());
055    putIfNotNull(b, DwcTerm.stateProvince, from.getStateOrProvince());
056    putIfNotNull(b, DwcTerm.county, from.getCounty());
057    putIfNotNull(b, DwcTerm.recordedBy, from.getCollectorName());
058    putIfNotNull(b, DwcTerm.locality, from.getLocality());
059    putIfNotNull(b, DwcTerm.year, from.getYear());
060    putIfNotNull(b, DwcTerm.month, from.getMonth());
061    putIfNotNull(b, DwcTerm.day, from.getDay());
062    putIfNotNull(b, DwcTerm.eventDate, from.getOccurrenceDate());
063    putIfNotNull(b, DwcTerm.basisOfRecord, from.getBasisOfRecord());
064    putIfNotNull(b, DwcTerm.identifiedBy, from.getIdentifierName());
065    putIfNotNull(b, DwcTerm.dateIdentified,
066      yearMonthDayDate(from.getYearIdentified(), from.getMonthIdentified(), from.getDayIdentified(), from.getDateIdentified()));
067    vo.setFields(b.build());
068    return vo;
069  }
070
071  private static String minMaxPrecision(String min, String max, String precision) {
072    List<String> s = Lists.newArrayList();
073    if (min != null) {
074      s.add("Min[" + min + "]");
075    }
076    if (max != null) {
077      s.add("Max[" + max + "]");
078    }
079    if (precision != null) {
080      s.add("Precision[" + precision + "]");
081    }
082
083    if (!s.isEmpty()) {
084      return Joiner.on(" ").join(s);
085    }
086    return null;
087  }
088
089
090  private static void putIfNotNull(Builder<Term, String> b, Term k, @Nullable String v) {
091    if (v != null) {
092      b.put(k, v);
093    }
094  }
095
096  private static void putIfNotNull(Builder<Term, String> b, Term k, @Nullable String v, String suffix) {
097    if (v != null) {
098      putIfNotNull(b, k, v + ' ' + suffix);
099    }
100  }
101
102  private static String yearMonthDayDate(String year, String month, String day, String date) {
103    List<String> s = Lists.newArrayList();
104    if (year != null) {
105      s.add("Year[" + year + ']');
106    }
107    if (month != null) {
108      s.add("Month[" + month + ']');
109    }
110    if (day != null) {
111      s.add("Day[" + day + ']');
112    }
113    if (date != null) {
114      s.add("Date[" + date + ']');
115    }
116
117    if (!s.isEmpty()) {
118      return Joiner.on(" ").join(s);
119    }
120    return null;
121  }
