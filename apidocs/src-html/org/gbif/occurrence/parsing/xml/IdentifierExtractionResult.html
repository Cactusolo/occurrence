001package org.gbif.occurrence.parsing.xml;
002
003import org.gbif.occurrencestore.common.model.UniqueIdentifier;
004
005import java.util.Set;
006import javax.annotation.Nullable;
007
008import com.google.common.base.Objects;
009
010import static com.google.common.base.Preconditions.checkNotNull;
011
012/**
013 * Contains the set of {@link UniqueIdentifier} that were extracted from a raw fragment that each uniquely identify
014 * that fragment. May also contain a unitQualifier for cases of multiple occurrences within a single fragment
015 * (e.g. ABCD 2.06).
016 */
017public class IdentifierExtractionResult {
018
019  private final Set<UniqueIdentifier> uniqueIdentifiers;
020  @Nullable
021  private final String unitQualifier;
022
023  public IdentifierExtractionResult(Set<UniqueIdentifier> uniqueIdentifiers, @Nullable String unitQualifier) {
024    this.uniqueIdentifiers = checkNotNull(uniqueIdentifiers, "uniqueIdentifiers can't be null");
025    this.unitQualifier = unitQualifier;
026  }
027
028  public String getUnitQualifier() {
029    return unitQualifier;
030  }
031
032  public Set<UniqueIdentifier> getUniqueIdentifiers() {
033    return uniqueIdentifiers;
034  }
035
036  @Override
037  public int hashCode() {
038    return Objects.hashCode(uniqueIdentifiers, unitQualifier);
039  }
040
041  @Override
042  public boolean equals(Object obj) {
043    if (this == obj) {
044      return true;
045    }
046    if (obj == null || getClass() != obj.getClass()) {
047      return false;
048    }
049    final IdentifierExtractionResult other = (IdentifierExtractionResult) obj;
050    return Objects.equal(this.uniqueIdentifiers, other.uniqueIdentifiers) && Objects
051      .equal(this.unitQualifier, other.unitQualifier);
052  }
053}
