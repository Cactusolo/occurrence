034/**
035 * This object is the one that gets populated by Digester when parsing raw xml records into RawOccurrenceRecords.
036 * Because some of the xml schemas allow multiple identification records, and we interpret each one as its own
037 * RawOccurrenceRecord, this class needs to generate the correct number of RawOccurrenceRecords based on the input
038 * xml.  In some schemas there are also different ways of representing the same data (eg decimal latitude vs text
039 * latitude) and we prefer the more accurate version when we can get it.  If more than one representation for a given
040 * property is populated we have to take the one with highest priority - a mechanism that is inherited from
041 * PropertyPrioritizer.
042 */
043public class RawOccurrenceRecordBuilder extends PropertyPrioritizer {

102  public List<RawOccurrenceRecord> generateRawOccurrenceRecords() {
103    List<RawOccurrenceRecord> records = new ArrayList<RawOccurrenceRecord>();
104
105    // reconcile year/month/date into an occurrenceDate string, if necessary
106    occurrenceDate = reconcileDate(occurrenceDate, year, month, day);
107    dateIdentified = reconcileDate(dateIdentified, yearIdentified, monthIdentified, dayIdentified);
108
109    // if multiple valid identifications, generate multiple RoRs and set UnitQualifier to Identification's SciName
110    if (identifications.size() > 0) {
111      if (identifications.size() == 1) {
112        RawOccurrenceRecord bareBones = generateBareRor();
113        Identification ident = identifications.get(0);
114        ident.populateRawOccurrenceRecord(bareBones);
115        records.add(bareBones);
116      } else {
117        // take any that have preferred flag set
118        // if no preferred flags are set, take them all
119        boolean gotPreferred = false;
120        List<RawOccurrenceRecord> preferred = new ArrayList<RawOccurrenceRecord>();
121        for (Identification ident : identifications) {
122          RawOccurrenceRecord bareBones = generateBareRor();
123          ident.populateRawOccurrenceRecord(bareBones, true);
124          if (ident.isPreferred()) {
125            gotPreferred = true;
126            preferred.add(bareBones);
127          } else {
128            records.add(bareBones);
129          }
130        }
131        if (gotPreferred) records = preferred;
132      }
133    } else {
134      // no identifications, take bare bones only
135      RawOccurrenceRecord bareBones = generateBareRor();
136      records.add(bareBones);
137    }
138
139    return records;
140  }
141
142  private String reconcileDate(String reconciledDate, String year, String month, String day) {
143    if (StringUtils.isEmpty(reconciledDate) && !StringUtils.isEmpty(year)) {
144      reconciledDate = year;
145      if (!StringUtils.isEmpty(month)) {
146        reconciledDate = reconciledDate + "-" + month;
147        if (!StringUtils.isEmpty(day)) reconciledDate = reconciledDate + "-" + day;
148      }
149    }
150
151    return reconciledDate;
152  }

158    bareBones.setBasisOfRecord(basisOfRecord);
159    bareBones.setCatalogueNumber(catalogueNumber);
160    bareBones.setCollectionCode(collectionCode);
161    bareBones.setCollectorName(collectorName);
162    bareBones.setContinentOrOcean(continentOrOcean);
163    bareBones.setCountry(country);
164    bareBones.setCounty(county);
165    bareBones.setDataProviderId(dataProviderId);
166    bareBones.setDataResourceId(dataResourceId);
167    bareBones.setYearIdentified(yearIdentified);
168    bareBones.setMonthIdentified(monthIdentified);
169    bareBones.setDayIdentified(dayIdentified);
170    bareBones.setDateIdentified(dateIdentified);
171    bareBones.setDepthPrecision(depthPrecision);
172    bareBones.setFamily(family);
173    bareBones.setGenus(genus);
174    bareBones.setIdentifierName(identifierName);
175    bareBones.setIdentifierRecords(identifierRecords);
176    bareBones.setImageRecords(images);
177    bareBones.setInstitutionCode(institutionCode);
178    bareBones.setKingdom(kingdom);
179    bareBones.setKlass(klass);
180    bareBones.setLatitude(latitude);
181    bareBones.setLatLongPrecision(latLongPrecision);
182    bareBones.setLinkRecords(links);
183    bareBones.setLocality(locality);
184    bareBones.setLongitude(longitude);
185    bareBones.setMaxAltitude(maxAltitude);
186    bareBones.setMaxDepth(maxDepth);
187    bareBones.setMinAltitude(minAltitude);
188    bareBones.setMinDepth(minDepth);
189    bareBones.setYear(year);
190    bareBones.setMonth(month);
191    bareBones.setDay(day);
192    bareBones.setOccurrenceDate(occurrenceDate);
193    bareBones.setOrder(order);
194    bareBones.setPhylum(phylum);
195    bareBones.setRank(rank);
196    bareBones.setResourceAccessPointId(resourceAccessPointId);
197    bareBones.setScientificName(scientificName);
198    bareBones.setSpecies(species);
199    bareBones.setStateOrProvince(stateOrProvince);
200    bareBones.setSubspecies(subspecies);
201    bareBones.setTypificationRecords(typificationRecords);
202    bareBones.setUnitQualifier(unitQualifier);
203
204    return bareBones;
205  }

211  public void addTypification(String scientificName, String publication, String typeStatus, String notes) {
212    LOG.debug(">> addTypification");
213
214    TypificationRecord typRec = new TypificationRecord(scientificName, publication, typeStatus, notes);
215
216    if (typRec.isEmpty()) {
217      LOG.debug("Got all nulls for new type - ignoring");
218    } else {
219      LOG.debug("Got new typRec:\n" + typRec.debugDump());
220      typificationRecords.add(typRec);
221    }
222
223    LOG.debug("<< addTypification");
224  }

226  private void addIdentifier(Integer identifierType, String identifier) {
227    LOG.debug("Attempting add of Identifier record with type [{}] and body [{}]", identifierType, identifier);
228    if (!identifier.trim().isEmpty()) {
229      IdentifierRecord idRec = new IdentifierRecord();
230      idRec.setIdentifier(identifier);
231      idRec.setIdentifierType(identifierType);
232      identifierRecords.add(idRec);
233    }
234  }

244  /**
245   * Once this object has been populated by a Digester, there may be several PrioritizedProperties that
246   * need to be resolved, and thereby set the final value of the corresponding field on this object.
247   */
248  @Override
249  public void resolvePriorities() {
250    for (PrioritizedPropertyNameEnum name : prioritizedProps.keySet()) {
251      String result = findHighestPriority(prioritizedProps.get(name));
252      switch (name) {
253        case CATALOGUE_NUMBER:
254          this.catalogueNumber = result;
255          break;
256        case COLLECTOR_NAME:
257          this.collectorName = result;
258          break;
259        case CONTINENT_OR_OCEAN:
260          this.continentOrOcean = result;
261          break;
262        case COUNTRY:
263          this.country = result;
264          break;
265        case DATE_COLLECTED:
266          this.occurrenceDate = result;
267          break;
268        case LATITUDE:
269          this.latitude = result;
270          break;
271        case LONGITUDE:
272          this.longitude = result;
273          break;
274        default:
275          LOG.warn("Fell through priority resolution for [{}]", name);
276      }
277    }
278
279    for (Identification ident : identifications) {
280      ident.resolvePriorities();
281    }
282    for (ImageRecord image : images) {
283      image.resolvePriorities();
284    }
285  }

604    // build our own "verbatim" string

609        result = result + "-" + month;
610        if (!StringUtils.isEmpty(day)) result = result + "-" + day;
611      }
612    }
613    if (result != null) this.dateIdentified = result;
614  }
