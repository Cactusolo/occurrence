049/**
050 * Methods for parsing {@link RawXmlOccurrence}s and {@link UniqueIdentifier}s from xml fragments.
051 */
052public class XmlFragmentParser {
053
054  private static final Logger LOG = LoggerFactory.getLogger(XmlFragmentParser.class);
055
056  private static final Map<OccurrenceSchemaType, RuleSet> RULE_SETS = Maps.newHashMap();
057
058  // static class, should never be instantiated
059  private XmlFragmentParser() {
060  }
061
062  static {
063    try {
064      RULE_SETS.put(OccurrenceSchemaType.ABCD_1_2, new Abcd12RuleSet());
065      RULE_SETS.put(OccurrenceSchemaType.ABCD_2_0_6, new Abcd206RuleSet());
066      RULE_SETS.put(OccurrenceSchemaType.DWC_1_0, new Dwc10RuleSet());
067      RULE_SETS.put(OccurrenceSchemaType.DWC_1_4, new Dwc14RuleSet());
068      RULE_SETS.put(OccurrenceSchemaType.DWC_2009, new Dwc2009RuleSet());
069      RULE_SETS.put(OccurrenceSchemaType.DWC_MANIS, new DwcManisRuleSet());
070    } catch (IOException e) {
071      LOG.warn("Unable to read properties files for parsing xml", e);
072    }
073  }
074
075  public static List<RawOccurrenceRecord> parseRecord(RawXmlOccurrence xmlRecord) {
076    return parseRecord(xmlRecord.getXml(), xmlRecord.getSchemaType());
077  }
078
079  public static List<RawOccurrenceRecord> parseRecord(String xml, OccurrenceSchemaType schemaType) {
080    List<RawOccurrenceRecord> records = null;
081    try {
082      InputSource inputSource = new InputSource(new StringReader(xml));
083      records = parseRecord(inputSource, schemaType);
084    } catch (IOException e) {
085      LOG.warn("IOException parsing xml string [{}]", xml, e);
086    } catch (SAXException e) {
087      LOG.warn("SAXException parsing xml string [{}]", xml, e);
088    }
089    return records;
090  }
091
092  public static List<RawOccurrenceRecord> parseRecord(byte[] xml, OccurrenceSchemaType schemaType) {
093    List<RawOccurrenceRecord> records = null;
094    try {
095      InputSource inputSource = new InputSource(new ByteArrayInputStream(xml));
096      records = parseRecord(inputSource, schemaType);
097    } catch (IOException e) {
098      LOG.warn("IOException parsing xml bytes", e);
099    } catch (SAXException e) {
100      LOG.warn("SAXException parsing xml bytes", e);
101    }
102    return records;
103  }
104
105  private static List<RawOccurrenceRecord> parseRecord(InputSource inputSource, OccurrenceSchemaType schemaType)
106    throws IOException, SAXException {
107    RawOccurrenceRecordBuilder builder = new RawOccurrenceRecordBuilder();
108    Digester digester = new Digester();
109    digester.setNamespaceAware(true);
110    digester.setValidating(false);
111    digester.push(builder);
112    digester.addRuleSet(RULE_SETS.get(schemaType));
113    digester.parse(inputSource);
114
115    builder.resolvePriorities();
116    return builder.generateRawOccurrenceRecords();
117  }
118
119  /**
120   * This method is a hack to return a single result where ScientificName matches the given unitQualifier. This
121   * behaviour is only relevant for ABCD 2.06 - the others all produce a single record anyway.
122   * TODO: refactor the parse/builder to return what we want, rather than hacking around
123   */
124  public static RawOccurrenceRecord parseRecord(byte[] xml, OccurrenceSchemaType schemaType, String unitQualifier) {
125    RawOccurrenceRecord result = null;
126    List<RawOccurrenceRecord> records = parseRecord(xml, schemaType);
127    if (records.isEmpty()) {
128      LOG.warn("Could not parse any records from given xml - returning null.");
129    } else if (records.size() == 1) {
130      result = records.get(0);
131    } else if (unitQualifier == null) {
132      LOG.warn("Got multiple records from given xml, but no unitQualifier set - returning first record as a guess.");
133      result = records.get(0);
134    } else {
135      for (RawOccurrenceRecord record : records) {
136        if (record.getScientificName().equals(unitQualifier)) {
137          result = record;
138          break;
139        }
140      }
141      if (result == null) {
142        LOG.warn("Got multiple records from xml but none matched unitQualifier - returning null");
143      }
144    }
145
146    return result;
147  }
148
149  /**
150   * Extract sets of UniqueIdentifiers from the xml snippet. In the usual case the set will contain a single
151   * result, which will in turn contain 1 or more UniqueIdentifiers for the given xml. In the ABCD 2 case there
152   * may be more than one occurrence represented by the given xml, in which case there will be an
153   * IdentifierExtractionResult (with UniqueIdentifiers) returned for each of the represented occurrences (e.g. if 3
154   * occurrences are in the xml snippet and each have one UniqueIdentifier the result will be a set of 3
155   * IdentifierExtractionResults, where each result contains a single UniqueIdentifier). If the passed in xml is
156   * somehow malformed there may be 0 UniqueIdentifiers found, in which case an empty set is returned.
157   *
158   * @param datasetKey      UUID for this dataset
159   * @param xml             snippet of xml representing one (or more, in ABCD) occurrence
160   * @param schemaType      the protocol that produced this xml (e.g. DWC, ABCD)
161   * @param useOccurrenceId @return a set of 0 or more IdentifierExtractionResults containing UniqueIdentifiers as found
162   *                        in the xml
163   *
164   * @see UniqueIdentifier
165   */
166  public static Set<IdentifierExtractionResult> extractIdentifiers(UUID datasetKey, byte[] xml,
167    OccurrenceSchemaType schemaType, boolean useTriplet, boolean useOccurrenceId) {
168    Set<IdentifierExtractionResult> results = Sets.newHashSet();
169
170    // this is somewhat wasteful, but a whole separate stack of parsing to extract triplet seems excessive
171    List<RawOccurrenceRecord> records = parseRecord(xml, schemaType);
172    if (records != null && !records.isEmpty()) {
173      for (RawOccurrenceRecord record : records) {
174        Set<UniqueIdentifier> ids = Sets.newHashSet();
175
176        if (useTriplet) {
177          HolyTriplet holyTriplet = null;
178          try {
179            holyTriplet = new HolyTriplet(datasetKey, record.getInstitutionCode(), record.getCollectionCode(),
180              record.getCatalogueNumber(), record.getUnitQualifier());
181          } catch (IllegalArgumentException e) {
182            // some of the triplet was null or empty, so it's not valid - that's highly suspicious, but could be ok...
183            LOG.info("No holy triplet for an xml snippet in dataset [{}] and schema [{}], got error [{}]",
184              new String[] {datasetKey.toString(), schemaType.toString(), e.getMessage()});
185          }
186          if (holyTriplet != null) ids.add(holyTriplet);
187        }
188
189        if (useOccurrenceId) {
190          if (record.getIdentifierRecords() != null && !record.getIdentifierRecords().isEmpty()) {
191            for (IdentifierRecord idRecord : record.getIdentifierRecords()) {
192              // TODO: this needs much better checking (ie can we trust that guid (type 1) and sourceid (type 7) are
193              // getting set and parsed properly?)
194              // TODO: identifier types need to be enums
195              if (idRecord.getIdentifierType() == 1 || idRecord.getIdentifierType() == 7) {
196                if (idRecord.getIdentifier() != null) {
197                  ids.add(new PublisherProvidedUniqueIdentifier(datasetKey, idRecord.getIdentifier()));
198                }
199              }
200            }
201          }
202        }
203
204        if (!ids.isEmpty()) {
205          results.add(new IdentifierExtractionResult(ids, record.getUnitQualifier()));
206        }
207      }
208    }
209
210    return results;
211  }
