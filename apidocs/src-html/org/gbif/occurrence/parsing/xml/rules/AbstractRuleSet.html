001package org.gbif.occurrence.parsing.xml.rules;
002
003import org.gbif.occurrence.constants.PrioritizedPropertyNameEnum;
004import org.gbif.occurrence.parsing.xml.PrioritizedProperty;
005
006import java.util.Properties;
007
008import org.apache.commons.digester.Digester;
009import org.apache.commons.digester.RuleSet;
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013public abstract class AbstractRuleSet implements RuleSet {
014
015  private static final Logger LOG = LoggerFactory.getLogger(AbstractRuleSet.class);
016
017  protected Properties mappingProps;
018
019  @Override
020  public void addRuleInstances(Digester digester) {
021    // common, simple fields
022    addNonNullMethod(digester, "institutionCode", "setInstitutionCode", 1);
023    addNonNullParam(digester, "institutionCode", 0);
024
025    addNonNullMethod(digester, "collectionCode", "setCollectionCode", 1);
026    addNonNullParam(digester, "collectionCode", 0);
027
028    addNonNullMethod(digester, "basisOfRecord", "setBasisOfRecord", 1);
029    addNonNullParam(digester, "basisOfRecord", 0);
030
031    addNonNullMethod(digester, "minDepth", "setMinDepth", 1);
032    addNonNullParam(digester, "minDepth", 0);
033
034    addNonNullMethod(digester, "maxDepth", "setMaxDepth", 1);
035    addNonNullParam(digester, "maxDepth", 0);
036
037    addNonNullMethod(digester, "latLongPrecision", "setLatLongPrecision", 1);
038    addNonNullParam(digester, "latLongPrecision", 0);
039
040    addNonNullMethod(digester, "minAltitude", "setMinAltitude", 1);
041    addNonNullParam(digester, "minAltitude", 0);
042
043    addNonNullMethod(digester, "maxAltitude", "setMaxAltitude", 1);
044    addNonNullParam(digester, "maxAltitude", 0);
045
046    // identifier records
047    addNonNullMethod(digester, "identifierType1", "setIdentifierType1", 1);
048    addNonNullParam(digester, "identifierType1", 0);
049
050    addNonNullMethod(digester, "identifierType2", "setIdentifierType2", 1);
051    addNonNullParam(digester, "identifierType2", 0);
052
053    addNonNullMethod(digester, "identifierType3", "setIdentifierType3", 1);
054    addNonNullParam(digester, "identifierType3", 0);
055
056    addNonNullMethod(digester, "identifierType4", "setIdentifierType4", 1);
057    addNonNullParam(digester, "identifierType4", 0);
058
059    addNonNullMethod(digester, "identifierType5", "setIdentifierType5", 1);
060    addNonNullParam(digester, "identifierType5", 0);
061
062    addNonNullMethod(digester, "identifierType6", "setIdentifierType6", 1);
063    addNonNullParam(digester, "identifierType6", 0);
064
065    addNonNullMethod(digester, "identifierType7", "setIdentifierType7", 1);
066    addNonNullParam(digester, "identifierType7", 0);
067
068    // possibly many typifications
069    addNonNullMethod(digester, "typificationElement", "addTypification", 4);
070    addNonNullParam(digester, "typeScientificName", 0);
071    addNonNullParam(digester, "typePublication", 1);
072    addNonNullParam(digester, "typeStatus", 2);
073    addNonNullParam(digester, "typeNotes", 3);
074  }
075
076  protected void addNonNullMethod(Digester digester, String property, String methodName, int argCount) {
077    String pattern = mappingProps.getProperty(property);
078    if (pattern != null) {
079      pattern = pattern.trim();
080      LOG.debug("adding call method [{}] for pattern [{}] from property [{}]",
081        new Object[] {methodName, pattern, property});
082      digester.addCallMethod(pattern, methodName, argCount);
083    }
084  }
085
086  protected void addNonNullParam(Digester digester, String property, int argPosition) {
087    String pattern = mappingProps.getProperty(property);
088    if (pattern != null) {
089      pattern = pattern.trim();
090      LOG.debug("adding call param for pattern [{}] from property [{}]", pattern, property);
091      digester.addCallParam(pattern, argPosition);
092    }
093  }
094
095  protected void addNonNullAttParam(Digester digester, String elementProperty, String attributeProperty,
096                                    int argPosition) {
097    String elemPattern = mappingProps.getProperty(elementProperty);
098    String attPattern = mappingProps.getProperty(attributeProperty);
099    if (elemPattern != null && attPattern != null) {
100      elemPattern = elemPattern.trim();
101      attPattern = attPattern.trim();
102      LOG.debug(
103        "adding call param from attribute for element pattern [{}] from property [{}], seeking attribute [{}] from property [{}]",
104        new Object[] {elemPattern, elementProperty, attPattern, attributeProperty});
105      digester.addCallParam(elemPattern, argPosition, attPattern);
106    }
107  }
108
109  protected void addNonNullPrioritizedProperty(Digester digester, String property, PrioritizedPropertyNameEnum name,
110                                               int paramCount) {
111    for (int i = 1; i <= paramCount; i++) {
112      String key = property + '.' + i;
113      String pattern = mappingProps.getProperty(key);
114      if (pattern != null) {
115        pattern = pattern.trim();
116        LOG.debug("adding prioritized property [{}] with pattern [{}]", key, pattern);
117        // note order of rule addition is critical
118        digester.addObjectCreate(pattern, PrioritizedProperty.class);
119        digester.addSetNext(pattern, "addPrioritizedProperty");
120        digester.addRule(pattern, new SetLiteralRule("setPriority", i));
121        digester.addRule(pattern, new SetLiteralRule("setName", name));
122        digester.addCallMethod(pattern, "setProperty", 1);
123        digester.addCallParam(pattern, 0);
124      }
125    }
126  }
127}
