033/**
034 * Primary purpose of this class is, given a raw occurrence record serialized to String from XML,
035 * determine which OccurrenceSchemaType it matches.
036 */
037public class ResponseSchemaDetector {
038
039  private static final Logger LOG = LoggerFactory.getLogger(ResponseSchemaDetector.class);
040
041  // a dumb way to guarantee that string searching will be
042  // done such that eg "CatalogNumberText" gets searched for before "CatalogNumber"
043  private List<OccurrenceSchemaType> schemaSearchOrder;
044  private Map<OccurrenceSchemaType, Map<ResponseElementEnum, String>> distinctiveElements;
045
046  public ResponseSchemaDetector() {
047    init();
048  }
049
050  private boolean checkElements(String xml, Collection<String> elements) {
051    boolean result = true;
052    for (String elem : elements) {
053      result = xml.contains(elem);
054      LOG.debug("Xml contains [{}] is [{}]", elem, result);
055      if (!result) return result;
056    }
057
058    return result;
059  }
060
061  public OccurrenceSchemaType detectSchema(String xml) {
062    OccurrenceSchemaType result = null;
063    for (OccurrenceSchemaType schema : schemaSearchOrder) {
064      LOG.debug("Checking for schema [{}]", schema);
065      boolean success = checkElements(xml, distinctiveElements.get(schema).values());
066      if (success) {
067        result = schema;
068        break;
069      }
070    }
071
072    if (result == null) LOG.warn("Could not determine schema for xml [{}]", xml);
073
074    return result;
075  }
076
077  public Map<ResponseElementEnum, String> getResponseElements(OccurrenceSchemaType schemaType) {
078    return distinctiveElements.get(schemaType);
079  }
080

134
135  private void printNode(Node node) {
136    LOG.info("Node name: [{}]", node.getNodeName());
137    if (node.hasChildNodes()) {
138      NodeList children = node.getChildNodes();
139      for (int i = 0; i < children.getLength(); i++) {
140        printNode(children.item(i));
141      }
142    }
143  }
