053/**
054 * Entry point into the parsing of raw occurrence records as retrieved from publishers.  Will attempt to determine
055 * both XML encodings and schema type.  Parse happens in two steps - first extracts each record element into a
056 * RawXmlOccurrence, and then parses each of those into RawOccurrenceRecords.
057 */
058public class OccurrenceParser {
059
060  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceParser.class);
061
062  public List<RawOccurrenceRecord> parseResponseFileToRor(File inputFile) {
063    List<RawXmlOccurrence> raws = parseResponseFileToRawXml(inputFile);
064    List<RawOccurrenceRecord> rors = parseRawXmlToRor(raws);
065
066    return rors;
067  }

069  /**
070   * This parses a stream of uncompressed ABCD or DwC Occurrences into {@link RawXmlOccurrence}s.
071   *
072   * No care is taken to handle wrong encodings or character sets in general. This might be changed later on.
073   *
074   * @param is stream to parse
075   * @return list of parsed occurrences
076   * @throws ParsingException if there were any problems during parsing the stream
077   */
078  // TODO: Optionally handle compressed streams
079  public List<RawXmlOccurrence> parseStream(InputStream is) throws ParsingException {
080    checkNotNull(is, "is can't be null");
081    try {
082      ParsedSearchResponse responseBody = new ParsedSearchResponse();
083      InputSource inputSource = new InputSource(is);
084
085      Digester digester = new Digester();
086      digester.setNamespaceAware(true);
087      digester.setValidating(false);
088      digester.push(responseBody);
089
090      NodeCreateRule rawAbcd = new NodeCreateRule();
091      digester.addRule(ExtractionSimpleXPaths.ABCD_RECORD_XPATH, rawAbcd);
092      digester.addSetNext(ExtractionSimpleXPaths.ABCD_RECORD_XPATH, "addRecordAsXml");
093
094      NodeCreateRule rawAbcd1Header = new NodeCreateRule();
095      digester.addRule(ExtractionSimpleXPaths.ABCD_HEADER_XPATH, rawAbcd1Header);
096      digester.addSetNext(ExtractionSimpleXPaths.ABCD_HEADER_XPATH, "setAbcd1Header");
097
098      NodeCreateRule rawDwc1_0 = new NodeCreateRule();
099      digester.addRule(ExtractionSimpleXPaths.DWC_1_0_RECORD_XPATH, rawDwc1_0);
100      digester.addSetNext(ExtractionSimpleXPaths.DWC_1_0_RECORD_XPATH, "addRecordAsXml");
101
102      NodeCreateRule rawDwc1_4 = new NodeCreateRule();
103      digester.addRule(ExtractionSimpleXPaths.DWC_1_4_RECORD_XPATH, rawDwc1_4);
104      digester.addSetNext(ExtractionSimpleXPaths.DWC_1_4_RECORD_XPATH, "addRecordAsXml");
105
106//      NodeCreateRule rawDwcManis = new NodeCreateRule();
107//      digester.addRule(ExtractionSimpleXPaths.DWC_MANIS_RECORD_XPATH, rawDwcManis);
108//      digester.addSetNext(ExtractionSimpleXPaths.DWC_MANIS_RECORD_XPATH, "addRecordAsXml");
109
110      NodeCreateRule rawDwc2009 = new NodeCreateRule();
111      digester.addRule(ExtractionSimpleXPaths.DWC_2009_RECORD_XPATH, rawDwc2009);
112      digester.addSetNext(ExtractionSimpleXPaths.DWC_2009_RECORD_XPATH, "addRecordAsXml");
113
114      digester.parse(inputSource);
115      return responseBody.getRecords();
116    } catch (ParserConfigurationException e) {
117      throw new ServiceUnavailableException("Error setting up Commons Digester", e);
118    } catch (SAXException e) {
119      throw new ParsingException("Parsing failed", e);
120    } catch (IOException e) {
121      throw new ParsingException("Parsing failed", e);
122    } catch (TransformerException e) {
123      throw new ServiceUnavailableException("Error setting up Commons Digester", e);
124    }
125  }

127  /**
128   * Parses a single response gzipFile and returns a List of the contained RawXmlOccurrences.
129   */
130  public List<RawXmlOccurrence> parseResponseFileToRawXml(File gzipFile) {
131    if (LOG.isDebugEnabled()) LOG.debug(">> parseResponseFileToRawXml [{}]", gzipFile.getAbsolutePath());
132    ParsedSearchResponse responseBody = null;
133    InputStreamReader inputStreamReader = null;
134    BufferedReader bufferedReader = null;
135    try {
136      responseBody = new ParsedSearchResponse();
137
138      FileInputStream fis = new FileInputStream(gzipFile);
139      GZIPInputStream inputStream = new GZIPInputStream(fis);
140
141      // charsets are a nightmare and users can't be trusted, so strategy
142      // is try these encodings in order until one of them (hopefully) works
143      // (note the last two could be repeats of the first two):
144      // - utf-8
145      // - latin1 (iso-8859-1)
146      // - the declared encoding from the parsing itself
147      // - a guess at detecting the charset from the raw gzipFile bytes
148
149      List<String> charsets = new ArrayList<String>();
150      charsets.add("UTF-8");
151      charsets.add("ISO-8859-1");
152
153      // read parsing declaration
154
155      inputStreamReader = new InputStreamReader(inputStream);
156      bufferedReader = new BufferedReader(inputStreamReader);
157      boolean gotEncoding = false;
158      String encoding = "";
159      int lineCount = 0;
160      while (bufferedReader.ready() && !gotEncoding && lineCount < 5) {
161        String line = bufferedReader.readLine();
162        lineCount++;
163        if (line != null && line.contains("encoding=")) {
164          encoding = line.split("encoding=")[1];
165          // drop trailing ?>
166          encoding = encoding.substring(0, encoding.length() - 2);
167          // drop quotes
168          encoding = encoding.replaceAll("\"", "").replaceAll("'", "").trim();
169          LOG.debug("Found encoding [{}] in parsing declaration", encoding);
170          try {
171            Charset.forName(encoding);
172            charsets.add(encoding);
173          } catch (Exception e) {
174            LOG.debug(
175              "Could not find supported charset matching detected encoding of [{}] - trying other guesses instead",
176              encoding);
177          }
178          gotEncoding = true;
179        }
180      }
181
182      // attempt detection from bytes
183      Charset charset = CharsetDetection.detectEncoding(gzipFile);
184      charsets.add(charset.name());
185      String goodCharset = null;
186      boolean encodingError = false;
187      for (String charsetName : charsets) {
188        LOG.debug("Trying charset [{}]", charsetName);
189        try {
190          // reset streams
191          fis = new FileInputStream(gzipFile);
192          inputStream = new GZIPInputStream(fis);
193
194          BufferedReader inputReader =
195            new BufferedReader(new XmlSanitizingReader(new InputStreamReader(inputStream, charsetName)));
196          InputSource inputSource = new InputSource(inputReader);
197
198          Digester digester = new Digester();
199          digester.setNamespaceAware(true);
200          digester.setValidating(false);
201          digester.push(responseBody);
202
203          NodeCreateRule rawAbcd = new NodeCreateRule();
204          digester.addRule(ExtractionSimpleXPaths.ABCD_RECORD_XPATH, rawAbcd);
205          digester.addSetNext(ExtractionSimpleXPaths.ABCD_RECORD_XPATH, "addRecordAsXml");
206
207          NodeCreateRule rawAbcd1Header = new NodeCreateRule();
208          digester.addRule(ExtractionSimpleXPaths.ABCD_HEADER_XPATH, rawAbcd1Header);
209          digester.addSetNext(ExtractionSimpleXPaths.ABCD_HEADER_XPATH, "setAbcd1Header");
210
211          NodeCreateRule rawDwc1_0 = new NodeCreateRule();
212          digester.addRule(ExtractionSimpleXPaths.DWC_1_0_RECORD_XPATH, rawDwc1_0);
213          digester.addSetNext(ExtractionSimpleXPaths.DWC_1_0_RECORD_XPATH, "addRecordAsXml");
214
215          NodeCreateRule rawDwc1_4 = new NodeCreateRule();
216          digester.addRule(ExtractionSimpleXPaths.DWC_1_4_RECORD_XPATH, rawDwc1_4);
217          digester.addSetNext(ExtractionSimpleXPaths.DWC_1_4_RECORD_XPATH, "addRecordAsXml");
218
219          // TODO: dwc_manis appears to work without a NodeCreateRule here - why?
220
221          NodeCreateRule rawDwc2009 = new NodeCreateRule();
222          digester.addRule(ExtractionSimpleXPaths.DWC_2009_RECORD_XPATH, rawDwc2009);
223          digester.addSetNext(ExtractionSimpleXPaths.DWC_2009_RECORD_XPATH, "addRecordAsXml");
224
225          digester.parse(inputSource);
226
227          LOG.debug("Success with charset [{}] - skipping any others", charsetName);
228          goodCharset = charsetName;
229          break;
230        } catch (SAXException e) {
231          String msg = "SAX exception when parsing parsing from response gzipFile [" + gzipFile.getAbsolutePath()
232                       + "] using encoding [" + charsetName + "] - trying another charset";
233          LOG.debug(msg, e);
234        } catch (IOException e) {
235          if (e instanceof MalformedByteSequenceException) {
236            LOG.debug("Malformed utf-8 byte when parsing with encoding [{}] - trying another charset", charsetName);
237            encodingError = true;
238          }
239        }
240      }
241
242      if (goodCharset == null) {
243        if (encodingError) {
244          LOG.warn(
245            "Could not parse gzipFile - none of the encoding attempts worked (failed with malformed utf8) - skipping gzipFile [{}]",
246            gzipFile.getAbsolutePath());
247        } else {
248          LOG.warn("Could not parse gzipFile (malformed parsing) - skipping gzipFile [{}]", gzipFile.getAbsolutePath());
249        }
250      }
251
252    } catch (FileNotFoundException e) {
253      LOG.warn("Could not find response gzipFile [{}] - skipping gzipFile", gzipFile.getAbsolutePath(), e);
254    } catch (IOException e) {
255      LOG.warn("Could not read response gzipFile [{}] - skipping gzipFile", gzipFile.getAbsolutePath(), e);
256    } catch (TransformerException e) {
257      LOG.warn("Could not create parsing transformer for [{}] - skipping gzipFile", gzipFile.getAbsolutePath(), e);
258    } catch (ParserConfigurationException e) {
259      LOG.warn("Failed to pull raw parsing from response gzipFile [{}] - skipping gzipFile", gzipFile.getAbsolutePath(), e);
260    } finally {
261      try {
262        if (bufferedReader != null) bufferedReader.close();
263        if (inputStreamReader != null) inputStreamReader.close();
264      } catch (IOException e) {
265        LOG.debug("Failed to close input files", e);
266      }
267    }
268
269    if (LOG.isDebugEnabled()) LOG.debug("<< parseResponseFileToRawXml [{}]", gzipFile.getAbsolutePath());
270    return (responseBody == null) ? null : responseBody.getRecords();
271  }
272
273  public List<RawOccurrenceRecord> parseRawXmlToRor(List<RawXmlOccurrence> raws) {
274    List<RawOccurrenceRecord> rors = new ArrayList<RawOccurrenceRecord>();
275    for (RawXmlOccurrence raw : raws) {
276      List<RawOccurrenceRecord> innerRors = XmlFragmentParser.parseRecord(raw);
277      rors.addAll(innerRors);
278    }
279    return rors;
280  }

282  /**
283   * Takes a dwca file and parses into RawOccurrenceRecords. The file can be any of:
284   * 1) valid dwca with .zip or .tar.gz compression and extension
285   * 2) single text file with delimiters (eg tab, space, comma)
286   * 3) as 2), but compressed using gzip or zip
287   *
288   * @param dwca the file containing dwc records
289   *
290   * @return an iterable collection of RawOccurrenceRecords
291   */
292  public Iterable<RawOccurrenceRecord> parseDwca(File dwca) {
293
294    File workingFile = dwca;
295
296    // uncompress any files with extensions suggesting compression into a temporary directory
297    if (dwca.getPath().endsWith(".zip") || dwca.getPath().endsWith(".gz")) {
298      try {
299        workingFile = File.createTempFile(System.currentTimeMillis() + "", "");
300        workingFile.delete();
301        workingFile.mkdir();
302        /** TODO: cleaning up on jvm shutdown may not be often enough! */
303        workingFile.deleteOnExit();
304      } catch (IOException e) {
305        LOG.warn("Could not create temporary directory for extracting archive file [{}]", dwca.getPath(), e);
306        return null;
307      }
308    }
309
310    if (dwca.getPath().endsWith(".zip")) {
311      try {
312        CompressionUtil.unzipFile(workingFile, dwca);
313      } catch (IOException e) {
314        LOG.warn("Could not open [{}] as a Zip archive", e);
315        return null;
316      }
317    } else if (dwca.getPath().endsWith(".tar.gz")) {
318      try {
319        CompressionUtil.ungzipFile(workingFile, dwca);
320      } catch (IOException e) {
321        LOG.warn("Could not open [{}] as a tar.gz archive", e);
322        return null;
323      }
324    } else if (dwca.getPath().endsWith(".gz")) {
325      try {
326        CompressionUtil.ungzipFile(workingFile, dwca, false);
327      } catch (IOException e) {
328        LOG.warn("Could not open [{}] as a gzip archive", e);
329        return null;
330      }
331    }
332
333    BufferedDwcaParser dwcaParser = new BufferedDwcaParser(workingFile, 10000);
334    LOG.info("Reading dwca from [{}]", workingFile.getAbsolutePath());
335
336    return dwcaParser;
337  }
