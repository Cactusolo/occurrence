001package org.gbif.occurrencestore.download.util;
002
003import org.gbif.occurrencestore.common.model.constants.FieldName;
004
005import java.io.ByteArrayInputStream;
006import java.io.File;
007import java.io.FileWriter;
008import java.io.IOException;
009import java.io.InputStream;
010import java.io.OutputStream;
011import java.util.List;
012
013import com.google.common.base.Charsets;
014import com.google.common.base.Joiner;
015import com.google.common.base.Strings;
016import com.google.common.collect.Lists;
017import com.google.common.io.ByteStreams;
018import com.google.common.io.Closer;
019
020import static org.gbif.occurrence.download.util.HiveFieldUtil.DOWNLOAD_COLUMNS;
021import static org.gbif.occurrence.download.util.HiveFieldUtil.getHiveField;
022
023/**
024 * Utility class that generates a headers file for occurrence downloads.
025 */
026public class HeadersFileUtil {
027
028  public static final String DEFAULT_FILE_NAME = "headers.txt";
029
030  public static final String HEADERS_FILE_PATH = "inc/" + DEFAULT_FILE_NAME;
031
032  public static final String OCCURRENCE_HEADER = Joiner.on('\t').join(getDownloadHeaders()) + '\n';
033
034  /**
035   * Empty private constructor.
036   */
037  private HeadersFileUtil() {
038    // empty constructor
039  }
040
041  /**
042   * Appends the occurrence headers line to the output file.
043   */
044  public static void appendHeaders(OutputStream fileWriter) throws IOException {
045    Closer resultCloser = Closer.create();
046    try {
047      InputStream headerInputStream =
048        resultCloser.register(new ByteArrayInputStream(OCCURRENCE_HEADER.getBytes(Charsets.UTF_8)));
049      ByteStreams.copy(headerInputStream, fileWriter);
050    } finally {
051      resultCloser.close();
052    }
053  }
054
055  /**
056   * Creates the headers file.
057   * The output file name can be specified as argument. If the file name is not specified the file is generated in the
058   * current directory with the name "headers.txt".
059   */
060  public static void generateHeadersFile(String outFileName) throws IOException {
061    Closer closer = Closer.create();
062    String fileName = Strings.isNullOrEmpty(outFileName) ? DEFAULT_FILE_NAME : outFileName;
063    try {
064      FileWriter fileWriter = closer.register(new FileWriter(new File(fileName)));
065      fileWriter.write(OCCURRENCE_HEADER);
066    } finally {
067      closer.close();
068    }
069  }
070
071  /**
072   * Returns the headers names of download columns.
073   */
074  public static String[] getColumnHeaders() {
075    List<String> headers = Lists.newArrayList();
076    for (FieldName field : DOWNLOAD_COLUMNS) {
077      headers.add(getHiveField(field));
078    }
079    return headers.toArray(new String[headers.size()]);
080  }
081
082  /**
083   * Returns the headers names of download columns.
084   */
085  public static String[] getDownloadHeaders() {
086    List<String> headers = Lists.newArrayList();
087    for (FieldName field : DOWNLOAD_COLUMNS) {
088      String col = getHiveField(field);
089      if (col.endsWith("_")) {
090        col = col.substring(0, col.length() - 1);
091      }
092      headers.add(col);
093    }
094    return headers.toArray(new String[headers.size()]);
095  }
096
097}
