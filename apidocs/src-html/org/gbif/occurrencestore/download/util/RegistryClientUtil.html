001package org.gbif.occurrencestore.download.util;
002
003import org.gbif.api.service.registry.DatasetOccurrenceDownloadUsageService;
004import org.gbif.api.service.registry.DatasetService;
005import org.gbif.api.service.registry.OccurrenceDownloadService;
006import org.gbif.occurrencestore.download.oozie.JacksonJsonContextResolver;
007import org.gbif.registry.ws.client.DatasetOccurrenceDownloadUsageWsClient;
008import org.gbif.registry.ws.client.DatasetWsClient;
009import org.gbif.registry.ws.client.OccurrenceDownloadWsClient;
010import org.gbif.utils.file.properties.PropertiesUtil;
011import org.gbif.ws.client.guice.SingleUserAuthModule;
012
013import java.io.IOException;
014import java.util.Properties;
015
016import com.google.inject.AbstractModule;
017import com.google.inject.Guice;
018import com.google.inject.Injector;
019import com.sun.jersey.api.client.config.ClientConfig;
020import com.sun.jersey.api.client.config.DefaultClientConfig;
021import com.sun.jersey.api.client.filter.ClientFilter;
022import com.sun.jersey.api.json.JSONConfiguration;
023import com.sun.jersey.client.apache.ApacheHttpClient;
024import org.codehaus.jackson.jaxrs.JacksonJsonProvider;
025
026/**
027 * Utility class to create registry web service clients.
028 */
029public class RegistryClientUtil {
030
031  private static final int REGISTRY_CLIENT_TO = 600000; // registry client default timeout
032
033  private static final String DOWNLOAD_USER_KEY = "occurrence-download.ws.username";
034  private static final String DOWNLOAD_PASSWORD_KEY = "occurrence-download.ws.password";
035
036  private final Injector injector;
037
038
039  /**
040   * Constructs an instance using properties class instance.
041   */
042  public RegistryClientUtil(Properties properties) {
043    injector = Guice.createInjector(createAuthModuleInstance(properties));
044  }
045
046
047  /**
048   * Constructs an instance using a properties file name.
049   */
050  public RegistryClientUtil(String propertiesFile) {
051    try {
052      injector = Guice.createInjector(createAuthModuleInstance(PropertiesUtil.loadProperties(propertiesFile)));
053    } catch (IllegalArgumentException e) {
054      throw e;
055    } catch (IOException e) {
056      throw new IllegalStateException(e);
057    }
058  }
059
060  /**
061   * Creates an HTTP client.
062   */
063  private static ApacheHttpClient createHttpClient() {
064    ClientConfig cc = new DefaultClientConfig();
065    cc.getClasses().add(JacksonJsonContextResolver.class);
066    cc.getClasses().add(JacksonJsonProvider.class);
067    cc.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, true);
068    cc.getProperties().put(ClientConfig.PROPERTY_CONNECT_TIMEOUT, REGISTRY_CLIENT_TO);
069    return ApacheHttpClient.create(cc);
070  }
071
072  /**
073   * Sets up a registry DatasetService client avoiding the use of guice as our gbif jackson libraries clash with the
074   * hadoop versions.
075   * Sets up an http client with a one minute timeout and http support only.
076   * 
077   * @throws IOException
078   */
079  public DatasetService setupDatasetService(final String uri) {
080    return new DatasetWsClient(createHttpClient().resource(uri), injector.getInstance(ClientFilter.class));
081  }
082
083  /**
084   * Sets up a DatasetOccurrenceDownloadUsageService client avoiding the use of guice as our gbif jackson libraries
085   * clash with the hadoop versions.
086   * Sets up an http client with a one minute timeout and http support only.
087   * 
088   * @throws IOException
089   */
090  public DatasetOccurrenceDownloadUsageService setupDatasetUsageService(final String uri) {
091    return new DatasetOccurrenceDownloadUsageWsClient(createHttpClient().resource(uri),
092      injector.getInstance(ClientFilter.class));
093  }
094
095
096  /**
097   * Sets up a OccurrenceDownloadService client avoiding the use of guice as our gbif jackson libraries
098   * clash with the hadoop versions.
099   * Sets up an http client with a one minute timeout and http support only.
100   * 
101   * @throws IOException
102   */
103  public OccurrenceDownloadService setupOccurrenceDownloadService(final String uri) {
104    return new OccurrenceDownloadWsClient(createHttpClient().resource(uri),
105      injector.getInstance(ClientFilter.class));
106  }
107
108
109  /**
110   * Creates a instance of the gbif authentication module.
111   */
112  private AbstractModule createAuthModuleInstance(Properties properties) {
113    return new SingleUserAuthModule(properties.getProperty(DOWNLOAD_USER_KEY),
114      properties.getProperty(DOWNLOAD_PASSWORD_KEY));
115  }
116}
