001package org.gbif.occurrencestore.download.service;
002
003import org.gbif.api.service.occurrence.DownloadRequestService;
004import org.gbif.occurrence.download.util.DownloadUtils;
005import org.gbif.service.guice.PrivateServiceModule;
006
007import java.util.Map;
008import java.util.Properties;
009
010import javax.mail.Session;
011
012import com.google.common.collect.ImmutableMap;
013import com.google.inject.Provides;
014import com.google.inject.Singleton;
015import com.google.inject.name.Named;
016import org.apache.oozie.client.OozieClient;
017
018public class OccurrenceDownloadServiceModule extends PrivateServiceModule {
019
020  private static final String PREFIX = "occurrence-download.";
021  private static final String JOB_TRACKER = "jobtracker";
022  private static final String NAME_NODE = "namenode";
023
024  public OccurrenceDownloadServiceModule(Properties properties) {
025    super(PREFIX, properties);
026  }
027
028  @Override
029  protected void configureService() {
030    bind(DownloadEmailUtils.class);
031    bind(DownloadRequestService.class).to(DownloadRequestServiceImpl.class);
032    expose(DownloadRequestService.class);
033
034    bind(CallbackService.class).to(DownloadRequestServiceImpl.class);
035    expose(CallbackService.class);
036  }
037
038  @Provides
039  @Singleton
040  Session providesMailSession(@Named("mail.smtp") String smtpServer,
041    @Named("mail.from") String fromAddress) {
042    Properties props = new Properties();
043    props.setProperty("mail.smtp.host", smtpServer);
044    props.setProperty("mail.from", fromAddress);
045    return Session.getInstance(props, null);
046  }
047
048  @Provides
049  @Singleton
050  OozieClient providesOozieClient(@Named("oozie.url") String url) {
051    return new OozieClient(url);
052  }
053
054  @Provides
055  @Singleton
056  @Named("oozie.default_properties")
057  Map<String, String> providesOozieDefaultProperties(@Named(NAME_NODE) String nameNode,
058    @Named(JOB_TRACKER) String jobTracker, @Named("oozie.workflow.path") String workflowPath,
059    @Named("hive.hdfs.out") String hdfsOutput, @Named("ws.url") String wsUrl, @Named("registry.ws.url") String regUrl,
060    @Named("hive.table") String occurrenceTable, @Named("oozie.mount") String oozieDownloadMount) {
061
062    ImmutableMap.Builder<String, String> builder = ImmutableMap.builder();
063
064    builder.put(OozieClient.APP_PATH, workflowPath)
065      .put(OozieClient.USER_NAME, Constants.OOZIE_USER)
066      .put(OozieClient.WORKFLOW_NOTIFICATION_URL,
067        DownloadUtils.concatUrlPaths(wsUrl, "occurrence/download/request/callback?job_id=$jobId&status=$status"))
068      .put(JOB_TRACKER, jobTracker)
069      .put(NAME_NODE, nameNode)
070      .put("hdfs_hive_path", hdfsOutput)
071      .put("occurrence_table", occurrenceTable)
072      .put("download_mount", oozieDownloadMount)
073      .put("mapreduce.task.classpath.user.precedence", "true")
074      .put("registry_ws", regUrl);
075    // we dont have a specific downloadId yet, submit a placeholder
076    String downloadLinkTemplate = DownloadUtils.concatUrlPaths(wsUrl,
077      "occurrence/download/" + DownloadUtils.DOWNLOAD_ID_PLACEHOLDER + ".zip");
078    builder.put("download_link", downloadLinkTemplate);
079
080    return builder.build();
081  }
082}
