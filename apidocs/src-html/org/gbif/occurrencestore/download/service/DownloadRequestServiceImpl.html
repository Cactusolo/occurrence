086  private static final Joiner EMAIL_JOINER = Joiner.on(';').skipNulls();
087  private static final CharMatcher REPL_INVALID_HIVE_CHARS = CharMatcher.inRange('a', 'z')

092  private static final String HIVE_SELECT;
093
094  static {
095    StringBuilder selectSB = new StringBuilder();
096
097    for (FieldName f : HiveFieldUtil.DOWNLOAD_COLUMNS) {
098      if (selectSB.length() > 0) {
099        selectSB.append(", \n");
100      }
101      String colName = HiveFieldUtil.getHiveField(f);
102      switch (f) {
103        case ID:
104        case DATASET_KEY:
105        case I_NUB_ID:
106        case I_KINGDOM_ID:
107        case I_PHYLUM_ID:
108        case I_CLASS_ID:
109        case I_ORDER_ID:
110        case I_FAMILY_ID:
111        case I_GENUS_ID:
112        case I_SPECIES_ID:
113        case I_ALTITUDE:
114        case I_DEPTH:
115        case I_LATITUDE:
116        case I_LONGITUDE:
117        case I_YEAR:
118        case I_MONTH:
119        case I_ISO_COUNTRY_CODE:
120          selectSB.append("cleanNull(" + colName + ")");
121          break;
122        case CREATED:
123        case MODIFIED:
124        case I_MODIFIED:
125        case I_OCCURRENCE_DATE:
126          selectSB.append("toISO8601(" + colName + ")");
127          break;
128        case I_BASIS_OF_RECORD:
129          // use the enum converter UDF from the workflow
130          selectSB.append("getBorEnum(" + colName + ")");
131          break;
132
133        default:
134          // escape tabs, new lines and line breaks
135          selectSB
136            .append("cleanDelimiters(" + colName + ")");
137          break;
138      }
139      selectSB.append(" AS " + colName);
140    }
141
142    HIVE_SELECT = selectSB.toString();
143  }

185  @Override
186  public void cancel(String downloadKey) {
187    try {
188      Download download = occurrenceDownloadService.get(downloadKey);
189      if (download != null) {
190        if (RUNNING_STATUSES.contains(download.getStatus())) {
191          updateDownloadStatus(download, Download.Status.CANCELLED);
192          client.kill(DownloadUtils.downloadToWorkflowId(downloadKey));
193          LOG.info("Download {} canceled", downloadKey);
194        }
195      } else {
196        throw new NotFoundException(String.format("Download %s not found", downloadKey));
197      }
198    } catch (OozieClientException e) {
199      throw new ServiceUnavailableException("Failed to cancel download " + downloadKey, e);
200    }
201  }

203  @Override
204  public String create(DownloadRequest request) {
205    LOG.debug(">> create");
206    Preconditions.checkNotNull(request);
207
208    HiveQueryVisitor hiveVisitor = new HiveQueryVisitor();
209    SolrQueryVisitor solrVisitor = new SolrQueryVisitor();
210    String hiveQuery;
211    String solrQuery;
212    try {
213      hiveQuery = StringEscapeUtils.escapeXml(hiveVisitor.getHiveQuery(request.getPredicate()));
214      solrQuery = StringEscapeUtils.escapeXml(solrVisitor.getQuery(request.getPredicate()));
215    } catch (QueryBuildingException e) {
216      throw new ServiceUnavailableException("Error building the hive query, attempting to continue", e);
217    }
218    LOG.debug("Attempting to download with hive query [{}]", hiveQuery);
219
220    final String uniqueId = REPL_INVALID_HIVE_CHARS.removeFrom(
221      request.getCreator() + '_' + UUID.randomUUID().toString());
222    final String tmpTable = "download_tmp_" + uniqueId;
223    final String citationTable = "download_tmp_citation_" + uniqueId;
224
225    Properties jobProps = new Properties();
226    jobProps.putAll(defaultProperties);
227    jobProps.setProperty("download_link", downloadLink(wsUrl, DownloadUtils.DOWNLOAD_ID_PLACEHOLDER)); // we dont have a
228    // downloadId yet, submit a placeholder
229    jobProps.setProperty("select", HIVE_SELECT);
230    jobProps.setProperty("query", hiveQuery);
231    jobProps.setProperty("solr_query", solrQuery);
232    jobProps.setProperty("query_result_table", tmpTable);
233    jobProps.setProperty("citation_table", citationTable);
234    jobProps.setProperty(Constants.USER_PROPERTY, request.getCreator());
235    if (request.getNotificationAddresses() != null && !request.getNotificationAddresses().isEmpty()) {
236      jobProps.setProperty(Constants.NOTIFICATION_PROPERTY, EMAIL_JOINER.join(request.getNotificationAddresses()));
237    }
238    // serialize the predicate filter into json
239    StringWriter writer = new StringWriter();
240    try {
241      mapper.writeValue(writer, request.getPredicate());
242      writer.flush();
243      jobProps.setProperty(Constants.FILTER_PROPERTY, writer.toString());
244    } catch (IOException e) {
245      throw new ServiceUnavailableException("Failed to serialize download filter " + request.getPredicate(), e);
246    }
247
248    LOG.debug("job properties: {}", jobProps);
249
250    try {
251      final String jobId = client.run(jobProps);
252      LOG.debug("oozie job id is: [{}], with tmpTable [{}]", jobId, tmpTable);
253      LOG.debug("<< create");
254      String downloadId = DownloadUtils.workflowToDownloadId(jobId);
255      persistDownload(request, downloadId);
256      return downloadId;
257    } catch (OozieClientException e) {
258      throw new ServiceUnavailableException("Failed to create download job", e);
259    }
260  }

263  @Override
264  public InputStream getResult(String downloadKey) {
265    Download d = occurrenceDownloadService.get(downloadKey);
266    if (!d.isAvailable()) {
267      throw new NotFoundException("Download " + downloadKey + " is not ready yet");
268    }
269
270    File localFile = new File(downloadMount, downloadKey + ".zip");
271    try {
272      return new FileInputStream(localFile);
273
274    } catch (IOException e) {
275      throw new IllegalStateException(
276        "Failed to read download " + downloadKey + " from " + localFile.getAbsolutePath(), e);
277    }
278  }

280  /**
281   * Processes a callback from Oozie which update the download status.
282   */
283  public void processCallback(String jobId, String status) {
284    Preconditions.checkNotNull(Strings.isNullOrEmpty(jobId), "<jobId> may not be null or empty");
285    Preconditions.checkNotNull(Strings.isNullOrEmpty(status), "<status> may not be null or empty");
286    final Optional<Job.Status> opStatus = Enums.getIfPresent(Job.Status.class, status.toUpperCase());
287    Preconditions.checkArgument(opStatus.isPresent(), "<status> the requested status is not valid");
288    final Status jobStatus = opStatus.get();
289    String downloadId = DownloadUtils.workflowToDownloadId(jobId);
290
291    LOG.debug("Processing callback for jobId [{}] with status [{}]", jobId, status);
292
293    Download download = occurrenceDownloadService.get(downloadId);
294    // Updates the job status in the registry DB
295    org.gbif.api.model.occurrence.Download.Status newStatus = STATUSES_MAP.get(jobStatus);
296
297    if (download == null) {
298      // Download can be null if the oozie reports status before the download is persisted
299      LOG.error(String.format("Download [%s] not found", downloadId));
300      return;
301    }
302    // Do nor kill a cancelled download
303    if (download.getStatus() == org.gbif.api.model.occurrence.Download.Status.CANCELLED
304      && org.gbif.api.model.occurrence.Download.Status.KILLED == newStatus) {
305      CANCELLED_DOWNLOADS.inc();
306      return;
307    } else {
308      updateDownloadStatus(download, newStatus);
309    }
310
311    // Catch failures
312    if (FAILURE_STATES.contains(jobStatus)) {
313      LOG.error(NOTIFY_ADMIN, "Got callback for unsuccessful query. JobId [{}], Status [{}]", jobId, status);
314      downloadEmailUtils.sendErrorNotificationMail(download);
315      FAILED_DOWNLOADS.inc();
316      return;
317    }
318
319    // Don't notify anyone for RUNNING jobs
320    if (Status.RUNNING == jobStatus) {
321      return;
322    }
323
324    // We got some invalid status
325    if (Status.SUCCEEDED != jobStatus) {
326      LOG.info("Got invalid callback status. JobId [{}], Status[{}]", jobId, status);
327      return;
328    }
329    SUCCESSFUL_DOWNLOADS.inc();
330
331    // notify about download
332    downloadEmailUtils.sendSuccessNotificationMail(download);
333  }

335  /**
336   * Returns the download size in bytes.
337   */
338  private Long getDownloadSize(String downloadKey) {
339    File downloadFile = new File(downloadMount, downloadKey + ".zip");
340    if (downloadFile.exists()) {
341      return downloadFile.length();
342    }
343    return 0L;
344  }

346  /**
347   * Persists the download information.
348   */
349  private void persistDownload(DownloadRequest request, String downloadId) {
350    Download download = new Download();
351    download.setKey(downloadId);
352    download.setRequest(request);
353    download.setStatus(Download.Status.PREPARING);
354    download.setDownloadLink(downloadLink(wsUrl, downloadId));
355    occurrenceDownloadService.create(download);
356  }
357
358
359  /**
360   * Updates the download status and file size.
361   */
362  private void updateDownloadStatus(Download download, Download.Status newStatus) {
363    download.setStatus(newStatus);
364    download.setSize(getDownloadSize(download.getKey()));
365    occurrenceDownloadService.update(download);
366  }
