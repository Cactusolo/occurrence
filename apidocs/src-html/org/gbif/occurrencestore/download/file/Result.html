001package org.gbif.occurrencestore.download.file;
002
003import java.util.Map;
004import java.util.UUID;
005
006import com.google.common.base.Objects;
007
008
009/**
010 * Communicates the result of a file creation job.
011 */
012public class Result implements Comparable<Result> {
013
014  // FileJob associated to this Result instance.
015  private final FileJob fileJob;
016
017  private final Map<UUID, Long> datasetUsages;
018
019  /**
020   * Default constructor.
021   */
022  public Result(FileJob fileJob, Map<UUID, Long> datasetUsages) {
023    this.fileJob = fileJob;
024    this.datasetUsages = datasetUsages;
025  }
026
027  /**
028   * Results are ordered by the fileJob field.
029   */
030  @Override
031  public int compareTo(Result that) {
032    return this.fileJob.compareTo(that.fileJob);
033  }
034
035  @Override
036  public boolean equals(Object obj) {
037    if (this == obj) {
038      return true;
039    }
040    if (!(obj instanceof Result)) {
041      return false;
042    }
043
044    Result that = (Result) obj;
045    return Objects.equal(this.fileJob, that.fileJob);
046  }
047
048  /**
049   * @return the datasetUsages map
050   */
051  public Map<UUID, Long> getDatasetUsages() {
052    return datasetUsages;
053  }
054
055  /**
056   * @return the fileJob
057   */
058  public FileJob getFileJob() {
059    return fileJob;
060  }
061
062  @Override
063  public int hashCode() {
064    return Objects.hashCode(fileJob);
065  }
066}
