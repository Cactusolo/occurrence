043/**
044 * Class that creates a single CSV file using a OccurrenceSearchRequest.
045 * The file is built with a set of worker threads that create a piece of the output file, once the jobs have finished
046 * their jobs a single file is created. The order of the records is respected from the search request by using the
047 * fileJob.startOffset reported by each job.
048 */
049public class OccurrenceFileWriter {
050
051
052  /**
053   * Utility class that holds the general settings of the OccurrenceFileWriter class.
054   */
055  public static class Configuration {
056
057    // Maximum number of workers
058    private final int nrOfWorkers;
059
060    // Minimum number of records per job
061    private final int minNrOfRecords;
062
063    // Limits the maximum number of records that can be processed.
064    // This parameters avoids use this class to create file with a size beyond a maximum.
065    private final int maximumNrOfRecords;
066
067    // Occurrence download lock/counter name
068    private final String lockName;
069
070    /**
071     * Default/full constructor.
072     */
073    @Inject
074    public Configuration(@Named("job.max_threads") int nrOfWorkers, @Named("job.min_records") int minNrOfRecords,
075      @Named("file.max_records") int maximumNrOfRecords, @Named("zookeeper.lock_name") String lockName) {
076      this.nrOfWorkers = nrOfWorkers;
077      this.minNrOfRecords = minNrOfRecords;
078      this.maximumNrOfRecords = maximumNrOfRecords;
079      this.lockName = lockName;
080    }

082  }

116  /**
117   * Utility method that creates a file, if the files exists it is deleted.
118   */
119  private static void createFile(String outFile) {
120    try {
121      File file = new File(outFile);
122      if (file.exists()) {
123        file.delete();
124      }
125      file.createNewFile();
126    } catch (IOException e) {
127      Throwables.propagate(e);
128    }
129
130  }

132
133  /**
134   * Entry point. This method: creates the output files, runs the jobs (OccurrenceFileWriterJob) and then collect the
135   * results. Individual files created by each job are deleted.
136   */
137  public void run(String outFile, String citationFileName, String query) {
138    try {
139      StopWatch stopwatch = new StopWatch();
140      stopwatch.start();
141      ExecutionContextExecutorService executionContext =
142        ExecutionContexts.fromExecutorService(Executors.newFixedThreadPool(conf.nrOfWorkers));
143      createFile(outFile);
144      collectResults(Futures.sequence(runJobs(executionContext, outFile, query), executionContext), outFile,
145        citationFileName);
146      stopwatch.stop();
147      executionContext.shutdownNow();
148      final long timeInSeconds = TimeUnit.MILLISECONDS.toSeconds(stopwatch.getTime());
149      System.out.println(String.format(FINISH_MSG_FMT, TimeUnit.SECONDS.toMinutes(timeInSeconds), timeInSeconds % 60));
150    } catch (IOException e) {
151      Throwables.propagate(e);
152    }
153  }

156  /**
157   * Appends a result file to the output file.
158   */
159  private void appendResult(Result result, OutputStream fileWriter) throws IOException {
160    Closer closer = Closer.create();
161    File dataFile = new File(result.getFileJob().getDataFile());
162    try {
163      FileInputStream fileReader = closer.register(new FileInputStream(dataFile));
164      ByteStreams.copy(fileReader, fileWriter);
165    } catch (FileNotFoundException e) {
166      Throwables.propagate(e);
167    } finally {
168      closer.close();
169      dataFile.delete();
170    }
171  }
172
173  /**
174   * Collects the results of each job.
175   * Iterates over the list of futures to collect individual results.
176   */
177  private void collectResults(Future<Iterable<Result>> futures, String outFile, String citationFileName)
178    throws IOException {
179    Closer outFileCloser = Closer.create();
180    try {
181      List<Result> results =
182        Lists.newArrayList(Await.result(futures, Duration.Inf()));
183      // Results are sorted to respect the original ordering
184      Collections.sort(results);
185      FileOutputStream fileWriter = outFileCloser.register(new FileOutputStream(outFile, true));
186      HeadersFileUtil.appendHeaders(fileWriter);
187      Map<UUID, Long> datasetUsages = Maps.newHashMap();
188      for (Result result : results) {
189        datasetUsages = sumUsages(datasetUsages, result.getDatasetUsages());
190        appendResult(result, fileWriter);
191      }
192      CitationsFileWriter.createCitationFile(datasetUsages, citationFileName, datasetOccUsageService, downloadId);
193    } catch (FileNotFoundException e) {
194      Throwables.propagate(e);
195    } catch (IOException e) {
196      Throwables.propagate(e);
197    } catch (Exception e) {
198      Throwables.propagate(e);
199    } finally {
200      outFileCloser.close();
201    }
202  }

204  /**
205   * Creates and gets a reference to a Lock.
206   */
207  private Lock getLock() {
208    return lockFactory.makeLock(conf.lockName);
209  }

212  /**
213   * Executes a search and get number of records.
214   */
215  private Long getSearchCount(String query) {
216    try {
217      SolrQuery solrQuery = new SolrQuery().setQuery(SolrConstants.DEFAULT_QUERY).setRows(0);
218      if (!Strings.isNullOrEmpty(query)) {
219        solrQuery.addFilterQuery(query);
220      }
221      QueryResponse queryResponse = solrServer.query(solrQuery);
222      return queryResponse.getResults().getNumFound();
223    } catch (SolrServerException e) {
224      logError("Error executing Solr", e);
225      return 0L;
226    }
227  }

233    System.err.println(message);

238  /**
239   * Run the list of jobs. The amount of records is assigned evenly among the worker threads.
240   * If the amount of records is not divisible by the nrOfWorkers the remaining records are assigned "evenly" among the
241   * first jobs.
242   */
243  private List<Future<Result>> runJobs(ExecutionContextExecutorService executionContext, String outFile, String query) {
244    final int nrOfRecords =
245      Math.min(getSearchCount(query).intValue(), conf.maximumNrOfRecords);
246    // Calculates the required workers.
247    int calcNrOfWorkers =
248      conf.minNrOfRecords >= nrOfRecords ? 1 : Math.min(conf.nrOfWorkers, nrOfRecords / conf.minNrOfRecords);
249
250    // Number of records that will be assigned to each job
251    final int sizeOfChunks = Math.max(nrOfRecords / calcNrOfWorkers, 1);
252
253    // Remaining jobs, that are not assigned to a job yet
254    final int remaining = nrOfRecords - (sizeOfChunks * calcNrOfWorkers);
255
256    // How many of the remaining jobs will be assigned to one job
257    int remainingPerJob = remaining > 0 ? Math.max(remaining / calcNrOfWorkers, 1) : 0;
258
259    List<Future<Result>> futures = Lists.newArrayList();
260    createFile(outFile);
261    int from;
262    int to = 0;
263    int additionalJobsCnt = 0;
264    for (int i = 0; i < calcNrOfWorkers; i++) {
265      from = i == 0 ? 0 : to;
266      to = from + sizeOfChunks + remainingPerJob;
267
268      // Calculates the remaining jobs that will be assigned to the new FileJob.
269      additionalJobsCnt += remainingPerJob;
270      if (remainingPerJob != 0 && additionalJobsCnt > remaining) {
271        remainingPerJob = additionalJobsCnt - remaining;
272      } else if (additionalJobsCnt == remaining) {
273        remainingPerJob = 0;
274      }
275      // Awaits for an available thread
276      Lock lock = getLock();
277      lock.lock();
278      // Adds the Job to the list. The file name is the output file name + the sequence i
279      futures.add(Futures.future(new OccurrenceFileWriterJob(new FileJob(from, to, outFile + i, query), lock,
280        solrServer, occurrenceHBaseReader),
281        executionContext));
282    }
283    return futures;
284  }

286  /**
287   * Adds the integer values of each input map.
288   */
289  private Map<UUID, Long> sumUsages(Map<UUID, Long> usages1, Map<UUID, Long> usages2) {
290    Map<UUID, Long> result = Maps.newHashMap();
291    for (Entry<UUID, Long> entry1 : usages1.entrySet()) {
292      Long valueIn2 = usages2.get(entry1.getKey());
293      if (valueIn2 == null) {
294        result.put(entry1.getKey(), entry1.getValue());
295      } else {
296        result.put(entry1.getKey(), entry1.getValue() + valueIn2);
297      }
298    }
299    result.putAll(Maps.difference(usages1, usages2).entriesOnlyOnRight());
300    return result;
301  }
