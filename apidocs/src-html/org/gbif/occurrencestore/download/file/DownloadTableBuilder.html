036/**
037 * Class that wraps the process of creating the occurrence and citation files.
038 * This class can be executed as jvm application that receives the following arguments:
039 * - occurrence data outputFile
040 * - citationFileName
041 * - solr query
042 * - hadoop name node, required to access the hdfs.
043 * - hadoop dfs output directory where the citation and data files will be copied
044 */
045public class DownloadTableBuilder {
046
047  /**
048   * Private guice module that provides bindings the required Modules and dependencies.
049   * The following class are exposed:
050   * - CuratorFramework: this class is exposed only to close the zookeeper connections properly.
051   * - OccurrenceFileWriter: class that creates the occurrence data and citations file.
052   */
053  private static final class DownloadTableBuilderModule extends PrivateServiceModule {
054
055    private static final String LOCKING_PATH = "/runningJobs/";
056
057    private final String downloadId;
058
059    private static final String PROPERTIES_PREFIX = "occurrence-download.";
060
061    /**
062     * Default constructor.
063     */
064    public DownloadTableBuilderModule(Properties properties, String downloadId) {
065      super(PROPERTIES_PREFIX, properties);
066      this.downloadId = downloadId;
067    }
068
069    @Override
070    protected void configureService() {
071      install(new SolrModule());
072      bind(Configuration.class).in(Scopes.SINGLETON);
073      bind(String.class).annotatedWith(Names.named("downloadId")).toInstance(downloadId);
074      bind(OccurrenceFileWriter.class);
075      bind(OccurrenceMapReader.class);
076      expose(CuratorFramework.class);
077      expose(OccurrenceFileWriter.class);
078      expose(DatasetOccurrenceDownloadUsageService.class);
079    }
080
081    @Provides
082    @Singleton
083    CuratorFramework provideCuratorFramework(@Named("zookeeper.namespace") String zookeeperNamespace,
084      @Named("zookeeper.quorum") String zookeeperConnection,
085      @Named("zookeeper.sleep_time") Integer sleepTime, @Named("zookeeper.max_retries") Integer maxRetries)
086      throws IOException {
087      CuratorFramework curator = CuratorFrameworkFactory.builder()
088        .namespace(zookeeperNamespace)
089        .retryPolicy(new ExponentialBackoffRetry(sleepTime, maxRetries))
090        .connectString(zookeeperConnection)
091        .build();
092      curator.start();
093      return curator;
094    }
095
096    @Provides
097    @Singleton
098    DatasetOccurrenceDownloadUsageService provideDatasetOccurrenceDownloadUsageService(
099      @Named("registry.ws.url") String registryWsUri) {
100      RegistryClientUtil registryClientUtil = new RegistryClientUtil(this.getVerbatimProperties());
101      return registryClientUtil.setupDatasetUsageService(registryWsUri);
102    }
103
104    @Provides
105    ExecutionContextExecutorService provideExecutionContextExecutorService(
106      @Named("job.max_threads") int maxThreads) {
107      return ExecutionContexts.fromExecutorService(Executors.newFixedThreadPool(maxThreads));
108    }
109
110    @Provides
111    HTablePool provideHTablePool(@Named("max_connection_pool") Integer maxConnectionPool) {
112      return new HTablePool(HBaseConfiguration.create(), maxConnectionPool);
113    }
114
115    @Provides
116    LockFactory provideLock(CuratorFramework curatorFramework,
117      @Named("max_global_threads") Integer maxGlobalThreads) {
118      return new ZooKeeperLockFactory(curatorFramework, maxGlobalThreads, LOCKING_PATH);
119    }
120
121  }
122
123  private static final String CONF_FILE = "occurrence-download.properties";
124
125
126  /**
127   * Entry point, receives the following arguments:
128   * - occurrence data outputFile
129   * - citationFileName
130   * - solr query
131   * - hadoop name node, required to access the hdfs.
132   * - hadoop dfs output directory where the citation and data files will be copied
133   */
134  public static void main(String[] args) throws IOException {
135    DownloadTableBuilder downloadTableBuilder = new DownloadTableBuilder();
136    downloadTableBuilder.createFiles(args[0], args[1], args[2], args[3], args[4],
137      DownloadUtils.workflowToDownloadId(args[5]));
138  }
139
140  /**
141   * Executes the file creation process.
142   * Citation and data files are created in the local file system and the moved to hadoop file system directory
143   * 'hdfsPath'.
144   */
145  public void createFiles(String outputFile, String citationFileName, String query, String nameNode, String hdfsPath,
146    String downloadId)
147    throws IOException {
148    final Injector injector = createInjector(downloadId);
149    CuratorFramework curator = injector.getInstance(CuratorFramework.class);
150    OccurrenceFileWriter occurrenceFileWriter = injector.getInstance(OccurrenceFileWriter.class);
151    occurrenceFileWriter.run(outputFile, citationFileName, query);
152    FileSystem fileSystem = getHadoopFileSystem(nameNode);
153    fileSystem.copyFromLocalFile(true, new Path(outputFile), buildDestinationPath(hdfsPath, outputFile));
154    fileSystem.copyFromLocalFile(true, new Path(citationFileName), buildDestinationPath(hdfsPath, citationFileName));
155    curator.close();
156  }
157
158  /**
159   * Build the destination path for occurrence data a citation files.
160   * The built path will have the form: hdfsRootPath Path.SEPARATOR + dataFile + Path.SEPARATOR + dataFile; which means
161   * that creates a directory with the same file name, this is the format expected later in the download workflow by the
162   * ArchiveBuilder class since the output files can be generated by DownloadTableBuilder class or else using Hive.
163   */
164  private Path buildDestinationPath(String hdfsRootPath, String dataFile) {
165    return new Path(hdfsRootPath + Path.SEPARATOR + dataFile + Path.SEPARATOR + dataFile);
166  }
167
168  /**
169   * Utility method that creates the Guice injector.
170   */
171  private Injector createInjector(String jobId) {
172    try {
173      return Guice.createInjector(new DownloadTableBuilderModule(PropertiesUtil.loadProperties(CONF_FILE), jobId));
174    } catch (IllegalArgumentException e) {
175      Throwables.propagate(e);
176    } catch (IOException e) {
177      Throwables.propagate(e);
178    }
179    throw new IllegalStateException("Guice couldn't be initialized");
180  }
181
182  private FileSystem getHadoopFileSystem(String nameNode) {
183    try {
184      org.apache.hadoop.conf.Configuration conf = new org.apache.hadoop.conf.Configuration();
185      conf.set(CommonConfigurationKeys.FS_DEFAULT_NAME_KEY, nameNode);
186      return FileSystem.get(conf);
187    } catch (IOException e) {
188      throw new IllegalArgumentException(e);
189    }
190  }
