036/**
037 * Reads a occurrence record from HBase and return it in a Map<String,Object>.
038 */
039public class OccurrenceMapReader {
040
041  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceMapReader.class);
042
043  private final String occurrenceTableName;
044  private final HTablePool tablePool;
045
046  private static final BasisOfRecordConverter BOR_CONVERTER = new BasisOfRecordConverter();
047
048
049  @Inject
050  public OccurrenceMapReader(@Named("occurrence_table_name") String occurrenceTableName, HTablePool tablePool) {
051    this.occurrenceTableName = occurrenceTableName;
052    this.tablePool = tablePool;
053  }
054
055  /**
056   * Creates a Map that contains an occurrence record.
057   * The key values are taken from the enumeration FieldName.
058   * 
059   * @return A Map, or an empty map if the result row parameter is null or empty.
060   */
061  public static Map<String, Object> buildOccurrence(@Nullable Result row) {
062    if (row != null && !row.isEmpty()) {
063      Map<String, Object> occ = Maps.newHashMap();
064      Integer key = Bytes.toInt(row.getRow());
065      occ.put(getHiveField(FieldName.ID), key);
066      occ.put(getHiveField(FieldName.DATASET_KEY), getUuid(row, FieldName.DATASET_KEY));
067      occ.put(getHiveField(FieldName.INSTITUTION_CODE), getCleanString(row, FieldName.INSTITUTION_CODE));
068      occ.put(getHiveField(FieldName.COLLECTION_CODE), getCleanString(row, FieldName.COLLECTION_CODE));
069      occ.put(getHiveField(FieldName.CATALOG_NUMBER), getCleanString(row, FieldName.CATALOG_NUMBER));
070      occ.put(getHiveField(FieldName.I_BASIS_OF_RECORD),
071        BOR_CONVERTER.toEnum(getInteger(row, FieldName.I_BASIS_OF_RECORD)));
072      occ.put(getHiveField(FieldName.I_SCIENTIFIC_NAME), getString(row, FieldName.I_SCIENTIFIC_NAME));
073      occ.put(getHiveField(FieldName.AUTHOR), getCleanString(row, FieldName.AUTHOR));
074      occ.put(getHiveField(FieldName.I_NUB_ID), getInteger(row, FieldName.I_NUB_ID));
075      occ.put(getHiveField(FieldName.I_KINGDOM), getString(row, FieldName.I_KINGDOM));
076      occ.put(getHiveField(FieldName.I_PHYLUM), getString(row, FieldName.I_PHYLUM));
077      occ.put(getHiveField(FieldName.I_CLASS), getString(row, FieldName.I_CLASS));
078      occ.put(getHiveField(FieldName.I_ORDER), getString(row, FieldName.I_ORDER));
079      occ.put(getHiveField(FieldName.I_FAMILY), getString(row, FieldName.I_FAMILY));
080      occ.put(getHiveField(FieldName.I_GENUS), getString(row, FieldName.I_GENUS));
081      occ.put(getHiveField(FieldName.I_SPECIES), getString(row, FieldName.I_SPECIES));
082      occ.put(getHiveField(FieldName.I_KINGDOM_ID), getInteger(row, FieldName.I_KINGDOM_ID));
083      occ.put(getHiveField(FieldName.I_PHYLUM_ID), getInteger(row, FieldName.I_PHYLUM_ID));
084      occ.put(getHiveField(FieldName.I_CLASS_ID), getInteger(row, FieldName.I_CLASS_ID));
085      occ.put(getHiveField(FieldName.I_ORDER_ID), getInteger(row, FieldName.I_ORDER_ID));
086      occ.put(getHiveField(FieldName.I_FAMILY_ID), getInteger(row, FieldName.I_FAMILY_ID));
087      occ.put(getHiveField(FieldName.I_GENUS_ID), getInteger(row, FieldName.I_GENUS_ID));
088      occ.put(getHiveField(FieldName.I_SPECIES_ID), getInteger(row, FieldName.I_SPECIES_ID));
089      occ.put(getHiveField(FieldName.I_ISO_COUNTRY_CODE), getString(row, FieldName.I_ISO_COUNTRY_CODE));
090      occ.put(getHiveField(FieldName.HOST_COUNTRY), getString(row, FieldName.HOST_COUNTRY));
091      occ.put(getHiveField(FieldName.OWNING_ORG_KEY), getUuid(row, FieldName.OWNING_ORG_KEY));
092      occ.put(getHiveField(FieldName.I_LATITUDE), getDouble(row, FieldName.I_LATITUDE));
093      occ.put(getHiveField(FieldName.I_LONGITUDE), getDouble(row, FieldName.I_LONGITUDE));
094      occ.put(getHiveField(FieldName.I_YEAR), getInteger(row, FieldName.I_YEAR));
095      occ.put(getHiveField(FieldName.I_MONTH), getInteger(row, FieldName.I_MONTH));
096      occ.put(getHiveField(FieldName.I_OCCURRENCE_DATE), toISO8601Date(getDate(row, FieldName.I_OCCURRENCE_DATE)));
097      occ.put(getHiveField(FieldName.I_ALTITUDE), getInteger(row, FieldName.I_ALTITUDE));
098      occ.put(getHiveField(FieldName.I_DEPTH), getInteger(row, FieldName.I_DEPTH));
099      occ.put(getHiveField(FieldName.SCIENTIFIC_NAME), getCleanString(row, FieldName.SCIENTIFIC_NAME));
100      occ.put(getHiveField(FieldName.RANK), getCleanString(row, FieldName.RANK));
101      occ.put(getHiveField(FieldName.KINGDOM), getCleanString(row, FieldName.KINGDOM));
102      occ.put(getHiveField(FieldName.PHYLUM), getCleanString(row, FieldName.PHYLUM));
103      occ.put(getHiveField(FieldName.CLASS), getCleanString(row, FieldName.CLASS));
104      occ.put(getHiveField(FieldName.ORDER), getCleanString(row, FieldName.ORDER));
105      occ.put(getHiveField(FieldName.FAMILY), getCleanString(row, FieldName.FAMILY));
106      occ.put(getHiveField(FieldName.GENUS), getCleanString(row, FieldName.GENUS));

115      occ.put(getHiveField(FieldName.MIN_DEPTH), getCleanString(row, FieldName.MIN_DEPTH));
116      occ.put(getHiveField(FieldName.MAX_DEPTH), getCleanString(row, FieldName.MAX_DEPTH));
117      occ.put(getHiveField(FieldName.DEPTH_PRECISION), getCleanString(row, FieldName.DEPTH_PRECISION));
118      occ.put(getHiveField(FieldName.CONTINENT_OCEAN), getCleanString(row, FieldName.CONTINENT_OCEAN));
119      occ.put(getHiveField(FieldName.STATE_PROVINCE), getCleanString(row, FieldName.STATE_PROVINCE));
120      occ.put(getHiveField(FieldName.COUNTY), getCleanString(row, FieldName.COUNTY));
121      occ.put(getHiveField(FieldName.COUNTRY), getCleanString(row, FieldName.COUNTRY));
122      occ.put(getHiveField(FieldName.COLLECTOR_NAME), getCleanString(row, FieldName.COLLECTOR_NAME));
123      occ.put(getHiveField(FieldName.LOCALITY), getCleanString(row, FieldName.LOCALITY));
124      occ.put(getHiveField(FieldName.YEAR), getCleanString(row, FieldName.YEAR));
125      occ.put(getHiveField(FieldName.MONTH), getCleanString(row, FieldName.MONTH));
126      occ.put(getHiveField(FieldName.DAY), getCleanString(row, FieldName.DAY));
127      occ.put(getHiveField(FieldName.BASIS_OF_RECORD), getCleanString(row, FieldName.BASIS_OF_RECORD));
128      occ.put(getHiveField(FieldName.IDENTIFIER_NAME), getCleanString(row, FieldName.IDENTIFIER_NAME));
129      occ.put(getHiveField(FieldName.IDENTIFICATION_DATE), toISO8601Date(getDate(row, FieldName.IDENTIFICATION_DATE)));
130      occ.put(getHiveField(FieldName.PROTOCOL), getString(row, FieldName.PROTOCOL));
131      occ.put(getHiveField(FieldName.CREATED), toISO8601Date(getDate(row, FieldName.CREATED)));
132      occ.put(getHiveField(FieldName.MODIFIED), toISO8601Date(getDate(row, FieldName.MODIFIED)));
133      return occ;
134    }
135    return null;
136  }
137
138  /**
139   * Cleans specials characters from a string value.
140   * Removes tabs, line breaks and new lines.
141   */
142  public static String getCleanString(Result row, FieldName column) {
143    String value = getString(row, column);
144    return value != null ? value.replaceAll(DownloadUtils.DELIMETERS_MATCH, " ") : value;
145  }
146
147  /**
148   * Converts a date object into a String in IS0 8601 format.
149   */
150  public static String toISO8601Date(Date date) {
151    if (date != null) {
152      return new SimpleDateFormat(DownloadUtils.ISO_8601_FORMAT).format(date);
153    }
154    return null;
155  }
156
157  /**
158   * Reads an occurrence record from HBase into Map.
159   * The occurrence record
160   */
161  public Map<String, Object> get(@Nonnull Integer key) throws IOException {
162    Preconditions.checkNotNull(key, "Ocurrence key can't be null");
163    Map<String, Object> occurrence = null;
164    HTableInterface table = null;
165    Closer closer = Closer.create();
166    try {
167      table = tablePool.getTable(occurrenceTableName);
168      closer.register(table);
169      Get get = new Get(Bytes.toBytes(key));
170      Result result = table.get(get);
171      if (result == null || result.isEmpty()) {
172        LOG.debug("Couldn't find occurrence for key [{}], returning null", key);
173        return null;
174      }
175      occurrence = buildOccurrence(result);
176    } catch (IOException e) {
177      throw new ServiceUnavailableException("Could not read from HBase", e);
178    } finally {
179      closer.close();
180    }
181    return occurrence;
182  }
