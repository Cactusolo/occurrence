001package org.gbif.occurrencestore.download.export;
002
003import org.gbif.api.model.occurrence.DownloadRequest;
004import org.gbif.api.model.occurrence.predicate.Predicate;
005import org.gbif.occurrencestore.download.service.Constants;
006
007import java.io.IOException;
008import java.util.Set;
009
010import com.google.common.base.Splitter;
011import com.google.common.base.Strings;
012import com.google.common.collect.Sets;
013import org.codehaus.jackson.map.ObjectMapper;
014import org.xml.sax.Attributes;
015import org.xml.sax.SAXException;
016import org.xml.sax.helpers.DefaultHandler;
017
018/**
019 * SAX Handler that can parse a single oozie download job definition and extract a download instance from it.
020 * Extracts download properties using the Hadoop configuration conventions.
021 * 
022 * <pre>
023 * {@code
024 * <configuration>
025 *   <property>
026 *     <name>property name</name>
027 *     <value>property value</value>
028 *   </property>
029 * </configuration>}
030 * </pre>
031 */
032public class OozieJobSaxHandler extends DefaultHandler {
033
034  private static final String CONFIGURATION_ELEMENT = "configuration";
035  private static final String PROPERTY_ELEMENT = "property";
036  private static final String NAME_ELEMENT = "name";
037  private static final String VALUE_ELEMENT = "value";
038  // ObjectMappers are thread safe if not reconfigured in code
039  private final ObjectMapper mapper = new ObjectMapper();
040  private static final Splitter EMAIL_SPLITTER = Splitter.on(';').omitEmptyStrings().trimResults();
041
042  // download data
043  private String creator;
044  private Predicate predicate;
045  private Set<String> notificationAddresses;
046
047  private boolean inConfiguration;
048  private boolean inProperty;
049  private StringBuffer chars;
050  private String propName;
051  private String propValue;
052
053  public DownloadRequest buildDownload() {
054    return new DownloadRequest(predicate, creator, notificationAddresses);
055  }
056
057  @Override
058  public void characters(char[] ch, int start, int length) {
059    chars.append(ch, start, length);
060  }
061
062  @Override
063  public void endElement(String uri, String localName, String qName) throws SAXException {
064    if (localName.equalsIgnoreCase(CONFIGURATION_ELEMENT)) {
065      inConfiguration = false;
066    }
067
068    if (inProperty) {
069      String content = Strings.emptyToNull(chars.toString().trim());
070      if (localName.equals(PROPERTY_ELEMENT)) {
071        inProperty = false;
072        // interpret name & value
073        if (propNameEquals(Constants.USER_PROPERTY)) {
074          creator = propValue.trim();
075
076        } else if (propNameEquals(Constants.NOTIFICATION_PROPERTY)) {
077          notificationAddresses = Sets.newHashSet(EMAIL_SPLITTER.split(Strings.nullToEmpty(propValue)));
078
079        } else if (propNameEquals(Constants.FILTER_PROPERTY) && !Strings.isNullOrEmpty(propValue)) {
080          try {
081            predicate = mapper.readValue(propValue, Predicate.class);
082          } catch (IOException e) {
083            // TODO: Handle exception
084          }
085        }
086
087      } else if (localName.equals(NAME_ELEMENT)) {
088        propName = content;
089
090      } else if (localName.equals(VALUE_ELEMENT)) {
091        propValue = content;
092      }
093
094    }
095  }
096
097  @Override
098  public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
099    chars = new StringBuffer();
100    if (localName.equalsIgnoreCase(CONFIGURATION_ELEMENT)) {
101      inConfiguration = true;
102    }
103    if (inConfiguration && localName.equalsIgnoreCase(PROPERTY_ELEMENT)) {
104      inProperty = true;
105      propName = null;
106      propValue = null;
107    }
108  }
109
110  /**
111   * Normalizes the current property name so that old and new property naming styles with . or _ are supported
112   * and then does a case insensitive comparison against the desired name.
113   */
114  private boolean propNameEquals(String wantedPropName) {
115    return propName.replace('.', '_').equalsIgnoreCase(wantedPropName);
116  }
117}
