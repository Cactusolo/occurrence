046/**
047 * Utility class that exports the downloads information from Oozie and Solr.
048 * 2 tab separated file are generated: one containing the downloads information and second one with dataset usages.
049 * The resulting files can be imported into PostgresSQL by using the command COPY, for example:
050 * COPY occurrence_download FROM 'downloads.txt';
051 * COPY dataset_occurrence_download FROM 'datasetusages.txt';
052 */
053public class DownloadsDataExporter {
054
055  /**
056   * Simple Guice module that creates the clients required.
057   */
058  private static class DownloadMetadataImporterModule extends AbstractModule {
059
060    private final String oozieUrl;
061    private final String solrUrl;
062
063    public DownloadMetadataImporterModule(String oozieUrl, String solrUrl) {
064      this.oozieUrl = oozieUrl;
065      this.solrUrl = solrUrl;
066    }
067
068    @Override
069    protected void configure() {
070      bind(SolrServer.class).toInstance(new HttpSolrServer(solrUrl));
071      bind(OozieClient.class).toInstance(new OozieClient(oozieUrl));
072    }
073  }
074
075  private static final String OOZIE_FILTER = "name=" + Constants.OOZIE_USER;
076  private static final String DT_FMT = "yyyy-MM-dd HH:mm:ss.SSSSSZ";
077  private static final int DEFAULT_LIMIT = 30;
078  private static final String ZIP_EXT = ".zip";
079
080  private static final ObjectMapper objectMapper = new ObjectMapper().configure(Feature.FAIL_ON_UNKNOWN_PROPERTIES,
081    false);
082
083  /**
084   * Map to provide conversions from oozie.Job.Status to Download.Status.
085   */
086  private static final ImmutableMap<WorkflowJob.Status, Download.Status> STATUSES_MAP =
087    new ImmutableMap.Builder<WorkflowJob.Status, Download.Status>()
088      .put(WorkflowJob.Status.PREP, Download.Status.PREPARING)
089      .put(WorkflowJob.Status.RUNNING, Download.Status.RUNNING)
090      .put(WorkflowJob.Status.KILLED, Download.Status.KILLED)
091      .put(WorkflowJob.Status.FAILED, Download.Status.FAILED)
092      .put(WorkflowJob.Status.SUCCEEDED, Download.Status.SUCCEEDED)
093      .put(WorkflowJob.Status.SUSPENDED, Download.Status.SUSPENDED).build();
094
095
096  /**
097   * Executable method that creates the files.
098   */
099  public static void export(String oozieUrl, String solrUrl, String downloadsFile, String datasetUsageFile)
100    throws IOException {
101    Injector injector = Guice.createInjector(new DownloadMetadataImporterModule(oozieUrl, solrUrl));
102    createDownloadsFile(injector.getInstance(OozieClient.class), injector.getInstance(SolrServer.class),
103      downloadsFile, datasetUsageFile);
104  }
105
106  /**
107   * Exports the occurrence downlods information from oozie into a tab separated file.
108   */
109  public static void main(String[] args) throws Exception {
110    export(args[0], args[1], args[2], args[3]);
111  }

113  /**
114   * Creates the downloads file.
115   */
116  private static void createDownloadsFile(OozieClient oozieClient, SolrServer solrServer, String downloadsFileName,
117    String datasetUsageFile) throws IOException {
118    int offset = 1;
119    int limit = DEFAULT_LIMIT;
120    Closer closer = Closer.create();
121    try {
122      Writer downloadsWriter =
123        new BufferedWriter(new OutputStreamWriter(new FileOutputStream(downloadsFileName), FileUtils.UTF8));
124      Writer datasetUsageWriter =
125        new BufferedWriter(new OutputStreamWriter(new FileOutputStream(datasetUsageFile), FileUtils.UTF8));
126      closer.register(downloadsWriter);
127      closer.register(datasetUsageWriter);
128      List<WorkflowJob> jobs = oozieClient.getJobsInfo(OOZIE_FILTER, offset, limit);
129      while (jobs.size() > 0) {
130        for (WorkflowJob job : jobs) {
131          // The job is retrieved again form Oozie to get the full configuration data
132          Download download = fromJob(oozieClient.getJobInfo(job.getId()));
133          downloadsWriter.write(toFileLine(download));
134          writeDatasetUsage(solrServer, download, datasetUsageFile, datasetUsageWriter);
135        }
136        offset += Math.min(limit, jobs.size());
137        jobs = oozieClient.getJobsInfo(OOZIE_FILTER, offset, limit);
138      }
139    } catch (Exception e) {
140      throw new IllegalStateException(e);
141    } finally {
142      closer.close();
143    }
144  }

147  /**
148   * Executes a search using a predicate.
149   */
150  private static QueryResponse executePredicateQuery(SolrServer solrServer, Predicate predicate) {
151    try {
152      SolrQueryVisitor solrQueryVisitor = new SolrQueryVisitor();
153      String query = solrQueryVisitor.getQuery(predicate);
154      SolrQuery solrQuery =
155        new SolrQuery()
156          .setQuery(SolrConstants.DEFAULT_QUERY).setRows(0)
157          .setFacet(true).setFacetMinCount(1).setFacetMissing(false).setFacetLimit(-1)
158          .addFacetField(OccurrenceSolrField.DATASET_KEY.getFieldName());
159      if (!Strings.isNullOrEmpty(query)) {
160        solrQuery.addFilterQuery(query);
161      }
162      QueryResponse queryResponse = solrServer.query(solrQuery);
163      return queryResponse;
164    } catch (SolrServerException e) {
165      throw new IllegalStateException(e);
166    } catch (QueryBuildingException e) {
167      throw new IllegalStateException(e);
168    }
169  }
170
171  /**
172   * Builds a download object from an oozie job.
173   */
174  private static Download fromJob(WorkflowJob job) {
175    try {
176      String downloadKey = DownloadUtils.workflowToDownloadId(job.getId());
177      Download download = new Download();
178      download.setRequest(parseDownloadRequest(job));
179      download.setCreated(job.getCreatedTime());
180      download.setDownloadLink(downloadKey + ZIP_EXT);
181      download.setModified(job.getLastModifiedTime());
182      download.setKey(downloadKey);
183      download.setStatus(STATUSES_MAP.get(job.getStatus()));
184      return download;
185    } catch (Exception e) {
186      throw new IllegalArgumentException("Failed to deserialize download filter for job " + job.getId(), e);
187    }
188  }
189
190  /**
191   * Builds a {@link DownloadRequest} object by parsing the job xml configuration.
192   */
193  private static DownloadRequest parseDownloadRequest(WorkflowJob job) {
194    try {
195      SAXParserFactory saxFactory = SAXParserFactory.newInstance();
196      saxFactory.setNamespaceAware(true);
197      saxFactory.setValidating(false);
198      OozieJobSaxHandler handler = new OozieJobSaxHandler();
199      SAXParser parser = saxFactory.newSAXParser();
200      XMLReader xmlReader = parser.getXMLReader();
201      xmlReader.setContentHandler(handler);
202      xmlReader.parse(new InputSource(new StringReader(job.getConf())));
203      return handler.buildDownload();
204    } catch (Exception e) {
205      throw new IllegalArgumentException("Failed to deserialize download filter for job " + job.getId(), e);
206    }
207  }

210  /**
211   * Creates a String with the the downloads fields separated by the tab character.
212   */
213  private static String toFileLine(Download download) throws Exception {
214    SimpleDateFormat dateFormat = new SimpleDateFormat(DT_FMT);
215    return download.getKey() + '\t'
216      + objectMapper.writeValueAsString(download.getRequest().getPredicate())
217      + '\t' + download.getStatus() + '\t' + download.getDownloadLink()
218      + '\t' + download.getRequest().getNotificationAddressesAsString() + '\t' + download.getRequest().getCreator()
219      + '\t' + dateFormat.format(download.getCreated()) + '\t' + dateFormat.format(download.getModified()) + '\n';
220  }

222  /**
223   * Executes the a Solr query using the download.request.predicate and collects the dataset usages for each download.
224   * For each Solr facet a line is added to the file.
225   */
226  private static void writeDatasetUsage(SolrServer solrServer, Download download, String fileName,
227    Writer datasetUsageWriter)
228    throws IOException {
229    try {
230      QueryResponse response = executePredicateQuery(solrServer, download.getRequest().getPredicate());
231      final FacetField facetField = response.getFacetField(OccurrenceSolrField.DATASET_KEY.getFieldName());
232      if (facetField != null) {
233        for (Count count : facetField.getValues()) {
234          datasetUsageWriter.write(download.getKey() + '\t' + count.getName() + '\t' + count.getCount() + '\n');
235        }
236      }
237    } catch (Exception e) {
238      throw new IllegalStateException(e);
239    }
240  }
