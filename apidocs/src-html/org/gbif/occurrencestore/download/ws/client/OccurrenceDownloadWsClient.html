001package org.gbif.occurrencestore.download.ws.client;
002
003import org.gbif.api.model.occurrence.Download;
004import org.gbif.api.model.occurrence.DownloadRequest;
005import org.gbif.api.service.occurrence.DownloadRequestService;
006import org.gbif.occurrencestore.download.ws.client.util.Constants;
007import org.gbif.ws.client.BaseWsGetClient;
008import org.gbif.ws.util.InputStreamUtils;
009
010import java.io.IOException;
011import java.io.InputStream;
012import javax.annotation.Nullable;
013import javax.ws.rs.core.MediaType;
014
015import com.google.inject.Inject;
016import com.sun.jersey.api.client.WebResource;
017import com.sun.jersey.api.client.filter.ClientFilter;
018import org.codehaus.jackson.map.ObjectMapper;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * Client-side implementation to the ChecklistService.
024 */
025public class OccurrenceDownloadWsClient extends BaseWsGetClient<Download, String> implements DownloadRequestService {
026
027  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceDownloadWsClient.class);
028  private static final String ZIP_EXT = ".zip";
029
030  private final ObjectMapper mapper = new ObjectMapper();
031
032  /**
033   * @param resource webapplication root url
034   */
035  @Inject
036  public OccurrenceDownloadWsClient(WebResource resource, @Nullable ClientFilter authFilter) {
037    super(Download.class, resource.path(Constants.OCCURRENCE_DOWNLOAD_PATH), authFilter);
038  }
039
040  @Override
041  public void cancel(String downloadKey) {
042    super.delete(downloadKey);
043  }
044
045
046  @Override
047  public String create(DownloadRequest download) {
048    String jobId;
049    try {
050      jobId = getResource().type(MediaType.APPLICATION_JSON).post(String.class, mapper.writeValueAsBytes(download));
051    } catch (IOException e) {
052      throw new IllegalStateException(e);
053    }
054    LOG.debug("Submitted download, got job id [{}]", jobId);
055    return jobId;
056  }
057
058  @Override
059  public InputStream getResult(String downloadKey) {
060    return InputStreamUtils.wrapStream(getResource(downloadKey + ZIP_EXT));
061  }
062
063}
