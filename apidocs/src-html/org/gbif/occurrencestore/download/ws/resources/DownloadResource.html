013package org.gbif.occurrencestore.download.ws.resources;
014
015import org.gbif.api.model.occurrence.DownloadRequest;
016import org.gbif.api.service.occurrence.DownloadRequestService;
017import org.gbif.api.service.registry.OccurrenceDownloadService;
018import org.gbif.occurrencestore.download.service.CallbackService;
019import org.gbif.ws.security.NotAllowedException;
020import org.gbif.ws.security.NotAuthenticatedException;
021import org.gbif.ws.util.ExtraMediaTypes;
022
023import java.io.InputStream;
024import java.security.AccessControlException;
025import java.security.Principal;
026
027import javax.servlet.http.HttpServletResponse;
028import javax.validation.Valid;
029import javax.ws.rs.Consumes;
030import javax.ws.rs.DELETE;
031import javax.ws.rs.GET;
032import javax.ws.rs.POST;
033import javax.ws.rs.Path;
034import javax.ws.rs.PathParam;
035import javax.ws.rs.Produces;
036import javax.ws.rs.QueryParam;
037import javax.ws.rs.core.Context;
038import javax.ws.rs.core.MediaType;
039import javax.ws.rs.core.Response;
040import javax.ws.rs.core.SecurityContext;
041
042import com.google.inject.Inject;
043import com.google.inject.Singleton;
044import org.apache.bval.guice.Validate;
045import org.slf4j.Logger;
046import org.slf4j.LoggerFactory;
047
048@Path("occurrence/download/request")
049@Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
050@Consumes(MediaType.APPLICATION_JSON)
051@Singleton
052public class DownloadResource {
053
054  private static final Logger LOG = LoggerFactory.getLogger(DownloadResource.class);
055
056  // low quality of source to default to JSON
057  private static final String OCT_STREAM_QS = ";qs=0.5";
058
059  private final DownloadRequestService requestService;
060
061  private final OccurrenceDownloadService occurrenceDownloadService;
062
063  private final CallbackService callbackService;
064
065  @Inject
066  public DownloadResource(DownloadRequestService service, CallbackService callbackService,
067    OccurrenceDownloadService occurrenceDownloadService) {
068    this.requestService = service;
069    this.callbackService = callbackService;
070    this.occurrenceDownloadService = occurrenceDownloadService;
071  }
072
073  @DELETE
074  @Path("{key}")
075  public void delDownload(@PathParam("key") String jobId, @Context SecurityContext security) {
076    // service.get returns a download or throws NotFoundException
077    assertLoginMatches(occurrenceDownloadService.get(jobId).getRequest(), security);
078    LOG.debug("Delete download: [{}]", jobId);
079    requestService.cancel(jobId);
080  }
081
082  @GET
083  @Path("{key}")
084  @Produces(MediaType.APPLICATION_OCTET_STREAM + OCT_STREAM_QS)
085  public InputStream getResult(@PathParam("key") String downloadKey, @Context HttpServletResponse response) {
086    LOG.debug("Get download data: [{}]", downloadKey);
087    // suggest filename for download in http headers
088    response.setHeader("Content-Disposition", "attachment; filename=" + downloadKey + ".zip");
089    return requestService.getResult(downloadKey);
090  }
091
092
093  @GET
094  @Path("callback")
095  public Response oozieCallback(@QueryParam("job_id") String jobId, @QueryParam("status") String status) {
096    LOG.debug("Received callback from Oozie for Job [{}] with status [{}]", jobId, status);
097    callbackService.processCallback(jobId, status);
098    return Response.ok().build();
099  }
100
101  @POST
102  @Validate
103  public String startDownload(@Valid DownloadRequest request, @Context SecurityContext security) {
104    LOG.debug("Download: [{}]", request);
105    // assert authenticated user is the same as in download
106    assertLoginMatches(request, security);
107    String downloadKey = requestService.create(request);
108    LOG.info("Created new download job with key [{}]", downloadKey);
109    return downloadKey;
110  }
111
112  /**
113   * Checks that a user is authenticated and the same user is the creator of the download.
114   * 
115   * @throws AccessControlException if no or wrong user is authenticated
116   */
117  private void assertLoginMatches(DownloadRequest request, SecurityContext security) {
118    // assert authenticated user is the same as in download
119    final Principal principal = security.getUserPrincipal();
120    if (principal == null) {
121      throw new NotAuthenticatedException("No user authenticated for creating a download");
122    } else if (!principal.getName().equals(request.getCreator())) {
123      LOG.warn("Different user authenticated [{}] than download specifies [{}]", principal.getName(),
124        request.getCreator());
125      throw new NotAllowedException(principal.getName() + " not allowed to create download with creator "
126        + request.getCreator());
127    }
128  }
129}
