001package org.gbif.occurrencestore.persistence;
002
003import org.gbif.service.exception.PersistenceException;
004
005import java.io.IOException;
006import java.util.Iterator;
007import java.util.NoSuchElementException;
008
009import org.apache.hadoop.hbase.client.HTableInterface;
010import org.apache.hadoop.hbase.client.HTablePool;
011import org.apache.hadoop.hbase.client.Result;
012import org.apache.hadoop.hbase.client.ResultScanner;
013import org.apache.hadoop.hbase.client.Scan;
014import org.apache.hadoop.hbase.util.Bytes;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018/**
019 * Provides an iterator over occurrence keys for a given HBase scan. To guarantee that all resources are released after
020 * use, please make sure to iterate over the entire result (until hasNext() returns false).
021 */
022public class OccurrenceKeyIterator implements Iterator<Integer> {
023
024  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceKeyIterator.class);
025  private final ResultScanner scanner;
026  private final Iterator<Result> iterator;
027  private final HTablePool tablePool;
028  private final HTableInterface table;
029  private boolean scannerClosed = false;
030
031  /**
032   * Create the iterator from a given tablePool, for the specified occurrence table, and the given scan. Only the row
033   * keys for the returned scan are returned in calls to next().
034   *
035   * @param tablePool the HBase tablePool that holds the connection to HBase
036   * @param occurrenceTableName the occurrence table to scan
037   * @param scan the scan (query) to execute
038   */
039  public OccurrenceKeyIterator(HTablePool tablePool, String occurrenceTableName, Scan scan) {
040    // TODO: heartbeat thread to shutdown/close resources if no activity after x seconds?
041    this.tablePool = tablePool;
042    this.table = tablePool.getTable(occurrenceTableName);
043    try {
044      this.scanner = table.getScanner(scan);
045      this.iterator = scanner.iterator();
046    } catch (IOException e) {
047      throw new PersistenceException("Could not read from HBase", e);
048    } finally {
049      if (table != null) {
050        try {
051          table.close();
052        } catch (IOException e) {
053          LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
054        }
055      }
056    }
057  }
058
059  @Override
060  public boolean hasNext() {
061    if (scannerClosed) return false;
062
063    if (iterator.hasNext()) {
064      return true;
065    } else {
066      scanner.close();
067      if (table != null) {
068        try {
069          table.close();
070        } catch (IOException e) {
071          LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
072        }
073      }
074      scannerClosed = true;
075      return false;
076    }
077  }
078
079  @Override
080  public Integer next() {
081    if (hasNext()) {
082     Result result = iterator.next();
083     return Bytes.toInt(result.getRow());
084    } else {
085      throw new NoSuchElementException();
086    }
087  }
088
089  @Override
090  public void remove() {
091    throw new UnsupportedOperationException("This iterator does not support removal.");
092  }
093}
