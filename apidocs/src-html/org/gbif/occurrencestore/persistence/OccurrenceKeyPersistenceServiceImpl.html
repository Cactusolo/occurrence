001package org.gbif.occurrencestore.persistence;
002
003import org.gbif.occurrencestore.common.model.UniqueIdentifier;
004import org.gbif.occurrencestore.persistence.api.KeyLookupResult;
005import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
006import org.gbif.occurrencestore.persistence.keygen.KeyPersistenceService;
007
008import java.util.Set;
009import java.util.concurrent.TimeUnit;
010
011import com.google.common.collect.Sets;
012import com.google.inject.Inject;
013import com.google.inject.Singleton;
014import com.yammer.metrics.Metrics;
015import com.yammer.metrics.core.Meter;
016import com.yammer.metrics.core.Timer;
017import com.yammer.metrics.core.TimerContext;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021
022@Singleton
023public class OccurrenceKeyPersistenceServiceImpl implements OccurrenceKeyPersistenceService {
024
025  private final KeyPersistenceService<Integer> keyPersistenceService;
026
027  private final Timer lockWaitTimer =
028    Metrics.newTimer(OccurrenceKeyPersistenceServiceImpl.class, "lock wait", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
029  private final Meter lockRequests =
030    Metrics.newMeter(OccurrenceKeyPersistenceServiceImpl.class, "lock req", "lock req", TimeUnit.SECONDS);
031
032  @Inject
033  public OccurrenceKeyPersistenceServiceImpl(KeyPersistenceService<Integer> keyPersistenceService) {
034    this.keyPersistenceService = checkNotNull(keyPersistenceService, "keyPersistenceService can't be null");
035  }
036
037  @Override
038  public KeyLookupResult findKey(Set<UniqueIdentifier> uniqueIdentifiers) {
039    checkNotNull(uniqueIdentifiers, "uniqueIds can't be null");
040    if (uniqueIdentifiers.isEmpty()) {
041      return null;
042    }
043
044    String datasetKey = uniqueIdentifiers.iterator().next().getDatasetKey().toString();
045    Set<String> uniqueStrings = Sets.newHashSet();
046    for (UniqueIdentifier uniqueIdentifier : uniqueIdentifiers) {
047      uniqueStrings.add(uniqueIdentifier.getUnscopedUniqueString());
048    }
049
050    return keyPersistenceService.findKey(uniqueStrings, datasetKey);
051  }
052
053  @Override
054  public Set<Integer> findKeysByDataset(String datasetKey) {
055    return keyPersistenceService.findKeysByScope(datasetKey);
056  }
057
058  /**
059   * Hands off to the member KeyPersistenceService to generate a key for the given uniqueIds.
060   *
061   * @param uniqueIdentifiers the identifiers that all refer to the same occurrence
062   *
063   * @return a KeyLookupResult with the key for this occurrence
064   *
065   * @throws IllegalArgumentException if the uniqueIdentifiers set is empty
066   */
067  @Override
068  public KeyLookupResult generateKey(Set<UniqueIdentifier> uniqueIdentifiers) {
069    checkArgument(!uniqueIdentifiers.isEmpty(), "uniqueIdentifiers can't be empty");
070
071    String datasetKey = uniqueIdentifiers.iterator().next().getDatasetKey().toString();
072    Set<String> uniqueStrings = Sets.newHashSet();
073    for (UniqueIdentifier uniqueIdentifier : uniqueIdentifiers) {
074      uniqueStrings.add(uniqueIdentifier.getUnscopedUniqueString());
075    }
076
077    lockRequests.mark();
078    final TimerContext context = lockWaitTimer.time();
079    try {
080      return keyPersistenceService.generateKey(uniqueStrings, datasetKey);
081    } finally {
082      context.stop();
083    }
084  }
085
086  /**
087   * Hands off to the member KeyPersistenceService to delete the key.
088   */
089  @Override
090  public void deleteKey(int occurrenceKey, String datasetKey) {
091    keyPersistenceService.deleteKey(occurrenceKey, datasetKey);
092  }
093
094  /**
095   * Hands off to the member KeyPersistenceService to delete the key(s).
096   *
097   * @param uniqueIdentifiers corresponding to the lookup entries to delete
098   */
099  @Override
100  public void deleteKeyByUniqueIdentifiers(Set<UniqueIdentifier> uniqueIdentifiers) {
101    String datasetKey = uniqueIdentifiers.iterator().next().getDatasetKey().toString();
102    Set<String> uniqueStrings = Sets.newHashSet();
103    for (UniqueIdentifier uniqueIdentifier : uniqueIdentifiers) {
104      uniqueStrings.add(uniqueIdentifier.getUnscopedUniqueString());
105    }
106    keyPersistenceService.deleteKeyByUniques(uniqueStrings, datasetKey);
107  }
108}
