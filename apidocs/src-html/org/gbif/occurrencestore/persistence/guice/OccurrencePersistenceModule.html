001package org.gbif.occurrencestore.persistence.guice;
002
003import org.gbif.api.service.occurrence.OccurrenceService;
004import org.gbif.occurrencestore.persistence.DatasetDeletionServiceImpl;
005import org.gbif.occurrencestore.persistence.FragmentPersistenceServiceImpl;
006import org.gbif.occurrencestore.persistence.OccurrenceKeyPersistenceServiceImpl;
007import org.gbif.occurrencestore.persistence.OccurrencePersistenceServiceImpl;
008import org.gbif.occurrencestore.persistence.VerbatimOccurrencePersistenceServiceImpl;
009import org.gbif.occurrencestore.persistence.api.DatasetDeletionService;
010import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
011import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
012import org.gbif.occurrencestore.persistence.api.OccurrencePersistenceService;
013import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
014import org.gbif.occurrencestore.persistence.keygen.HBaseLockingKeyService;
015import org.gbif.occurrencestore.persistence.keygen.KeyPersistenceService;
016import org.gbif.occurrencestore.persistence.zookeeper.ZookeeperLockManager;
017import org.gbif.service.guice.PrivateServiceModule;
018
019import java.util.Properties;
020
021import com.google.inject.Provides;
022import com.google.inject.Singleton;
023import com.google.inject.TypeLiteral;
024import com.google.inject.name.Named;
025import com.netflix.curator.framework.CuratorFramework;
026import com.netflix.curator.framework.CuratorFrameworkFactory;
027import com.netflix.curator.retry.RetryNTimes;
028import org.apache.hadoop.hbase.HBaseConfiguration;
029import org.apache.hadoop.hbase.client.HTablePool;
030
031/**
032 * A convenience module to include the OccurrencePersistenceServiceImpl via Guice. See the README for needed
033 * properties.
034 */
035public class OccurrencePersistenceModule extends PrivateServiceModule {
036
037  private static final String PREFIX = "occurrencestore.db.";
038
039  public OccurrencePersistenceModule(Properties properties) {
040    super(PREFIX, properties);
041  }
042
043  @Override
044  protected void configureService() {
045    bind(OccurrenceService.class).to(OccurrencePersistenceServiceImpl.class);
046    bind(OccurrencePersistenceService.class).to(OccurrencePersistenceServiceImpl.class);
047    bind(OccurrenceKeyPersistenceService.class).to(OccurrenceKeyPersistenceServiceImpl.class);
048    bind(VerbatimOccurrencePersistenceService.class).to(VerbatimOccurrencePersistenceServiceImpl.class);
049    bind(FragmentPersistenceService.class).to(FragmentPersistenceServiceImpl.class);
050    bind(ZookeeperLockManager.class).toProvider(ThreadLocalLockProvider.class);
051    bind(new TypeLiteral<KeyPersistenceService<Integer>>(){}).to(HBaseLockingKeyService.class);
052    bind(DatasetDeletionService.class).to(DatasetDeletionServiceImpl.class);
053
054    expose(OccurrenceService.class);
055    expose(OccurrencePersistenceService.class);
056    expose(OccurrenceKeyPersistenceService.class);
057    expose(VerbatimOccurrencePersistenceService.class);
058    expose(FragmentPersistenceService.class);
059    expose(ZookeeperLockManager.class);
060    expose(new TypeLiteral<KeyPersistenceService<Integer>>(){});
061    expose(DatasetDeletionService.class);
062  }
063
064  @Provides
065  public HTablePool provideHTablePool(@Named("max_connection_pool") Integer maxConnectionPool) {
066    return new HTablePool(HBaseConfiguration.create(), maxConnectionPool);
067  }
068
069  @Provides
070  @Singleton
071  public ThreadLocalLockProvider provideLockProvider(@Named("zookeeper.connection_string") String zkUrl) {
072    CuratorFramework curator = CuratorFrameworkFactory.builder().namespace("hbasePersistence").connectString(zkUrl)
073      .retryPolicy(new RetryNTimes(5, 1000)).build();
074    curator.start();
075
076    ThreadLocalLockProvider provider = new ThreadLocalLockProvider(curator);
077
078    return provider;
079  }
080}
