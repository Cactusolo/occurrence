001package org.gbif.occurrencestore.persistence.keygen;
002
003import java.util.Set;
004
005import com.google.common.collect.Sets;
006
007/**
008 * An implementation of KeyBuilder to provide keys for use as secondary indexes to occurrences in HBase.
009 */
010public class OccurrenceKeyBuilder implements KeyBuilder {
011
012  private static final String DELIM = "|";
013
014  @Override
015  public Set<String> buildKeys(Set<String> uniqueStrings, String scope) {
016    Set<String> results = Sets.newHashSet();
017    for (String uniqueString : uniqueStrings) {
018      results.add(buildKey(uniqueString, scope));
019    }
020
021    return results;
022  }
023
024  @Override
025  public String buildKey(String uniqueString, String scope) {
026    return scope + DELIM + uniqueString;
027  }
028}
