001package org.gbif.occurrencestore.persistence.keygen;
002
003import java.util.Set;
004
005/**
006 * This interface describes methods that can be used to build unique keys given a string and its "scope". It is
007 * expected that implementations of this class would incorporate the scope in the resulting "key" (e.g. by
008 * concatenating it with the input uniqueString). It's expected that this class will move outside occurrence
009 * persistence at some point.
010 */
011public interface KeyBuilder {
012
013  /**
014   * Combine the given uniqueStrings with the given scope to produce a set of strings that are now (hopefully)
015   * globally unique.
016   *
017   * @param uniqueStrings the strings unique within the scope
018   * @param scope the scope of the given strings' uniqueness
019   *
020   * @return the uniqueStrings combined with the scope
021   */
022  Set<String> buildKeys(Set<String> uniqueStrings, String scope);
023
024  /**
025   * Combine the given uniqueString with the given scope to produce a string that is now (hopefully) globally unique.
026   *
027   * @param uniqueString the string unique within the scope
028   * @param scope the scope of the given string's uniqueness
029   *
030   * @return the uniqueString combined with the scope
031   */
032  String buildKey(String uniqueString, String scope);
033}
