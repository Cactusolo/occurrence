001package org.gbif.occurrencestore.persistence.keygen;
002
003import org.gbif.occurrencestore.persistence.api.KeyLookupResult;
004import org.gbif.occurrencestore.persistence.constants.HBaseTableConstants;
005import org.gbif.occurrencestore.persistence.guice.ThreadLocalLockProvider;
006
007import java.util.Set;
008
009import com.google.inject.Inject;
010import com.google.inject.name.Named;
011import org.apache.hadoop.hbase.client.HTablePool;
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015import static com.google.common.base.Preconditions.checkArgument;
016
017/**
018 * An extension of AbstracHBaseKeyPersistenceService with a generateKey implementation that uses a thread local
019 * ZookeeperLockManager to ensure thread-safe key gen.
020 *
021 * NOTE: this class is currently unused, as it becomes very slow when used concurrently by many threads.
022 */
023public class ZkLockingKeyService extends AbstractHBaseKeyPersistenceService {
024
025  private static final Logger LOG = LoggerFactory.getLogger(ZkLockingKeyService.class);
026
027  private final ThreadLocalLockProvider zookeeperLockManagerProvider;
028
029  @Inject
030  public ZkLockingKeyService(@Named("id_lookup_table_name") String occurrenceIdTableName,
031    @Named("counter_table_name") String counterTableName, @Named("table_name") String occurrenceTableName,
032    HTablePool tablePool, ThreadLocalLockProvider zookeeperLockManagerProvider) {
033    super(occurrenceIdTableName, counterTableName, occurrenceTableName, tablePool, new OccurrenceKeyBuilder());
034    this.zookeeperLockManagerProvider = zookeeperLockManagerProvider;
035  }
036
037  /**
038   * Takes out a lock in zookeeper for the dataset of this occurrence before generating a key for it. It first checks
039   * to see if a key already exists for this set of uniqueIdentifiers before generating a new key.
040   *
041   * @param uniqueStrings the identifiers that all refer to the same occurrence
042   *
043   * @return a KeyLookupResult with the key for this occurrence
044   *
045   * @throws IllegalArgumentException if the uniqueIdentifiers set is empty
046   * @throws IllegalStateException    if the next available key is greater than can be held in long
047   */
048  @Override
049  public KeyLookupResult generateKey(Set<String> uniqueStrings, String scope) {
050    checkArgument(!uniqueStrings.isEmpty(), "uniqueIdentifiers can't be empty");
051
052    // if it already exists, return it right away
053    KeyLookupResult findResult = findKey(uniqueStrings, scope);
054    if (findResult != null) {
055      LOG.debug("Asked to generate, but found existing.");
056      return findResult;
057    }
058
059    LOG.debug("Waiting for lock");
060    zookeeperLockManagerProvider.get().waitForLock(scope);
061    LOG.debug("Got lock");
062
063    // check again if it already exists, although now we expect it to be null - we have to have the lock for this
064    // because if we check for existence before getting the lock then someone could write a new key and drop the lock
065    // between us checking (finding nothing) and getting the lock and then writing a new (and now incorrect) key
066    try {
067      findResult = findKey(uniqueStrings, scope);
068      if (findResult != null) {
069        LOG.debug(Thread.currentThread().getName() + " Asked to generate, but found existing.");
070        return findResult;
071      }
072
073      // generate new key from counter table
074      Long longKey =
075        counterTableStore.incrementColumnValue(HBaseTableConstants.COUNTER_ROW, HBaseTableConstants.COUNTER_COLUMN, 1);
076      if (longKey > Integer.MAX_VALUE) {
077        throw new IllegalStateException(
078          "The next available occurrence id is greater than what Integer can handle. This is fatal!");
079      }
080      int newOccurrenceKey = longKey.intValue();
081
082      // build the lookup keys from the uniqueIdentifiers
083      Set<String> lookupKeys = keyBuilder.buildKeys(uniqueStrings, scope);
084
085      // write the new id to each of the lookup keys
086      for (String key : lookupKeys) {
087        lookupTableStore.putInt(key, HBaseTableConstants.LOOKUP_KEY_COLUMN, newOccurrenceKey);
088      }
089
090      return new KeyLookupResult(newOccurrenceKey, true);
091    } finally {
092      // in all cases we want to release the lock
093      LOG.debug("Releasing lock");
094      zookeeperLockManagerProvider.get().releaseLock(scope);
095    }
096  }
097}
