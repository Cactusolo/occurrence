001package org.gbif.occurrencestore.persistence.keygen;
002
003import org.gbif.hbase.util.ResultReader;
004import org.gbif.occurrencestore.persistence.api.KeyLookupResult;
005import org.gbif.occurrencestore.persistence.constants.HBaseTableConstants;
006
007import java.util.Map;
008import java.util.Random;
009import java.util.Set;
010import java.util.UUID;
011import java.util.concurrent.TimeUnit;
012
013import com.google.common.collect.Maps;
014import com.google.inject.Inject;
015import com.google.inject.name.Named;
016import com.yammer.metrics.Metrics;
017import com.yammer.metrics.core.Meter;
018import org.apache.hadoop.hbase.client.HTablePool;
019import org.apache.hadoop.hbase.client.Result;
020import org.apache.hadoop.hbase.util.Bytes;
021import org.slf4j.Logger;
022import org.slf4j.LoggerFactory;

024/**
025 * An extension of AbstractHBaseKeyPersistenceService with a generateKey implementation that uses an HBase
026 * implementation of the algorithm described at
027 * <a href="http://dev.gbif.org/code/snippet/CR-OCC-5">http://dev.gbif.org/code/snippet/CR-OCC-5</a>.
028 */
029public class HBaseLockingKeyService extends AbstractHBaseKeyPersistenceService {
030
031  private static final Logger LOG = LoggerFactory.getLogger(HBaseLockingKeyService.class);
032
033  private static final long WAIT_BEFORE_RETRY_MS = 5000;
034  private static final int WAIT_SKEW = 4000;
035  private static final long STALE_LOCK_TIME = 5 * 60 * 1000;
036
037  // The number of IDs to reserve at a time in batch
038  private static final int BATCHED_ID_SIZE = 100;
039  // the next available key to allocate
040  private int currentKey;
041  // our reserved upper key limit for the current batch
042  private int maxReservedKeyInclusive;
043
044  private final Meter reattempts =
045    Metrics.newMeter(HBaseLockingKeyService.class, "reattempts", "reattempts", TimeUnit.SECONDS);
046
047  @Inject
048  public HBaseLockingKeyService(@Named("id_lookup_table_name") String occurrenceIdTableName,
049    @Named("counter_table_name") String counterTableName, @Named("table_name") String occurrenceTableName,
050    HTablePool tablePool) {
051
052    super(occurrenceIdTableName, counterTableName, occurrenceTableName, tablePool, new OccurrenceKeyBuilder());
053  }
054
055  @Override
056  public KeyLookupResult generateKey(Set<String> uniqueStrings, String scope) {
057    Map<String, KeyStatus> statusMap = Maps.newHashMap();
058    Map<String, Integer> existingKeyMap = Maps.newHashMap();
059    byte[] lockId = Bytes.toBytes(UUID.randomUUID().toString());
060
061    // lookupTable schema: lookupKey | status | lock | key
062
063    // all of our locks will have the same timestamp
064    long now = System.currentTimeMillis();
065
066    Set<String> lookupKeys = keyBuilder.buildKeys(uniqueStrings, scope);
067    boolean failed = false;
068    Integer key = null;
069    Integer foundKey = null;
070    for (String lookupKey : lookupKeys) {
071      Result row = lookupTableStore.getRow(lookupKey);
072      LOG.debug("Lookup for [{}] produced [{}]", lookupKey, row);
073      KeyStatus status = null;
074      byte[] existingLock = null;
075      if (row != null) {
076        String rawStatus = ResultReader
077          .getString(row, HBaseTableConstants.LOOKUP_COLUMN_FAMILY, HBaseTableConstants.LOOKUP_STATUS_COLUMN, null);
078        if (rawStatus != null) {
079          status = KeyStatus.valueOf(rawStatus);
080        }
081        existingLock = ResultReader
082          .getBytes(row, HBaseTableConstants.LOOKUP_COLUMN_FAMILY, HBaseTableConstants.LOOKUP_LOCK_COLUMN, null);
083        key = ResultReader
084          .getInteger(row, HBaseTableConstants.LOOKUP_COLUMN_FAMILY, HBaseTableConstants.LOOKUP_KEY_COLUMN, null);
085        LOG.debug("Got existing status [{}] existingLock [{}] key [{}]", status, existingLock, key);
086      }
087
088      if (status == KeyStatus.ALLOCATED) {
089        LOG.debug("Status ALLOCATED, using found key [{}]", key);
090        // even if existingLock is != null, ALLOCATED means the key exists and is final
091        statusMap.put(lookupKey, KeyStatus.ALLOCATED);
092        existingKeyMap.put(lookupKey, key);
093        if (foundKey == null) {
094          foundKey = key;
095        } else {
096          // we've found conflicting keys for our lookupKeys - this is fatal
097          if (foundKey.intValue() != key.intValue()) {
098            failWithConflictingLookup(existingKeyMap);
099          }
100        }
101        LOG.debug("Status ALLOCATED, using found key [{}]", foundKey);
102      } else if (existingLock == null) {
103        // lock is ours for the taking - checkAndPut lockId, expecting null for lockId
104        boolean gotLock = lookupTableStore.checkAndPut(lookupKey, HBaseTableConstants.LOOKUP_LOCK_COLUMN, lockId,
105          HBaseTableConstants.LOOKUP_LOCK_COLUMN, null, now);
106        if (gotLock) {
107          statusMap.put(lookupKey, KeyStatus.ALLOCATING);
108          LOG.debug("Grabbed free lock, now ALLOCATING [{}]", lookupKey);
109        } else {
110          failed = true;
111          LOG.debug("Failed to grab free lock for [{}], breaking", lookupKey);
112          break;
113        }
114      } else {
115        // somebody has written their lockId and so has the lock, but they haven't finished yet (status != ALLOCATED)
116        Long existingLockTs = ResultReader
117          .getTimestamp(row, HBaseTableConstants.LOOKUP_COLUMN_FAMILY, HBaseTableConstants.LOOKUP_LOCK_COLUMN);
118        if (now - existingLockTs > STALE_LOCK_TIME) {
119          LOG.debug("Found stale lock for [{}]", lookupKey);
120          // Someone died before releasing lock.
121          // Note that key could be not null here - this means that thread had the lock, wrote the key, but then
122          // died before releasing lock.
123          // checkandPut our lockId, expecting lock to match the existing lock
124          boolean gotLock = lookupTableStore.checkAndPut(lookupKey, HBaseTableConstants.LOOKUP_LOCK_COLUMN, lockId,
125            HBaseTableConstants.LOOKUP_LOCK_COLUMN, existingLock, now);
126          if (gotLock) {
127            statusMap.put(lookupKey, KeyStatus.ALLOCATING);
128            LOG.debug("Reset stale lock, now ALLOCATING [{}]", lookupKey);
129          } else {
130            // someone beat us to this lock, in one of two ways
131            // 1) they grabbed lock, wrote new id, and released lock, so now status is ALLOCATED and id is final
132            // 2) they grabbed lock so status is a newer lock uuid with recent timestamp
133            // in either case we're toast - abort and try again
134            failed = true;
135            LOG.debug("Failed to reset stale lock for [{}], breaking", lookupKey);
136            break;
137          }
138        } else {
139          // someone has a current lock, we need to give up and try again
140          failed = true;
141          LOG.debug("Hit valid, current lock for [{}], breaking", lookupKey);
142          break;
143        }
144      }
145    }
146
147    if (failed) {
148      LOG.debug("Failed to get lock. Releasing held locks and trying again.");
149      reattempts.mark();
150      releaseLocks(statusMap);
151      try {
152        Random random = new Random();
153        TimeUnit.MILLISECONDS.sleep(WAIT_BEFORE_RETRY_MS + random.nextInt(WAIT_SKEW) - random.nextInt(WAIT_SKEW));
154      } catch (InterruptedException e) {
155        // do nothing
156      }
157      // recurse
158      return generateKey(uniqueStrings, scope);
159    }
160
161    // now we have map of every lookupKey to either ALLOCATED or ALLOCATING, and locks on all ALLOCATING
162    KeyLookupResult lookupResult;
163    if (foundKey == null) {
164      key = getNextKey();
165      LOG.debug("Now assigning new key [{}]", key);
166      lookupResult = new KeyLookupResult(key, true);
167    } else {
168      lookupResult = new KeyLookupResult(foundKey, false);
169    }
170
171    // write the key and update status to ALLOCATED
172    for (Map.Entry<String, KeyStatus> entry : statusMap.entrySet()) {
173      if (entry.getValue() == KeyStatus.ALLOCATING) {
174        // TODO: combine into one put
175        lookupTableStore.putInt(entry.getKey(), HBaseTableConstants.LOOKUP_KEY_COLUMN, key);
176        lookupTableStore
177          .putString(entry.getKey(), HBaseTableConstants.LOOKUP_STATUS_COLUMN, KeyStatus.ALLOCATED.toString());
178      }
179    }
180
181    releaseLocks(statusMap);
182
183    LOG.debug("<< generateKey (generated? [{}] key [{}])", !key.equals(foundKey), key);
184
185    return lookupResult;
186
187  }
188
189  /**
190   * Provides the next available key. Because throughput of an incrementColumnValue is limited by HBase to a few
191   * thousand calls per second, this implementation reserves a batch of IDs at a time, and then allocates them to
192   * the calling threads, until they are exhausted, when it will go and reserve another batch. Failure scenarios
193   * will therefore mean IDs go unused. This is expected to be a rare scenario and therefore acceptable.
194   *
195   * @return the next key
196   */
197  private synchronized int getNextKey() {
198    // if we have exhausted our reserved keys, get a new batch of them
199    if (currentKey == maxReservedKeyInclusive) {
200      // get batch
201      Long longKey = counterTableStore
202        .incrementColumnValue(HBaseTableConstants.COUNTER_ROW, HBaseTableConstants.COUNTER_COLUMN, BATCHED_ID_SIZE);
203      if (longKey > Integer.MAX_VALUE) {
204        throw new IllegalStateException("HBase issuing keys larger than Integer can support");
205      }
206      maxReservedKeyInclusive = longKey.intValue();
207      // safer to calculate our guaranteed safe range than rely on what nextKey was set to
208      currentKey = maxReservedKeyInclusive - BATCHED_ID_SIZE;
209    }
210    currentKey++;
211    return currentKey;
212  }
213
214  private void releaseLocks(Map<String, KeyStatus> statusMap) {
215    for (Map.Entry<String, KeyStatus> entry : statusMap.entrySet()) {
216      if (entry.getValue() == KeyStatus.ALLOCATING) {
217        lookupTableStore.delete(entry.getKey(), HBaseTableConstants.LOOKUP_LOCK_COLUMN);
218      }
219    }
220  }
221
222  private static enum KeyStatus {
223    ALLOCATING, ALLOCATED
224  }
225}
