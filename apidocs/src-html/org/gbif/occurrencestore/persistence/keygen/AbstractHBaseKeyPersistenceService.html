037/**
038 * An abstract implementation of KeyPersistenceService that handles the finding and deleting of keys in HBase, but
039 * leaves the generation of keys to sub-classes.
040 */
041public abstract class AbstractHBaseKeyPersistenceService implements KeyPersistenceService<Integer> {
042
043  private static final Logger LOG = LoggerFactory.getLogger(AbstractHBaseKeyPersistenceService.class);
044
045  private final HTablePool tablePool;
046  private final String lookupTableName;
047  private final HBaseStore<Integer> occurrenceTableStore;
048  protected final HBaseStore<String> lookupTableStore;
049  protected final HBaseStore<Integer> counterTableStore;
050  protected final KeyBuilder keyBuilder;
051
052  public AbstractHBaseKeyPersistenceService(String occurrenceIdTableName, String counterTableName,
053    String occurrenceTableName, HTablePool tablePool, KeyBuilder keyBuilder) {
054    this.lookupTableName = checkNotNull(occurrenceIdTableName, "occurrenceIdTableName can't be null");
055    this.tablePool = checkNotNull(tablePool, "tablePool can't be null");
056    this.keyBuilder = checkNotNull(keyBuilder, "keyBuilder can't be null");
057    this.lookupTableStore =
058      new HBaseStore<String>(occurrenceIdTableName, HBaseTableConstants.LOOKUP_COLUMN_FAMILY, tablePool);
059    this.counterTableStore =
060      new HBaseStore<Integer>(counterTableName, HBaseTableConstants.COUNTER_COLUMN_FAMILY, tablePool);
061    this.occurrenceTableStore =
062      new HBaseStore<Integer>(occurrenceTableName, HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY, tablePool);
063  }
064
065  @Override
066  public abstract KeyLookupResult generateKey(Set<String> uniqueStrings, String scope);
067
068  @Override
069  public KeyLookupResult findKey(Set<String> uniqueStrings, String scope) {
070    checkNotNull(uniqueStrings, "uniqueStrings can't be null");
071    checkNotNull(scope, "scope can't be null");
072    if (uniqueStrings.isEmpty()) {
073      return null;
074    }
075
076    Set<String> lookupKeys = keyBuilder.buildKeys(uniqueStrings, scope);
077    Map<String, Integer> foundOccurrenceKeys = Maps.newHashMap();
078
079    // get the occurrenceKey for each lookupKey, and set a flag if we find any null
080    boolean gotNulls = false;
081    for (String uniqueString : lookupKeys) {
082      Integer occurrenceKey = lookupTableStore.getInt(uniqueString, HBaseTableConstants.LOOKUP_KEY_COLUMN);
083      if (occurrenceKey == null) {
084        gotNulls = true;
085      } else {
086        foundOccurrenceKeys.put(uniqueString, occurrenceKey);
087      }
088    }
089
090    // go through all the returned keys and make sure they're all the same - if not, fail loudly (this means
091    // an inconsistency in the db that we can't resolve here)
092    Integer resultKey = null;
093    for (String uniqueString : lookupKeys) {
094      Integer occurrenceKey = foundOccurrenceKeys.get(uniqueString);
095      if (occurrenceKey != null) {
096        if (resultKey == null) {
097          resultKey = occurrenceKey;
098        } else if (resultKey.intValue() != occurrenceKey.intValue()) {
099          failWithConflictingLookup(foundOccurrenceKeys);
100        }
101      }
102    }
103
104    // if we got an occurrenceKey as well as nulls, then we need to fill in the lookup table with the missing entries
105    if (resultKey != null && gotNulls) {
106      fillMissingKeys(lookupKeys, foundOccurrenceKeys, resultKey);
107    }
108
109    KeyLookupResult result = null;
110    if (resultKey != null) {
111      result = new KeyLookupResult(resultKey, false);
112    }
113
114    return result;
115  }
116
117  @Override
118  public Set<Integer> findKeysByScope(String scope) {
119    Set<Integer> keys = Sets.newHashSet();
120    // note HTableStore isn't capable of ad hoc scans
121    HTableInterface table = null;
122    try {
123      table = tablePool.getTable(lookupTableName);
124      Scan scan = new Scan();
125      scan.setCacheBlocks(false);
126      scan.setCaching(200);
127      scan.setFilter(new PrefixFilter(Bytes.toBytes(scope)));
128      ResultScanner results = table.getScanner(scan);
129      for (Result result : results) {
130        byte[] rawKey = result.getValue(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
131          Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN));
132        if (rawKey != null) {
133          keys.add(Bytes.toInt(rawKey));
134        }
135      }
136    } catch (IOException e) {
137      throw new ServiceUnavailableException("Could not read from HBase", e);
138    } finally {
139      try {
140        if (table != null) {
141          table.close();
142        }
143      } catch (IOException e) {
144        LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
145      }
146    }
147
148    return keys;
149  }

152  /**
153   * Scans the lookup table for instances of the occurrenceKey and deletes those rows. It attempts to scope the scan
154   * for this occurrenceKey within the dataset of the original occurrence, but note that there is no guarantee that the
155   * original occurrence corresponding to this occurrenceKey still exists, so in the worst case this method will do a
156   * full table scan of the lookup table.
157   *
158   * @param occurrenceKey the key to delete
159   * @param datasetKey    the optional "scope" for the lookup (without it this method is very slow)
160   */
161  @Override
162  public void deleteKey(Integer occurrenceKey, @Nullable String datasetKey) {
163    checkNotNull(occurrenceKey, "occurrenceKey can't be null");
164
165    // get the dataset for this occurrence if not handed in as scope
166    String rawDatasetKey = datasetKey;
167    if (rawDatasetKey == null) {
168      rawDatasetKey = occurrenceTableStore
169        .getString(occurrenceKey, HBaseFieldUtil.getHBaseColumn(FieldName.DATASET_KEY).getColumnName());
170    }
171
172    // scan the lookup table for all rows where the key matches our dataset prefix and the cell value is our
173    // target occurrenceKey, then delete those rows
174    Scan scan = new Scan();
175    scan.addColumn(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
176      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN));
177    // TODO: this is still too slow even with prefix - lease timeouts in logs
178    List<Filter> filters = Lists.newArrayList();
179    if (rawDatasetKey == null) {
180      LOG.warn("About to scan lookup table with no datasetKey prefix - target key for deletion is [{}]", occurrenceKey);
181    } else {
182      filters.add(new PrefixFilter(Bytes.toBytes(OccurrenceKeyHelper.buildKeyPrefix(rawDatasetKey))));
183    }
184    Filter valueFilter = new SingleColumnValueFilter(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
185      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN), CompareFilter.CompareOp.EQUAL,
186      Bytes.toBytes(occurrenceKey));
187    filters.add(valueFilter);
188    Filter filterList = new FilterList(FilterList.Operator.MUST_PASS_ALL, filters);
189    scan.setFilter(filterList);
190    HTableInterface lookupTable = tablePool.getTable(lookupTableName);
191    List<Delete> keysToDelete = Lists.newArrayList();
192    try {
193      ResultScanner resultScanner = lookupTable.getScanner(scan);
194      for (Result result : resultScanner) {
195        Delete delete = new Delete(result.getRow());
196        keysToDelete.add(delete);
197      }
198      if (!keysToDelete.isEmpty()) {
199        lookupTable.delete(keysToDelete);
200      }
201    } catch (IOException e) {
202      throw new ServiceUnavailableException("Failure accessing HBase", e);
203    } finally {
204      if (lookupTable != null) {
205        try {
206          lookupTable.close();
207        } catch (IOException e) {
208          LOG.warn("Couldn't return table to pool, continuing with possible memory leak", e);
209        }
210      }
211    }
212  }

214  @Override
215  public void deleteKeyByUniques(Set<String> uniqueStrings, String scope) {
216    checkNotNull(uniqueStrings, "uniqueStrings can't be null");
217    checkNotNull(scope, "scope can't be null");
218
219    // craft a delete for every uniqueString
220    Set<String> lookupKeys = keyBuilder.buildKeys(uniqueStrings, scope);
221    HTableInterface lookupTable = tablePool.getTable(lookupTableName);
222    List<Delete> keysToDelete = Lists.newArrayList();
223    for (String lookupKey : lookupKeys) {
224      keysToDelete.add(new Delete(Bytes.toBytes(lookupKey)));
225    }
226    try {
227      if (!keysToDelete.isEmpty()) {
228        lookupTable.delete(keysToDelete);
229      }
230    } catch (IOException e) {
231      throw new ServiceUnavailableException("Failure accessing HBase", e);
232    } finally {
233      if (lookupTable != null) {
234        try {
235          lookupTable.close();
236        } catch (IOException e) {
237          LOG.warn("Couldn't return table to pool, continuing with possible memory leak", e);
238        }
239      }
240    }
241  }
242
243  protected void failWithConflictingLookup(Map<String, Integer> conflictingKeys) {
244    StringBuilder sb = new StringBuilder("Found inconsistent occurrence keys in looking up unique identifiers:");
245    for (Map.Entry<String, Integer> entry : conflictingKeys.entrySet()) {
246      sb.append("[").append(entry.getKey()).append("]=[").append(entry.getValue()).append("]");
247    }
248    throw new RuntimeException(sb.toString());
249  }
250
251
252  private void fillMissingKeys(Set<String> lookupKeys, Map<String, Integer> foundOccurrenceKeys,
253    Integer occurrenceKey) {
254    for (String lookupKey : lookupKeys) {
255      if (!foundOccurrenceKeys.containsKey(lookupKey)) {
256        lookupTableStore.putInt(lookupKey, HBaseTableConstants.LOOKUP_KEY_COLUMN, occurrenceKey);
257      }
258    }
259  }
