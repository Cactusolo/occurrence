001package org.gbif.occurrencestore.persistence.keygen;
002
003import org.gbif.occurrencestore.persistence.api.KeyLookupResult;
004
005import java.util.Set;
006import javax.annotation.Nullable;
007
008/**
009 * Implementations of this interface will be responsible for finding, generating and deleting the secondary index
010 * keys that are used to lookup entries by their unique identifiers. The initial target implementation is for
011 * occurrence records looked up by their {@link org.gbif.occurrencestore.common.model.UniqueIdentifier}. The type
012 * parameter refers to the type of the entry key in the primary index (e.g. for occurrences this is an integer). It's
013 * expected that this class will move outside of occurrence persistence at some point.
014  */
015public interface KeyPersistenceService<T> {
016
017  /**
018   * Given the strings that are unique within the given scope, find the key for the single record that matches all of
019   * the given strings.
020   *
021   * @param uniqueStrings strings unique within the scope
022   * @param scope a scope for the strings' uniqueness (e.g. a dataset key/name)
023   *
024   * @return the result of the lookup - the key within the KeyLookupResult or null if it is not found
025   */
026  @Nullable
027  KeyLookupResult findKey(Set<String> uniqueStrings, String scope);
028
029  /**
030   * Find the keys for all records within the given scope (e.g. all occurrence keys for a dataset).
031   *
032   * @param scope         a scope for the strings' uniqueness (e.g. a dataset key/name)
033   *
034   * @return the result of the lookup - a List with 0 or more keys of type T
035   */
036  Set<T> findKeysByScope(String scope);
037
038  /**
039   * Generates a new key for the given unique strings and their scope. If an existing key is found for these strings
040   * that is returned instead. The {@link KeyLookupResult} will indicate whether the key was generated or not.
041   *
042   * @param uniqueStrings strings unique within the scope
043   * @param scope a scope for the strings' uniqueness (e.g. a dataset key/name)
044   *
045   * @return the result of the lookup which will contain the key and whether it was found or generated
046   */
047  KeyLookupResult generateKey(Set<String> uniqueStrings, String scope);
048
049  /**
050   * Delete all existing indexes for the given key.
051   *
052   * @param key the entry for which to delete secondary indexes
053   * @param scope search only within the given scope (e.g. dataset) for the given key
054   */
055  void deleteKey(T key, @Nullable String scope);
056
057  /**
058   * Delete the entries in the secondary index corresponding to the passed in uniqueStrings and scope (e.g. delete
059   * occurrence lookup by triplet and dataset).
060   *
061   * @param uniqueStrings strings unique within the scope
062   * @param scope a scope for the strings' uniqueness (e.g. a dataset key/name)
063   */
064  void deleteKeyByUniques(Set<String> uniqueStrings, String scope);
065}
