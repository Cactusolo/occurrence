001package org.gbif.occurrencestore.persistence;
002
003import org.gbif.api.exception.ServiceUnavailableException;
004import org.gbif.occurrencestore.common.model.UniqueIdentifier;
005import org.gbif.occurrencestore.common.model.constants.FieldName;
006import org.gbif.occurrencestore.persistence.api.Fragment;
007import org.gbif.occurrencestore.persistence.api.FragmentCreationResult;
008import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
009import org.gbif.occurrencestore.persistence.api.KeyLookupResult;
010import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
011import org.gbif.occurrencestore.persistence.constants.HBaseTableConstants;
012import org.gbif.occurrencestore.persistence.hbase.HBaseHelper;
013import org.gbif.occurrencestore.persistence.util.OccurrenceBuilder;
014
015import java.io.IOException;
016import java.util.Set;
017import javax.annotation.Nullable;
018
019import com.google.inject.Inject;
020import com.google.inject.Singleton;
021import com.google.inject.name.Named;
022import org.apache.hadoop.hbase.client.Delete;
023import org.apache.hadoop.hbase.client.Get;
024import org.apache.hadoop.hbase.client.HTableInterface;
025import org.apache.hadoop.hbase.client.HTablePool;
026import org.apache.hadoop.hbase.client.Put;
027import org.apache.hadoop.hbase.client.Result;
028import org.apache.hadoop.hbase.util.Bytes;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import static com.google.common.base.Preconditions.checkNotNull;
033
034/**
035 * Reads and writes Fragments to/from HBase.
036 */
037@Singleton
038public class FragmentPersistenceServiceImpl implements FragmentPersistenceService {
039
040  private static final Logger LOG = LoggerFactory.getLogger(FragmentPersistenceServiceImpl.class);
041
042  private final String occurrenceTableName;
043  private final HTablePool tablePool;
044  private final OccurrenceKeyPersistenceService keyService;
045
046  @Inject
047  public FragmentPersistenceServiceImpl(@Named("table_name") String tableName, HTablePool tablePool,
048    OccurrenceKeyPersistenceService keyService) {
049    this.occurrenceTableName = tableName;
050    this.tablePool = tablePool;
051    this.keyService = keyService;
052  }
053
054  /**
055   * Simple HBase get for given key. Note that OccurrenceBuilder could throw ValidationException if the underlying
056   * fragment is missing required fields.
057   *
058   * @param key the key of the fragment (Integer rather than int for use in methods/classes using generic types)
059   *
060   * @return the fragment or null if not found
061   */
062  @Override
063  public Fragment get(@Nullable Integer key) {
064    if (key == null) {
065      return null;
066    }
067
068    Fragment frag = null;
069    HTableInterface table = null;
070    try {
071      table = tablePool.getTable(occurrenceTableName);
072      Get get = new Get(Bytes.toBytes(key));
073      Result result = table.get(get);
074      if (result == null || result.isEmpty()) {
075        LOG.debug("Couldn't find fragment for occurrence key [{}], returning null", key);
076      } else {
077        frag = OccurrenceBuilder.buildFragment(result);
078      }
079    } catch (IOException e) {
080      throw new ServiceUnavailableException("Could not read from HBase", e);
081    } finally {
082      if (table != null) {
083        try {
084          table.close();
085        } catch (IOException e) {
086          LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
087        }
088      }
089    }
090
091    return frag;
092  }
093
094  @Override
095  public FragmentCreationResult insert(Fragment fragment, Set<UniqueIdentifier> uniqueIds) {
096    checkNotNull(fragment, "fragment can't be null");
097    checkNotNull(uniqueIds, "uniqueIds can't be null");
098
099    fragment.setCreated(System.currentTimeMillis());
100    boolean keyCreated = write(fragment, true, uniqueIds);
101
102    return new FragmentCreationResult(fragment, keyCreated);
103  }
104
105  @Override
106  public void update(Fragment fragment) {
107    checkNotNull(fragment, "fragment can't be null");
108    checkNotNull(fragment.getKey(), "fragment's key can't be null for update");
109
110    write(fragment, false, null);
111  }
112
113
114  /**
115   * Returns true if a key was created, false otherwise.
116   */
117  private boolean write(Fragment fragment, boolean createKey, @Nullable Set<UniqueIdentifier> uniqueIds) {
118    if (createKey) {
119      checkNotNull(uniqueIds, "uniqueIds can't be null if createKey is true");
120    }
121
122    boolean keyCreated = false;
123    HTableInterface occTable = null;
124    try {
125      occTable = tablePool.getTable(occurrenceTableName);
126      byte[] cf = Bytes.toBytes(HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY);
127      if (createKey) {
128        KeyLookupResult keyLookupResult = keyService.generateKey(uniqueIds);
129        keyCreated = keyLookupResult.isCreated();
130        fragment.setKey(keyLookupResult.getKey());
131      }
132      writeFields(occTable, cf, fragment, true);
133    } catch (IOException e) {
134      throw new ServiceUnavailableException("Could not access HBase", e);
135    } finally {
136      if (occTable != null) {
137        try {
138          occTable.close();
139        } catch (IOException e) {
140          LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
141        }
142      }
143    }
144
145    return keyCreated;
146  }
147
148  /**
149   * Writes all fields of Fragment to the HBase table. If dn (deleteNulls) is true, any field in the fragment that is
150   * empty will be deleted from the corresponding "column" in the HBase row.  If dn is false, any data already existing
151   * in the HBase table for empty fields in the Fragment will remain.
152   *
153   * @param occTable the HBase table to write to
154   * @param cf the column family of the table to write to
155   * @param frag the fragment to persist
156   * @param dn deleteNulls?
157   *
158   * @throws IOException if communicating with HBase fails
159   */
160  private static void writeFields(HTableInterface occTable, byte[] cf, Fragment frag, boolean dn) throws IOException {
161    Put put = new Put(Bytes.toBytes(frag.getKey()));
162    Delete del = null;
163    if (dn) del = new Delete(Bytes.toBytes(frag.getKey()));
164
165    HBaseHelper.writeField(FieldName.DATASET_KEY,
166      frag.getDatasetKey() == null ? null : Bytes.toBytes(frag.getDatasetKey().toString()), dn, cf, put, del);
167    HBaseHelper.writeField(FieldName.CRAWL_ID,
168      frag.getCrawlId() == null ? null : Bytes.toBytes(frag.getCrawlId()), dn, cf, put, del);
169    HBaseHelper.writeField(FieldName.UNIT_QUALIFIER,
170      frag.getUnitQualifier() == null ? null : Bytes.toBytes(frag.getUnitQualifier()), dn, cf, put, del);
171    HBaseHelper.writeField(FieldName.HARVESTED_DATE,
172      frag.getHarvestedDate() == null ? null : Bytes.toBytes(frag.getHarvestedDate().getTime()), dn, cf, put, del);
173    HBaseHelper.writeField(FieldName.FRAGMENT, frag.getData(), dn, cf, put, del);
174    HBaseHelper.writeField(FieldName.FRAGMENT_HASH, frag.getDataHash(), dn, cf, put, del);
175    HBaseHelper.writeField(FieldName.XML_SCHEMA,
176      frag.getXmlSchema() == null ? null : Bytes.toBytes(frag.getXmlSchema().toString()), dn, cf, put, del);
177    HBaseHelper.writeField(FieldName.PROTOCOL,
178      frag.getProtocol() == null ? null : Bytes.toBytes(frag.getProtocol().toString()), dn, cf, put, del);
179    HBaseHelper.writeField(FieldName.CREATED,
180      frag.getCreated() == null ? null : Bytes.toBytes(frag.getCreated()), dn, cf, put, del);
181
182    occTable.put(put);
183    if (dn && !del.isEmpty()) occTable.delete(del);
184    occTable.flushCommits();
185  }
186}
