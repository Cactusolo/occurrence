048  private static final Logger LOG = LoggerFactory.getLogger(OccurrencePersistenceServiceImpl.class);
049  private static final int SCANNER_BATCH_SIZE = 50;
050  private static final int SCANNER_CACHE_SIZE = 50;
051
052  private final String occurrenceTableName;
053  private final HTablePool tablePool;
054  private static final BasisOfRecordConverter BR_CONVERTER = new BasisOfRecordConverter();

056  @Inject
057  public OccurrencePersistenceServiceImpl(@Named("table_name") String tableName, HTablePool tablePool) {
058    this.occurrenceTableName = checkNotNull(tableName, "tableName can't be null");
059    this.tablePool = checkNotNull(tablePool, "tablePool can't be null");
060  }
061
062  @Override
063  public Occurrence get(@Nullable Integer key) {
064    if (key == null) {
065      return null;
066    }
067    Occurrence occ = null;
068    HTableInterface table = null;
069    try {
070      table = tablePool.getTable(occurrenceTableName);
071      Get get = new Get(Bytes.toBytes(key));
072      Result result = table.get(get);
073      if (result == null || result.isEmpty()) {
074        LOG.debug("Couldn't find occurrence for key [{}], returning null", key);
075        return null;
076      }
077      occ = OccurrenceBuilder.buildOccurrence(result);
078    } catch (IOException e) {
079      throw new ServiceUnavailableException("Could not read from HBase", e);
080    } finally {
081      closeTable(table);
082    }
083
084    return occ;
085  }

087  /**
088   * Note that the returned fragment here is a String that holds the actual xml or json snippet for this occurrence,
089   * and not the Fragment object that is used elsewhere.
090   *
091   * @param key that identifies an occurrence
092   *
093   * @return a String holding the original xml or json snippet for this occurrence
094   */
095  @Override
096  public String getFragment(int key) {
097    String fragment = null;
098    HTableInterface table = null;
099    try {
100      table = tablePool.getTable(occurrenceTableName);
101      Get get = new Get(Bytes.toBytes(key));
102      Result result = table.get(get);
103      if (result == null || result.isEmpty()) {
104        LOG.info("Couldn't find occurrence for id [{}], returning null", key);
105        return null;
106      }
107      byte[] rawFragment = OccurrenceResultReader.getBytes(result, FieldName.FRAGMENT);
108      if (rawFragment != null) {
109        fragment = Bytes.toString(rawFragment);
110      }
111    } catch (IOException e) {
112      throw new ServiceUnavailableException("Could not read from HBase", e);
113    } finally {
114      closeTable(table);
115    }
116
117    return fragment;
118  }

120  @Override
121  public Iterator<Integer> getKeysByColumn(byte[] columnValue, FieldName columnName) {
122    byte[] cf = Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(columnName).getColumnFamilyName());
123    byte[] col = Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(columnName).getColumnName());
124    Scan scan = new Scan();
125    scan.setBatch(SCANNER_BATCH_SIZE);
126    scan.setCaching(SCANNER_CACHE_SIZE);
127    scan.addColumn(cf, col);
128    SingleColumnValueFilter filter =
129      new SingleColumnValueFilter(cf, col, CompareFilter.CompareOp.EQUAL, columnValue);
130    scan.setFilter(filter);
131
132    return new OccurrenceKeyIterator(tablePool, occurrenceTableName, scan);
133  }

135  @Override
136  public void update(Occurrence occ) {
137    checkNotNull(occ, "occurrence can't be null");
138    checkNotNull(occ.getKey(), "occurrence's key can't be null");
139
140    // TODO bean validation of occ
141
142    HTableInterface occTable = null;
143    try {
144      occTable = tablePool.getTable(occurrenceTableName);
145      byte[] cf = Bytes.toBytes(HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY);
146      write(occTable, cf, occ, true);
147    } catch (IOException e) {
148      throw new ServiceUnavailableException("Could not access HBase", e);
149    } finally {
150      closeTable(occTable);
151    }
152  }

154  @Override
155  public Occurrence delete(int occurrenceKey) {
156    Occurrence occurrence = get(occurrenceKey);
157    if (occurrence == null) {
158      LOG.debug("Occurrence for key [{}] not found, ignoring delete request.", occurrenceKey);
159    } else {
160      delete(new ImmutableList.Builder<Integer>().add(occurrenceKey).build());
161    }
162
163    LOG.debug("<< delete [{}]", occurrenceKey);
164    return occurrence;
165  }
166
167  @Override
168  public void delete(List<Integer> occurrenceKeys) {
169    checkNotNull(occurrenceKeys, "occurrenceKeys can't be null");
170
171    HTableInterface occTable = null;
172    try {
173      occTable = tablePool.getTable(occurrenceTableName);
174      List<Delete> deletes = Lists.newArrayList();
175      for (Integer occurrenceKey : occurrenceKeys) {
176        if (occurrenceKey != null) {
177          deletes.add(new Delete(Bytes.toBytes(occurrenceKey)));
178        }
179      }
180      LOG.debug("Deleting [{}] occurrences", occurrenceKeys.size());
181      occTable.delete(deletes);
182    } catch (IOException e) {
183      throw new ServiceUnavailableException("Could not access HBase", e);
184    } finally {
185      closeTable(occTable);
186    }
187  }

304  private static void addIdentifiersToPut(Put put, byte[] columnFamily, Collection<Identifier> records) {
305    put.add(columnFamily, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.IDENTIFIER_COUNT).getColumnName()),
306      Bytes.toBytes(records.size()));
307    int count = -1;
308    for (Identifier record : records) {
309      count++;
310      String idCol = HBaseTableConstants.IDENTIFIER_COLUMN + count;
311      String idTypeCol = HBaseTableConstants.IDENTIFIER_TYPE_COLUMN + count;
312      put.add(columnFamily, Bytes.toBytes(idCol), Bytes.toBytes(record.getIdentifier()));
313      put.add(columnFamily, Bytes.toBytes(idTypeCol), Bytes.toBytes(record.getType().toString()));
314    }
315  }

317  private static void deleteOldIdentifiers(HTableInterface occTable, int id, byte[] columnFamily) throws IOException {
318    Get get = new Get(Bytes.toBytes(id));
319    get.addColumn(columnFamily,
320      Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.IDENTIFIER_COUNT).getColumnName()));
321    Result result = occTable.get(get);
322    Integer maxCount = OccurrenceResultReader.getInteger(result, FieldName.IDENTIFIER_COUNT);
323    if (maxCount != null && maxCount > 0) {
324      Delete delete = new Delete(Bytes.toBytes(id));
325      for (int count = 0; count < maxCount; count++) {
326        String idCol = HBaseTableConstants.IDENTIFIER_COLUMN + count;
327        delete.deleteColumn(columnFamily, Bytes.toBytes(idCol));
328        String idTypeCol = HBaseTableConstants.IDENTIFIER_TYPE_COLUMN + count;
329        delete.deleteColumn(columnFamily, Bytes.toBytes(idTypeCol));
330      }
331      occTable.delete(delete);
332      // set count to 0
333      Put put = new Put(Bytes.toBytes(id));
334      put.add(columnFamily, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.IDENTIFIER_COUNT).getColumnName()),
335        Bytes.toBytes(0));
336      occTable.put(put);
337    }
338  }

340  private static void closeTable(HTableInterface table) {
341    if (table != null) {
342      try {
343        table.close();
344      } catch (IOException e) {
345        LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
346      }
347    }
