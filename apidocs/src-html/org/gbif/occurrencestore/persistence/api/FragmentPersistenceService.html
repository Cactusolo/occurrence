001package org.gbif.occurrencestore.persistence.api;
002
003import org.gbif.occurrencestore.common.model.UniqueIdentifier;
004
005import java.util.Set;
006
007/**
008 * A service for writing and retrieving Fragments.
009 */
010public interface FragmentPersistenceService {
011
012  /**
013   * Get the Fragment with the given key.
014   *
015   * @param key the key of the fragment (Integer rather than int for use in methods/classes using generic types)
016   *
017   * @return the Fragment
018   *
019   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
020   */
021  Fragment get(Integer key);
022
023  /**
024   * Insert a new Fragment, given its UniqueIdentifiers, generating a key for it in the process.
025   *
026   * @param fragment to be persisted
027   * @param uniqueIds the set of UniqueIdentifiers that identify this fragment
028   *
029   * @return the FragmentCreationResult with the Fragment's key set
030   *
031   * @throws IllegalArgumentException if fragment's key field is not null or uniqueIds is empty
032   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
033   */
034  FragmentCreationResult insert(Fragment fragment, Set<UniqueIdentifier> uniqueIds);
035
036  /**
037   * Update an existing Fragment.
038   *
039   * @param fragment the Fragment to update
040   *
041   * @throws IllegalArgumentException if fragment's key field is null
042   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
043   */
044  void update(Fragment fragment);
045}
