001package org.gbif.occurrencestore.persistence.api;
002
003import javax.annotation.Nullable;
004
005/**
006 * An interface to govern the creating and updating of VerbatimOccurrence objects.
007 */
008public interface VerbatimOccurrencePersistenceService {
009
010  /**
011   * Get the VerbatimOccurrence with the given key.
012   *
013   * @param key the key of the occurrence (Integer rather than int for use in methods/classes using generic types)
014   *
015   * @return the VerbatimOccurrence, or null if it is not found
016   *
017   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
018   */
019  @Nullable
020  VerbatimOccurrence get(Integer key);
021
022  /**
023   * Updates an existing VerbatimOccurrence. Note that VerbatimOccurrences can never be inserted as new - they
024   * must always have come from a Fragment originally, and therefore already have an occurrence key.
025   *
026   * @param occurrence the occurrence to update
027   *
028   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
029   * @throws IllegalArgumentException if the key field of the occurrence is null
030   */
031  void update(VerbatimOccurrence occurrence);
032}
