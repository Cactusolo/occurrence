001package org.gbif.occurrencestore.persistence.api;
002
003import org.gbif.api.vocabulary.EndpointType;
004import org.gbif.api.vocabulary.OccurrenceSchemaType;
005
006import java.util.Arrays;
007import java.util.Date;
008import java.util.UUID;
009import javax.annotation.Nullable;
010
011import com.google.common.base.Objects;
012
013import static com.google.common.base.Preconditions.checkNotNull;
014
015/**
016 * Represents an occurrence "fragment" from one of our publishers, either XML from any of our supported schemas (listed
017 * in OccurrenceSchemaType) or JSON from a DarwinCore Archive. In most cases it represents one occurrence record - the
018 * exception is ABCD 2, in which multiple Identification elements may exist (within the xml data) which we interpret as
019 * distinct, multiple occurrences within this one fragment. In the ABCD 2 multiples case the resulting Fragments can
020 * only be differentiated by the scientific name within each Identification, which is stored here and in subsequent
021 * occurrence objects (VerbatimOccurrence and Occurrence) as the UnitQualifier.
022 *
023 * @see OccurrenceSchemaType
024 */
025public class Fragment {
026
027  private Integer key;
028  private final UUID datasetKey;
029  private byte[] data;
030  private byte[] dataHash;
031  private String unitQualifier;
032  private FragmentType fragmentType;
033  private Date harvestedDate;
034  private Integer crawlId;
035  private OccurrenceSchemaType xmlSchema;
036  private EndpointType protocol;
037  private Long created;
038
039  public Fragment(
040    UUID datasetKey,
041    @Nullable byte[] data,
042    @Nullable byte[] dataHash,
043    @Nullable FragmentType fragmentType,
044    @Nullable EndpointType protocol,
045    @Nullable Date harvestedDate,
046    @Nullable Integer crawlId,
047    @Nullable OccurrenceSchemaType xmlSchema,
048    @Nullable String unitQualifier,
049    @Nullable Long created)
050  {
051    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
052    if (data != null) {
053      this.data = Arrays.copyOf(data, data.length);
054    }
055    if (dataHash != null) {
056      this.dataHash = Arrays.copyOf(dataHash, dataHash.length);
057    }
058    this.fragmentType = fragmentType;
059    this.harvestedDate = harvestedDate;
060    this.crawlId = crawlId;
061    this.xmlSchema = xmlSchema;
062    this.protocol = protocol;
063    this.unitQualifier = unitQualifier;
064    this.created = created;
065  }
066
067  public Fragment(UUID datasetKey) {
068    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
069  }
070
071  public Integer getKey() {
072    return key;
073  }
074
075  public void setKey(Integer key) {
076    this.key = key;
077  }
078
079  public UUID getDatasetKey() {
080    return datasetKey;
081  }
082
083  public byte[] getData() {
084    return data;
085  }
086
087  public void setData(byte[] data) {
088    if (data != null) {
089      this.data = Arrays.copyOf(data, data.length);
090    }
091  }
092
093  public byte[] getDataHash() {
094    return dataHash;
095  }
096
097  public void setDataHash(byte[] dataHash) {
098    if (dataHash != null) {
099      this.dataHash = Arrays.copyOf(dataHash, dataHash.length);
100    }
101  }
102
103  public FragmentType getFragmentType() {
104    return fragmentType;
105  }
106
107  public void setFragmentType(FragmentType fragmentType) {
108    this.fragmentType = fragmentType;
109  }
110
111  public Date getHarvestedDate() {
112    return harvestedDate;
113  }
114
115  public void setHarvestedDate(Date harvestedDate) {
116    this.harvestedDate = harvestedDate;
117  }
118
119  public Integer getCrawlId() {
120    return crawlId;
121  }
122
123  public void setCrawlId(Integer crawlId) {
124    this.crawlId = crawlId;
125  }
126
127  public OccurrenceSchemaType getXmlSchema() {
128    return xmlSchema;
129  }
130
131  public void setXmlSchema(OccurrenceSchemaType xmlSchema) {
132    this.xmlSchema = xmlSchema;
133  }
134
135  public String getUnitQualifier() {
136    return unitQualifier;
137  }
138
139  public void setUnitQualifier(String unitQualifier) {
140    this.unitQualifier = unitQualifier;
141  }
142
143  public EndpointType getProtocol() {
144    return protocol;
145  }
146
147  public void setProtocol(EndpointType protocol) {
148    this.protocol = protocol;
149  }
150
151  public Long getCreated() {
152    return created;
153  }
154
155  public void setCreated(Long created) {
156    this.created = created;
157  }
158
159  @Override
160  public int hashCode() {
161    return Objects
162      .hashCode(key, datasetKey, data, dataHash, unitQualifier, fragmentType, harvestedDate, crawlId, xmlSchema,
163        protocol, created);
164  }
165
166  @Override
167  public boolean equals(Object obj) {
168    if (this == obj) {
169      return true;
170    }
171    if (obj == null || getClass() != obj.getClass()) {
172      return false;
173    }
174    final Fragment other = (Fragment) obj;
175    return Objects.equal(this.key, other.key) && Objects.equal(this.datasetKey, other.datasetKey) && Objects
176      .equal(this.data, other.data) && Objects.equal(this.dataHash, other.dataHash) && Objects
177             .equal(this.unitQualifier, other.unitQualifier) && Objects.equal(this.fragmentType, other.fragmentType)
178           && Objects.equal(this.harvestedDate, other.harvestedDate) && Objects.equal(this.crawlId, other.crawlId)
179           && Objects.equal(this.xmlSchema, other.xmlSchema) && Objects.equal(this.protocol, other.protocol) && Objects
180      .equal(this.created, other.created);
181  }
182
183  @Override
184  public String toString() {
185    return Objects.toStringHelper(this).add("key", key).add("datasetKey", datasetKey).add("data", data)
186      .add("dataHash", dataHash).add("unitQualifier", unitQualifier).add("fragmentType", fragmentType)
187      .add("harvestedDate", harvestedDate).add("crawlId", crawlId).add("xmlSchema", xmlSchema).add("protocol", protocol)
188      .add("created", created).toString();
189  }
190
191  public enum FragmentType {
192    XML, JSON
193  }
194}
