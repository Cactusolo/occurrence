001package org.gbif.occurrencestore.persistence.api;
002
003import org.gbif.api.model.occurrence.Occurrence;
004
005import java.util.List;
006import javax.annotation.Nullable;
007
008/**
009 * An interface to govern the updating and deleting of Occurrences. There is no insert because an Occurrence must
010 * always first exist as a Fragment, thereby having an assigned key. With the key any changes to the record are
011 * therefore updates.
012 */
013public interface OccurrenceWriter {
014
015  /**
016   * Updates an existing occurrence. If the key field of the occurrence is null an IllegalArgumentException is thrown.
017   *
018   * @param occurrence the occurrence to update
019   *
020   * @throws IllegalArgumentException if the occurrence key is null
021   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
022   */
023  void update(Occurrence occurrence);
024
025  /**
026   * Deletes the occurrence having the given key. Returns the deleted occurrence, or null if the occurrence was not
027   * found.
028   *
029   * @param occurrenceKey the key of the occurrence
030   *
031   * @return the deleted Occurrence, or null if it was not found
032   *
033   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
034   */
035  @Nullable
036  Occurrence delete(int occurrenceKey);
037
038
039  /**
040   * Deletes the occurrences given in the collection (in the order given).
041   *
042   * @param occurrenceKeys the occurrence keys to delete
043   *
044   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
045   */
046  void delete(List<Integer> occurrenceKeys);
047}
