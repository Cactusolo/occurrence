001package org.gbif.occurrencestore.persistence.api;
002
003import org.gbif.api.service.occurrence.OccurrenceService;
004import org.gbif.occurrencestore.common.model.constants.FieldName;
005
006import java.util.Iterator;
007
008/**
009 * A convenience interface to avoid exposing writing and other, expensive methods at the public api level.
010 */
011public interface OccurrencePersistenceService extends OccurrenceService, OccurrenceWriter {
012
013  /**
014   * Returns an iterator over the occurrence keys that match the given column name and value. Note that this Iterator
015   * will probably be backed by a connection to a datasource (e.g. HBase) and there is the possibility that the
016   * connection will fail and/or timeout. If that should happen a runtime
017   * {@link org.gbif.api.exception.ServiceUnavailableException} will be thrown. To guarantee that any held resources
018   * are properly released, make sure to iterate until getNext() returns false.
019   *
020   * @param columnValue the value to match
021   * @param columnName the column to check for the columnValue
022   *
023   * @return an Iterator over the occurrence keys matching the requested column
024   *
025   * @throws org.gbif.api.exception.ServiceUnavailableException if the underlying data connection fails
026   */
027  Iterator<Integer> getKeysByColumn(byte[] columnValue, FieldName columnName);
028}
