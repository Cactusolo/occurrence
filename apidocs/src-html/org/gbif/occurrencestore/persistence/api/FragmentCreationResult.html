001package org.gbif.occurrencestore.persistence.api;
002
003import com.google.common.base.Objects;
004
005/**
006 * The result of attempting to insert a new Fragment. The usual case will be that the key is created as expected. It is
007 * possible that a race could mean that another snippet with the same unique identifiers creates the key first, making
008 * this an update.
009 */
010public class FragmentCreationResult {
011
012  private final Fragment fragment;
013  private final boolean keyCreated;
014
015  public FragmentCreationResult(Fragment fragment, boolean keyCreated) {
016    this.fragment = fragment;
017    this.keyCreated = keyCreated;
018  }
019
020  public Fragment getFragment() {
021    return fragment;
022  }
023
024  public boolean isKeyCreated() {
025    return keyCreated;
026  }
027
028  @Override
029  public int hashCode() {
030    return Objects.hashCode(fragment, keyCreated);
031  }
032
033  @Override
034  public boolean equals(Object obj) {
035    if (this == obj) {
036      return true;
037    }
038    if (obj == null || getClass() != obj.getClass()) {
039      return false;
040    }
041    final FragmentCreationResult other = (FragmentCreationResult) obj;
042    return Objects.equal(this.fragment, other.fragment) && Objects.equal(this.keyCreated, other.keyCreated);
043  }
044
045  @Override
046  public String toString() {
047    return Objects.toStringHelper(this).add("fragment", fragment).add("keyCreated", keyCreated).toString();
048  }
049}
