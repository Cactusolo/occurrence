001package org.gbif.occurrencestore.persistence.api;
002
003import java.util.UUID;
004
005/**
006 * Describes a service dedicated to the sole purpose of deleting all occurrences for an entire dataset.
007 */
008public interface DatasetDeletionService {
009
010  /**
011   * Deletes the occurrences of the indicated dataset as well as any secondary indexes that may be based on them.
012   *
013   * @param datasetKey the dataset to delete
014   *
015   * @throws org.gbif.api.exception.ServiceUnavailableException
016   *          if the underlying data connection fails
017   */
018  void deleteDataset(UUID datasetKey);
019
020  /**
021   * Deletes the occurrences of the indicated data resource (the legacy dataset identifier from the MySQL portal) as
022   * well as any secondary indexes that may be based on them.
023   *
024   * @param dataResourceId the dataset to delete
025   *
026   * @throws org.gbif.api.exception.ServiceUnavailableException
027   *          if the underlying data connection fails
028   */
029  void deleteDataResource(int dataResourceId);
030}
