042  /**
043   * Builds a Fragment object from the given result, assigning the passed in key.
044   *
045   * @param result an HBase scan/get Result
046   * @return the Fragment or null if the passed in Result is null
047   * @throws ValidationException if the fragment as stored in the table is invalid
048   */
049  public static Fragment buildFragment(@Nullable Result result) {
050    if (result == null) {
051      return null;
052    }
053
054    int key = Bytes.toInt(result.getRow());
055
056    String rawDatasetKey = OccurrenceResultReader.getString(result, FieldName.DATASET_KEY);
057    if (rawDatasetKey == null) {
058      throw new ValidationException("Fragment with key [" + key + "] has no datasetKey.");
059    }
060    UUID datasetKey = UUID.fromString(rawDatasetKey);
061
062    Integer crawlId = OccurrenceResultReader.getInteger(result, FieldName.CRAWL_ID);
063    if (crawlId == null) {
064      throw new ValidationException("Fragment with key [" + key + "] has no crawlId.");
065    }
066    Long harvested = OccurrenceResultReader.getLong(result, FieldName.HARVESTED_DATE);
067    if (harvested == null) {
068      throw new ValidationException("Fragment with key [" + key + "] has no harvestedDate.");
069    }
070    Date harvestedDate = new Date(harvested);
071    String unitQualifier = OccurrenceResultReader.getString(result, FieldName.UNIT_QUALIFIER);
072    byte[] data = OccurrenceResultReader.getBytes(result, FieldName.FRAGMENT);
073    byte[] dataHash = OccurrenceResultReader.getBytes(result, FieldName.FRAGMENT_HASH);
074    Long created = OccurrenceResultReader.getLong(result, FieldName.CREATED);
075    String rawSchema = OccurrenceResultReader.getString(result, FieldName.XML_SCHEMA);
076    OccurrenceSchemaType schema;
077    if (rawSchema == null) {
078      // this is typically called just before updating the fragment, meaning schemaType will then be correctly set
079      LOG.debug("Fragment with key [{}] has no schema type - assuming DWCA.", key);
080      schema = OccurrenceSchemaType.DWCA;
081    } else {
082      schema = OccurrenceSchemaType.valueOf(rawSchema);
083    }
084    String rawProtocol = OccurrenceResultReader.getString(result, FieldName.PROTOCOL);
085    EndpointType protocol = rawProtocol == null ? null : EndpointType.valueOf(rawProtocol);
086
087    Fragment frag;
088    if (schema == null || schema == OccurrenceSchemaType.DWCA) {
089      frag =
090        new Fragment(datasetKey, data, dataHash, Fragment.FragmentType.JSON, protocol, harvestedDate, crawlId, schema,
091          null, created);
092    } else {
093      frag =
094        new Fragment(datasetKey, data, dataHash, Fragment.FragmentType.XML, protocol, harvestedDate, crawlId, schema,
095          unitQualifier, created);
096    }
097    frag.setKey(key);
098
099    return frag;
100  }

102  /**
103   * Utility to build an API Occurrence from an HBase row.
104   *
105   * @return A complete occurrence, or null
106   */
107  public static Occurrence buildOccurrence(@Nullable Result row) {
108    if (row == null || row.isEmpty()) {
109      return null;
110    } else {
111      Occurrence occ = new Occurrence();
112      Integer key = Bytes.toInt(row.getRow());
113      occ.setKey(key);
114      occ.setAltitude(OccurrenceResultReader.getInteger(row, FieldName.I_ALTITUDE));
115      occ.setBasisOfRecord(BOR_CONVERTER.toEnum(OccurrenceResultReader.getInteger(row, FieldName.I_BASIS_OF_RECORD)));
116//      occ.setCatalogNumber(OccurrenceResultReader.getString(row, FieldName.CATALOG_NUMBER));
117      occ.setClassKey(OccurrenceResultReader.getInteger(row, FieldName.I_CLASS_ID));
118      occ.setClazz(OccurrenceResultReader.getString(row, FieldName.I_CLASS));
119//      occ.setCollectionCode(OccurrenceResultReader.getString(row, FieldName.COLLECTION_CODE));
120//      occ.setDataProviderId(OccurrenceResultReader.getInteger(row, FieldName.DATA_PROVIDER_ID));
121//      occ.setDataResourceId(OccurrenceResultReader.getInteger(row, FieldName.DATA_RESOURCE_ID));
122      occ.setDatasetKey(OccurrenceResultReader.getUuid(row, FieldName.DATASET_KEY));
123      occ.setDepth(OccurrenceResultReader.getInteger(row, FieldName.I_DEPTH));
124//      occ.setOccurrenceId(OccurrenceResultReader.getString(row, FieldName.DWC_OCCURRENCE_ID));
125      occ.setFamily(OccurrenceResultReader.getString(row, FieldName.I_FAMILY));
126      occ.setFamilyKey(OccurrenceResultReader.getInteger(row, FieldName.I_FAMILY_ID));
127      occ.setGenus(OccurrenceResultReader.getString(row, FieldName.I_GENUS));
128      occ.setGenusKey(OccurrenceResultReader.getInteger(row, FieldName.I_GENUS_ID));
129//      occ.setGeospatialIssue(OccurrenceResultReader.getInteger(row, FieldName.I_GEOSPATIAL_ISSUE));
130//      occ.setHostCountry(Country.fromIsoCode(OccurrenceResultReader.getString(row, FieldName.HOST_COUNTRY)));
131//      occ.setInstitutionCode(OccurrenceResultReader.getString(row, FieldName.INSTITUTION_CODE));
132      occ.setCountry(Country.fromIsoCode(OccurrenceResultReader.getString(row, FieldName.I_ISO_COUNTRY_CODE)));
133      occ.setKingdom(OccurrenceResultReader.getString(row, FieldName.I_KINGDOM));
134      occ.setKingdomKey(OccurrenceResultReader.getInteger(row, FieldName.I_KINGDOM_ID));
135      occ.setLatitude(OccurrenceResultReader.getDouble(row, FieldName.I_LATITUDE));
136      occ.setLongitude(OccurrenceResultReader.getDouble(row, FieldName.I_LONGITUDE));
137      occ.setModified(OccurrenceResultReader.getDate(row, FieldName.I_MODIFIED));
138//      occ.setOccurrenceMonth(OccurrenceResultReader.getInteger(row, FieldName.I_MONTH));
139//      occ.setNubKey(OccurrenceResultReader.getInteger(row, FieldName.I_NUB_ID));
140//      occ.setOccurrenceDate(OccurrenceResultReader.getDate(row, FieldName.I_OCCURRENCE_DATE));
141      occ.setOrder(OccurrenceResultReader.getString(row, FieldName.I_ORDER));
142      occ.setOrderKey(OccurrenceResultReader.getInteger(row, FieldName.I_ORDER_ID));
143//      occ.setOtherIssue(OccurrenceResultReader.getInteger(row, FieldName.I_OTHER_ISSUE));
144//      occ.setOwningOrgKey(OccurrenceResultReader.getUuid(row, FieldName.OWNING_ORG_KEY));
145      occ.setPhylum(OccurrenceResultReader.getString(row, FieldName.I_PHYLUM));
146      occ.setPhylumKey(OccurrenceResultReader.getInteger(row, FieldName.I_PHYLUM_ID));
147      String rawEndpointType = OccurrenceResultReader.getString(row, FieldName.PROTOCOL);
148      if (rawEndpointType == null) {
149        LOG.warn("EndpointType is null for occurrence [{}] - possibly corrupt record.", key);
150      } else {
151        EndpointType endpointType = EndpointType.valueOf(rawEndpointType);
152        occ.setProtocol(endpointType);
153      }
154//      occ.setResourceAccessPointId(OccurrenceResultReader.getInteger(row, FieldName.RESOURCE_ACCESS_POINT_ID));
155      occ.setScientificName(OccurrenceResultReader.getString(row, FieldName.I_SCIENTIFIC_NAME));
156      occ.setSpecies(OccurrenceResultReader.getString(row, FieldName.I_SPECIES));
157      occ.setSpeciesKey(OccurrenceResultReader.getInteger(row, FieldName.I_SPECIES_ID));
158//      occ.setTaxonomicIssue(OccurrenceResultReader.getInteger(row, FieldName.I_TAXONOMIC_ISSUE));
159//      occ.setUnitQualifier(OccurrenceResultReader.getString(row, FieldName.UNIT_QUALIFIER));
160//      occ.setOccurrenceYear(OccurrenceResultReader.getInteger(row, FieldName.I_YEAR));
161//      occ.setLocality(OccurrenceResultReader.getString(row, FieldName.LOCALITY));
162//      occ.setCounty(OccurrenceResultReader.getString(row, FieldName.COUNTY));
163      occ.setStateProvince(OccurrenceResultReader.getString(row, FieldName.STATE_PROVINCE));
164//      occ.setContinent(OccurrenceResultReader.getString(row, FieldName.CONTINENT_OCEAN)); // no enums in hbase
165//      occ.setCollectorName(OccurrenceResultReader.getString(row, FieldName.COLLECTOR_NAME));
166//      occ.setIdentifierName(OccurrenceResultReader.getString(row, FieldName.IDENTIFIER_NAME));
167      occ.setIdentificationDate(OccurrenceResultReader.getDate(row, FieldName.IDENTIFICATION_DATE));
168      occ.setIdentifiers(extractIdentifiers(key, row, HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY));
169      return occ;
170    }

178  public static VerbatimOccurrence buildVerbatimOccurrence(@Nullable Result row) {
179    if (row == null || row.isEmpty()) {
180      return null;
181    } else {
182      String rawDatasetKey = OccurrenceResultReader.getString(row, FieldName.DATASET_KEY);

184
185      String rawProtocol = OccurrenceResultReader.getString(row, FieldName.PROTOCOL);
186      EndpointType protocol = rawProtocol == null ? null : EndpointType.valueOf(rawProtocol);
187
188      VerbatimOccurrence occ = VerbatimOccurrence.builder().key(Bytes.toInt(row.getRow())).datasetKey(datasetKey)
189        .dataProviderId(OccurrenceResultReader.getInteger(row, FieldName.DATA_PROVIDER_ID))

224        .latitude(OccurrenceResultReader.getString(row, FieldName.LATITUDE))

255  private static List<Identifier> extractIdentifiers(Integer key, Result result, String columnFamily) {
256    List<Identifier> records = Lists.newArrayList();
257    Integer maxCount = OccurrenceResultReader.getInteger(result, FieldName.IDENTIFIER_COUNT);
258    if (maxCount != null) {
259      for (int count = 0; count < maxCount; count++) {
260        String idCol = HBaseTableConstants.IDENTIFIER_COLUMN + count;
261        String idTypeCol = HBaseTableConstants.IDENTIFIER_TYPE_COLUMN + count;
262        String id = ResultReader.getString(result, columnFamily, idCol, null);
263        String rawType = ResultReader.getString(result, columnFamily, idTypeCol, null);
264        if (id != null && rawType != null) {
265          IdentifierType idType = null;
266          try {
267            idType = IdentifierType.valueOf(rawType);
268          } catch (IllegalArgumentException e) {
269            LOG.warn("Unrecognized value for IdentifierType from field [{}] - data is corrupt.", rawType);
270          }
271          if (idType != null) {
272            Identifier record = new Identifier();
273            record.setEntityKey(key);
274            record.setIdentifier(id);
275            record.setType(idType);
276            records.add(record);
277          }
278        }
279      }
280    }
281    return records;
282  }
