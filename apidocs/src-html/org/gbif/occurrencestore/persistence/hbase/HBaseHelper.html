001package org.gbif.occurrencestore.persistence.hbase;
002
003import org.gbif.occurrencestore.common.model.constants.FieldName;
004
005import javax.annotation.Nullable;
006
007import org.apache.hadoop.hbase.client.Delete;
008import org.apache.hadoop.hbase.client.Put;
009import org.apache.hadoop.hbase.util.Bytes;
010
011import static com.google.common.base.Preconditions.checkNotNull;
012
013/**
014 * Helper methods for working with HBase.
015 */
016public class HBaseHelper {
017
018  // shouldn't be instantiated
019  private HBaseHelper() {
020  }
021
022  /**
023   * A convenience method for writing a new value (or deleting an existing value) for a given Occurrence HBase field.
024   * The write operation is added to the passed in Put or Delete.
025   *
026   * @param fieldName   the occurrence field to modify
027   * @param fieldData   the data to write
028   * @param deleteNulls if this is true, and fieldData is null, the fieldName will be added to the passed in Delete
029   * @param cf          the HBase column family for this modification
030   * @param put         the put to be added to
031   * @param del         the delete to be added to
032   */
033  public static void writeField(FieldName fieldName, @Nullable byte[] fieldData, boolean deleteNulls, byte[] cf,
034    Put put, @Nullable Delete del) {
035    checkNotNull(fieldName, "fieldName may not be null");
036    checkNotNull(cf, "cf may not be null");
037    checkNotNull(put, "put may not be null");
038    if (deleteNulls) checkNotNull(del, "del may not be null if deleteNulls is true");
039
040    if (fieldData != null) {
041      put.add(cf, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnName()), fieldData);
042    } else if (deleteNulls) {
043      del.deleteColumn(cf, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnName()));
044    }
045  }
046}
