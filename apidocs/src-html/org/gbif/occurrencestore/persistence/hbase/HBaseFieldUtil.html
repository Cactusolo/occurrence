116  /**
117   * Should never be instantiated.
118   */
119  private HBaseFieldUtil() {
120  }
121
122  public static HBaseColumn getHBaseColumn(FieldName field) {
123    return NAME_MAP.get(field);
124  }
125
126  public static class HBaseColumn {
127    private final String columnFamilyName;
128    private final String columnName;
129
130    public HBaseColumn(String columnFamilyName, String columnName) {
131      this.columnFamilyName = columnFamilyName;
132      this.columnName = columnName;
133    }
134
135    public String getColumnFamilyName() {
136      return columnFamilyName;
137    }
138
139    public String getColumnName() {
140      return columnName;
141    }
142  }
