041/**
042 * Utility class for building Solr queries from supported parameters for occurrences search.
043 */
044public class OccurrenceSearchRequestBuilder {
045
046  // This is a placeholder to map from the JSON definition ID to the query field
047  public static final ImmutableMap<OccurrenceSearchParameter, OccurrenceSolrField> QUERY_FIELD_MAPPING =
048    ImmutableMap.<OccurrenceSearchParameter, OccurrenceSolrField>builder()
049      .put(OccurrenceSearchParameter.LATITUDE, OccurrenceSolrField.LATITUDE)
050      .put(OccurrenceSearchParameter.LONGITUDE, OccurrenceSolrField.LONGITUDE)
051      .put(OccurrenceSearchParameter.YEAR, OccurrenceSolrField.YEAR)
052      .put(OccurrenceSearchParameter.MONTH, OccurrenceSolrField.MONTH)
053      .put(OccurrenceSearchParameter.CATALOG_NUMBER, OccurrenceSolrField.CATALOG_NUMBER)
054      .put(OccurrenceSearchParameter.COLLECTOR_NAME, OccurrenceSolrField.COLLECTOR_NAME)
055      .put(OccurrenceSearchParameter.COLLECTION_CODE, OccurrenceSolrField.COLLECTION_CODE)
056      .put(OccurrenceSearchParameter.INSTITUTION_CODE, OccurrenceSolrField.INSTITUTION_CODE)
057      .put(OccurrenceSearchParameter.DEPTH, OccurrenceSolrField.DEPTH)
058      .put(OccurrenceSearchParameter.ALTITUDE, OccurrenceSolrField.ALTITUDE)
059      .put(OccurrenceSearchParameter.BASIS_OF_RECORD, OccurrenceSolrField.BASIS_OF_RECORD)
060      .put(OccurrenceSearchParameter.DATASET_KEY, OccurrenceSolrField.DATASET_KEY)
061      .put(OccurrenceSearchParameter.SPATIAL_ISSUES, OccurrenceSolrField.GEOSPATIAL_ISSUE)
062      .put(OccurrenceSearchParameter.GEOREFERENCED, OccurrenceSolrField.GEOREFERENCED)
063      .put(OccurrenceSearchParameter.DATE, OccurrenceSolrField.DATE)
064      .put(OccurrenceSearchParameter.MODIFIED, OccurrenceSolrField.MODIFIED)
065      .put(OccurrenceSearchParameter.COUNTRY, OccurrenceSolrField.COUNTRY)
066      .put(OccurrenceSearchParameter.PUBLISHING_COUNTRY, OccurrenceSolrField.PUBLISHING_COUNTRY)
067      .put(OccurrenceSearchParameter.TAXON_KEY, OccurrenceSolrField.TAXON_KEY)
068      .build();
069
070  private static final BasisOfRecordConverter BASIS_OF_RECORD_CONVERTER = new BasisOfRecordConverter();
071
072
073  // Holds the value used for an optional sort order applied to a search via param "sort"
074  private final Map<String, SolrQuery.ORDER> sortOrder;
075
076  // Solr request handler.
077  private final String requestHandler;
078
079  public final static int MAX_OFFSET = 1000000;
080  public final static int MAX_PAGE_SIZE = 300;
081
082  /**
083   * Default constructor.
084   */
085  public OccurrenceSearchRequestBuilder(String requestHandler, Map<String, SolrQuery.ORDER> sortOrder) {
086    this.requestHandler = requestHandler;
087    this.sortOrder = sortOrder;
088  }
089
090  /**
091   * Parses a geometry parameter in WKT format.
092   * If the parsed geometry is a polygon the produced query will be in INTERSECTS(wkt parameter) format.
093   * If the parsed geometry is a rectangle, the query is transformed into a range query using the southmost and
094   * northmost points.
095   */
096  protected static String parseGeometryParam(String wkt) {
097    try {
098      Geometry geometry = new WKTReader().read(wkt);
099      if (geometry.isRectangle()) {
100        Envelope bbox = geometry.getEnvelopeInternal();
101        return String
102          .format(RANGE_FORMAT, bbox.getMinY() + "," + bbox.getMinX(), bbox.getMaxY() + "," + bbox.getMaxX());
103      }
104      return String.format(GEO_INTERSECTS_QUERY_FMT, wkt);
105    } catch (ParseException e) {
106      throw new IllegalArgumentException(e);
107    }
108  }
109
110
111  public SolrQuery build(@Nullable OccurrenceSearchRequest request) {
112    final int maxOffset = MAX_OFFSET - request.getLimit();
113    Preconditions.checkArgument(request.getOffset() <= maxOffset, "maximum offset allowed is %s", MAX_OFFSET);
114
115    SolrQuery solrQuery = new SolrQuery();
116    // q param
117    solrQuery.setQuery(DEFAULT_QUERY);
118    // paging
119    setQueryPaging(request, solrQuery, MAX_PAGE_SIZE);
120    // sets the filters
121    setFilterParameters(request, solrQuery);
122    // sorting
123    setSortOrder(solrQuery, sortOrder);
124    // set the request handler
125    setRequestHandler(solrQuery, requestHandler);
126
127    return solrQuery;
128  }
129
130  /**
131   * Adds an occurrence date parameter: DATE or MODIFIED.
132   */
133  private void addDateQuery(Multimap<OccurrenceSearchParameter, String> params,
134    OccurrenceSearchParameter dateParam, OccurrenceSolrField solrField, List<String> filterQueries) {
135    if (params.containsKey(dateParam)) {
136      List<String> dateParams = new ArrayList<String>();
137      for (String value : params.get(dateParam)) {
138        dateParams.add(PARAMS_JOINER.join(solrField.getFieldName(), toDateQuery(value)));
139      }
140      filterQueries.add(toParenthesesQuery(PARAMS_OR_JOINER.join(dateParams)));
141    }
142  }
143
144  /**
145   * Add the occurrence bounding box and polygon parameters.
146   * Those 2 parameters are returned in 1 filter expression because both refer to same Solr field: coordinate.
147   */
148  private void addLocationQuery(Multimap<OccurrenceSearchParameter, String> params, List<String> filterQueries) {
149    if (params.containsKey(OccurrenceSearchParameter.GEOMETRY)) {
150      List<String> locationParams = new ArrayList<String>();
151      for (String value : params.get(OccurrenceSearchParameter.GEOMETRY)) {
152        locationParams
153          .add(PARAMS_JOINER.join(OccurrenceSolrField.COORDINATE.getFieldName(), parseGeometryParam(value)));
154      }
155      filterQueries.add(toParenthesesQuery(PARAMS_OR_JOINER.join(locationParams)));
156    }
157  }
158
159  /**
160   * Adds the filter query to SolrQuery object.
161   * Creates a conjunction of disjunctions: disjunctions(ORs) are created for the filter applied to the same field;
162   * The those disjunctions are joint in a big conjunction.
163   */
164  private void setFilterParameters(OccurrenceSearchRequest request, SolrQuery solrQuery) {
165    Multimap<OccurrenceSearchParameter, String> params = request.getParameters();
166    if (params != null && !params.isEmpty()) {
167      List<String> filterQueries = Lists.newArrayList();
168      for (OccurrenceSearchParameter param : params.keySet()) {
169        List<String> aFieldParameters = Lists.newArrayList();
170        for (String value : params.get(param)) {
171          SearchTypeValidator.validate(param, value);
172          OccurrenceSolrField solrField = QUERY_FIELD_MAPPING.get(param);
173          if (solrField != null && param.type() != Date.class) {
174            String parsedValue;
175            if (solrField == OccurrenceSolrField.BASIS_OF_RECORD) {
176              parsedValue = BASIS_OF_RECORD_CONVERTER.fromEnum(BasisOfRecord.valueOf(value)).toString();
177            } else {
178              parsedValue = QueryUtils.parseQueryValue(value);
179            }
180            if (QueryUtils.isRangeQuery(parsedValue)) {
181              parsedValue = parsedValue.replace(",", " TO ");
182            }
183            aFieldParameters.add(PARAMS_JOINER.join(solrField.getFieldName(), parsedValue));
184          }
185        }
186        if (!aFieldParameters.isEmpty()) {
187          filterQueries.add(toParenthesesQuery(PARAMS_OR_JOINER.join(aFieldParameters)));
188        }
189      }
190      addLocationQuery(params, filterQueries);
191      addDateQuery(params, OccurrenceSearchParameter.DATE, OccurrenceSolrField.DATE, filterQueries);
192      addDateQuery(params, OccurrenceSearchParameter.MODIFIED, OccurrenceSolrField.MODIFIED, filterQueries);
193
194      if (!filterQueries.isEmpty()) {
195        solrQuery.addFilterQuery(PARAMS_AND_JOINER.join(filterQueries));
196      }
197    }
198  }
