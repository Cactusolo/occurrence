001package org.gbif.occurrencestore.search.solr;
002
003
004/**
005 * Enum that contains the mapping of symbolic names and field names of valid Solr fields.
006 */
007public enum OccurrenceSolrField {
008  KEY("key"), LATITUDE("latitude"), LONGITUDE("longitude"), COORDINATE("coordinate"),
009  COUNTRY("country"), PUBLISHING_COUNTRY("publishing_country"), YEAR("year"), MONTH("month"),
010  CATALOG_NUMBER("catalog_number"), COLLECTOR_NAME("collector_name"),
011  BASIS_OF_RECORD("basis_of_record"), DATASET_KEY("dataset_key"), TAXON_KEY("taxon_key"),
012  COLLECTION_CODE("collection_code"), ALTITUDE("altitude"), DEPTH("depth"), INSTITUTION_CODE("institution_code"),
013  GEOSPATIAL_ISSUE("geospatial_issue"), GEOREFERENCED("georeferenced"), DATE("date"), MODIFIED("modified");
014
015  private final String fieldName;
016
017
018  private OccurrenceSolrField(String fieldName) {
019    this.fieldName = fieldName;
020  }
021
022  /**
023   * @return the fieldName
024   */
025  public String getFieldName() {
026    return fieldName;
027  }
028
029}
