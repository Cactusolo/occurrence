001package org.gbif.occurrencestore.search;
002
003import org.gbif.api.util.IsoDateParsingUtils;
004import org.gbif.api.util.IsoDateParsingUtils.IsoDateFormat;
005import org.gbif.api.util.SearchTypeValidator;
006
007import java.util.Date;
008
009import com.google.common.collect.Range;
010
011import static org.gbif.api.util.IsoDateParsingUtils.getFirstDateFormatMatch;
012import static org.gbif.api.util.IsoDateParsingUtils.parseDateRange;
013import static org.gbif.api.util.IsoDateParsingUtils.toLastDayOfMonth;
014import static org.gbif.api.util.IsoDateParsingUtils.toLastDayOfYear;
015import static org.gbif.common.search.util.QueryUtils.toDateQueryFormat;
016import static org.gbif.common.search.util.QueryUtils.toPhraseQuery;
017import static org.gbif.common.search.util.SolrConstants.DEFAULT_FILTER_QUERY;
018import static org.gbif.common.search.util.SolrConstants.RANGE_FORMAT;
019
020/**
021 * Utility class that contains functions for building occurrence date queries.
022 */
023public class OccurrenceSearchDateUtils {
024
025
026  /**
027   * Converts the value parameter into a date range query.
028   */
029  public static String toDateQuery(String value) {
030    if (SearchTypeValidator.isRange(value)) {
031      return toRangeDateQueryFormat(value);
032    } else {
033      return toSingleDateQueryFormat(value);
034    }
035  }
036
037  private static String toRangeDateQueryFormat(String value) {
038    final Range<Date> dateRange = parseDateRange(value);
039    if (!dateRange.hasLowerBound() && !dateRange.hasUpperBound()) {
040      return DEFAULT_FILTER_QUERY;
041    } else if (dateRange.hasLowerBound() && !dateRange.hasUpperBound()) {
042      return String.format(RANGE_FORMAT, toDateQueryFormat(dateRange.lowerEndpoint()),
043        DEFAULT_FILTER_QUERY);
044    } else if (!dateRange.hasLowerBound() && dateRange.hasUpperBound()) {
045      return String.format(RANGE_FORMAT, DEFAULT_FILTER_QUERY,
046        toDateQueryFormat(dateRange.upperEndpoint()));
047    } else {
048      return String.format(RANGE_FORMAT, toDateQueryFormat(dateRange.lowerEndpoint()),
049        toDateQueryFormat(dateRange.upperEndpoint()));
050    }
051  }
052
053  /**
054   * Transforms a single ISO date value into a solr query being a range query depending on the precision of the
055   * ISO date. For example a single date 1989-09 results in a solr range for the entire month.
056   */
057  private static String toSingleDateQueryFormat(String value) {
058    final IsoDateFormat occDateFormat = getFirstDateFormatMatch(value);
059    final Date lowerDate = IsoDateParsingUtils.parseDate(value);
060    final String lowerDateStr = toDateQueryFormat(lowerDate);
061    if (occDateFormat == IsoDateFormat.YEAR_MONTH) {
062      // Generated query: [yyyy-MM-01T00:00:00Z TO yyyy-'MM-LAST_DATE_OF_THE_MONTH'T00:00:00Z]
063      String upperDayeStr = toDateQueryFormat(toLastDayOfMonth(lowerDate));
064      return String.format(RANGE_FORMAT, lowerDateStr, upperDayeStr);
065    } else if (occDateFormat == IsoDateFormat.YEAR) {
066      // Generated query: [yyyy-01-01T00:00:00Z TO yyyy-'LAST_DATE_OF_THE_YEAR'T00:00:00Z]
067      String upperDayeStr = toDateQueryFormat(toLastDayOfYear(lowerDate));
068      return String.format(RANGE_FORMAT, lowerDateStr, upperDayeStr);
069    }
070    return toPhraseQuery(lowerDateStr);
071  }
072}
