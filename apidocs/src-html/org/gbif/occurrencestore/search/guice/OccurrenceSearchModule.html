001package org.gbif.occurrencestore.search.guice;
002
003import org.gbif.api.service.occurrence.OccurrenceSearchService;
004import org.gbif.common.search.inject.SolrModule;
005import org.gbif.occurrencestore.search.OccurrenceSearchImpl;
006import org.gbif.service.guice.PrivateServiceModule;
007
008import java.util.Properties;
009
010/**
011 * Occurrence search guice module.
012 */
013public class OccurrenceSearchModule extends PrivateServiceModule {
014
015  private static final String PREFIX = "occurrencestore.search.";
016
017  public OccurrenceSearchModule(Properties properties) {
018    super(PREFIX, properties);
019  }
020
021  @Override
022  protected void configureService() {
023    install(new SolrModule());
024    bind(OccurrenceSearchService.class).to(OccurrenceSearchImpl.class);
025    expose(OccurrenceSearchService.class);
026  }
027
028}
