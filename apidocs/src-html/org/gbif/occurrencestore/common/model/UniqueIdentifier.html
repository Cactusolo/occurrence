001package org.gbif.occurrencestore.common.model;
002
003import java.util.UUID;
004
005/**
006 * This interface is meant to be used for classes that can uniquely identify occurrence records.
007 */
008public interface UniqueIdentifier {
009
010  /**
011   * Every unique identifier must be scoped within the dataset key.
012   *
013   * @return the UUID of the dataset for the identified occurrence
014   */
015  public UUID getDatasetKey();
016
017  /**
018   * A string that uniquely identifies the occurrence (e.g. a concatenation of it's unique fields) that could be used
019   * as a key for maps or databases.
020   *
021   * @return a unique String representing the unique identifier
022   */
023  public String getUniqueString();
024
025  /**
026   * A string that uniquely identifies the occurrence within a dataset (e.g. a concatenation of it's unique fields) but
027   * does not incorporate the datasetKey.
028   *
029   * @return a unique String representing the unique identifier within the dataset (but does not include the datasetKey)
030   */
031  public String getUnscopedUniqueString();
032}
