001package org.gbif.occurrencestore.common.model;
002
003import org.gbif.occurrencestore.util.OccurrenceKeyHelper;
004
005import java.util.UUID;
006import javax.annotation.Nullable;
007
008import com.google.common.base.Objects;
009
010import static com.google.common.base.Preconditions.checkArgument;
011import static com.google.common.base.Preconditions.checkNotNull;
012
013/**
014 * The original and still heavily used unique identifier for occurrence records, consisting of InstitutionCode,
015 * CollectionCode, and CatalogNumber. It additionally includes an optional unitQualifier, a GBIF specific term used
016 * for differentiating between multiple occurrences within a single ABCD 2.06 record.
017 */
018public class HolyTriplet implements UniqueIdentifier {
019
020  private final UUID datasetKey;
021  private final String institutionCode;
022  private final String collectionCode;
023  private final String catalogNumber;
024  private final String unitQualifier;
025
026  /**
027   * Constructs the immutable triplet.
028   *
029   * @throws IllegalArgumentException if institutionCode, collectionCode, or catalogNumber are null or empty
030   * @throws NullPointerException if datasetKey is null
031   */
032  public HolyTriplet(UUID datasetKey, String institutionCode, String collectionCode, String catalogNumber,
033    @Nullable String unitQualifier) {
034    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
035    checkArgument(institutionCode != null && !institutionCode.isEmpty(), "institutionCode can't be null or empty");
036    checkArgument(collectionCode != null && !collectionCode.isEmpty(), "collectionCode can't be null or empty");
037    checkArgument(catalogNumber != null && !catalogNumber.isEmpty(), "catalogNumber can't be null or empty");
038    this.institutionCode = institutionCode;
039    this.collectionCode = collectionCode;
040    this.catalogNumber = catalogNumber;
041    this.unitQualifier = unitQualifier;
042  }
043
044  public String getInstitutionCode() {
045    return institutionCode;
046  }
047
048  public String getCollectionCode() {
049    return collectionCode;
050  }
051
052  public String getCatalogNumber() {
053    return catalogNumber;
054  }
055
056  public String getUnitQualifier() {
057    return unitQualifier;
058  }
059
060  @Override
061  public UUID getDatasetKey() {
062    return datasetKey;
063  }
064
065  @Override
066  public String getUniqueString() {
067    return OccurrenceKeyHelper.buildKey(this);
068  }
069
070  @Override
071  public String getUnscopedUniqueString() {
072    return OccurrenceKeyHelper.buildUnscopedKey(this);
073  }
074
075  @Override
076  public int hashCode() {
077    return Objects.hashCode(datasetKey, institutionCode, collectionCode, catalogNumber, unitQualifier);
078  }
079
080  @Override
081  public boolean equals(Object obj) {
082    if (obj == null) {
083      return false;
084    }
085    if (getClass() != obj.getClass()) {
086      return false;
087    }
088    final HolyTriplet other = (HolyTriplet) obj;
089    return Objects.equal(this.datasetKey, other.datasetKey) && Objects
090      .equal(this.institutionCode, other.institutionCode) && Objects.equal(this.collectionCode, other.collectionCode)
091           && Objects.equal(this.catalogNumber, other.catalogNumber) && Objects
092      .equal(this.unitQualifier, other.unitQualifier);
093  }
094}
