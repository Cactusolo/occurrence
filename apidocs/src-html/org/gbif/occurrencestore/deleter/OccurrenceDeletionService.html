001package org.gbif.occurrencestore.deleter;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.occurrencestore.common.model.HolyTriplet;
005import org.gbif.occurrencestore.common.model.UniqueIdentifier;
006import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
007import org.gbif.occurrencestore.persistence.api.OccurrencePersistenceService;
008import org.gbif.occurrencestore.persistence.api.VerbatimOccurrence;
009import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
010
011import java.util.concurrent.TimeUnit;
012
013import com.google.common.collect.Sets;
014import com.yammer.metrics.Metrics;
015import com.yammer.metrics.core.Meter;
016import org.slf4j.Logger;
017import org.slf4j.LoggerFactory;
018
019import static com.google.common.base.Preconditions.checkArgument;
020import static com.google.common.base.Preconditions.checkNotNull;
021
022/**
023 * A simple service that can handle the deletion of a single occurrence (including its secondary index entry).
024 */
025public class OccurrenceDeletionService {
026
027  private static final Logger LOG = LoggerFactory.getLogger(OccurrenceDeletionService.class);
028
029  private final OccurrencePersistenceService occurrenceService;
030  private final OccurrenceKeyPersistenceService occurrenceKeyService;
031  private final VerbatimOccurrencePersistenceService verbatimService;
032
033  private final Meter occurrencesDeleted =
034    Metrics.newMeter(OccurrenceDeletionService.class, "deletes", "deletes", TimeUnit.SECONDS);
035
036  public OccurrenceDeletionService(OccurrencePersistenceService occurrenceService,
037    OccurrenceKeyPersistenceService occurrenceKeyService,
038    VerbatimOccurrencePersistenceService verbatimService) {
039    this.occurrenceService = checkNotNull(occurrenceService, "occurrenceService can't be null");
040    this.occurrenceKeyService = checkNotNull(occurrenceKeyService, "occurrenceKeyService can't be null");
041    this.verbatimService = checkNotNull(verbatimService, "verbatimService can't be null");
042  }
043
044  public Occurrence deleteOccurrence(int occurrenceKey) {
045    checkArgument(occurrenceKey > 0, "occurrenceKey must be > 0");
046    LOG.debug("Deleting occurrence for key [{}]", occurrenceKey);
047
048    // TODO: include dwcOccurrenceId lookup deletion (requires occ id on verbatim object)
049    VerbatimOccurrence verbatim = verbatimService.get(occurrenceKey);
050    if (verbatim == null) {
051      LOG.info("No occurrence for key [{}], ignoring deletion request", occurrenceKey);
052      return null;
053    }
054
055    UniqueIdentifier triplet = null;
056    try {
057      if (verbatim.getDatasetKey() != null) {
058        triplet = new HolyTriplet(verbatim.getDatasetKey(), verbatim.getInstitutionCode(), verbatim.getCollectionCode(),
059          verbatim.getCatalogNumber(), verbatim.getUnitQualifier());
060      }
061    } catch (IllegalArgumentException e) {
062      LOG.debug("No valid triplet for occurrenceKey [{}]", occurrenceKey, e);
063    }
064    if (triplet == null) {
065      LOG.info("No valid triplet for occurrenceKey [{}] therefore can't delete triplet lookup", occurrenceKey);
066    } else {
067      occurrenceKeyService.deleteKeyByUniqueIdentifiers(Sets.newHashSet(triplet));
068    }
069
070    // return the deleted occurrence
071    Occurrence deleted = occurrenceService.delete(occurrenceKey);
072    if (deleted != null) {
073      occurrencesDeleted.mark();
074    }
075    return deleted;
076  }
077}
