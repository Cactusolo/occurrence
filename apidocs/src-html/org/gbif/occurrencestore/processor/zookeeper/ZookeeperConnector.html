001package org.gbif.occurrencestore.processor.zookeeper;
002
003import java.util.Collections;
004import java.util.HashSet;
005import java.util.Set;
006import java.util.UUID;
007import java.util.concurrent.TimeUnit;
008import javax.annotation.Nullable;
009
010import com.google.inject.Singleton;
011import com.netflix.curator.framework.CuratorFramework;
012import com.netflix.curator.framework.recipes.atomic.DistributedAtomicLong;
013import com.netflix.curator.retry.RetryNTimes;
014import com.yammer.metrics.Metrics;
015import com.yammer.metrics.core.Timer;
016import com.yammer.metrics.core.TimerContext;
017import org.slf4j.Logger;
018import org.slf4j.LoggerFactory;
019
020import static com.google.common.base.Preconditions.checkNotNull;
021
022/**
023 * A connector to Zookeeper that provides convenience methods for incrementing and reading counts.
024 */
025@Singleton
026public class ZookeeperConnector {
027
028  private static final String CRAWL_PREFIX = "/crawls/";
029  private static final int BATCH_FLUSH_INTERVAL = 1000;
030
031  private final CuratorFramework curator;
032
033  private static final Logger LOG = LoggerFactory.getLogger(ZookeeperConnector.class);
034
035  private final Timer addTimer =
036    Metrics.newTimer(ZookeeperConnector.class, "counter add time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
037  private final Timer incrementTimer =
038    Metrics.newTimer(ZookeeperConnector.class, "dal inc time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
039
040  private final BatchingDalWrapper dalWrapper;
041  private static final Set<String> ENSURED_PATHS = Collections.synchronizedSet(new HashSet<String>());
042
043
044  /**
045   * Builds the connector with the given curator. Note that if the curator is not started before being passed in it
046   * will be started here.
047   *
048   * @param curator the CuratorFramework providing underlying connections to zookeeper
049   */
050  public ZookeeperConnector(final CuratorFramework curator) {
051    checkNotNull(curator, "curator can't be null");
052    if (!curator.isStarted()) {
053      curator.start();
054    }
055    this.curator = curator;
056    dalWrapper = new BatchingDalWrapper(curator, BATCH_FLUSH_INTERVAL);
057  }
058
059  /**
060   * Atomically adds to the count for this dataset and counterName.
061   *
062   * @param datasetUuid the dataset key
063   * @param counterName the counterName representing a processing phase
064   */
065  public void addCounter(UUID datasetUuid, CounterName counterName) {
066    checkNotNull(datasetUuid);
067    checkNotNull(counterName);
068    final TimerContext context = addTimer.time();
069    try {
070      try {
071        String path = CRAWL_PREFIX + datasetUuid.toString() + counterName.getPath();
072        LOG.debug("Updating DAL at path [{}]", path);
073        if (!ENSURED_PATHS.contains(path)) {
074          curator.newNamespaceAwareEnsurePath(path).ensure(curator.getZookeeperClient());
075          ENSURED_PATHS.add(path);
076        }
077        final TimerContext incContext = incrementTimer.time();
078        try {
079          dalWrapper.increment(path);
080        } finally {
081          incContext.stop();
082        }
083      } catch (Exception e) {
084        LOG.warn("Error updating zookeeper", e);
085      }
086    } finally {
087      context.stop();
088    }
089  }
090
091  /**
092   * Read the count for this dataset and counterName.
093   *
094   * @param datasetUuid the dataset key
095   * @param counterName the counterName representing a processing phase
096   *
097   * @return the count or null if it could not be read
098   */
099  @Nullable
100  public Long readCounter(UUID datasetUuid, CounterName counterName) {
101    checkNotNull(datasetUuid);
102    checkNotNull(counterName);
103    Long result = null;
104    try {
105      String path = CRAWL_PREFIX + datasetUuid.toString() + counterName.getPath();
106      LOG.debug("Reading DAL at path [{}]", path);
107      curator.newNamespaceAwareEnsurePath(path).ensure(curator.getZookeeperClient());
108      DistributedAtomicLong dal = new DistributedAtomicLong(curator, path, new RetryNTimes(1, 1000));
109      result = dal.get().preValue();
110    } catch (Exception e) {
111      LOG.warn("Error reading from zookeeper", e);
112    }
113    return result;
114  }
115
116  public enum CounterName {
117    FRAGMENT_RECEIVED("/fragmentsReceived"),
118    RAW_OCCURRENCE_PERSISTED_NEW("/rawOccurrencesPersisted/new"),
119    RAW_OCCURRENCE_PERSISTED_UPDATED("/rawOccurrencesPersisted/updated"),
120    RAW_OCCURRENCE_PERSISTED_UNCHANGED("/rawOccurrencesPersisted/unchanged"),
121    RAW_OCCURRENCE_PERSISTED_ERROR("/rawOccurrencesPersisted/error"),
122    FRAGMENT_PROCESSED("/fragmentsProcessed"),
123    VERBATIM_OCCURRENCE_PERSISTED_SUCCESS("/verbatimOccurrencesPersisted/successful"),
124    VERBATIM_OCCURRENCE_PERSISTED_ERROR("/verbatimOccurrencesPersisted/error"),
125    INTERPRETED_OCCURRENCE_PERSISTED_SUCCESS("/interpretedOccurrencesPersisted/successful"),
126    INTERPRETED_OCCURRENCE_PERSISTED_ERROR("/interpretedOccurrencesPersisted/error");
127
128    private final String path;
129
130    private CounterName(String path) {
131      this.path = path;
132    }
133
134    public static CounterName fromPath(String path) {
135      for (CounterName counterName : CounterName.values()) {
136        if (counterName.getPath().equals(path)) {
137          return counterName;
138        }
139      }
140      throw new IllegalArgumentException("No CounterName for path [" + path + "]");
141    }
142
143    public String getPath() {
144      return this.path;
145    }
146  }
147}
