001package org.gbif.occurrencestore.processor;
002
003import org.gbif.api.model.occurrence.OccurrencePersistenceStatus;
004import org.gbif.common.messaging.api.MessagePublisher;
005import org.gbif.common.messaging.api.messages.VerbatimPersistedMessage;
006import org.gbif.occurrencestore.persistence.api.Fragment;
007import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
008import org.gbif.occurrencestore.persistence.api.VerbatimOccurrence;
009import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
010import org.gbif.occurrencestore.processor.parsing.FragmentParser;
011import org.gbif.occurrencestore.processor.zookeeper.ZookeeperConnector;
012
013import java.io.IOException;
014import java.util.UUID;
015import java.util.concurrent.TimeUnit;
016import javax.annotation.Nullable;
017
018import com.google.inject.Inject;
019import com.google.inject.Singleton;
020import com.yammer.metrics.Metrics;
021import com.yammer.metrics.core.Meter;
022import com.yammer.metrics.core.Timer;
023import com.yammer.metrics.core.TimerContext;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027import static com.google.common.base.Preconditions.checkArgument;
028import static com.google.common.base.Preconditions.checkNotNull;
029
030/**
031 * Takes occurrence Fragments and parses them to produce and persist VerbatimOccurrence objects.
032 */
033@Singleton
034public class VerbatimProcessor {
035
036  private final FragmentPersistenceService fragmentPersister;
037  private final VerbatimOccurrencePersistenceService verbatimPersister;
038  private final MessagePublisher messagePublisher;
039  private final ZookeeperConnector zookeeperConnector;
040
041  private final Meter verbProcessed = Metrics.newMeter(VerbatimProcessor.class, "verbs", "verbs", TimeUnit.SECONDS);
042  private final Timer msgTimer =
043    Metrics.newTimer(VerbatimProcessor.class, "msg send time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
044
045  private static final Logger LOG = LoggerFactory.getLogger(VerbatimProcessor.class);
046
047  @Inject
048  public VerbatimProcessor(FragmentPersistenceService fragmentPersister,
049    VerbatimOccurrencePersistenceService verbatimPersister, MessagePublisher messagePublisher,
050    ZookeeperConnector zookeeperConnector) {
051    this.fragmentPersister = checkNotNull(fragmentPersister, "fragmentPersister can't be null");
052    this.verbatimPersister = checkNotNull(verbatimPersister, "verbatimPersister can't be null");
053    this.messagePublisher = checkNotNull(messagePublisher, "messagePublisher can't be null");
054    this.zookeeperConnector = checkNotNull(zookeeperConnector, "zookeeperConnector can't be null");
055  }
056
057  /**
058   * Builds and persists a VerbatimOccurrence object by parsing an existing Fragment with the given occurrenceKey.
059   * Updated zookeeper with success/error counts and sends a VerbatimPersistedMessage when successfully completed. Note
060   * that UNCHANGED Fragments are ignored.
061   *
062   * @param occurrenceKey the key of the existing Fragment to be parsed
063   * @param status        whether the Fragment is NEW, UPDATED, or UNCHANGED
064   * @param fromCrawl     true if this method is called as part of a crawl
065   * @param attemptId     the crawl attempt id, only used for passing along in logs and subsequent messages.
066   * @param datasetKey    the dataset that this occurrence belongs to (must not be null if fromCrawl is true)
067   */
068  public void buildVerbatim(int occurrenceKey, OccurrencePersistenceStatus status, boolean fromCrawl,
069    @Nullable Integer attemptId, @Nullable UUID datasetKey) {
070    checkArgument(occurrenceKey > 0, "occurrenceKey must be greater than 0");
071    checkNotNull(status, "status can't be null");
072    if (fromCrawl) {
073      checkNotNull(datasetKey, "datasetKey can't be null if fromCrawl is true");
074      checkArgument(attemptId != null && attemptId > 0, "attemptId must be greater than 0 if fromCrawl is true");
075    }
076
077    if (status == OccurrencePersistenceStatus.UNCHANGED) {
078      LOG.debug("Ignoring fragment of status UNCHANGED.");
079      return;
080    }
081
082    Fragment fragment = fragmentPersister.get(occurrenceKey);
083
084    if (fragment == null) {
085      logError("Could not find", occurrenceKey, datasetKey, fromCrawl);
086      return;
087    }
088
089    int localAttemptId = fromCrawl ? attemptId : fragment.getCrawlId();
090    LOG.debug("Fragment for key [{}] and UUID [{}] crawl [{}] is [{}]", occurrenceKey, fragment.getDatasetKey(),
091      localAttemptId, status);
092
093    VerbatimOccurrence verbatim = FragmentParser.parse(fragment);
094    if (verbatim == null) {
095      // parsing has failed, skip this fragment
096      logError("Could not parse", occurrenceKey, datasetKey, fromCrawl);
097      return;
098    }
099
100    verbatimPersister.update(verbatim);
101
102    if (fromCrawl) {
103      LOG.debug("Updating zookeeper for VerbatimOccurrencePersistedSuccess");
104      zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_SUCCESS);
105    }
106
107    VerbatimPersistedMessage verbMsg =
108      new VerbatimPersistedMessage(verbatim.getDatasetKey(), localAttemptId, status, verbatim.getKey());
109    final TimerContext msgContext = msgTimer.time();
110    try {
111      messagePublisher.send(verbMsg);
112    } catch (IOException e) {
113      LOG.warn("Could not send VerbatimPersistedMessage for successful [{}]", status, e);
114    } finally {
115      msgContext.stop();
116    }
117
118    verbProcessed.mark();
119  }
120
121  private void logError(String message, int occurrenceKey, UUID datasetKey, boolean fromCrawl) {
122    // TODO: send msg?
123    LOG.warn(message + " fragment with key [{}] - skipping.", occurrenceKey);
124    if (fromCrawl) {
125      LOG.debug("Updating zookeeper for VerbatimOccurrencePersistedError");
126      zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_ERROR);
127    }
128  }
129}
