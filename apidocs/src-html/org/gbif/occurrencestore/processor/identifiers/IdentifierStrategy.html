001package org.gbif.occurrencestore.processor.identifiers;
002
003import org.gbif.api.model.crawler.DwcaValidationReport;
004import org.gbif.api.vocabulary.OccurrenceSchemaType;
005
006import javax.annotation.Nullable;
007
008import com.google.common.base.Objects;
009
010import static com.google.common.base.Preconditions.checkNotNull;
011
012/**
013 * A simple class to determine whether triplets and/or occurrenceIds are valid as identifiers for the given dataset.
014 */
015public class IdentifierStrategy {
016
017  private final boolean tripletsValid;
018  private final boolean occurrenceIdsValid;
019
020  public IdentifierStrategy(OccurrenceSchemaType schemaType, @Nullable DwcaValidationReport validationReport) {
021    checkNotNull(schemaType, "schemaType can't be null");
022    if (schemaType == OccurrenceSchemaType.DWCA) {
023      checkNotNull(validationReport, "validationReport can't be null if schema is DWCA");
024    }
025    tripletsValid = tripletsValid(schemaType, validationReport);
026    occurrenceIdsValid = occurrenceIdsValid(schemaType, validationReport);
027  }
028
029  /**
030   * For XML datasets triplets are always valid. For DwC-A datasets triplets are valid if there are more than 0 unique
031   * triplets in the dataset, and exactly 0 triplets referenced by more than one record.
032   */
033  private static boolean tripletsValid(OccurrenceSchemaType schemaType, DwcaValidationReport validationReport) {
034    boolean valid = true;
035    if (schemaType == OccurrenceSchemaType.DWCA) {
036      valid = validationReport.getUniqueTriplets() > 0
037              && validationReport.getCheckedRecords() - validationReport.getRecordsWithInvalidTriplets()
038                 == validationReport.getUniqueTriplets();
039    }
040
041    return valid;
042  }
043
044  /**
045   * For XML datasets occurrenceIds are ignored. For DwC-A datasets occurrenceIds are valid if there are more than 0
046   * unique occurrenceIds in the dataset, and exactly 0 occurrenceIds referenced by more than one record.
047   */
048  private static boolean occurrenceIdsValid(OccurrenceSchemaType schemaType, DwcaValidationReport validationReport) {
049    boolean valid = false;
050    if (schemaType == OccurrenceSchemaType.DWCA) {
051      valid = validationReport.getUniqueOccurrenceIds() > 0
052              && validationReport.getCheckedRecords() - validationReport.getRecordsMissingOccurrenceId()
053                 == validationReport.getUniqueOccurrenceIds();
054    }
055
056    return valid;
057  }
058
059  public boolean isTripletsValid() {
060    return tripletsValid;
061  }
062
063  public boolean isOccurrenceIdsValid() {
064    return occurrenceIdsValid;
065  }
066
067  @Override
068  public int hashCode() {
069    return Objects.hashCode(tripletsValid, occurrenceIdsValid);
070  }
071
072  @Override
073  public boolean equals(Object obj) {
074    if (this == obj) {
075      return true;
076    }
077    if (obj == null || getClass() != obj.getClass()) {
078      return false;
079    }
080    final IdentifierStrategy other = (IdentifierStrategy) obj;
081    return Objects.equal(this.tripletsValid, other.tripletsValid) && Objects
082      .equal(this.occurrenceIdsValid, other.occurrenceIdsValid);
083  }
084}
