001package org.gbif.occurrencestore.processor.messaging;
002
003import org.gbif.common.messaging.AbstractMessageCallback;
004import org.gbif.common.messaging.api.messages.OccurrenceFragmentedMessage;
005import org.gbif.occurrencestore.processor.FragmentProcessor;
006
007import java.util.concurrent.TimeUnit;
008
009import com.google.inject.Inject;
010import com.google.inject.Singleton;
011import com.yammer.metrics.Metrics;
012import com.yammer.metrics.core.Timer;
013import com.yammer.metrics.core.TimerContext;
014
015import static com.google.common.base.Preconditions.checkNotNull;
016
017/**
018 * A MessageCallback implementation for OccurrenceFragmentedMessage. Hands off to the FragmentProcessor passed in
019 * during construction. This is the start of the processing chain.
020 */
021@Singleton
022public class OccurrenceFragmentedListener extends AbstractMessageCallback<OccurrenceFragmentedMessage> {
023
024  private final FragmentProcessor processor;
025
026  private final Timer processTimer =
027    Metrics.newTimer(FragmentProcessor.class, "frag process time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
028
029  @Inject
030  public OccurrenceFragmentedListener(FragmentProcessor processor) {
031    checkNotNull(processor, "processor can't be null");
032    this.processor = processor;
033  }
034
035  @Override
036  public void handleMessage(OccurrenceFragmentedMessage message) {
037    final TimerContext context = processTimer.time();
038    try {
039      processor.buildFragments(message.getDatasetUuid(), message.getFragment(), message.getSchemaType(),
040        message.getEndpointType(), message.getAttempt(), message.getValidationReport());
041    } finally {
042      context.stop();
043    }
044  }
045}
