029/**
030 * Produces Occurrence objects by interpreting the verbatim fields of a VerbatimOccurrence.
031 */
032@Singleton
033public class VerbatimOccurrenceInterpreter {
034
035  private static final Logger LOG = LoggerFactory.getLogger(VerbatimOccurrenceInterpreter.class);
036
037  private final OccurrencePersistenceService occurrenceService;
038  private final ZookeeperConnector zookeeperConnector;
039
040  @Inject
041  public VerbatimOccurrenceInterpreter(OccurrencePersistenceService occurrenceService,
042    ZookeeperConnector zookeeperConnector) {
043    this.occurrenceService = checkNotNull(occurrenceService, "occurrenceService can't be null");
044    this.zookeeperConnector = checkNotNull(zookeeperConnector, "zookeeperConnector can't be null");
045  }
046
047  /**
048   * Interpret all the verbatim fields into our standard Occurrence fields.
049   * TODO: send messages/write logs for interpretation errors
050   *
051   * @param verbatim the verbatim occurrence to interpret
052   *
053   * @return an InterpretationResult that contains an "updated" Occurrence with interpreted fields and an "original"
054   *         occurrence iff this was an update to an existing record (will be null otherwise)
055   */
056  public InterpretationResult interpret(VerbatimOccurrence verbatim, OccurrencePersistenceStatus status,
057    boolean fromCrawl) {
058    Occurrence occ = occurrenceFrom(verbatim);
059
060    try {
061      ExecutorService threadPool = Executors.newFixedThreadPool(3);
062      Future<?> coordFuture = threadPool.submit(new CoordBasedInterpreter(verbatim, occ));
063      Future<?> taxonomyFuture = threadPool.submit(new TaxonomyInterpreter(verbatim, occ));
064      Future<?> hostCountryFuture = threadPool.submit(new OwningOrgInterpreter(occ));
065      threadPool.shutdown();
066
067      interpretAltitude(verbatim, occ);
068
069      interpretBor(verbatim, occ);
070
071      interpretDate(verbatim, occ);
072
073      interpretDepth(verbatim, occ);
074
075      // TODO: if any errors, update messages
076      //   possible errors:
077      //     GEO: lat/lng out of bounds, lat/lng reversed for given country, others?
078      //     TAXONOMY: name not found, others?
079      //     BASIS: basis not interpreted, basis not provided, others?
080      //     TEMPORAL: dates uninterpretable (eg 10-11-12)
081      //     ALT_DEPTH: any errors?
082
083      occ.setModified(new Date());
084
085      // wait for the ws calls
086      if (!threadPool.awaitTermination(2, TimeUnit.MINUTES)) {
087        if (coordFuture.isCancelled()) {
088          LOG.warn("Coordinate lookup timed out - geospatial will be wrong for occurrence [{}]", occ.getKey());
089        }
090        if (taxonomyFuture.isCancelled()) {
091          LOG.warn("Nub lookup timed out - taxonomy will be wrong for occurrence [{}]", occ.getKey());
092        }
093        if (hostCountryFuture.isCancelled()) {
094          LOG.warn("Host country lookup timed out - host country will be null for occurrence [{}]", occ.getKey());
095        }
096      }
097
098    } catch (Exception e) {
099      // TODO: tidy up exception handling to not be Exception
100      LOG.warn("Caught a runtime exception during interpretation", e);
101    }
102
103    // populate the returned class
104    Occurrence original = null;
105    if (status == OccurrencePersistenceStatus.UPDATED) {
106      original = occurrenceService.get(verbatim.getKey());
107      // TODO: compare original with interp - if identical then return unchanged flag on result, so downstream
108      // doesn't need to do anything
109    }
110    InterpretationResult result = new InterpretationResult(original, occ);
111
112    // persist the record (considered an update in all cases because the key must already exist on verbatim)
113    LOG.debug("Persisting interpreted occurrence");
114    occurrenceService.update(occ);
115
116    if (fromCrawl) {
117      LOG.debug("Updating zookeeper for OccurrenceInterpretedPersistedSuccess");
118      zookeeperConnector
119        .addCounter(occ.getDatasetKey(), ZookeeperConnector.CounterName.INTERPRETED_OCCURRENCE_PERSISTED_SUCCESS);
120    }
121
122    return result;
123  }
124
125  private static void interpretDepth(VerbatimOccurrence verbatim, Occurrence occ) {
126    Integer depth =
127      DepthInterpreter.interpretDepth(verbatim.getMinDepth(), verbatim.getMaxDepth(), verbatim.getDepthPrecision());
128    occ.setDepth(depth);
129    LOG.debug("Got depth [{}]", depth);
130  }
131
132  private static void interpretDate(VerbatimOccurrence verbatim, Occurrence occ) {
133    DateInterpretationResult dateResult = DateInterpreter
134      .interpretDate(verbatim.getYear(), verbatim.getMonth(), verbatim.getDay(), verbatim.getOccurrenceDate());
135    if (dateResult == null) {
136      LOG.debug("Got date [null]");
137    } else {
138      occ.setEventDate(dateResult.getDate());
139      occ.setMonth(dateResult.getMonth());
140      occ.setYear(dateResult.getYear());
141      LOG.debug("Got date [{}] month [{}] year [{}]",
142        dateResult.getDate() == null ? "null" : dateResult.getDate().toString(),
143        dateResult.getMonth() == null ? "null" : dateResult.getMonth().toString(),
144        dateResult.getYear() == null ? "null" : dateResult.getYear().toString());
145    }
146  }
147
148  private static void interpretAltitude(VerbatimOccurrence verbatim, Occurrence occ) {
149    Integer alt = AltitudeInterpreter
150      .interpretAltitude(verbatim.getMinAltitude(), verbatim.getMaxAltitude(), verbatim.getAltitudePrecision());
151    occ.setAltitude(alt);
152    LOG.debug("Got altitude [{}]", alt);
153  }
154
155  private static void interpretBor(VerbatimOccurrence verbatim, Occurrence occ) {
156    // TODO: interpretation should produce enum
157    BasisOfRecordConverter borConv = new BasisOfRecordConverter();
158    Integer basisOfRecord = BasisOfRecordInterpreter.interpretBasisOfRecord(verbatim.getBasisOfRecord());
159    BasisOfRecord bor = borConv.toEnum(basisOfRecord);
160    occ.setBasisOfRecord(bor);
161    LOG.debug("Got BOR [{}]", bor.toString());
162  }
163
164  private Occurrence occurrenceFrom(VerbatimOccurrence verbatim) {
165    Occurrence occ = new Occurrence();
166    occ.setKey(verbatim.getKey());
167//    occ.setInstitutionCode(verbatim.getInstitutionCode());
168//    occ.setCollectionCode(verbatim.getCollectionCode());
169//    occ.setCatalogNumber(verbatim.getCatalogNumber());
170//    occ.setUnitQualifier(verbatim.getUnitQualifier());
171    occ.setDatasetKey(verbatim.getDatasetKey());
172    occ.setProtocol(verbatim.getProtocol());
173
174    // these are the as-yet uninterpreted fields
175//    occ.setCollectorName(verbatim.getCollectorName());
176//    occ.setContinent(verbatim.getContinentOrOcean());
177//    occ.setCounty(verbatim.getCounty());
178//    occ.setDataProviderId(verbatim.getDataProviderId());
179//    occ.setDataResourceId(verbatim.getDataResourceId());
180//    occ.setIdentifierName(verbatim.getIdentifierName());
181//    occ.setLocality(verbatim.getLocality());
182//    occ.setOwningOrgKey(verbatim.getOwningOrgKey());
183//    occ.setResourceAccessPointId(verbatim.getResourceAccessPointId());
184    occ.setStateProvince(verbatim.getStateOrProvince());
185
186    return occ;
187  }
