001package org.gbif.occurrencestore.processor.guice;
002
003import org.gbif.api.service.occurrence.OccurrenceService;
004import org.gbif.common.messaging.api.MessagePublisher;
005import org.gbif.common.messaging.guice.PostalServiceModule;
006import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
007import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
008import org.gbif.occurrencestore.persistence.api.OccurrencePersistenceService;
009import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
010import org.gbif.occurrencestore.persistence.guice.OccurrencePersistenceModule;
011import org.gbif.occurrencestore.processor.zookeeper.ZookeeperConnector;
012import org.gbif.service.guice.PrivateServiceModule;
013
014import java.util.Properties;
015
016import com.google.inject.Provides;
017import com.google.inject.name.Named;
018import com.netflix.curator.framework.CuratorFramework;
019import com.netflix.curator.framework.CuratorFrameworkFactory;
020import com.netflix.curator.retry.RetryNTimes;
021
022/**
023 * The Guice module that configures everything needed for the processing to start up. See the README for needed
024 * properties. Only needed when using the Startup class - see also the occurrence-cli project.
025 */
026public class OccurrenceProcessorModule extends PrivateServiceModule {
027
028  private static final String PREFIX = "occurrencestore.processor.";
029
030  public OccurrenceProcessorModule(Properties properties) {
031    super(PREFIX, properties);
032  }
033
034  @Override
035  protected void configureService() {
036    install(new OccurrencePersistenceModule(getVerbatimProperties()));
037    expose(OccurrenceService.class);
038    expose(OccurrencePersistenceService.class);
039    expose(OccurrenceKeyPersistenceService.class);
040    expose(VerbatimOccurrencePersistenceService.class);
041    expose(FragmentPersistenceService.class);
042    install(new PostalServiceModule("occurrencestore", getVerbatimProperties()));
043    expose(MessagePublisher.class);
044    expose(ZookeeperConnector.class);
045  }
046
047  @Provides
048  public ZookeeperConnector provideZookeeperConnector(@Named("zookeeper.connection_string") String zkUrl)
049    throws Exception {
050    CuratorFramework curator =
051      CuratorFrameworkFactory.builder().namespace("crawler").connectString(zkUrl).retryPolicy(new RetryNTimes(1, 1000))
052        .build();
053    curator.start();
054    return new ZookeeperConnector(curator);
055  }
056}
