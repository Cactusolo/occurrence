001package org.gbif.occurrencestore.processor.parsing;
002
003import org.gbif.occurrence.model.RawOccurrenceRecord;
004import org.gbif.occurrence.parsing.xml.XmlFragmentParser;
005import org.gbif.occurrencestore.persistence.api.Fragment;
006import org.gbif.occurrencestore.persistence.api.VerbatimOccurrence;
007
008import java.util.concurrent.TimeUnit;
009import javax.annotation.Nullable;
010
011import com.yammer.metrics.Metrics;
012import com.yammer.metrics.core.Timer;
013import com.yammer.metrics.core.TimerContext;
014
015/**
016 * A thin wrapper around the XmlFragmentParser from the occurrence-parser project, and the local JsonFragmentParser, in
017 * order to produce VerbatimOccurrence objects from Fragments.
018 */
019public class FragmentParser {
020
021  private static final Timer xmlParsingTimer =
022    Metrics.newTimer(FragmentParser.class, "verb xml parse time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
023  private static final Timer jsonParsingTimer =
024    Metrics.newTimer(FragmentParser.class, "verb json parse time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
025
026  // should not be instantiated
027  private FragmentParser() {
028  }
029
030  /**
031   * Parse the given fragment into a VerbatimOccurrence object.
032   * TODO: there is a lot of hacking here that should be refactored out with general rework of parsing project
033   *
034   * @param fragment containing parsing to be parsed
035   *
036   * @return a VerbatimOccurrence or null if the fragment could not be parsed
037   */
038  @Nullable
039  public static VerbatimOccurrence parse(Fragment fragment) {
040    VerbatimOccurrence verbatim = null;
041
042    switch (fragment.getFragmentType()) {
043      case XML:
044        final TimerContext xmlContext = xmlParsingTimer.time();
045        try {
046          RawOccurrenceRecord ror =
047            XmlFragmentParser.parseRecord(fragment.getData(), fragment.getXmlSchema(), fragment.getUnitQualifier());
048          verbatim = buildVerbatim(ror, fragment);
049        } finally {
050          xmlContext.stop();
051        }
052        break;
053      case JSON:
054        final TimerContext jsonContext = jsonParsingTimer.time();
055        try {
056          verbatim = JsonFragmentParser.parseRecord(fragment);
057        } finally {
058          jsonContext.stop();
059        }
060        break;
061    }
062
063    if (verbatim != null) {
064      verbatim.setProtocol(fragment.getProtocol());
065    }
066
067    return verbatim;
068  }
069
070  // TODO: identifier records
071  private static VerbatimOccurrence buildVerbatim(RawOccurrenceRecord ror, Fragment frag) {
072    VerbatimOccurrence v = VerbatimOccurrence.builder().key(frag.getKey()).datasetKey(frag.getDatasetKey())
073      .altitudePrecision(ror.getAltitudePrecision()).author(ror.getAuthor()).basisOfRecord(ror.getBasisOfRecord())
074      .catalogNumber(ror.getCatalogueNumber()).collectionCode(ror.getCollectionCode())
075      .collectorName(ror.getCollectorName()).continentOrOcean(ror.getContinentOrOcean()).country(ror.getCountry())
076      .county(ror.getCounty()).dataProviderId(ror.getDataProviderId()).dataResourceId(ror.getDataResourceId())
077      .dateIdentified(ror.getDateIdentified()).day(ror.getDay()).dayIdentified(ror.getDayIdentified())
078      .dateIdentified(ror.getDateIdentified()).depthPrecision(ror.getDepthPrecision()).family(ror.getFamily())
079      .genus(ror.getGenus()).identifierName(ror.getIdentifierName()).institutionCode(ror.getInstitutionCode())
080      .kingdom(ror.getKingdom()).klass(ror.getKlass()).latitude(ror.getLatitude()).longitude(ror.getLongitude())
081      .latLongPrecision(ror.getLatLongPrecision()).locality(ror.getLocality()).maxAltitude(ror.getMaxAltitude())
082      .maxDepth(ror.getMaxDepth()).minAltitude(ror.getMinAltitude()).minDepth(ror.getMinDepth())
083      .modified(ror.getModified()).month(ror.getMonth()).monthIdentified(ror.getMonthIdentified())
084      .occurrenceDate(ror.getOccurrenceDate()).order(ror.getOrder()).phylum(ror.getPhylum()).rank(ror.getRank())
085      .resourceAccessPointId(ror.getResourceAccessPointId()).scientificName(ror.getScientificName())
086      .species(ror.getSpecies()).stateOrProvince(ror.getStateOrProvince()).subspecies(ror.getSubspecies())
087      .unitQualifier(ror.getUnitQualifier()).year(ror.getYear()).yearIdentified(ror.getYearIdentified()).build();
088
089    return v;
090  }
091}
