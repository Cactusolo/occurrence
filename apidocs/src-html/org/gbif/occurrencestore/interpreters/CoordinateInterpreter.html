041/**
042 * Attempts to parse given String lat and long into Doubles, and compares the given country (if any) to a reverse
043 * lookup of the parsed coordinates. If no country was given and the lookup produced something, that looked up result
044 * is returned. If the lookup result and passed in country don't match, a "GeospatialIssue" is noted.
045 * TODO: return something better than int for GeospatialIssue (enum or split into many fields)
046 */
047public class CoordinateInterpreter {
048
049  private static final Logger LOG = LoggerFactory.getLogger(CoordinateInterpreter.class);
050
051  private static final Map<GeospatialIssue, Integer[]> TRANSFORMS =
052    new EnumMap<GeospatialIssue, Integer[]>(GeospatialIssue.class);
053
054  // if the WS is not responding, we drop into a retry count
055  private static final int NUM_RETRIES = 5;
056  private static final int RETRY_PERIOD_MSEC = 2000;
057
058  private static final String WEB_SERVICE_URL;
059  private static final String WEB_SERVICE_URL_PROPERTY = "occurrencestore.geo.ws.url";
060
061  private static final String FUZZY_COUNTRY_FILE;
062  private static final String FUZZY_COUNTRY_FILE_PROPERTY = "occurrencestore.geo.fuzzycountryfile";
063
064  // The repetitive nature of our data encourages use of a light cache to reduce WS load
065  private static final LoadingCache<WebResource, Location[]> CACHE =
066    CacheBuilder.newBuilder().maximumSize(1000).expireAfterAccess(1, TimeUnit.MINUTES)
067      .build(RetryingWebserviceClient.newInstance(Location[].class, 10, 2000));
068
069  private static final WebResource RESOURCE;
070
071  /*
072   Some countries are commonly mislabeled and are close to correct, so we want to accommodate them, e.g. Northern
073   Ireland mislabeled as Ireland (should be GB). Add comma separated pairs of acceptable country swaps in the
074   fuzzy_country_pairs.txt file. Entries will be made in both directions (e.g. IE->GB, GB->IE). Multiple entries per
075   country are allowed (e.g. AB,CD and AB,EF). This is only for setting of geospatial flags.
076  */
077  private static final Map<String, Set<String>> FUZZY_COUNTRIES = Maps.newHashMap();
078
079  static {
080    TRANSFORMS.put(GeospatialIssue.PRESUMED_NEGATED_LATITUDE, new Integer[] {-1, 1});
081    TRANSFORMS.put(GeospatialIssue.PRESUMED_NEGATED_LONGITUDE, new Integer[] {1, -1});
082    TRANSFORMS.put(GeospatialIssue.PRESUMED_INVERTED_COORDINATES, new Integer[] {-1, -1});
083
084    try {
085      InputStream is =
086        NubLookupInterpreter.class.getClassLoader().getResourceAsStream("occurrence-interpreter.properties");
087      if (is == null) {
088        throw new RuntimeException("Can't load properties file [occurrence-interpreter.properties]");
089      }
090      try {
091        Properties props = new Properties();
092        props.load(is);
093        WEB_SERVICE_URL = props.getProperty(WEB_SERVICE_URL_PROPERTY);
094        FUZZY_COUNTRY_FILE = props.getProperty(FUZZY_COUNTRY_FILE_PROPERTY);
095      } finally {
096        is.close();
097      }
098    } catch (IOException e) {
099      throw new RuntimeException("Can't load properties file [occurrence-interpreter.properties]", e);
100    }
101
102    ClientConfig cc = new DefaultClientConfig();
103    cc.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, true);
104
105    // We need to manually register it here because we're building an assembly and Jackson can't find it automatically
106    cc.getClasses().add(JacksonJsonProvider.class);
107    Client client = ApacheHttpClient.create(cc);
108    RESOURCE = client.resource(WEB_SERVICE_URL);
109
110    if (FUZZY_COUNTRY_FILE == null) {
111      LOG.warn("Fuzzy country file property [{}] not set, so not loading any fuzzy country pairs.",
112        FUZZY_COUNTRY_FILE_PROPERTY);
113    } else {
114      InputStream in = CoordinateInterpreter.class.getClassLoader().getResourceAsStream(FUZZY_COUNTRY_FILE);
115      BufferedReader reader = new BufferedReader(new InputStreamReader(in));
116      try {
117        String nextLine;
118        while ((nextLine = reader.readLine()) != null) {
119          if (!nextLine.startsWith("#")) {
120            String[] countries = nextLine.split(",");
121            String countryA = countries[0].trim().toUpperCase();
122            String countryB = countries[1].trim().toUpperCase();
123            LOG.info("Adding [{}][{}] pair to fuzzy country matches.", countryA, countryB);
124            addFuzzyCountry(countryA, countryB);
125            addFuzzyCountry(countryB, countryA);
126          }
127        }
128      } catch (IOException e) {
129        throw new RuntimeException("Can't read [" + FUZZY_COUNTRY_FILE + "] - aborting", e);
130      } finally {
131        try {
132          reader.close();
133          if (in != null) {
134            in.close();
135          }
136        } catch (IOException e) {
137          LOG.warn("Couldn't close [{}] - continuing anyway", FUZZY_COUNTRY_FILE, e);
138        }
139      }
140    }
141  }
142
143  private static void addFuzzyCountry(String countryA, String countryB) {
144    Set<String> fuzzyA = FUZZY_COUNTRIES.get(countryA);
145    if (fuzzyA == null) {
146      fuzzyA = Sets.newHashSet();
147      fuzzyA.add(countryA);
148    }
149    fuzzyA.add(countryB);
150    FUZZY_COUNTRIES.put(countryA, fuzzyA);
151  }

159  /**
160   * Attempts to convert the given lat and long into Doubles, and the given country string into an ISO country code.
161   *
162   * @param latitude    decimal latitude as string
163   * @param longitude   decimal longitude as string
164   * @param countryCode country name
165   *
166   * @return the latitude and longitude as doubles, the country as an ISO code, and a "geospatial issue" flag if any
167   *         known errors were encountered in the interpretation (e.g. lat/lng reversed), or all fields set to null if
168   *         latitude
169   *         or longitude are null
170   */
171  public static CoordinateInterpretationResult interpretCoordinates(String latitude, String longitude,
172    String countryCode) {
173    if (latitude == null || longitude == null) return new CoordinateInterpretationResult();
174
175    ParseResult<LatLngStatus> parseResult = GeospatialParseUtils.parseLatLng(latitude, longitude);
176
177    // round to 5 decimals (~1m precision) since no way we're getting anything legitimately more precise
178    Double lat = parseResult.getPayload() == null || parseResult.getPayload().getLat() == null ? null
179      : Math.round(parseResult.getPayload().getLat() * Math.pow(10, 5)) / Math.pow(10, 5);
180    Double lng = parseResult.getPayload() == null || parseResult.getPayload().getLng() == null ? null
181      : Math.round(parseResult.getPayload().getLng() * Math.pow(10, 5)) / Math.pow(10, 5);
182
183    String finalCountryCode = countryCode;
184
185    // the utils do a basic sanity check - even if it suggests success, we have to check that the lat/long
186    // actually falls in the country given in the record. If it doesn't, try common mistakes and note the issue
187    if (parseResult.getStatus() == ParseResult.STATUS.SUCCESS) {
188      List<String> isoCountries = getCountryForLatLng(lat, lng);
189      if (finalCountryCode == null) {
190        // we have nothing to say about coord accuracy, but can try to fetch the right country
191        if (!isoCountries.isEmpty()) {
192          finalCountryCode = isoCountries.get(0);
193        }
194      } else if (matchCountry(countryCode, isoCountries)) {
195        // in cases where fuzzy match we want to use the lookup value, not the fuzzy one
196        finalCountryCode = isoCountries.get(0);
197      } else {
198        boolean match = false;
199        for (Map.Entry<GeospatialIssue, Integer[]> geospatialIssueEntry : TRANSFORMS.entrySet()) {
200          Integer[] transform = geospatialIssueEntry.getValue();
201          if (matchCountry(countryCode, getCountryForLatLng(lat * transform[0], lng * transform[1]))) {
202            parseResult = ParseResult.fail(new LatLngStatus(lat, lng, geospatialIssueEntry.getKey()));
203            match = true;
204            break;
205          }
206        }
207
208        // if we made it here no transforms worked and the point is either in international waters or really weird
209        parseResult = match ? parseResult
210          : ParseResult.fail(new LatLngStatus(lat, lng, GeospatialIssue.COUNTRY_COORDINATE_MISMATCH));
211      }
212    }
213
214    CoordinateInterpretationResult result = new CoordinateInterpretationResult();
215    if (parseResult.getStatus() == ParseResult.STATUS.SUCCESS || parseResult.getStatus() == ParseResult.STATUS.FAIL) {
216      result = new CoordinateInterpretationResult(lat, lng, finalCountryCode,
217        parseResult.getPayload().getIssue().getIssueCode());
218    }
219
220    return result;
221  }
222
223  private static boolean matchCountry(String iso2CharCountry, Collection<String> isoCountries) {
224    String searchCountry = iso2CharCountry.toUpperCase();
225    Set<String> fuzzyCountries = FUZZY_COUNTRIES.get(searchCountry);
226    if (fuzzyCountries == null) {
227      fuzzyCountries = Sets.newHashSet();
228      fuzzyCountries.add(searchCountry);
229      FUZZY_COUNTRIES.put(searchCountry, fuzzyCountries);
230    }
231
232    boolean match = false;
233    if (!isoCountries.isEmpty()) {
234      for (String country : isoCountries) {
235        if (fuzzyCountries.contains(country)) {
236          match = true;
237          break;
238        }
239      }
240    }
241
242    return match;
243  }

245  /**
246   * It's theoretically possible that the webservice could respond with more than one country, though it's not
247   * known under what conditions that might happen.
248   */
249  private static List<String> getCountryForLatLng(Double lat, Double lng) {
250    List<String> countries = Lists.newArrayList();
251
252    MultivaluedMap<String, String> queryParams = new MultivaluedMapImpl();
253    queryParams.add("lat", lat.toString());
254    queryParams.add("lng", lng.toString());
255
256    for (int i = 0; i < NUM_RETRIES; i++) {
257      LOG.debug("Attempt [{}] to lookup lat [{}] lng [{}]", i, lat, lng);
258      try {
259        Location[] lookups = CACHE.get(RESOURCE.queryParams(queryParams));
260        if (lookups != null && lookups.length > 0) {
261          LOG.debug("Successfully retrieved [{}] locations for lat [{}] lng [{}]", lookups.length, lat, lng);
262          for (Location loc : lookups) {
263            if (loc.getIsoCountryCode2Digit() != null) {
264              countries.add(loc.getIsoCountryCode2Digit());
265            }
266          }
267        }
268        break; // from retry loop
269      } catch (ExecutionException e) {
270        // Log the error
271        StringBuilder sb = new StringBuilder("Failed WS call with: ");
272        for (Map.Entry<String, List<String>> en : queryParams.entrySet()) {
273          sb.append(en.getKey()).append('=');
274          for (String s : en.getValue()) {
275            sb.append(s);
276          }
277          sb.append('&');
278        }
279        LOG.error(sb.toString(), e);
280
281        // have we exhausted our attempts?
282        if (i >= NUM_RETRIES) {
283          throw new RuntimeException(e);
284        }
285
286        try {
287          Thread.sleep(RETRY_PERIOD_MSEC);
288        } catch (InterruptedException e1) {
289        }
290      }
291    } // retry loop
292
293    return countries;
294  }
