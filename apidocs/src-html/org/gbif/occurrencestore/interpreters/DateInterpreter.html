001package org.gbif.occurrencestore.interpreters;
002
003import org.gbif.common.parsers.ParseResult;
004import org.gbif.common.parsers.date.DateParseUtils;
005import org.gbif.common.parsers.date.YearMonthDay;
006import org.gbif.occurrencestore.interpreters.result.DateInterpretationResult;
007
008import java.text.ParseException;
009import java.text.SimpleDateFormat;
010import java.util.Calendar;
011import java.util.Date;
012
013import org.slf4j.Logger;
014import org.slf4j.LoggerFactory;
015
016/**
017 * Interprets date representations into a Date.
018 */
019public class DateInterpreter {
020
021  private static final Logger LOG = LoggerFactory.getLogger(DateInterpreter.class);
022
023  private static final String OUTPUT_DATE_FORMAT = "yyyy-MM-dd";
024  private static final Integer EARLIEST_YEAR = 1700;
025
026  private DateInterpreter() {
027  }
028
029  /**
030   * Given possibly both of year, month, day and a dateString, produces a single date. When year, month and day are
031   * all populated and parseable they are given priority, but if any field is missing and dateString is parseable,
032   * dateString is preferred.
033   */
034  public static DateInterpretationResult interpretDate(String year, String month, String day, String dateString) {
035    if (year == null && month == null && day == null && dateString == null) return new DateInterpretationResult();
036
037    boolean tryDateString = (dateString != null);
038    Date finalDate = null;
039    // first, attempt year, month, day parsing
040
041    // note that ymd filters out any years greater than current calendar year
042    YearMonthDay ymd = DateParseUtils.normalize(year, month, day);
043    ParseResult<Date> parseResult = DateParseUtils.parse(ymd.getYear(), ymd.getMonth(), ymd.getDay());
044    // DateFormats are not thread safe, hence a new one for each call. Returning a formatted string is itself a
045    // legacy holdover.
046    SimpleDateFormat sdf = new SimpleDateFormat(OUTPUT_DATE_FORMAT);
047
048    /**
049     * If the parse result is SUCCESS it means that a whole date could be extracted (with year,
050     * month and day). If it is a failure but the normalizer returned a meaningful result (e.g. it could extract just
051     * a year) we're going to return a result with all the fields set that we could parse.
052     */
053    DateInterpretationResult result = new DateInterpretationResult();
054    switch (parseResult.getStatus()) {
055      case SUCCESS:
056        tryDateString = false;
057        finalDate = parseResult.getPayload();
058        break;
059      case FAIL:
060        // TODO: check requirements - seems we should return null here if we have null/illegal year
061        if (!ymd.representsNull()) {
062          result = new DateInterpretationResult(ymd.getIntegerYear(), ymd.getIntegerMonth(), null, null);
063          break;
064        }
065    }
066
067    // will be true when dateString is not null and the ymd parse ended in FAIL or ERROR
068    if (tryDateString) {
069      Date fromString = parseDateString(dateString);
070      if (fromString != null) {
071        finalDate = fromString;
072      }
073    }
074
075    // finalDate will be null in the ymd FAIL case where tryDateString didn't work
076    if (finalDate != null) {
077      Calendar c = Calendar.getInstance();
078      c.setTime(finalDate);
079
080      if (c.get(Calendar.YEAR) >= EARLIEST_YEAR) {
081        // months are 0 based
082        result = new DateInterpretationResult(c.get(Calendar.YEAR), c.get(Calendar.MONTH) + 1, c.getTime(),
083          sdf.format(c.getTime()));
084      }
085    }
086
087    return result;
088  }
089
090  // TODO this is very naive string parsing - make it better
091  private static Date parseDateString(String dateString) {
092    Date result = null;
093    String[] dateFormats =
094      new String[] {"yyyy-MM-dd", "dd-MM-yyyy", "yyyy/MM/dd", "dd/MM/yyyy", "dd.MM.yyyy", "yyyy.MM.dd"};
095
096    for (String dateFormat : dateFormats) {
097      try {
098        SimpleDateFormat sdf = new SimpleDateFormat(dateFormat);
099        // lenient sdf will happily match 24-12-2001 to yyyy-MM-dd with completely wrong results
100        sdf.setLenient(false);
101        result = sdf.parse(dateString);
102        break;
103      } catch (ParseException e) {
104        LOG.debug("Failed to parse dateString [{}] with pattern [{}], trying other patterns.", dateString, dateFormat);
105      }
106    }
107
108    if (result == null) LOG.debug("Failed to parse dateString [{}] - giving up.", dateString);
109    return result;
110  }
111}
