001package org.gbif.occurrencestore.interpreters;
002
003import org.gbif.api.model.registry.Dataset;
004import org.gbif.api.model.registry.Organization;
005import org.gbif.api.vocabulary.Country;
006import org.gbif.registry.ws.client.DatasetWsClient;
007import org.gbif.registry.ws.client.OrganizationWsClient;
008import org.gbif.ws.json.JacksonJsonContextResolver;
009
010import java.io.IOException;
011import java.io.InputStream;
012import java.util.Properties;
013import java.util.UUID;
014import java.util.concurrent.ExecutionException;
015import java.util.concurrent.TimeUnit;
016import javax.annotation.Nullable;
017
018import com.google.common.cache.CacheBuilder;
019import com.google.common.cache.CacheLoader;
020import com.google.common.cache.LoadingCache;
021import com.google.common.util.concurrent.UncheckedExecutionException;
022import com.sun.jersey.api.client.Client;
023import com.sun.jersey.api.client.WebResource;
024import com.sun.jersey.api.client.config.ClientConfig;
025import com.sun.jersey.api.client.config.DefaultClientConfig;
026import com.sun.jersey.api.json.JSONConfiguration;
027import com.sun.jersey.client.apache.ApacheHttpClient;
028import org.codehaus.jackson.jaxrs.JacksonJsonProvider;
029import org.slf4j.Logger;
030import org.slf4j.LoggerFactory;
031
032import static com.google.common.base.Preconditions.checkNotNull;
033
034/**
035 * Static utility to return fields from an organization.
036 */
037public class OrganizationLookup {
038
039  private static final Logger LOG = LoggerFactory.getLogger(OrganizationLookup.class);
040
041  private static final String WEB_SERVICE_URL;
042  private static final String WEB_SERVICE_URL_PROPERTY = "occurrencestore.registry.ws.url";
043  private static final DatasetWsClient DATASET_CLIENT;
044  private static final OrganizationWsClient ORG_CLIENT;
045
046  // The repetitive nature of our data encourages use of a light cache to reduce WS load
047  private static final LoadingCache<UUID, Dataset> DATASET_CACHE =
048    CacheBuilder.newBuilder().maximumSize(1000).expireAfterAccess(1, TimeUnit.MINUTES)
049      .build(new CacheLoader<UUID, Dataset>() {
050
051        @Override
052        public Dataset load(UUID key) throws Exception {
053          return DATASET_CLIENT.get(key);
054        }
055      });
056
057  private static final LoadingCache<UUID, Organization> ORG_CACHE =
058    CacheBuilder.newBuilder().maximumSize(1000).expireAfterAccess(1, TimeUnit.MINUTES)
059      .build(new CacheLoader<UUID, Organization>() {
060
061        @Override
062        public Organization load(UUID key) throws Exception {
063          return ORG_CLIENT.get(key);
064        }
065      });
066
067  private static final WebResource RESOURCE;
068
069  static {
070    try {
071      InputStream is =
072        NubLookupInterpreter.class.getClassLoader().getResourceAsStream("occurrence-interpreter.properties");
073      try {
074        Properties props = new Properties();
075        props.load(is);
076        WEB_SERVICE_URL = props.getProperty(WEB_SERVICE_URL_PROPERTY);
077      } finally {
078        is.close();
079      }
080    } catch (IOException e) {
081      throw new RuntimeException("Can't load properties file [occurrence-interpreter.properties]", e);
082    }
083
084    ClientConfig cc = new DefaultClientConfig();
085    cc.getClasses().add(JacksonJsonContextResolver.class);
086    cc.getClasses().add(JacksonJsonProvider.class);
087    cc.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, Boolean.TRUE);
088    Client client = ApacheHttpClient.create(cc);
089    RESOURCE = client.resource(WEB_SERVICE_URL);
090    DATASET_CLIENT = new DatasetWsClient(RESOURCE, null);
091    ORG_CLIENT = new OrganizationWsClient(RESOURCE, null);
092  }
093
094  private OrganizationLookup() {
095  }
096
097  /**
098   * Find and return the organization which owns the dataset for the given datasetKey.
099   *
100   * @param datasetKey the dataset owner to find
101   * @return the organization that owns the dataset
102   */
103  @Nullable
104  public static Organization getOrgByDataset(UUID datasetKey) {
105    checkNotNull(datasetKey, "datasetKey can't be null");
106
107    Organization org = null;
108    try {
109      Dataset dataset = DATASET_CACHE.get(datasetKey);
110      if (dataset != null && dataset.getOwningOrganizationKey() != null) {
111        org = ORG_CACHE.get(dataset.getOwningOrganizationKey());
112      }
113    } catch (UncheckedExecutionException e) {
114      LOG.warn("WS failure while looking up org for dataset [{}]", datasetKey, e);
115    } catch (ExecutionException e) {
116      LOG.warn("WS failure while looking up org for dataset [{}]", datasetKey, e);
117    }
118
119    return org == null ? null : org;
120  }
121
122  /**
123   * Find and return the country of the organization.
124   *
125   * @param organizationKey the organization for which to find the country
126   * @return the org's country (the "hostCountry") or null if the organization couldn't be found (or loaded)
127   */
128  @Nullable
129  public static Country getOrgCountry(UUID organizationKey) {
130    checkNotNull(organizationKey, "organizationKey can't be null");
131
132    Organization org = null;
133    try {
134      org = ORG_CACHE.get(organizationKey);
135    } catch (UncheckedExecutionException e) {
136      LOG.warn("WS failure while looking up country for org [{}]", organizationKey, e);
137    } catch (ExecutionException e) {
138      LOG.warn("WS failure while looking up country for org [{}]", organizationKey, e);
139    }
140
141    return org == null ? null : org.getCountry();
142  }
143}
