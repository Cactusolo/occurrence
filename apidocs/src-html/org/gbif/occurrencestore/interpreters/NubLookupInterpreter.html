001package org.gbif.occurrencestore.interpreters;
002
003import org.gbif.api.model.checklistbank.NameUsageMatch;
004import org.gbif.occurrencestore.interpreters.result.NubLookupInterpretationResult;
005import org.gbif.occurrencestore.interpreters.util.RetryingWebserviceClient;
006
007import java.io.IOException;
008import java.io.InputStream;
009import java.util.List;
010import java.util.Map;
011import java.util.Properties;
012import java.util.concurrent.ExecutionException;
013import java.util.concurrent.TimeUnit;
014import javax.ws.rs.core.MultivaluedMap;
015
016import com.google.common.cache.CacheBuilder;
017import com.google.common.cache.LoadingCache;
018import com.sun.jersey.api.client.Client;
019import com.sun.jersey.api.client.UniformInterfaceException;
020import com.sun.jersey.api.client.WebResource;
021import com.sun.jersey.api.client.config.ClientConfig;
022import com.sun.jersey.api.client.config.DefaultClientConfig;
023import com.sun.jersey.api.json.JSONConfiguration;
024import com.sun.jersey.client.apache.ApacheHttpClient;
025import com.sun.jersey.core.util.MultivaluedMapImpl;
026import org.codehaus.jackson.jaxrs.JacksonJsonProvider;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 * Attempts to lookup the nub key and nub classification for a given classification.
032 */
033public class NubLookupInterpreter {
034
035  private static final Logger LOG = LoggerFactory.getLogger(NubLookupInterpreter.class);
036
037  private static final String WEB_SERVICE_URL;
038  private static final String WEB_SERVICE_URL_PROPERTY = "occurrencestore.nub.ws.url";
039
040  // The repetitive nature of our data encourages use of a light cache to reduce WS load
041  private static final LoadingCache<WebResource, NameUsageMatch> CACHE =
042    CacheBuilder.newBuilder().maximumSize(10000).expireAfterAccess(10, TimeUnit.MINUTES)
043      .build(RetryingWebserviceClient.newInstance(NameUsageMatch.class, 10, 2000));
044
045  private static final WebResource RESOURCE;
046
047  static {
048    try {
049      InputStream is =
050        NubLookupInterpreter.class.getClassLoader().getResourceAsStream("occurrence-interpreter.properties");
051      if (is == null) {
052        throw new RuntimeException("Can't load properties file [occurrence-interpreter.properties]");
053      }
054      try {
055        Properties props = new Properties();
056        props.load(is);
057        WEB_SERVICE_URL = props.getProperty(WEB_SERVICE_URL_PROPERTY);
058      } finally {
059        is.close();
060      }
061    } catch (IOException e) {
062      throw new RuntimeException("Can't load properties file [occurrence-interpreter.properties]", e);
063    }
064    ClientConfig cc = new DefaultClientConfig();
065    cc.getFeatures().put(JSONConfiguration.FEATURE_POJO_MAPPING, true);
066    cc.getClasses().add(JacksonJsonProvider.class);
067    Client client = ApacheHttpClient.create(cc);
068    RESOURCE = client.resource(WEB_SERVICE_URL);
069  }
070
071  // if the WS is not responding, we drop into a retry count
072  private static final int NUM_RETRIES = 3;
073  private static final int RETRY_PERIOD_MSEC = 2000;
074
075  private NubLookupInterpreter() {
076  }
077
078  public static NubLookupInterpretationResult nubLookup(String kingdom, String phylum, String clazz, String order,
079    String family, String genus, String scientificName, String author) {
080
081    if (kingdom == null && phylum == null && clazz == null && order == null && family == null && genus == null
082        && scientificName == null && author == null) {
083      return new NubLookupInterpretationResult();
084    }
085
086    NubLookupInterpretationResult result = null;
087    MultivaluedMap<String, String> queryParams = new MultivaluedMapImpl();
088    queryParams.add("kingdom", kingdom);
089    queryParams.add("phylum", phylum);
090    queryParams.add("class", clazz);
091    queryParams.add("order", order);
092    queryParams.add("family", family);
093    queryParams.add("genus", genus);
094    queryParams.add("name", scientificName);
095    /** TODO: include author in query */
096
097    if (scientificName != null) {
098      for (int i = 0; i < NUM_RETRIES; i++) {
099        LOG.debug("Attempt [{}] to lookup sci name [{}]", i, scientificName);
100        try {
101          NameUsageMatch lookup = CACHE.get(RESOURCE.queryParams(queryParams));
102          if (lookup != null) {
103            if (lookup.getMatchType() == NameUsageMatch.MatchType.NONE) {
104              LOG.info("Nub lookup for [{}] returned no match. Lookup note: [{}]", scientificName, lookup.getNote());
105            } else {
106              if (lookup.getMatchType() != NameUsageMatch.MatchType.EXACT) {
107                LOG.debug("Nub lookup for [{}] was not exact. Match type [{}] with note: [{}]", scientificName,
108                  lookup.getMatchType(), lookup.getNote());
109              }
110              result =
111                new NubLookupInterpretationResult(lookup.getUsageKey(), lookup.getKingdomKey(), lookup.getPhylumKey(),
112                  lookup.getClassKey(), lookup.getOrderKey(), lookup.getFamilyKey(), lookup.getGenusKey(),
113                  lookup.getSpeciesKey(), lookup.getKingdom(), lookup.getPhylum(), lookup.getClazz(), lookup.getOrder(),
114                  lookup.getFamily(), lookup.getGenus(), lookup.getSpecies(), lookup.getScientificName(),
115                  lookup.getConfidence());
116            }
117          }
118
119          break; // from retry loop
120        } catch (ExecutionException e) {
121          // Log the error
122          LOG.error("Failed WS call with: {}", recreateQueryString(queryParams));
123
124          // have we exhausted our attempts?
125          if (i >= NUM_RETRIES) {
126            throw new RuntimeException(e);
127          }
128
129          try {
130            Thread.sleep(RETRY_PERIOD_MSEC);
131          } catch (InterruptedException e1) {
132          }
133        } catch (UniformInterfaceException e) {
134          LOG.info("Got unexpected result for scientific name '{}', Response: {}", scientificName, e.getResponse());
135          // have we exhausted our attempts?
136          if (i >= NUM_RETRIES) {
137            throw e;
138          }
139
140          try {
141            Thread.sleep(RETRY_PERIOD_MSEC);
142          } catch (InterruptedException e1) {
143          }
144        }
145      } // retry loop
146    } // scientific name exists
147
148    return result;
149  }
150
151  private static String recreateQueryString(MultivaluedMap<String, String> queryParams) {
152    StringBuilder sb = new StringBuilder("?");
153    for (Map.Entry<String, List<String>> en : queryParams.entrySet()) {
154      sb.append(en.getKey()).append('=');
155      for (String s : en.getValue()) {
156        sb.append(s);
157      }
158      sb.append('&');
159    }
160
161    return sb.toString();
162  }
163}
