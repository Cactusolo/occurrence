001package org.gbif.occurrencestore.interpreters;
002
003import org.gbif.common.parsers.ParseResult;
004import org.gbif.common.parsers.basisofrecord.BasisOfRecordParser;
005import org.gbif.common.parsers.utils.MappingUtils;
006
007/**
008 * Interpret basis of records and return controlled values.
009 * TODO: return a java enum rather than int
010 */
011public class BasisOfRecordInterpreter {
012
013  private static final Integer DEFAULT_RESULT = 0;
014  private static final BasisOfRecordParser PARSER = BasisOfRecordParser.getInstance();
015
016  private BasisOfRecordInterpreter() {
017  }
018
019  /**
020   * Compares given string to controlled vocabulary and then looks up a corresponding int value for that term.
021   *
022   * @param basisOfRecord string to be interpreted
023   *
024   * @return integer value corresponding to matching basis of record from controlled vocabulary
025   */
026  public static Integer interpretBasisOfRecord(String basisOfRecord) {
027    if (basisOfRecord == null) {
028      return DEFAULT_RESULT;
029    }
030
031    ParseResult<String> parseResult = PARSER.parse(basisOfRecord.toString());
032    if (!parseResult.isSuccessful()) {
033      return DEFAULT_RESULT;
034    }
035
036    Integer result = DEFAULT_RESULT;
037    Integer id = MappingUtils.mapBasisOfRecord(parseResult.getPayload());
038    if (id != null) {
039      result = id;
040    }
041
042    return result;
043  }
044}
