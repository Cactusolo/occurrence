001package org.gbif.occurrencestore.interpreters.util;
002
003import java.net.URI;
004import java.net.URL;
005import java.util.concurrent.TimeUnit;
006
007import org.slf4j.Logger;
008import org.slf4j.LoggerFactory;
009
010import com.google.common.base.Preconditions;
011import com.google.common.cache.CacheBuilder;
012import com.google.common.cache.CacheLoader;
013import com.sun.jersey.api.client.GenericType;
014import com.sun.jersey.api.client.WebResource;
015
016
017/**
018 * A cache loader that will call a webservice, falling back into a retry mechanism should any
019 * errors occur in the web service call.  Developers are urged to consider carefully their use
020 * before adopting this approach.  This is particularly suitable for operations such as batch
021 * processing, where one knows the client will be making a large number of identical web service
022 * calls in a short space of time, and changes during that time are not of high importance.
023 * Note that {@link WebResource} is suitable as the key
024 * since it uses {@link URI} and not {@link URL} which the JVM does not try to resolve at runtime.
025 * Construction is forced through static factories to ensure that generics are handled correctly,
026 * removing the need for casting at use.
027 * The following illustrates the intended use of this class:
028 * <pre>
029 *   LoadingCache<WebResource, NameUsageMatch> cache = CacheBuilder.newBuilder()
030 *     .maximumSize(1000)
031 *     .expireAfterAccess(1, TimeUnit.MINUTES)
032 *     .build(RetryingWebserviceClient.newInstance(NameUsageMatch.class, NUM_RETRIES, RETRY_PERIOD_MSEC));
033 *   NameUsageMatch lookup = cache.get(RESOURCE.queryParams(queryParams));
034 * </pre>
035 *
036 * TODO: this is a copy from occurrence-store - move to somewhere common
037 */
038public class RetryingWebserviceClient<T> extends CacheLoader<WebResource, T> {
039
040  private static final Logger LOGGER = LoggerFactory.getLogger(RetryingWebserviceClient.class);
041  private final Class<T> type;
042  private final GenericType<T> genericType;
043  private final int numberOfAttempts;
044  private final int retryDelayMsec;
045  public static final int DEFAULT_NUMBER_OF_ATTEMPTS = 1;
046  public static final int DEFAULT_DELAY_MSECS = 0;
047
048  /**
049   * Creates a new instance using the default configuration {@link #DEFAULT_NUMBER_OF_ATTEMPTS}
050   * and {@link #DEFAULT_DELAY_MSECS}.
051   *
052   * @param type Of resource to be returned by the web service
053   *
054   * @return A cacheloader suitable for using in a Google {@link CacheBuilder}
055   */
056  public static <T> RetryingWebserviceClient<T> newInstance(Class<T> type) {
057    return new RetryingWebserviceClient<T>(type, DEFAULT_NUMBER_OF_ATTEMPTS, DEFAULT_DELAY_MSECS);
058  }
059
060  /**
061   * Creates a new instance using the provided configuration.
062   *
063   * @param type             Of resource to be returned by the web service
064   * @param numberOfAttempts Which must be 1 or more
065   * @param retryDelayMsec   Which must be 0 or more
066   *
067   * @return A cacheloader suitable for using in a Google {@link CacheBuilder}
068   */
069  public static <T> RetryingWebserviceClient<T> newInstance(Class<T> type, int numberOfAttempts, int retryDelayMsec) {
070    return new RetryingWebserviceClient<T>(type, numberOfAttempts, retryDelayMsec);
071  }
072
073  /**
074   * Creates a new instance using the default configuration {@link #DEFAULT_NUMBER_OF_ATTEMPTS}
075   * and {@link #DEFAULT_DELAY_MSECS}.
076   * This allows implementation to use the likes of:
077   * <pre>
078   *   new GenericType<PagingResponse<Dataset>>() {};
079   * </pre>
080   *
081   * @param genericType Of the response object
082   *
083   * @return A cacheloader suitable for using in a Google {@link CacheBuilder}
084   */
085  public static <T> RetryingWebserviceClient<T> newInstance(GenericType<T> genericType) {
086    return new RetryingWebserviceClient<T>(genericType, DEFAULT_NUMBER_OF_ATTEMPTS, DEFAULT_DELAY_MSECS);
087  }
088
089  /**
090   * Creates a new instance using the provided configuratio, which supports the generic typing of the response object.
091   * This allows implementation to use the likes of:
092   * <pre>
093   *   new GenericType<PagingResponse<Dataset>>() {};
094   * </pre>
095   *
096   * @param genericType      Of the response object
097   * @param numberOfAttempts Which must be 1 or more
098   * @param retryDelayMsec   Which must be 0 or more
099   *
100   * @return A cacheloader suitable for using in a Google {@link CacheBuilder}
101   */
102  public static <T> RetryingWebserviceClient<T> newInstance(GenericType<T> genericType, int numberOfAttempts, int retryDelayMsec) {
103    return new RetryingWebserviceClient<T>(genericType, numberOfAttempts, retryDelayMsec);
104  }
105
106  /**
107   * Force the static factory, to ensure generics are respected.
108   */
109  private RetryingWebserviceClient(Class<T> type, int numberOfAttempts, int retryDelayMsec) {
110    Preconditions.checkArgument(numberOfAttempts > 0, "Number of attempts must be a positive number");
111    Preconditions.checkArgument(retryDelayMsec >= 0, "Retry delay must be 0 or more milliseconds");
112    this.type = Preconditions.checkNotNull(type);
113    this.genericType = null;
114    this.numberOfAttempts = numberOfAttempts;
115    this.retryDelayMsec = retryDelayMsec;
116  }
117
118  /**
119   * Force the static factory, to ensure generics are respected.
120   */
121  private RetryingWebserviceClient(GenericType<T> genericType, int numberOfAttempts, int retryDelayMsec) {
122    Preconditions.checkArgument(numberOfAttempts > 0, "Number of attempts must be a positive number");
123    Preconditions.checkArgument(retryDelayMsec >= 0, "Retry delay must be 0 or more milliseconds");
124    this.type = null;
125    this.genericType = Preconditions.checkNotNull(genericType);
126    this.numberOfAttempts = numberOfAttempts;
127    this.retryDelayMsec = retryDelayMsec;
128  }
129
130  @Override
131  public T load(WebResource resource) throws Exception {
132    for (int attempt = 1; attempt <= numberOfAttempts; attempt++) {
133      try {
134        if (type != null) {
135          return resource.get(type);
136        } else {
137          return resource.get(genericType);
138        }
139
140      } catch (Exception e) {
141        LOGGER.error("Error looking up resource in attempt[{}] of max[{}]", new Object[] {attempt, numberOfAttempts, e});
142        if (attempt >= numberOfAttempts) {
143          throw e;
144        }
145        if (retryDelayMsec > 0) {
146          TimeUnit.MILLISECONDS.sleep(retryDelayMsec);
147        }
148      }
149    }
150    // Should not ever happen as we propagate the underlying exception above
151    throw new IllegalStateException("Retry count exhausted");
152  }
153}
