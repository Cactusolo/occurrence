001package org.gbif.occurrencestore.hive.udf;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004import org.gbif.occurrencestore.util.BasisOfRecordConverter;
005
006import org.apache.hadoop.hive.ql.exec.Description;
007import org.apache.hadoop.hive.ql.exec.UDF;
008import org.apache.hadoop.io.Text;
009
010/**
011 * Converts the basis of record int stored in hbase into the textual BasisOfRecord enum value.
012 */
013@Description(
014  name = "getBorEnum",
015  value = "_FUNC_(borAsInt)")
016public class BasisOfRecordLookupUDF extends UDF {
017
018  private static final BasisOfRecordConverter BOR_CONVERTER = new BasisOfRecordConverter();
019  private static final Text BOR = new Text();
020
021  public Text evaluate(Text borAsInt) {
022    // default to nothing in case anything goes "wrong" below
023    BOR.set("");
024
025    if (borAsInt != null && borAsInt.getLength() > 0) {
026      try {
027        Integer code = Integer.parseInt(borAsInt.toString());
028        if (code != null) {
029          BasisOfRecord bor = BOR_CONVERTER.toEnum(code);
030          if (bor != null) {
031            BOR.set(bor.name());
032          } else {
033            System.err.println("Unknown BasisOfRecord code: " + borAsInt.toString());
034          }
035        }
036      } catch (NumberFormatException e) {
037        // swallow it all and log
038        System.err.println("Non numerical representation of BasisOfRecord: " + borAsInt.toString());
039      }
040    }
041
042    return BOR;
043  }
044
045}
