001package org.gbif.occurrencestore.hive.udf;
002
003import org.gbif.occurrence.download.util.DownloadUtils;
004
005import java.text.SimpleDateFormat;
006import java.util.Date;
007
008import org.apache.hadoop.hive.ql.exec.Description;
009import org.apache.hadoop.hive.ql.exec.UDF;
010import org.apache.hadoop.io.Text;
011
012/**
013 * A simple UDF for Hive to convert a date/long to an string in ISO 8601 format.
014 * If the input value is null or can't be parsed, and empty string is returned.
015 */
016@Description(
017  name = "toISO8601",
018  value = "_FUNC_(field)")
019public class ToISO8601UDF extends UDF {
020
021  private final Text text = new Text();
022
023
024  public Text evaluate(Text field) {
025    if (field == null) {
026      text.set("");
027    } else {
028      try {
029        text
030          .set(new SimpleDateFormat(DownloadUtils.ISO_8601_FORMAT).format(new Date(Long.parseLong(field.toString()))));
031      } catch (NumberFormatException e) {
032        text.set("");
033      }
034    }
035
036    return text;
037  }
038}
