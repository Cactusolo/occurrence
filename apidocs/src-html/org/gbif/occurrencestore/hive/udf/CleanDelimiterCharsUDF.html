001package org.gbif.occurrencestore.hive.udf;
002
003import org.gbif.occurrence.download.util.DownloadUtils;
004
005import org.apache.hadoop.hive.ql.exec.Description;
006import org.apache.hadoop.hive.ql.exec.UDF;
007import org.apache.hadoop.io.Text;
008
009/**
010 * A simple UDF for Hive that replaces specials characters with blanks.
011 * The characters replaced by this UDF can break a download format and those are: tabs, line breaks and new lines.
012 * If the input value is null or can't be parsed, and empty string is returned.
013 */
014@Description(
015  name = "cleanDelimiters",
016  value = "_FUNC_(field)")
017public class CleanDelimiterCharsUDF extends UDF {
018
019  private final Text text = new Text();
020
021  public Text evaluate(Text field) {
022    if (field == null) {
023      text.set("");
024    } else {
025      text.set(field.toString().replaceAll(DownloadUtils.DELIMETERS_MATCH, " "));
026    }
027    return text;
028  }
029}
