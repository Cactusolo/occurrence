001package org.gbif.occurrencestore.util;
002
003import org.gbif.api.vocabulary.BasisOfRecord;
004
005import java.util.Map;
006
007import com.google.common.collect.ImmutableMap;
008
009/**
010 * Converter turning a basis of record enum into an integer and vice versa.
011 * The integer value is the one used by HBase persistence.
012 */
013public class BasisOfRecordConverter {
014
015  // TODO: 4 used to be germplams and is hangover from mysql days - once the dictionary parser (gbif-parsers) is fixed
016  // to return only enums and the only int handling is here, this class can hold 1:1 mapping enum:int.
017  // See: GBIFCOM-141
018  private final Map<Integer, BasisOfRecord> mapToEnum =
019    ImmutableMap.<Integer, BasisOfRecord>builder()
020      .put(0, BasisOfRecord.UNKNOWN)
021      .put(1, BasisOfRecord.OBSERVATION)
022      .put(2, BasisOfRecord.PRESERVED_SPECIMEN)
023      .put(3, BasisOfRecord.LIVING_SPECIMEN)
024      .put(4, BasisOfRecord.LIVING_SPECIMEN)
025      .put(5, BasisOfRecord.FOSSIL_SPECIMEN)
026      .put(6, BasisOfRecord.LITERATURE)
027      .build();
028
029  private final Map<BasisOfRecord, Integer> mapFromEnum =
030    ImmutableMap.<BasisOfRecord, Integer>builder()
031      .put(BasisOfRecord.UNKNOWN, 0)
032      .put(BasisOfRecord.OBSERVATION, 1)
033      .put(BasisOfRecord.PRESERVED_SPECIMEN, 2)
034      .put(BasisOfRecord.LIVING_SPECIMEN, 3)
035      .put(BasisOfRecord.FOSSIL_SPECIMEN, 5)
036      .put(BasisOfRecord.LITERATURE, 6)
037      .put(BasisOfRecord.HUMAN_OBSERVATION, 7)
038      .put(BasisOfRecord.MACHINE_OBSERVATION, 8)
039      .build();
040
041
042  public BasisOfRecord toEnum(Integer key) {
043    if (key == null) return BasisOfRecord.UNKNOWN;
044    BasisOfRecord val = mapToEnum.get(key);
045    return val == null ? BasisOfRecord.UNKNOWN : val;
046  }
047
048  public Integer fromEnum(BasisOfRecord value) {
049    if (value == null) return null;
050    if (mapFromEnum.containsKey(value)) {
051      return mapFromEnum.get(value);
052    }
053    throw new IllegalArgumentException("Enumeration value " + value.name() + " unknown");
054  }
055}
