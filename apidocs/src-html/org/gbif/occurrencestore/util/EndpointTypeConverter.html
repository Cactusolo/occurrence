001package org.gbif.occurrencestore.util;
002
003import org.gbif.api.vocabulary.EndpointType;
004
005import java.util.Map;
006
007import com.google.common.collect.ImmutableMap;
008
009/**
010 * Converter turning an EndpointType enum into an integer and vice versa.
011 * The integer value is the one used by HBase persistence.
012 */
013public class EndpointTypeConverter {
014
015  private final Map<Integer, EndpointType> mapToEnum =
016    ImmutableMap.<Integer, EndpointType>builder()
017      .put(0, EndpointType.BIOCASE)
018      .put(1, EndpointType.DIGIR)
019      .put(2, EndpointType.DIGIR_MANIS)
020      .put(3, EndpointType.DWC_ARCHIVE)
021      // here used to be DWC_ARCHIVE_OCCURRENCE & CHECKLIST (4 and 5)
022      .put(6, EndpointType.EML)
023      .put(7, EndpointType.FEED)
024      .put(8, EndpointType.OAI_PMH)
025      .put(9, EndpointType.OTHER)
026      .put(10, EndpointType.TAPIR)
027      .put(11, EndpointType.TCS_RDF)
028      .put(12, EndpointType.TCS_XML)
029      .put(13, EndpointType.WFS)
030      .put(14, EndpointType.WMS)
031      .build();
032
033  private final Map<EndpointType, Integer> mapFromEnum =
034    ImmutableMap.<EndpointType, Integer>builder()
035      .put(EndpointType.BIOCASE, 0)
036      .put(EndpointType.DIGIR, 1)
037      .put(EndpointType.DIGIR_MANIS, 2)
038      .put(EndpointType.DWC_ARCHIVE, 3)
039        // here used to be DWC_ARCHIVE_OCCURRENCE & CHECKLIST (4 and 5)
040      .put(EndpointType.EML, 6)
041      .put(EndpointType.FEED, 7)
042      .put(EndpointType.OAI_PMH, 8)
043      .put(EndpointType.OTHER, 9)
044      .put(EndpointType.TAPIR, 10)
045      .put(EndpointType.TCS_RDF, 11)
046      .put(EndpointType.TCS_XML, 12)
047      .put(EndpointType.WFS, 13)
048      .put(EndpointType.WMS, 14)
049      .build();
050
051
052  public Integer fromEnum(EndpointType value) {
053    if (value == null) {
054      return null;
055    }
056    if (mapFromEnum.containsKey(value)) {
057      return mapFromEnum.get(value);
058    }
059    throw new IllegalArgumentException("Enumeration value " + value.name() + " unknown");
060  }
061
062  public EndpointType toEnum(Integer key) {
063    return mapToEnum.get(key);
064  }
065}
