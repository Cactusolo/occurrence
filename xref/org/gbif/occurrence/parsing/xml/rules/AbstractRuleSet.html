76    protected void addNonNullMethod(Digester digester, String property, String methodName, int argCount) {
77      String pattern = mappingProps.getProperty(property);
78      if (pattern != null) {
79        pattern = pattern.trim();
80        LOG.debug("adding call method [{}] for pattern [{}] from property [{}]",
81          new Object[] {methodName, pattern, property});
82        digester.addCallMethod(pattern, methodName, argCount);

95    protected void addNonNullAttParam(Digester digester, String elementProperty, String attributeProperty,

97      String elemPattern = mappingProps.getProperty(elementProperty);

99      if (elemPattern != null && attPattern != null) {

102       LOG.debug(
103         "adding call param from attribute for element pattern [{}] from property [{}], seeking attribute [{}] from property [{}]",

105       digester.addCallParam(elemPattern, argPosition, attPattern);

109   protected void addNonNullPrioritizedProperty(Digester digester, String property, PrioritizedPropertyNameEnum name,
110                                                int paramCount) {
111     for (int i = 1; i <= paramCount; i++) {
112       String key = property + '.' + i;
113       String pattern = mappingProps.getProperty(key);
114       if (pattern != null) {
115         pattern = pattern.trim();
116         LOG.debug("adding prioritized property [{}] with pattern [{}]", key, pattern);
117         // note order of rule addition is critical
118         digester.addObjectCreate(pattern, PrioritizedProperty.class);
119         digester.addSetNext(pattern, "addPrioritizedProperty");
120         digester.addRule(pattern, new SetLiteralRule("setPriority", i));
121         digester.addRule(pattern, new SetLiteralRule("setName", name));
122         digester.addCallMethod(pattern, "setProperty", 1);
123         digester.addCallParam(pattern, 0);
124       }
125     }
