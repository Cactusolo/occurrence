33  /**
34   * Primary purpose of this class is, given a raw occurrence record serialized to String from XML,
35   * determine which OccurrenceSchemaType it matches.
36   */
37  public class ResponseSchemaDetector {
38  
39    private static final Logger LOG = LoggerFactory.getLogger(ResponseSchemaDetector.class);
40  
41    // a dumb way to guarantee that string searching will be
42    // done such that eg "CatalogNumberText" gets searched for before "CatalogNumber"
43    private List<OccurrenceSchemaType> schemaSearchOrder;
44    private Map<OccurrenceSchemaType, Map<ResponseElementEnum, String>> distinctiveElements;
45  
46    public ResponseSchemaDetector() {
47      init();
48    }
49  
50    private boolean checkElements(String xml, Collection<String> elements) {
51      boolean result = true;
52      for (String elem : elements) {
53        result = xml.contains(elem);
54        LOG.debug("Xml contains [{}] is [{}]", elem, result);
55        if (!result) return result;
56      }
57  
58      return result;
59    }
60  
61    public OccurrenceSchemaType detectSchema(String xml) {
62      OccurrenceSchemaType result = null;
63      for (OccurrenceSchemaType schema : schemaSearchOrder) {
64        LOG.debug("Checking for schema [{}]", schema);
65        boolean success = checkElements(xml, distinctiveElements.get(schema).values());
66        if (success) {
67          result = schema;
68          break;
69        }
70      }
71  
72      if (result == null) LOG.warn("Could not determine schema for xml [{}]", xml);
73  
74      return result;
75    }
76  
77    public Map<ResponseElementEnum, String> getResponseElements(OccurrenceSchemaType schemaType) {
78      return distinctiveElements.get(schemaType);
79    }

135   private void printNode(Node node) {
136     LOG.info("Node name: [{}]", node.getNodeName());
137     if (node.hasChildNodes()) {
138       NodeList children = node.getChildNodes();
139       for (int i = 0; i < children.getLength(); i++) {
140         printNode(children.item(i));
141       }
142     }
