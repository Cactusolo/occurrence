102   public RawOccurrenceRecord(DarwinCoreRecord dwcr) {
103     this.basisOfRecord = dwcr.getBasisOfRecord();
104     this.catalogueNumber = dwcr.getCatalogNumber();
105     this.klass = dwcr.getClasss();
106     this.collectionCode = dwcr.getCollectionCode();
107     this.continentOrOcean = dwcr.getContinent();
108     this.country =
109       (dwcr.getCountry() == null || dwcr.getCountry().equals("")) ? dwcr.getCountryCode() : dwcr.getCountry();
110     this.county = dwcr.getCounty();
111     this.dateIdentified = dwcr.getDateIdentified();
112     this.latitude = (dwcr.getVerbatimLatitude() == null) ? dwcr.getDecimalLatitude() : dwcr.getVerbatimLatitude();
113     this.longitude = (dwcr.getVerbatimLongitude() == null) ? dwcr.getDecimalLongitude() : dwcr.getVerbatimLongitude();
114     this.family = dwcr.getFamily();
115     this.scientificName =
116       (dwcr.getScientificName() == null || dwcr.getScientificName().equals("")) ? dwcr.getFullScientificName()
117         : dwcr.getScientificName();
118     this.genus = dwcr.getGenus();
119     this.identifierName = dwcr.getIdentifiedBy();
120     this.institutionCode = dwcr.getInstitutionCode();
121     this.kingdom = dwcr.getKingdom();
122     this.maxDepth = dwcr.getMaximumDepthInMeters();
123     this.minDepth = dwcr.getMinimumDepthInMeters();
124     this.maxAltitude = dwcr.getMaximumElevationInMeters();
125     this.minAltitude = dwcr.getMinimumElevationInMeters();
126     this.order = dwcr.getOrder();
127     this.phylum = dwcr.getPhylum();
128     this.occurrenceDate = dwcr.getYear() + "-" + dwcr.getMonth() + "-" + dwcr.getDay();
129 
130     /** TODO: this probably won't always work - check and fix */
131     //    this.id = (dwcr.getIdentifier() != null) ? dwcr.getIdentifier() : dwcr.getInstitutionCode() + "_"
132     //        + dwcr.getCollectionCode() + "_" + dwcr.getCatalogNumber();;
133   }
