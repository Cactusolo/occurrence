18  import org.gbif.occurrence.constants.PrioritizedPropertyNameEnum;
19  import org.gbif.occurrence.parsing.xml.PrioritizedProperty;

21  import java.util.HashMap;
22  import java.util.HashSet;

29  /**
30   * In schemas that have multiple representations of the same field (eg decimal latitude vs text latitude) this class
31   * gives a framework for setting the order of preference of fields for resolving cases where more than one of them
32   * is populated.
33   */
34  public abstract class PropertyPrioritizer {
35  
36    private static final Logger LOG = LoggerFactory.getLogger(PropertyPrioritizer.class);
37  
38  
39    protected Map<PrioritizedPropertyNameEnum, Set<PrioritizedProperty>> prioritizedProps =
40      new HashMap<PrioritizedPropertyNameEnum, Set<PrioritizedProperty>>();
41  
42    public abstract void resolvePriorities();
43  
44    public void addPrioritizedProperty(PrioritizedProperty prop) {
45      if (LOG.isDebugEnabled()) LOG.debug(">> addPrioritizedProperty [{}]", prop.debugDump());
46  
47      if (prop.getName() != null) {
48        Set<PrioritizedProperty> nameProps = prioritizedProps.get(prop.getName());
49        if (nameProps == null) nameProps = new HashSet<PrioritizedProperty>();
50        nameProps.add(prop);
51        prioritizedProps.put(prop.getName(), nameProps);
52      } else {
53        LOG.warn("Attempting add of null PrioritizedProperty");
54      }
55  
56      LOG.debug("<< addPrioritizedProperty");
57    }
58  
59    /** Highest priority is 1. */
60    protected String findHighestPriority(Set<PrioritizedProperty> props) {
61      String result = null;
62      int highestPriority = Integer.MAX_VALUE;
63      for (PrioritizedProperty prop : props) {
64        if (prop.getPriority().intValue() < highestPriority) {
65          highestPriority = prop.getPriority();
66          result = prop.getProperty();
67        }
68      }
69  
70      return result;
71    }
72  }
