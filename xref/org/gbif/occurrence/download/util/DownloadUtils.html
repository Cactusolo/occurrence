27     * @param path the path to be added without a prefix /
28     * @return a string with the form base/path

30    public static String concatUrlPaths(String base, String path) {
31      StringBuilder sb = new StringBuilder();
32      sb.append(base);
33      if (!base.endsWith("/")) {
34        sb.append('/');
35      }
36      sb.append(path);
37      return sb.toString();
38    }
