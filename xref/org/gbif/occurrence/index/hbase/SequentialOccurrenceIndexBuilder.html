50    /**
51     * Main method.
52     * Requires 3 parameters: <hbaseInputTable> <solrHomeDir> <solrDataDir>.
53     * hbaseInputTable: name of the hbase table.
54     */
55    public static void main(String[] args) throws IOException {
56      if (args.length < 2) {
57        System.err
58          .println("Usage: SequentialOccurrenceIndexBuilder <hbaseInputTable> <solrHomeDir> <solrDataDir>");
59        return;
60      }
61      Properties p = new Properties();
62      p.setProperty(KEY_SOURCE_TABLE, args[0]);
63      p.setProperty(SOLR_HOME, args[1]);
64      if (args[2] != null) {
65        p.setProperty(NR_OF_RECORDS, args[2]);
66      }
67      SequentialOccurrenceIndexBuilder builder = new SequentialOccurrenceIndexBuilder(p);
68      builder.build();

72    /**
73     * Builds the Occurrence Solr index.
74     */
75    private void build() throws IOException {
76      HTable table = null;
77      ResultScanner scanner = null;
78      Closer closer = Closer.create();
79      SolrServer solrServer = EmbeddedSolrServerBuilder.build(props.getProperty(SOLR_HOME));
80      long nrOfRecords = -1;
81      if (props.containsKey(NR_OF_RECORDS)) {
82        nrOfRecords = Long.parseLong(props.getProperty(NR_OF_RECORDS));
83      }
84      int numDocs = 0;
85      // The doc instance is re-used
86      SolrInputDocument doc = new SolrInputDocument();
87      LOG.info("Initiating indexing job");
88      try {
89        Configuration conf = HBaseConfiguration.create();
90        table = new HTable(conf, props.getProperty(KEY_SOURCE_TABLE));
91        closer.register(table);
92        Scan scan = buildOccurrenceScan();
93        scan.setCaching(1200);
94        scanner = table.getScanner(scan);
95        closer.register(scanner);
96  
97        Iterator<Result> iter = scanner.iterator();
98        while (iter.hasNext() && (nrOfRecords == -1 || numDocs < nrOfRecords)) {
99          buildOccSolrDocument(iter.next(), doc);
100         addDocumentQuietly(solrServer, doc);
101         numDocs++;
102       }
103     } catch (Exception ex) {
104       LOG.error("A general error has occurred", ex);
105       LOG.error("Last occurrence record visited", doc.getFieldValue("key"));
106     } finally {
107       closer.close();
108       LOG.info("Commiting {} documents", numDocs);
109       commitQuietly(solrServer);
110       LOG.info("Commit has finished");
111       shutdownSolrServer(solrServer);
112       LOG.info("Indexing job has finished, # of documents indexed: {}", numDocs);
113     }
114   }

116   /**
117    * Commits changes to the solr server.
118    * Exceptions {@link SolrServerException} and {@link IOException} are swallowed.
119    */
120   private void commitQuietly(SolrServer solrServer) {
121     try {
122       solrServer.commit();
123     } catch (SolrServerException e) {
124       LOG.error("Solr error commiting on Solr server", e);
125     } catch (IOException e) {
126       LOG.error("I/O error commiting on Solr server", e);
127     }
128   }
