29    public int run(String[] args) throws Exception {
30      if (args.length < 2) {
31        System.err
32          .println("Usage: SolrOccIndexer  -solr <solrHome> -htable <hbaseTable>[-shards NNN] [-compress_output]");
33        System.err.println("\tinputDir\tinput directory(-ies) containing occurrence files");
34        System.err.println("\toutputDir\toutput directory containing Solr indexes.");
35        System.err.println("\tsolr <solrHome>\tlocal directory containing Solr conf/ and lib/");
36        System.err.println("\tshards NNN\tset the number of output shards to NNN");
37        System.err.println("\t\t(default: the default number of reduce tasks)");
38        System.err.println("\tcompress_output\tto compress the output of the reducer tasks (create .zip file)");
39        return -1;
40      }
41      Job job = new Job(getConf());
42      job.setJarByClass(HbaseToSolrOccIndexer.class);
43      job.getConfiguration().set("mapred.map.tasks.speculative.execution", "false");
44      job.getConfiguration().set("mapred.reduce.tasks.speculative.execution", "false");
45      job.getConfiguration().set("mapred.task.timeout", "900000"); // 10 mins
46      job.getConfiguration().set("hbase.regionserver.lease.period", "900000");
47      job.getConfiguration().set("mapred.compress.map.output", "true");
48      job.getConfiguration().set("mapred.output.compress", "true");
49  
50  
51      int shards = 1;
52      boolean compressOutput = false;
53      String solrHome = null;
54      Path out = new Path(args[0]);
55      String inputTable = null;
56      for (int i = 1; i < args.length; i++) {
57        if (args[i] == null) {
58          continue;
59        }
60        if (args[i].equals("-shards")) {
61          shards = Integer.parseInt(args[++i]);
62        } else if (args[i].equals("-compress_output")) {
63          compressOutput = true;
64        } else if (args[i].equals("-solr")) {
65          solrHome = args[++i];
66          continue;
67        } else if (args[i].equals("-htable")) {
68          inputTable = args[++i];
69          continue;
70        }
71      }
72      if (solrHome == null || !new File(solrHome).exists()) {
73        throw new IOException("You must specify a valid solr.home directory!");
74      }
75      TableMapReduceUtil.initTableMapperJob(inputTable, IndexingUtils.buildOccurrenceScan(),
76        OccurrenceTableMapper.class, OccurrenceWritable.class,
77        IntWritable.class, job);
78  
79      job.setReducerClass(SolrReducer.class);
80      job.setOutputFormatClass(SolrOutputFormat.class);
81      SolrOutputFormat.setupSolrHomeCache(new File(solrHome), job.getConfiguration());
82      if (shards > 0) {
83        job.setNumReduceTasks(shards);
84      }
85      job.setOutputKeyClass(OccurrenceWritable.class);
86      job.setOutputValueClass(IntWritable.class);
87  
88      SolrDocumentConverter.setSolrDocumentConverter(OccurrenceObjectConverter.class, job.getConfiguration());
89  
90      FileOutputFormat.setOutputPath(job, out);
91  
92      SolrOutputFormat.setOutputZipFormat(compressOutput, job.getConfiguration());
93  
94  
95      return job.waitForCompletion(true) ? 0 : -1;
96    }
