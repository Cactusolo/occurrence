24   * Job that creates a part of CSV file. The file is generated according to the fileJob field.

26  class OccurrenceIndexingJob implements Callable<Integer> {

31    private final String hTable;
32    private final int stopRow;
33    private final int startRow;

36     * Default constructor.
37     */
38    public OccurrenceIndexingJob(int startRow, int stopRow, SolrServer solrServer, String hTable) {
39      this.solrServer = solrServer;
40      this.startRow = startRow;
41      this.stopRow = stopRow;
42      this.hTable = hTable;
43    }

45  
46    /**
47     * Executes the job.query and creates a data file that will contains the records from job.from to job.to positions.
48     */
49    @Override
50    public Integer call() throws IOException {
51      HTable table = null;
52      ResultScanner scanner = null;
53      int numDocs = 0;
54      // The doc instance is re-used
55      SolrInputDocument doc = new SolrInputDocument();
56      LOG.info("Initiating indexing job from " + startRow + " to " + stopRow);
57      Closer closer = Closer.create();
58      try {
59        Configuration conf = HBaseConfiguration.create();
60        table = new HTable(conf, hTable);
61        closer.register(table);
62        Scan scan = buildOccurrenceScan();
63        scan.setStartRow(Bytes.toBytes(startRow));
64        scan.setStopRow(Bytes.toBytes(stopRow));
65        scan.setCaching(200);
66        scanner = table.getScanner(scan);
67        closer.register(scanner);
68        for (Result result : scanner) {
69          IndexingUtils.buildOccSolrDocument(result, doc);
70          IndexingUtils.addDocumentQuietly(solrServer, doc);
71          numDocs++;
72        }
73      } catch (Exception ex) {
74        LOG.error("A general error has occurred", ex);
75        LOG.error("Last occurrence record visited", doc.getFieldValue("key"));
76      } finally {
77        closer.close();
78        LOG.info("Commiting {} documents", numDocs);
79        LOG.info("Indexing job has finished, # of documents indexed: {}", numDocs);
80      }
81      return numDocs;
82    }
