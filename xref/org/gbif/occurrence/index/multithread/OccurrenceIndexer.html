35  /**
36   * Class that creates a single CSV file using a OccurrenceSearchRequest.
37   * The file is built with a set of worker threads that create a piece of the output file, once the jobs have finished
38   * their jobs a single file is created. The order of the records is respected from the search request by using the
39   * fileJob.startOffset reported by each job.
40   */

77    /**
78     * Entry point. This method: creates the output files, runs the jobs (OccurrenceFileWriterJob) and then collect the
79     * results. Individual files created by each job are deleted.
80     */
81    public void run() {
82      try {
83        StopWatch stopwatch = new StopWatch();
84        stopwatch.start();
85        ExecutionContextExecutorService executionContext =
86          ExecutionContexts.fromExecutorService(Executors.newFixedThreadPool(nrOfWorkers));
87        collectResults(Futures.sequence(runHBaseJobs(executionContext), executionContext));
88        stopwatch.stop();
89        executionContext.shutdownNow();
90        commitQuietly(solrServer);
91        shutdownSolrServer(solrServer);
92        final long timeInSeconds = TimeUnit.MILLISECONDS.toSeconds(stopwatch.getTime());
93        System.out.println(String.format(FINISH_MSG_FMT, TimeUnit.SECONDS.toMinutes(timeInSeconds), timeInSeconds % 60));
94      } catch (IOException e) {
95        Throwables.propagate(e);
96      }
97    }

99    /**
100    * Collects the results of each job.
101    * Iterates over the list of futures to collect individual results.
102    */
103   private void collectResults(Future<Iterable<Integer>> futures) throws IOException {
104     Closer outFileCloser = Closer.create();
105     try {
106       List<Integer> results =
107         Lists.newArrayList(Await.result(futures, new Timeout(AWAIT_TIME, TimeUnit.MINUTES).duration()));
108       Integer total = 0;
109       for (Integer result : results) {
110         total += result;
111       }
112       System.out.println(total + " documents indexed");
113     } catch (FileNotFoundException e) {
114       Throwables.propagate(e);
115     } catch (IOException e) {
116       Throwables.propagate(e);
117     } catch (Exception e) {
118       Throwables.propagate(e);
119     } finally {
120       outFileCloser.close();
121     }
122   }

139   /**
140    * Run the list of jobs. The amount of records is assigned evenly among the worker threads.
141    * If the amount of records is not divisible by the nrOfWorkers the remaining records are assigned "evenly" among the
142    * first jobs.
143    */
144   private List<Future<Integer>> runHBaseJobs(ExecutionContextExecutorService executionContext) throws IOException {
145 
146     HTable table = null;
147     ResultScanner scanner = null;
148     // Number of records that will be assigned to each job
149     final long sizeOfChunks = Math.max(nrOfRecords / nrOfWorkers, 1);
150 
151     // Remaining jobs, that are not assigned to a job yet
152     final long remaining = nrOfRecords - (sizeOfChunks * nrOfWorkers);
153 
154     // How many of the remaining jobs will be assigned to one job
155     long remainingPerJob = remaining > 0 ? Math.max(remaining / nrOfWorkers, 1) : 0;
156 
157     List<Future<Integer>> futures = Lists.newArrayList();
158     int numDocs = 0;
159     // The doc instance is re-used
160 
161     Closer closer = Closer.create();
162     try {
163       Configuration conf = HBaseConfiguration.create();
164       table = new HTable(conf, hTable);
165       closer.register(table);
166       Scan scan = new Scan();
167       scan.setCaching(200);
168       scanner = table.getScanner(scan);
169       closer.register(scanner);
170       long from;
171       long to = 0;
172       long additionalJobsCnt = 0;
173       for (int i = 0; i < nrOfWorkers; i++) {
174         from = i == 0 ? 0 : to;
175         to = from + sizeOfChunks + remainingPerJob;
176 
177         // Calculates the remaining jobs that will be assigned to the new FileJob.
178         additionalJobsCnt += remainingPerJob;
179         if (remainingPerJob != 0 && additionalJobsCnt > remaining) {
180           remainingPerJob = additionalJobsCnt - remaining;
181         } else if (additionalJobsCnt == remaining) {
182           remainingPerJob = 0;
183         }
184         Result startRow = scanner.next();
185         if (startRow != null) {
186           Result stopRow = startRow;
187           long iJob = 0;
188           for (iJob = from + 1; stopRow != null && iJob <= to; iJob++) {
189             stopRow = scanner.next();
190           }
191           System.out.println(iJob);
192           System.out.println("Assiging occ records from " + from + " to " + to);
193           futures
194             .add(Futures.future(
195               new OccurrenceIndexingJob(Bytes.toInt(startRow.getRow()), Bytes.toInt(stopRow.getRow()), solrServer,
196                 hTable), executionContext));
197         }
198         numDocs++;
199       }
200     } catch (Exception ex) {
201       LOG.error("A general error has occurred", ex);
202     } finally {
203       closer.close();
204       LOG.info("Commiting {} documents", numDocs);
205       LOG.info("Indexing job has finished, # of documents indexed: {}", numDocs);
206     }
207     return futures;
208   }
