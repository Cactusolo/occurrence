73      public void run(RecordReader<LongWritable, Text> reader, OutputCollector<Text, Text> output, final Reporter report)
74        throws IOException {
75        LongWritable key = reader.createKey();
76        Text value = reader.createValue();
77  
78        String tmp = _conf.get("hadoop.tmp.dir");
79        long millis = System.currentTimeMillis();
80        String shardName = "" + millis + "-" + new Random().nextInt();
81        File file = new File(tmp, shardName);
82        report.progress();
83        Analyzer analyzer = getAnalyzer(_conf);
84        Directory directory = FSDirectory.open(file);
85        IndexWriterConfig idxConfig = new IndexWriterConfig(Version.LUCENE_41, new StandardAnalyzer(Version.LUCENE_41));
86        IndexWriter indexWriter = new IndexWriter(directory, idxConfig);
87        report.setStatus("Adding documents...");
88        while (reader.next(key, value)) {
89          report.progress();
90          Document doc = new Document();
91          String text = "" + value.toString();
92          Field contentField = new Field("content", text, StringField.TYPE_STORED);
93          doc.add(contentField);
94          indexWriter.addDocument(doc);
95        }
96  
97        report.setStatus("Done adding documents.");
98        Thread t = new Thread() {
99  
100         public boolean stop = false;
101 
102         @Override
103         public void run() {
104           while (!stop) {
105             // Makes sure hadoop is not killing the task in case the
106             // optimization
107             // takes longer than the task timeout.
108             report.progress();
109             try {
110               sleep(10000);
111             } catch (InterruptedException e) {
112               // don't need to do anything.
113               stop = true;
114             }
115           }
116         }
117       };
118       t.start();
119       report.setStatus("Optimizing index...");
120       report.setStatus("Done optimizing!");
121       report.setStatus("Closing index...");
122       indexWriter.close();
123       report.setStatus("Closing done!");
124       FileSystem fileSystem = FileSystem.get(_conf);
125 
126       report.setStatus("Starting copy to final destination...");
127       Path destination = new Path(_conf.get("finalDestination"));
128       fileSystem.copyFromLocalFile(new Path(file.getAbsolutePath()), destination);
129       report.setStatus("Copy to final destination done!");
130       report.setStatus("Deleting tmp files...");
131       FileUtil.fullyDelete(file);
132       report.setStatus("Delteing tmp files done!");
133       t.interrupt();
134     }
135   }

153   public void startIndexer(String path, String finalDestination, int numOfShards) throws IOException {
154     // create job conf with class pointing into job jar.
155     JobConf jobConf = new JobConf(IndexerJob.class);
156     jobConf.setJobName("indexer");
157     jobConf.setMapRunnerClass(Indexer.class);
158     // alternative use a text file and a TextInputFormat
159     jobConf.setInputFormat(SequenceFileInputFormat.class);
160 
161     Path input = new Path(path);
162     SequenceFileInputFormat.setInputPaths(jobConf, input);
163     // the output path is set just is because is required
164     TextOutputFormat.setOutputPath(jobConf, new Path(finalDestination));
165     // lucene indexes folder
166     jobConf.set("finalDestination", finalDestination);
167     // important to switch spec exec off to avoid duplicated documents
168     jobConf.setSpeculativeExecution(false);
169 
170     // The num of map tasks is equal to the num of input splits.
171     // The num of input splits by default is equal to the num of hdf blocks
172     // for the input file(s). To get the right num of shards we need to
173     // calculate the best input split size.
174 
175     FileSystem fs = FileSystem.get(input.toUri(), jobConf);
176     FileStatus[] status = fs.globStatus(input);
177     long size = 0;
178     for (FileStatus fileStatus : status) {
179       size += fileStatus.getLen();
180     }
181     long optimalSplisize = size / numOfShards;
182     jobConf.set("mapred.min.split.size", "" + optimalSplisize);
183 
184     // give more mem to lucene tasks.
185     jobConf.set("mapred.child.java.opts", "-Xmx2G");
186     jobConf.setNumMapTasks(1);
187     jobConf.setNumReduceTasks(0);
188     JobClient.runJob(jobConf);
189   }
