17  /**
18   * A cache loader that will call a webservice, falling back into a retry mechanism should any
19   * errors occur in the web service call.  Developers are urged to consider carefully their use
20   * before adopting this approach.  This is particularly suitable for operations such as batch
21   * processing, where one knows the client will be making a large number of identical web service
22   * calls in a short space of time, and changes during that time are not of high importance.
23   * Note that {@link WebResource} is suitable as the key
24   * since it uses {@link URI} and not {@link URL} which the JVM does not try to resolve at runtime.
25   * Construction is forced through static factories to ensure that generics are handled correctly,
26   * removing the need for casting at use.
27   * The following illustrates the intended use of this class:
28   * <pre>
29   *   LoadingCache<WebResource, NameUsageMatch> cache = CacheBuilder.newBuilder()
30   *     .maximumSize(1000)
31   *     .expireAfterAccess(1, TimeUnit.MINUTES)
32   *     .build(RetryingWebserviceClient.newInstance(NameUsageMatch.class, NUM_RETRIES, RETRY_PERIOD_MSEC));
33   *   NameUsageMatch lookup = cache.get(RESOURCE.queryParams(queryParams));
34   * </pre>
35   *
36   * TODO: this is a copy from occurrence-store - move to somewhere common
37   */

130   @Override
131   public T load(WebResource resource) throws Exception {
132     for (int attempt = 1; attempt <= numberOfAttempts; attempt++) {
133       try {
134         if (type != null) {
135           return resource.get(type);
136         } else {
137           return resource.get(genericType);
138         }
139 
140       } catch (Exception e) {
141         LOGGER.error("Error looking up resource in attempt[{}] of max[{}]", new Object[] {attempt, numberOfAttempts, e});
142         if (attempt >= numberOfAttempts) {
143           throw e;
144         }
145         if (retryDelayMsec > 0) {
146           TimeUnit.MILLISECONDS.sleep(retryDelayMsec);
147         }
148       }
149     }
150     // Should not ever happen as we propagate the underlying exception above
151     throw new IllegalStateException("Retry count exhausted");
152   }
