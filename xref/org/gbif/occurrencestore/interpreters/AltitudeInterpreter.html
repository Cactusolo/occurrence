13  public class AltitudeInterpreter {

20    /**
21     * Calculate a single altitude value (in m) as representative for given min, max, and precision.
22     *
23     * @param min       minimum altitude (in m)
24     * @param max       maximum altitude (in m)
25     * @param precision min and max are precise to +/- this value (in m)
26     *
27     * @return a single representative altitude (in m), or null if min and max are null
28     */
29    public static Integer interpretAltitude(String min, String max, String precision) {
30      if (min == null && max == null) return null;
31  
32      Long altitude = null;
33      ParseResult<LongPrecisionStatus> result = GeospatialParseUtils.parseAltitude(min, max, precision);
34      if (result.isSuccessful() && result.getPayload().getValue() != null) {
35        altitude = result.getPayload().getValue();
36      }
37  
38      // long for altitude doesn't make sense - max int metres is 2 million km
39      Integer intAlt = null;
40      if (altitude != null) {
41        if (altitude > Integer.MAX_VALUE) {
42          LOG.warn(
43            "Bad altitude value of [" + altitude + "] - reducing to max integer value of [" + Integer.MAX_VALUE + "]");
44          intAlt = Integer.MAX_VALUE;
45        } else {
46          intAlt = altitude.intValue();
47        }
48      }
49  
50      return intAlt;
51    }
