78    public static NubLookupInterpretationResult nubLookup(String kingdom, String phylum, String clazz, String order,
79      String family, String genus, String scientificName, String author) {
80  
81      if (kingdom == null && phylum == null && clazz == null && order == null && family == null && genus == null
82          && scientificName == null && author == null) {
83        return new NubLookupInterpretationResult();
84      }
85  
86      NubLookupInterpretationResult result = null;
87      MultivaluedMap<String, String> queryParams = new MultivaluedMapImpl();
88      queryParams.add("kingdom", kingdom);
89      queryParams.add("phylum", phylum);
90      queryParams.add("class", clazz);
91      queryParams.add("order", order);
92      queryParams.add("family", family);
93      queryParams.add("genus", genus);
94      queryParams.add("name", scientificName);
95      /** TODO: include author in query */
96  
97      if (scientificName != null) {
98        for (int i = 0; i < NUM_RETRIES; i++) {
99          LOG.debug("Attempt [{}] to lookup sci name [{}]", i, scientificName);
100         try {
101           NameUsageMatch lookup = CACHE.get(RESOURCE.queryParams(queryParams));
102           if (lookup != null) {
103             if (lookup.getMatchType() == NameUsageMatch.MatchType.NONE) {
104               LOG.info("Nub lookup for [{}] returned no match. Lookup note: [{}]", scientificName, lookup.getNote());
105             } else {
106               if (lookup.getMatchType() != NameUsageMatch.MatchType.EXACT) {
107                 LOG.debug("Nub lookup for [{}] was not exact. Match type [{}] with note: [{}]", scientificName,
108                   lookup.getMatchType(), lookup.getNote());
109               }
110               result =
111                 new NubLookupInterpretationResult(lookup.getUsageKey(), lookup.getKingdomKey(), lookup.getPhylumKey(),
112                   lookup.getClassKey(), lookup.getOrderKey(), lookup.getFamilyKey(), lookup.getGenusKey(),
113                   lookup.getSpeciesKey(), lookup.getKingdom(), lookup.getPhylum(), lookup.getClazz(), lookup.getOrder(),
114                   lookup.getFamily(), lookup.getGenus(), lookup.getSpecies(), lookup.getScientificName(),
115                   lookup.getConfidence());
116             }
117           }
118 
119           break; // from retry loop
120         } catch (ExecutionException e) {
121           // Log the error
122           LOG.error("Failed WS call with: {}", recreateQueryString(queryParams));
123 
124           // have we exhausted our attempts?
125           if (i >= NUM_RETRIES) {
126             throw new RuntimeException(e);
127           }
128 
129           try {
130             Thread.sleep(RETRY_PERIOD_MSEC);
131           } catch (InterruptedException e1) {
132           }
133         } catch (UniformInterfaceException e) {
134           LOG.info("Got unexpected result for scientific name '{}', Response: {}", scientificName, e.getResponse());
135           // have we exhausted our attempts?
136           if (i >= NUM_RETRIES) {
137             throw e;
138           }
139 
140           try {
141             Thread.sleep(RETRY_PERIOD_MSEC);
142           } catch (InterruptedException e1) {
143           }
144         }
145       } // retry loop
146     } // scientific name exists
147 
148     return result;
149   }

151   private static String recreateQueryString(MultivaluedMap<String, String> queryParams) {
152     StringBuilder sb = new StringBuilder("?");
153     for (Map.Entry<String, List<String>> en : queryParams.entrySet()) {
154       sb.append(en.getKey()).append('=');
155       for (String s : en.getValue()) {
156         sb.append(s);
157       }
158       sb.append('&');
159     }
160 
161     return sb.toString();
162   }
