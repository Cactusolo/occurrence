12  /**
13   * A simple class to determine whether triplets and/or occurrenceIds are valid as identifiers for the given dataset.
14   */
15  public class IdentifierStrategy {
16  
17    private final boolean tripletsValid;
18    private final boolean occurrenceIdsValid;
19  
20    public IdentifierStrategy(OccurrenceSchemaType schemaType, @Nullable DwcaValidationReport validationReport) {
21      checkNotNull(schemaType, "schemaType can't be null");
22      if (schemaType == OccurrenceSchemaType.DWCA) {
23        checkNotNull(validationReport, "validationReport can't be null if schema is DWCA");
24      }
25      tripletsValid = tripletsValid(schemaType, validationReport);
26      occurrenceIdsValid = occurrenceIdsValid(schemaType, validationReport);
27    }
28  
29    /**
30     * For XML datasets triplets are always valid. For DwC-A datasets triplets are valid if there are more than 0 unique
31     * triplets in the dataset, and exactly 0 triplets referenced by more than one record.
32     */
33    private static boolean tripletsValid(OccurrenceSchemaType schemaType, DwcaValidationReport validationReport) {
34      boolean valid = true;
35      if (schemaType == OccurrenceSchemaType.DWCA) {
36        valid = validationReport.getUniqueTriplets() > 0
37                && validationReport.getCheckedRecords() - validationReport.getRecordsWithInvalidTriplets()
38                   == validationReport.getUniqueTriplets();
39      }
40  
41      return valid;
42    }
43  
44    /**
45     * For XML datasets occurrenceIds are ignored. For DwC-A datasets occurrenceIds are valid if there are more than 0
46     * unique occurrenceIds in the dataset, and exactly 0 occurrenceIds referenced by more than one record.
47     */
48    private static boolean occurrenceIdsValid(OccurrenceSchemaType schemaType, DwcaValidationReport validationReport) {
49      boolean valid = false;
50      if (schemaType == OccurrenceSchemaType.DWCA) {
51        valid = validationReport.getUniqueOccurrenceIds() > 0
52                && validationReport.getCheckedRecords() - validationReport.getRecordsMissingOccurrenceId()
53                   == validationReport.getUniqueOccurrenceIds();
54      }
55  
56      return valid;
57    }
58  
59    public boolean isTripletsValid() {
60      return tripletsValid;
61    }
62  
63    public boolean isOccurrenceIdsValid() {
64      return occurrenceIdsValid;
65    }
66  
67    @Override
68    public int hashCode() {
69      return Objects.hashCode(tripletsValid, occurrenceIdsValid);
70    }
71  
72    @Override
73    public boolean equals(Object obj) {
74      if (this == obj) {
75        return true;
76      }
77      if (obj == null || getClass() != obj.getClass()) {
78        return false;
79      }
80      final IdentifierStrategy other = (IdentifierStrategy) obj;
81      return Objects.equal(this.tripletsValid, other.tripletsValid) && Objects
82        .equal(this.occurrenceIdsValid, other.occurrenceIdsValid);
83    }
