64    /**
65     * Builds and persists an Occurrence record by interpreting the fields of the VerbatimOccurrence identified by the
66     * passed in occurrenceKey. Note that UNCHANGED occurrences are ignored.
67     *
68     * @param occurrenceKey the key of the VerbatimOccurrence that will be fetched from HBase and interpreted
69     * @param status        whether this is a NEW, UPDATED, or UNCHANGED occurrence
70     * @param fromCrawl     true if this method is called as part of a crawl
71     * @param attemptId     the crawl attempt id, only used for passing along in logs and subsequent messages.
72     * @param datasetKey    the dataset that this occurrence belongs to (must not be null if fromCrawl is true)
73     */
74    public void buildInterpreted(int occurrenceKey, OccurrencePersistenceStatus status, boolean fromCrawl,
75      @Nullable Integer attemptId, @Nullable UUID datasetKey) {
76      checkArgument(occurrenceKey > 0, "occurrenceKey must be greater than 0");
77      checkNotNull(status, "status can't be null");
78      if (fromCrawl) {
79        checkNotNull(datasetKey, "datasetKey can't be null if fromCrawl is true");
80        checkArgument(attemptId != null && attemptId > 0, "attemptId must be greater than 0 if fromCrawl is true");
81      }
82  
83      if (status == OccurrencePersistenceStatus.UNCHANGED) {
84        LOG.debug("Ignoring verbatim of status UNCHANGED.");
85        return;
86      }
87  
88      int localAttemptId;
89      if (fromCrawl) {
90        localAttemptId = attemptId;
91      } else {
92        Fragment fragment = fragmentPersister.get(occurrenceKey);
93        if (fragment == null) {
94          LOG.warn(
95            "Could not find fragment with key [{}] when looking up attemptId for non-crawl interpretation - skipping.",
96            occurrenceKey);
97          return;
98        }
99        localAttemptId = fragment.getCrawlId();
100     }
101 
102     VerbatimOccurrence verbatim = verbatimPersister.get(occurrenceKey);
103     if (verbatim == null) {
104       logError("Could not find", occurrenceKey, datasetKey, fromCrawl);
105       return;
106     }
107 
108     InterpretationResult interpretationResult;
109     final TimerContext interpContext = interpTimer.time();
110     try {
111       interpretationResult = verbatimInterpreter.interpret(verbatim, status, fromCrawl);
112     } finally {
113       interpContext.stop();
114     }
115 
116     if (interpretationResult.getUpdated() == null) {
117       logError("Could not interpret", occurrenceKey, datasetKey, fromCrawl);
118       return;
119     }
120 
121     Occurrence interpreted = interpretationResult.getUpdated();
122 
123     LOG.debug("sending messages");
124     OccurrenceMutatedMessage interpMsg;
125     // can only be NEW or UPDATED
126     if (status == OccurrencePersistenceStatus.NEW) {
127       interpMsg = OccurrenceMutatedMessage.buildNewMessage(interpreted.getDatasetKey(), interpreted, localAttemptId);
128     } else {
129       interpMsg = OccurrenceMutatedMessage
130         .buildUpdateMessage(interpreted.getDatasetKey(), interpretationResult.getOriginal(), interpreted,
131           localAttemptId);
132     }
133 
134     final TimerContext context = msgTimer.time();
135     try {
136       messagePublisher.send(interpMsg);
137     } catch (IOException e) {
138       LOG.warn("Could not send OccurrencePersistedMessage for successful [{}]", status.toString(), e);
139     } finally {
140       context.stop();
141     }
142 
143     interpProcessed.mark();
144   }

146   private void logError(String message, int occurrenceKey, UUID datasetKey, boolean fromCrawl) {
147     // TODO: send msg?
148     LOG.warn("{} verbatim occurrence with key [{}] - skipping.", message, occurrenceKey);
149     if (fromCrawl) {
150       LOG.debug("Updating zookeeper for InterpretedOccurrencePersistedError");
151       zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.INTERPRETED_OCCURRENCE_PERSISTED_ERROR);
152     }
