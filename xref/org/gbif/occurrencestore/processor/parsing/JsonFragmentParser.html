38    /**
39     * Expects a fragment encoded in JSON (so, DwCA) and parses it into a VerbatimOccurrence.
40     *
41     * @param fragment of type JSON from a DwCA
42     *
43     * @return a VerbatimOccurrence parsed from the fragment's JSON
44     */
45    @Nullable
46    public static VerbatimOccurrence parseRecord(Fragment fragment) {
47      checkNotNull(fragment, "fragment can't be null");
48      checkArgument(fragment.getFragmentType() == Fragment.FragmentType.JSON, "fragment must be of type JSON");
49  
50      VerbatimOccurrence verbatim = null;
51  
52      // jackson untyped bind of data to object
53      ObjectMapper mapper = new ObjectMapper();
54      try {
55        Map<String, Object> jsonMap = mapper.readValue(new ByteArrayInputStream(fragment.getData()), Map.class);
56  
57        String verbatimLatitude = (String) jsonMap.get(DwcTerm.verbatimLatitude.simpleName());
58        String verbatimLongitude = (String) jsonMap.get(DwcTerm.verbatimLongitude.simpleName());
59        String decimalLatitude = (String) jsonMap.get(DwcTerm.decimalLatitude.simpleName());
60        String decimalLongitude = (String) jsonMap.get(DwcTerm.decimalLongitude.simpleName());
61  
62        String continent = (String) jsonMap.get(DwcTerm.continent.simpleName());
63        String waterBody = (String) jsonMap.get(DwcTerm.waterBody.simpleName());
64  
65        String country = (String) jsonMap.get(DwcTerm.country.simpleName());
66        String countryCode = (String) jsonMap.get(DwcTerm.countryCode.simpleName());
67  
68        verbatim = VerbatimOccurrence.builder().key(fragment.getKey()).datasetKey(fragment.getDatasetKey())
69          .unitQualifier(fragment.getUnitQualifier())
70          .institutionCode((String) jsonMap.get(DwcTerm.institutionCode.simpleName()))
71          .collectionCode((String) jsonMap.get(DwcTerm.collectionCode.simpleName()))
72          .catalogNumber((String) jsonMap.get(DwcTerm.catalogNumber.simpleName()))
73          .scientificName((String) jsonMap.get(DwcTerm.scientificName.simpleName()))
74          .author((String) jsonMap.get(DwcTerm.scientificNameAuthorship.simpleName()))
75          .rank((String) jsonMap.get(DwcTerm.taxonRank.simpleName()))
76          .kingdom((String) jsonMap.get(DwcTerm.kingdom.simpleName()))
77          .phylum((String) jsonMap.get(DwcTerm.phylum.simpleName()))
78          .klass((String) jsonMap.get(DwcTerm.class_.simpleName()))
79          .order((String) jsonMap.get(DwcTerm.order.simpleName()))
80          .family((String) jsonMap.get(DwcTerm.family.simpleName()))
81          .genus((String) jsonMap.get(DwcTerm.genus.simpleName()))
82          .species((String) jsonMap.get(DwcTerm.specificEpithet.simpleName()))
83          .subspecies((String) jsonMap.get(DwcTerm.infraspecificEpithet.simpleName()))
84            // prefer the decimal versions if both exist
85          .latitude(decimalLatitude == null ? verbatimLatitude : decimalLatitude)
86          .longitude(decimalLongitude == null ? verbatimLongitude : decimalLongitude)
87          .latLongPrecision((String) jsonMap.get(DwcTerm.coordinatePrecision.simpleName()))
88          .maxAltitude((String) jsonMap.get(DwcTerm.maximumElevationInMeters.simpleName()))
89          .minAltitude((String) jsonMap.get(DwcTerm.minimumElevationInMeters.simpleName()))
90            // no altitude precision
91          .maxDepth((String) jsonMap.get(DwcTerm.maximumDepthInMeters.simpleName()))
92          .minDepth((String) jsonMap.get(DwcTerm.minimumDepthInMeters.simpleName()))
93            // no depth precision
94          .continentOrOcean(continent == null ? waterBody : continent)
95          .country(countryCode == null ? country : countryCode)
96          .stateOrProvince((String) jsonMap.get(DwcTerm.stateProvince.simpleName()))
97          .county((String) jsonMap.get(DwcTerm.county.simpleName()))
98          .collectorName((String) jsonMap.get(DwcTerm.recordedBy.simpleName()))
99          .locality((String) jsonMap.get(DwcTerm.locality.simpleName()))
100         .year((String) jsonMap.get(DwcTerm.year.simpleName())).month((String) jsonMap.get(DwcTerm.month.simpleName()))
101         .day((String) jsonMap.get(DwcTerm.day.simpleName()))
102         .occurrenceDate((String) jsonMap.get(DwcTerm.eventDate.simpleName()))
103         .basisOfRecord((String) jsonMap.get(DwcTerm.basisOfRecord.simpleName()))
104         .identifierName((String) jsonMap.get(DwcTerm.identifiedBy.simpleName()))
105           // no day-, month- or year-identified in DwcTerm
106         .dateIdentified((String) jsonMap.get(DwcTerm.dateIdentified.simpleName())).modified(System.currentTimeMillis())
107         .build();
108     } catch (IOException e) {
109       LOG.warn("Unable to parse JSON data, returning null VerbatimOccurrence", e);
110     }
111 
112     return verbatim;
113   }

115   /**
116    * Extracts the unique identifiers from this json snippet.
117    *
118    * @param datasetKey      the UUID of the dataset
119    * @param jsonData        the raw byte array of the snippet's json
120    * @param useOccurrenceId @return a set of
121    */
122   @Nullable
123   public static IdentifierExtractionResult extractIdentifiers(UUID datasetKey, byte[] jsonData, boolean useTriplet,
124     boolean useOccurrenceId) {
125     checkNotNull(datasetKey, "datasetKey can't be null");
126     checkNotNull(jsonData, "jsonData can't be null");
127 
128     IdentifierExtractionResult result = null;
129 
130     // jackson untyped bind of data to object
131     ObjectMapper mapper = new ObjectMapper();
132     try {
133       Map<String, Object> jsonMap = mapper.readValue(new ByteArrayInputStream(jsonData), Map.class);
134       Set<UniqueIdentifier> uniqueIds = Sets.newHashSet();
135 
136       // look for triplet
137       if (useTriplet) {
138         String ic = (String) jsonMap.get(DwcTerm.institutionCode.simpleName());
139         String cc = (String) jsonMap.get(DwcTerm.collectionCode.simpleName());
140         String cn = (String) jsonMap.get(DwcTerm.catalogNumber.simpleName());
141 
142         HolyTriplet holyTriplet = null;
143         try {
144           holyTriplet = new HolyTriplet(datasetKey, ic, cc, cn, null);
145         } catch (IllegalArgumentException e) {
146           // some of the triplet was null or empty, so it's not valid - that's highly suspicious, but could be ok...
147           LOG.info("No holy triplet for json snippet in dataset [{}] and schema [{}], got error [{}]",
148             new String[] {datasetKey.toString(), OccurrenceSchemaType.DWCA.toString(), e.getMessage()});
149         }
150         if (holyTriplet != null) {
151           uniqueIds.add(holyTriplet);
152         }
153       }
154 
155       // look for occurrence id
156       if (useOccurrenceId) {
157         String occId = (String) jsonMap.get(DwcTerm.occurrenceID.simpleName());
158         if (occId != null) {
159           uniqueIds.add(new PublisherProvidedUniqueIdentifier(datasetKey, occId));
160         }
161       }
162 
163       if (!uniqueIds.isEmpty()) {
164         result = new IdentifierExtractionResult(uniqueIds, null);
165       }
166     } catch (IOException e) {
167       LOG.warn("Unable to parse JSON data, returning null IdentifierExtractionResult", e);
168     }
169 
170     return result;
171   }
