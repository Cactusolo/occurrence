15  /**
16   * A thin wrapper around the XmlFragmentParser from the occurrence-parser project, and the local JsonFragmentParser, in
17   * order to produce VerbatimOccurrence objects from Fragments.
18   */
19  public class FragmentParser {
20  
21    private static final Timer xmlParsingTimer =
22      Metrics.newTimer(FragmentParser.class, "verb xml parse time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
23    private static final Timer jsonParsingTimer =
24      Metrics.newTimer(FragmentParser.class, "verb json parse time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);
25  
26    // should not be instantiated
27    private FragmentParser() {
28    }

30    /**
31     * Parse the given fragment into a VerbatimOccurrence object.
32     * TODO: there is a lot of hacking here that should be refactored out with general rework of parsing project
33     *
34     * @param fragment containing parsing to be parsed
35     *
36     * @return a VerbatimOccurrence or null if the fragment could not be parsed
37     */
38    @Nullable
39    public static VerbatimOccurrence parse(Fragment fragment) {
40      VerbatimOccurrence verbatim = null;
41  
42      switch (fragment.getFragmentType()) {
43        case XML:
44          final TimerContext xmlContext = xmlParsingTimer.time();
45          try {
46            RawOccurrenceRecord ror =
47              XmlFragmentParser.parseRecord(fragment.getData(), fragment.getXmlSchema(), fragment.getUnitQualifier());
48            verbatim = buildVerbatim(ror, fragment);
49          } finally {
50            xmlContext.stop();
51          }
52          break;
53        case JSON:
54          final TimerContext jsonContext = jsonParsingTimer.time();
55          try {
56            verbatim = JsonFragmentParser.parseRecord(fragment);
57          } finally {
58            jsonContext.stop();
59          }
60          break;
61      }
62  
63      if (verbatim != null) {
64        verbatim.setProtocol(fragment.getProtocol());
65      }
66  
67      return verbatim;
68    }
69

78        .dateIdentified(ror.getDateIdentified()).depthPrecision(ror.getDepthPrecision()).family(ror.getFamily())
79        .genus(ror.getGenus()).identifierName(ror.getIdentifierName()).institutionCode(ror.getInstitutionCode())
80        .kingdom(ror.getKingdom()).klass(ror.getKlass()).latitude(ror.getLatitude()).longitude(ror.getLongitude())
