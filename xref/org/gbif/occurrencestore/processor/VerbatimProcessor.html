31   * Takes occurrence Fragments and parses them to produce and persist VerbatimOccurrence objects.
32   */

43      Metrics.newTimer(VerbatimProcessor.class, "msg send time", TimeUnit.MILLISECONDS, TimeUnit.SECONDS);

48    public VerbatimProcessor(FragmentPersistenceService fragmentPersister,

51      this.fragmentPersister = checkNotNull(fragmentPersister, "fragmentPersister can't be null");
52      this.verbatimPersister = checkNotNull(verbatimPersister, "verbatimPersister can't be null");
53      this.messagePublisher = checkNotNull(messagePublisher, "messagePublisher can't be null");

57    /**
58     * Builds and persists a VerbatimOccurrence object by parsing an existing Fragment with the given occurrenceKey.
59     * Updated zookeeper with success/error counts and sends a VerbatimPersistedMessage when successfully completed. Note
60     * that UNCHANGED Fragments are ignored.
61     *
62     * @param occurrenceKey the key of the existing Fragment to be parsed
63     * @param status        whether the Fragment is NEW, UPDATED, or UNCHANGED
64     * @param fromCrawl     true if this method is called as part of a crawl
65     * @param attemptId     the crawl attempt id, only used for passing along in logs and subsequent messages.
66     * @param datasetKey    the dataset that this occurrence belongs to (must not be null if fromCrawl is true)
67     */
68    public void buildVerbatim(int occurrenceKey, OccurrencePersistenceStatus status, boolean fromCrawl,
69      @Nullable Integer attemptId, @Nullable UUID datasetKey) {
70      checkArgument(occurrenceKey > 0, "occurrenceKey must be greater than 0");
71      checkNotNull(status, "status can't be null");
72      if (fromCrawl) {
73        checkNotNull(datasetKey, "datasetKey can't be null if fromCrawl is true");
74        checkArgument(attemptId != null && attemptId > 0, "attemptId must be greater than 0 if fromCrawl is true");
75      }
76  
77      if (status == OccurrencePersistenceStatus.UNCHANGED) {
78        LOG.debug("Ignoring fragment of status UNCHANGED.");
79        return;
80      }
81  
82      Fragment fragment = fragmentPersister.get(occurrenceKey);
83  
84      if (fragment == null) {
85        logError("Could not find", occurrenceKey, datasetKey, fromCrawl);
86        return;
87      }
88  
89      int localAttemptId = fromCrawl ? attemptId : fragment.getCrawlId();
90      LOG.debug("Fragment for key [{}] and UUID [{}] crawl [{}] is [{}]", occurrenceKey, fragment.getDatasetKey(),
91        localAttemptId, status);
92  
93      VerbatimOccurrence verbatim = FragmentParser.parse(fragment);
94      if (verbatim == null) {
95        // parsing has failed, skip this fragment
96        logError("Could not parse", occurrenceKey, datasetKey, fromCrawl);
97        return;
98      }
99  
100     verbatimPersister.update(verbatim);
101 
102     if (fromCrawl) {
103       LOG.debug("Updating zookeeper for VerbatimOccurrencePersistedSuccess");
104       zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_SUCCESS);
105     }
106 
107     VerbatimPersistedMessage verbMsg =
108       new VerbatimPersistedMessage(verbatim.getDatasetKey(), localAttemptId, status, verbatim.getKey());
109     final TimerContext msgContext = msgTimer.time();
110     try {
111       messagePublisher.send(verbMsg);
112     } catch (IOException e) {
113       LOG.warn("Could not send VerbatimPersistedMessage for successful [{}]", status, e);
114     } finally {
115       msgContext.stop();
116     }
117 
118     verbProcessed.mark();
119   }
120 
121   private void logError(String message, int occurrenceKey, UUID datasetKey, boolean fromCrawl) {
122     // TODO: send msg?
123     LOG.warn(message + " fragment with key [{}] - skipping.", occurrenceKey);
124     if (fromCrawl) {
125       LOG.debug("Updating zookeeper for VerbatimOccurrencePersistedError");
126       zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_ERROR);
127     }
128   }
