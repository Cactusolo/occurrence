88    /**
89     * Takes the given response/dwca fragment and parses it into Fragment object(s) and then persists it/them.
90     * Responsible for identifying duplicates of existing Fragments and assigning ids to new Fragments. The only case
91     * where multiple Fragments are returned from this message is ABCD 2.06 snippets that are differentiated with
92     * UnitQualifiers.
93     *
94     * @param datasetKey       the dataset to which this fragment belongs
95     * @param data             a snippet of xml or json from a crawler response, representing a single "record"
96     * @param schema           the crawler response type (e.g. ABCD_2_06, DWC_1_4) or DWCA for DwC-A as JSON
97     * @param protocol         the endpoint type of this occurrence's dataset (e.g. DIGIR, BIOCASE, DWC_ARCHIVE)
98     * @param crawlId          the crawl attempt that produced the snippet
99     * @param validationReport the DwcaValidationReport if this fragment came from a DwC-A, null otherwise
100    */
101   public void buildFragments(UUID datasetKey, byte[] data, OccurrenceSchemaType schema, EndpointType protocol,
102     Integer crawlId, @Nullable DwcaValidationReport validationReport) {
103     checkNotNull(datasetKey, "datasetKey can't be null");
104     checkNotNull(data, "data can't be null");
105     checkNotNull(crawlId, "crawlId can't be null");
106     checkNotNull(schema, "xmlSchema can't be null");
107     checkNotNull(protocol, "protocol can't be null");
108     if (schema == OccurrenceSchemaType.DWCA) {
109       checkNotNull(validationReport, "validationReport can't be null if schema is DWCA");
110     }
111 
112     MDC.put("datasetUuid", datasetKey.toString());
113     MDC.put("attempt", crawlId.toString());
114 
115     boolean isXml = schema != OccurrenceSchemaType.DWCA;
116     IdentifierStrategy idStrategy = new IdentifierStrategy(schema, validationReport);
117     LOG.debug("For dataset [{}] of type [{}] useTriplet is [{}] and useOccurrenceId is [{}]",
118       new Object[] {datasetKey, schema, idStrategy.isTripletsValid(), idStrategy.isOccurrenceIdsValid()});
119 
120     // update zookeeper - fragment received but not yet processed
121     zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.FRAGMENT_RECEIVED);
122 
123     Set<IdentifierExtractionResult> extractionResults;
124     if (isXml) {
125       final TimerContext context = xmlParsingTimer.time();
126       try {
127         extractionResults = XmlFragmentParser.extractIdentifiers(datasetKey, data, schema, idStrategy.isTripletsValid(),
128           idStrategy.isOccurrenceIdsValid());
129       } finally {
130         context.stop();
131       }
132     } else {
133       final TimerContext context = jsonParsingTimer.time();
134       try {
135         extractionResults = Sets.newHashSet();
136         IdentifierExtractionResult result = JsonFragmentParser
137           .extractIdentifiers(datasetKey, data, idStrategy.isTripletsValid(), idStrategy.isOccurrenceIdsValid());
138         if (result != null) {
139           extractionResults.add(result);
140         }
141       } finally {
142         context.stop();
143       }
144     }
145 
146     if (extractionResults.isEmpty()) {
147       LOG.debug(
148         "Given snippet for dataset [{}] does not contain enough information to uniquely identify it - can not parse or "
149         + "persist it", datasetKey.toString());
150       LOG.debug("Raw snippet that couldn't be uniquely identified: [\n{}\n]",
151         new String(data, Charset.forName("UTF-8")));
152       zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_ERROR);
153     } else {
154       for (IdentifierExtractionResult extractionResult : extractionResults) {
155         processSingleFragment(extractionResult.getUniqueIdentifiers(), isXml, data, schema, protocol, crawlId,
156           extractionResult.getUnitQualifier());
157       }
158     }
159 
160     // update zookeeper with a single increment for this snippet (not for each Fragment)
161     LOG.debug("Updating zookeeper for FragmentProcessed");
162     zookeeperConnector.addCounter(datasetKey, ZookeeperConnector.CounterName.FRAGMENT_PROCESSED);
163   }

165   /**
166    * Process a single record into a single Fragment and persist it, generating a new key for it if necessary.
167    */
168   private void processSingleFragment(Set<UniqueIdentifier> uniqueIds, boolean isXml, byte[] data,
169     OccurrenceSchemaType schema, EndpointType protocol, Integer crawlId, String unitQualifier) {
170 
171     UUID datasetKey = uniqueIds.iterator().next().getDatasetKey();
172 
173     // check for an existing key for these uniqueIds
174     KeyLookupResult keyResult = null;
175     final TimerContext fetchContext = findTimer.time();
176     try {
177       keyResult = occurrenceKeyPersister.findKey(uniqueIds);
178     } catch (ServiceUnavailableException e) {
179       failForServiceUnavailable(datasetKey, e);
180       return;
181     } finally {
182       fetchContext.stop();
183     }
184 
185     Fragment fragment;
186     OccurrencePersistenceStatus status;
187     if (keyResult == null) {
188       // this is a new record, prepare to insert a new Fragment
189       fragment = new Fragment(datasetKey);
190       status = OccurrencePersistenceStatus.NEW;
191     } else {
192       // this is an existing record - fetch fragment from hbase and see if we need to do an update
193       int key = keyResult.getKey();
194       int attempts = 0;
195       do {
196         attempts++;
197         LOG.debug("Attempt [{}] to fetch fragment for key [{}]", attempts, key);
198         try {
199           fragment = fragmentPersister.get(key);
200         } catch (ValidationException v) {
201           LOG.warn("The fragment for key [{}] is invalid. Ignoring this update.", v);
202           updateZookeeper(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_ERROR);
203           return;
204         } catch (ServiceUnavailableException e) {
205           failForServiceUnavailable(datasetKey, e);
206           return;
207         }
208         if (fragment == null) {
209           /* this means either
210               1) the key we're trying was only just assigned and the owner hasn't written their fragment yet, or
211               2) our index is out of sync - it has an index entry pointing to nothing, so delete the index
212                 entry(ies) and proceed as if this is an insert
213              So we retry a few times to get clear of 1), and then act on 2).
214           */
215           LOG.debug("Fragment for key [{}] was null, sleeping and trying again", key);
216           try {
217             Thread.sleep(NULL_FRAG_RETRY_WAIT);
218           } catch (InterruptedException e) {
219             LOG.info("Woken up from sleep while waiting to retry fragment get", e);
220           }
221         }
222       } while (fragment == null && attempts <= MAX_NULL_FRAG_RETRIES);
223 
224       if (fragment == null) {
225         // we're in case 2) from above
226         LOG.info("Could not retrieve Fragment with key [{}] even though it exists in lookup table - deleting lookup "
227                  + "and inserting Fragment as NEW.", key);
228         try {
229           occurrenceKeyPersister.deleteKeyByUniqueIdentifiers(uniqueIds);
230         } catch (ServiceUnavailableException e) {
231           failForServiceUnavailable(datasetKey, e);
232           return;
233         }
234         fragment = new Fragment(datasetKey);
235         status = OccurrencePersistenceStatus.NEW;
236       } else {
237         status = OccurrencePersistenceStatus.UPDATED;
238       }
239     }
240 
241     // check hashes to see if this is an existing and unchanged record
242     byte[] newHash = DigestUtils.md5(data);
243     if (status == OccurrencePersistenceStatus.UPDATED && Arrays.equals(newHash, fragment.getDataHash())) {
244       // the new record and existing record don't differ
245       status = OccurrencePersistenceStatus.UNCHANGED;
246     } else {
247       // this is an UPDATE or NEW, but not UNCHANGED
248       fragment.setData(data);
249       fragment.setXmlSchema(schema);
250       fragment.setProtocol(protocol);
251       fragment.setDataHash(newHash);
252       fragment.setHarvestedDate(new Date());
253       fragment.setCrawlId(crawlId);
254       fragment.setUnitQualifier(unitQualifier);
255       if (isXml) {
256         fragment.setFragmentType(Fragment.FragmentType.XML);
257       } else {
258         fragment.setFragmentType(Fragment.FragmentType.JSON);
259       }
260     }
261 
262     // update these in every case (including UNCHANGED)
263     fragment.setHarvestedDate(new Date());
264     fragment.setCrawlId(crawlId);
265 
266     //    if (LOG.isDebugEnabled()) {
267     //      if (status != OccurrencePersistenceStatus.NEW) {
268     StringBuilder uniques = new StringBuilder(64);
269 
270     for (UniqueIdentifier uniqueIdentifier : uniqueIds) {
271       if (uniques.length() > 0) {
272         uniques.append(" ||| ");
273       }
274       uniques.append(uniqueIdentifier.getUniqueString());
275     }
276     LOG.debug("Persisting fragment of status [{}] for key [{}] uniqueIds [{}]", status.toString(),
277       fragment.getKey() == null ? "null" : fragment.getKey().toString(), uniques.toString());
278     //      }
279     //    }
280 
281     final TimerContext context = persistenceTimer.time();
282     try {
283       if (status == OccurrencePersistenceStatus.NEW) {
284         FragmentCreationResult creationResult = fragmentPersister.insert(fragment, uniqueIds);
285         if (!creationResult.isKeyCreated()) {
286           // we lost a race to generate the key for these uniqueIds, and now this is an update
287           status = OccurrencePersistenceStatus.UPDATED;
288           LOG.info(
289             "Fragment creation did not generate new key - lost race and now using existing [{}] as status [UPDATE]",
290             fragment.getKey());
291         }
292       } else {
293         fragmentPersister.update(fragment);
294       }
295     } catch (ServiceUnavailableException e) {
296       failForServiceUnavailable(datasetKey, e);
297       return;
298     } finally {
299       context.stop();
300     }
301 
302     ZookeeperConnector.CounterName counterToUpdate = null;
303     switch (status) {
304       case UNCHANGED:
305         counterToUpdate = ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_UNCHANGED;
306         break;
307       case NEW:
308         counterToUpdate = ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_NEW;
309         break;
310       case UPDATED:
311         counterToUpdate = ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_UPDATED;
312         break;
313     }
314     updateZookeeper(fragment.getDatasetKey(), counterToUpdate);
315 
316     Message fragMsg = new FragmentPersistedMessage(fragment.getDatasetKey(), crawlId, status, fragment.getKey());
317     final TimerContext msgContext = msgTimer.time();
318     try {
319       LOG.debug("Sending FragmentPersistedMessage for key [{}]", fragment.getKey());
320       messagePublisher.send(fragMsg);
321       msgsSent.mark();
322     } catch (IOException e) {
323       LOG.warn("Could not send FragmentPersistedMessage for successful persist of status [{}]", status, e);
324     } finally {
325       msgContext.stop();
326     }
327 
328     fragmentsProcessed.mark();
329   }

331   private void failForServiceUnavailable(UUID datasetKey, Throwable e) {
332     LOG.warn("Caught service unavailable from HBase: this fragment for dataset [{}] won't be processed", datasetKey, e);
333     updateZookeeper(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_ERROR);
334   }
335 
336   private void updateZookeeper(UUID datasetKey, ZookeeperConnector.CounterName counterName) {
337     LOG.debug("Updating zookeeper for RawOccurrencePersisted");
338     zookeeperConnector.addCounter(datasetKey, counterName);
339   }
