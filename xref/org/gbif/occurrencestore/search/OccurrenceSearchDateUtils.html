23  public class OccurrenceSearchDateUtils {

26    /**
27     * Converts the value parameter into a date range query.
28     */
29    public static String toDateQuery(String value) {
30      if (SearchTypeValidator.isRange(value)) {
31        return toRangeDateQueryFormat(value);
32      } else {
33        return toSingleDateQueryFormat(value);
34      }
35    }
36  
37    private static String toRangeDateQueryFormat(String value) {
38      final Range<Date> dateRange = parseDateRange(value);
39      if (!dateRange.hasLowerBound() && !dateRange.hasUpperBound()) {
40        return DEFAULT_FILTER_QUERY;
41      } else if (dateRange.hasLowerBound() && !dateRange.hasUpperBound()) {
42        return String.format(RANGE_FORMAT, toDateQueryFormat(dateRange.lowerEndpoint()),
43          DEFAULT_FILTER_QUERY);
44      } else if (!dateRange.hasLowerBound() && dateRange.hasUpperBound()) {
45        return String.format(RANGE_FORMAT, DEFAULT_FILTER_QUERY,
46          toDateQueryFormat(dateRange.upperEndpoint()));
47      } else {
48        return String.format(RANGE_FORMAT, toDateQueryFormat(dateRange.lowerEndpoint()),
49          toDateQueryFormat(dateRange.upperEndpoint()));
50      }
51    }
52  
53    /**
54     * Transforms a single ISO date value into a solr query being a range query depending on the precision of the
55     * ISO date. For example a single date 1989-09 results in a solr range for the entire month.
56     */
57    private static String toSingleDateQueryFormat(String value) {
58      final IsoDateFormat occDateFormat = getFirstDateFormatMatch(value);
59      final Date lowerDate = IsoDateParsingUtils.parseDate(value);
60      final String lowerDateStr = toDateQueryFormat(lowerDate);
61      if (occDateFormat == IsoDateFormat.YEAR_MONTH) {
62        // Generated query: [yyyy-MM-01T00:00:00Z TO yyyy-'MM-LAST_DATE_OF_THE_MONTH'T00:00:00Z]
63        String upperDayeStr = toDateQueryFormat(toLastDayOfMonth(lowerDate));
64        return String.format(RANGE_FORMAT, lowerDateStr, upperDayeStr);
65      } else if (occDateFormat == IsoDateFormat.YEAR) {
66        // Generated query: [yyyy-01-01T00:00:00Z TO yyyy-'LAST_DATE_OF_THE_YEAR'T00:00:00Z]
67        String upperDayeStr = toDateQueryFormat(toLastDayOfYear(lowerDate));
68        return String.format(RANGE_FORMAT, lowerDateStr, upperDayeStr);
69      }
70      return toPhraseQuery(lowerDateStr);
71    }
