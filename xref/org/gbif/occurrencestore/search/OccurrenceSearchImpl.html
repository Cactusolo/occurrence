79    /**
80     * Builds a SearchResponse instance using the current builder state.
81     * 
82     * @return a new instance of a SearchResponse.
83     */
84    public SearchResponse<Occurrence, OccurrenceSearchParameter> buildResponse(QueryResponse queryResponse,
85      Pageable request) {
86      // Create response
87      SearchResponse<Occurrence, OccurrenceSearchParameter> response =
88        new SearchResponse<Occurrence, OccurrenceSearchParameter>(request);
89      SolrDocumentList results = queryResponse.getResults();
90      // set total count
91      response.setCount(results.getNumFound());
92      // Populates the results
93      List<Occurrence> occurrences = Lists.newArrayList();
94      for (SolrDocument doc : results) {
95        // Only field key is returned in the result
96        Integer occKey = (Integer) doc.getFieldValue(OccurrenceSolrField.KEY.getFieldName());
97        Occurrence occ = occurrenceService.get(occKey);
98        if (occ == null || occ.getKey() == null) {
99          LOG.warn("Occurrence {} not found in store, but present in solr", occKey);
100       } else {
101         occurrences.add(occ);
102       }
103     }
104     if (request.getLimit() > OccurrenceSearchRequestBuilder.MAX_PAGE_SIZE) {
105       response.setLimit(OccurrenceSearchRequestBuilder.MAX_PAGE_SIZE);
106     }
107     response.setResults(occurrences);
108     return response;
109   }

149   /**
150    * Searches a indexed terms of a field that matched against the prefix parameter.
151    * 
152    * @param prefix search term
153    * @param parameter mapped field to be searched
154    * @param limit of maximum matches
155    * @return a list of elements that matched against the prefix
156    */
157   public List<String> suggestTermByField(String prefix, OccurrenceSearchParameter parameter, Integer limit) {
158     List<String> suggestions = Lists.newArrayList();
159     try {
160       String solrField = QUERY_FIELD_MAPPING.get(parameter).getFieldName();
161       SolrQuery solrQuery = buildTermQuery(prefix, solrField, Objects.firstNonNull(limit, DEFAULT_SUGGEST_LIMIT));
162       QueryResponse queryResponse = solrServer.query(solrQuery);
163       TermsResponse termsResponse = queryResponse.getTermsResponse();
164       List<Term> terms = termsResponse.getTerms(solrField);
165       for (Term term : terms) {
166         suggestions.add(term.getTerm());
167       }
168       return suggestions;
169     } catch (SolrServerException e) {
170       LOG.error("Error executing/building the request", e);
171       throw new SearchException(e);
172     }
173   }

176   /**
177    * Tries to get the corresponding name usage keys from the scientific_name parameter values.
178    * 
179    * @return true: if the request doesn't contain any scientific_name parameter or if any scientific name was found
180    *         false: if none scientific name was found
181    */
182   private boolean replaceScientificNames(OccurrenceSearchRequest request) {
183     boolean hasValidReplaces = true;
184     if (request.getParameters().containsKey(OccurrenceSearchParameter.SCIENTIFIC_NAME)) {
185       hasValidReplaces = false;
186       for (String value : request.getParameters().get(OccurrenceSearchParameter.SCIENTIFIC_NAME)) {
187         NameUsageMatch nameUsageMatch = nameUsageMatchingService.match(value, null, null, true, false);
188         if (nameUsageMatch.getMatchType() == MatchType.EXACT) {
189           hasValidReplaces = true;
190           request.addParameter(OccurrenceSearchParameter.TAXON_KEY, nameUsageMatch.getUsageKey());
191         }
192       }
193     }
194     return hasValidReplaces;
195   }
