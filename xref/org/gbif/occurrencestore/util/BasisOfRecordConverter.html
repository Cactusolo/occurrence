9   /**
10   * Converter turning a basis of record enum into an integer and vice versa.
11   * The integer value is the one used by HBase persistence.
12   */
13  public class BasisOfRecordConverter {
14  
15    // TODO: 4 used to be germplams and is hangover from mysql days - once the dictionary parser (gbif-parsers) is fixed
16    // to return only enums and the only int handling is here, this class can hold 1:1 mapping enum:int.
17    // See: GBIFCOM-141
18    private final Map<Integer, BasisOfRecord> mapToEnum =
19      ImmutableMap.<Integer, BasisOfRecord>builder()
20        .put(0, BasisOfRecord.UNKNOWN)
21        .put(1, BasisOfRecord.OBSERVATION)
22        .put(2, BasisOfRecord.PRESERVED_SPECIMEN)
23        .put(3, BasisOfRecord.LIVING_SPECIMEN)
24        .put(4, BasisOfRecord.LIVING_SPECIMEN)
25        .put(5, BasisOfRecord.FOSSIL_SPECIMEN)
26        .put(6, BasisOfRecord.LITERATURE)
27        .build();
28  
29    private final Map<BasisOfRecord, Integer> mapFromEnum =
30      ImmutableMap.<BasisOfRecord, Integer>builder()
31        .put(BasisOfRecord.UNKNOWN, 0)
32        .put(BasisOfRecord.OBSERVATION, 1)
33        .put(BasisOfRecord.PRESERVED_SPECIMEN, 2)
34        .put(BasisOfRecord.LIVING_SPECIMEN, 3)
35        .put(BasisOfRecord.FOSSIL_SPECIMEN, 5)
36        .put(BasisOfRecord.LITERATURE, 6)
37        .put(BasisOfRecord.HUMAN_OBSERVATION, 7)
38        .put(BasisOfRecord.MACHINE_OBSERVATION, 8)
39        .build();
40  
41  
42    public BasisOfRecord toEnum(Integer key) {
43      if (key == null) return BasisOfRecord.UNKNOWN;
44      BasisOfRecord val = mapToEnum.get(key);
45      return val == null ? BasisOfRecord.UNKNOWN : val;
46    }
47  
48    public Integer fromEnum(BasisOfRecord value) {
49      if (value == null) return null;
50      if (mapFromEnum.containsKey(value)) {
51        return mapFromEnum.get(value);
52      }
53      throw new IllegalArgumentException("Enumeration value " + value.name() + " unknown");
54    }
55  }
