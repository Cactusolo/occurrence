26     * @param fieldName   the occurrence field to modify
27     * @param fieldData   the data to write
28     * @param deleteNulls if this is true, and fieldData is null, the fieldName will be added to the passed in Delete
29     * @param cf          the HBase column family for this modification
30     * @param put         the put to be added to
31     * @param del         the delete to be added to
32     */
33    public static void writeField(FieldName fieldName, @Nullable byte[] fieldData, boolean deleteNulls, byte[] cf,
34      Put put, @Nullable Delete del) {
35      checkNotNull(fieldName, "fieldName may not be null");
36      checkNotNull(cf, "cf may not be null");
37      checkNotNull(put, "put may not be null");
38      if (deleteNulls) checkNotNull(del, "del may not be null if deleteNulls is true");
39  
40      if (fieldData != null) {
41        put.add(cf, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnName()), fieldData);
42      } else if (deleteNulls) {
43        del.deleteColumn(cf, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(fieldName).getColumnName()));
44      }
