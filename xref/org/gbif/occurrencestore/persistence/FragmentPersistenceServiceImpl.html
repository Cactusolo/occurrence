54    /**
55     * Simple HBase get for given key. Note that OccurrenceBuilder could throw ValidationException if the underlying
56     * fragment is missing required fields.
57     *
58     * @param key the key of the fragment (Integer rather than int for use in methods/classes using generic types)
59     *
60     * @return the fragment or null if not found
61     */
62    @Override
63    public Fragment get(@Nullable Integer key) {
64      if (key == null) {
65        return null;
66      }
67  
68      Fragment frag = null;
69      HTableInterface table = null;
70      try {
71        table = tablePool.getTable(occurrenceTableName);
72        Get get = new Get(Bytes.toBytes(key));
73        Result result = table.get(get);
74        if (result == null || result.isEmpty()) {
75          LOG.debug("Couldn't find fragment for occurrence key [{}], returning null", key);
76        } else {
77          frag = OccurrenceBuilder.buildFragment(result);
78        }
79      } catch (IOException e) {
80        throw new ServiceUnavailableException("Could not read from HBase", e);
81      } finally {
82        if (table != null) {
83          try {
84            table.close();
85          } catch (IOException e) {
86            LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
87          }
88        }
89      }
90  
91      return frag;
92    }

114   /**
115    * Returns true if a key was created, false otherwise.
116    */
117   private boolean write(Fragment fragment, boolean createKey, @Nullable Set<UniqueIdentifier> uniqueIds) {
118     if (createKey) {
119       checkNotNull(uniqueIds, "uniqueIds can't be null if createKey is true");
120     }
121 
122     boolean keyCreated = false;
123     HTableInterface occTable = null;
124     try {
125       occTable = tablePool.getTable(occurrenceTableName);
126       byte[] cf = Bytes.toBytes(HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY);
127       if (createKey) {
128         KeyLookupResult keyLookupResult = keyService.generateKey(uniqueIds);
129         keyCreated = keyLookupResult.isCreated();
130         fragment.setKey(keyLookupResult.getKey());
131       }
132       writeFields(occTable, cf, fragment, true);
133     } catch (IOException e) {
134       throw new ServiceUnavailableException("Could not access HBase", e);
135     } finally {
136       if (occTable != null) {
137         try {
138           occTable.close();
139         } catch (IOException e) {
140           LOG.warn("Couldn't return table to pool - continuing with possible memory leak", e);
141         }
142       }
143     }
144 
145     return keyCreated;
146   }

148   /**
149    * Writes all fields of Fragment to the HBase table. If dn (deleteNulls) is true, any field in the fragment that is
150    * empty will be deleted from the corresponding "column" in the HBase row.  If dn is false, any data already existing
151    * in the HBase table for empty fields in the Fragment will remain.
152    *
153    * @param occTable the HBase table to write to
154    * @param cf the column family of the table to write to
155    * @param frag the fragment to persist
156    * @param dn deleteNulls?
157    *
158    * @throws IOException if communicating with HBase fails
159    */
160   private static void writeFields(HTableInterface occTable, byte[] cf, Fragment frag, boolean dn) throws IOException {
161     Put put = new Put(Bytes.toBytes(frag.getKey()));
162     Delete del = null;
163     if (dn) del = new Delete(Bytes.toBytes(frag.getKey()));
164 
165     HBaseHelper.writeField(FieldName.DATASET_KEY,
166       frag.getDatasetKey() == null ? null : Bytes.toBytes(frag.getDatasetKey().toString()), dn, cf, put, del);
167     HBaseHelper.writeField(FieldName.CRAWL_ID,
168       frag.getCrawlId() == null ? null : Bytes.toBytes(frag.getCrawlId()), dn, cf, put, del);
169     HBaseHelper.writeField(FieldName.UNIT_QUALIFIER,
170       frag.getUnitQualifier() == null ? null : Bytes.toBytes(frag.getUnitQualifier()), dn, cf, put, del);
171     HBaseHelper.writeField(FieldName.HARVESTED_DATE,
172       frag.getHarvestedDate() == null ? null : Bytes.toBytes(frag.getHarvestedDate().getTime()), dn, cf, put, del);
173     HBaseHelper.writeField(FieldName.FRAGMENT, frag.getData(), dn, cf, put, del);
174     HBaseHelper.writeField(FieldName.FRAGMENT_HASH, frag.getDataHash(), dn, cf, put, del);
175     HBaseHelper.writeField(FieldName.XML_SCHEMA,
176       frag.getXmlSchema() == null ? null : Bytes.toBytes(frag.getXmlSchema().toString()), dn, cf, put, del);
177     HBaseHelper.writeField(FieldName.PROTOCOL,
178       frag.getProtocol() == null ? null : Bytes.toBytes(frag.getProtocol().toString()), dn, cf, put, del);
179     HBaseHelper.writeField(FieldName.CREATED,
180       frag.getCreated() == null ? null : Bytes.toBytes(frag.getCreated()), dn, cf, put, del);
181 
182     occTable.put(put);
183     if (dn && !del.isEmpty()) occTable.delete(del);
184     occTable.flushCommits();
185   }
