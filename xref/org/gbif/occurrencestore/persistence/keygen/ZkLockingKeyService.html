17  /**
18   * An extension of AbstracHBaseKeyPersistenceService with a generateKey implementation that uses a thread local
19   * ZookeeperLockManager to ensure thread-safe key gen.
20   *
21   * NOTE: this class is currently unused, as it becomes very slow when used concurrently by many threads.
22   */
23  public class ZkLockingKeyService extends AbstractHBaseKeyPersistenceService {
24  
25    private static final Logger LOG = LoggerFactory.getLogger(ZkLockingKeyService.class);
26  
27    private final ThreadLocalLockProvider zookeeperLockManagerProvider;
28  
29    @Inject
30    public ZkLockingKeyService(@Named("id_lookup_table_name") String occurrenceIdTableName,
31      @Named("counter_table_name") String counterTableName, @Named("table_name") String occurrenceTableName,
32      HTablePool tablePool, ThreadLocalLockProvider zookeeperLockManagerProvider) {
33      super(occurrenceIdTableName, counterTableName, occurrenceTableName, tablePool, new OccurrenceKeyBuilder());
34      this.zookeeperLockManagerProvider = zookeeperLockManagerProvider;
35    }
36  
37    /**
38     * Takes out a lock in zookeeper for the dataset of this occurrence before generating a key for it. It first checks
39     * to see if a key already exists for this set of uniqueIdentifiers before generating a new key.
40     *
41     * @param uniqueStrings the identifiers that all refer to the same occurrence
42     *
43     * @return a KeyLookupResult with the key for this occurrence
44     *
45     * @throws IllegalArgumentException if the uniqueIdentifiers set is empty
46     * @throws IllegalStateException    if the next available key is greater than can be held in long
47     */
48    @Override
49    public KeyLookupResult generateKey(Set<String> uniqueStrings, String scope) {
50      checkArgument(!uniqueStrings.isEmpty(), "uniqueIdentifiers can't be empty");
51  
52      // if it already exists, return it right away
53      KeyLookupResult findResult = findKey(uniqueStrings, scope);
54      if (findResult != null) {
55        LOG.debug("Asked to generate, but found existing.");
56        return findResult;
57      }
58  
59      LOG.debug("Waiting for lock");
60      zookeeperLockManagerProvider.get().waitForLock(scope);
61      LOG.debug("Got lock");
62  
63      // check again if it already exists, although now we expect it to be null - we have to have the lock for this
64      // because if we check for existence before getting the lock then someone could write a new key and drop the lock
65      // between us checking (finding nothing) and getting the lock and then writing a new (and now incorrect) key
66      try {
67        findResult = findKey(uniqueStrings, scope);
68        if (findResult != null) {
69          LOG.debug(Thread.currentThread().getName() + " Asked to generate, but found existing.");
70          return findResult;
71        }
72  
73        // generate new key from counter table
74        Long longKey =
75          counterTableStore.incrementColumnValue(HBaseTableConstants.COUNTER_ROW, HBaseTableConstants.COUNTER_COLUMN, 1);
76        if (longKey > Integer.MAX_VALUE) {
77          throw new IllegalStateException(
78            "The next available occurrence id is greater than what Integer can handle. This is fatal!");
79        }
80        int newOccurrenceKey = longKey.intValue();
81  
82        // build the lookup keys from the uniqueIdentifiers
83        Set<String> lookupKeys = keyBuilder.buildKeys(uniqueStrings, scope);
84  
85        // write the new id to each of the lookup keys
86        for (String key : lookupKeys) {
87          lookupTableStore.putInt(key, HBaseTableConstants.LOOKUP_KEY_COLUMN, newOccurrenceKey);
88        }
89  
90        return new KeyLookupResult(newOccurrenceKey, true);
91      } finally {
92        // in all cases we want to release the lock
93        LOG.debug("Releasing lock");
94        zookeeperLockManagerProvider.get().releaseLock(scope);
95      }
96    }
97  }
