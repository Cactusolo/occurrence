47    @Override
48    public VerbatimOccurrence get(@Nullable Integer key) {
49      if (key == null) {
50        return null;
51      }
52      VerbatimOccurrence occ = null;
53      HTableInterface table = null;
54      try {
55        table = tablePool.getTable(occurrenceTableName);
56        Get get = new Get(Bytes.toBytes(key));
57        Result result = table.get(get);
58        if (result == null || result.isEmpty()) {
59          LOG.info("Couldn't find occurrence for key [{}], returning null", key);
60          return null;
61        }
62        occ = OccurrenceBuilder.buildVerbatimOccurrence(result);
63      } catch (IOException e) {
64        throw new ServiceUnavailableException("Could not read from HBase", e);
65      } finally {
66        closeTable(table);
67      }
68  
69      return occ;
70    }

72    @Override
73    public void update(VerbatimOccurrence occ) {
74      checkNotNull(occ, "occurrence can't be null");
75      checkNotNull(occ.getKey(), "occurrence's key can't be null");
76  
77      HTableInterface occTable = null;
78      try {
79        occTable = tablePool.getTable(occurrenceTableName);
80        byte[] cf = Bytes.toBytes(HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY);
81        occ.setModified(new Date().getTime());
82        write(occTable, cf, occ, true);
83      } catch (IOException e) {
84        throw new ServiceUnavailableException("Could not access HBase", e);
85      } finally {
86        closeTable(occTable);
87      }
88    }
