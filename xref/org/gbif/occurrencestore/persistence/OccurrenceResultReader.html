116   public static Object get(Result row, FieldName column) {
117     checkNotNull(row, "row can't be null");
118     checkNotNull(column, "column can't be null");
119     Object result = null;
120 
121     switch (column.getType()) {
122       case INT:
123         result = getInteger(row, column);
124         break;
125       case STRING:
126         result = getString(row, column);
127         break;
128       case DOUBLE:
129         result = getDouble(row, column);
130         break;
131       case LONG:
132         result = getLong(row, column);
133         break;
134       case BYTES:
135         result = getBytes(row, column);
136         break;
137       default:
138         LOG.info("FieldType [{}] is not supported.", column.getType().toString());
139     }
