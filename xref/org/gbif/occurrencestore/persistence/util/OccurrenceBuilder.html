42    /**
43     * Builds a Fragment object from the given result, assigning the passed in key.
44     *
45     * @param result an HBase scan/get Result
46     * @return the Fragment or null if the passed in Result is null
47     * @throws ValidationException if the fragment as stored in the table is invalid
48     */
49    public static Fragment buildFragment(@Nullable Result result) {
50      if (result == null) {
51        return null;
52      }
53  
54      int key = Bytes.toInt(result.getRow());
55  
56      String rawDatasetKey = OccurrenceResultReader.getString(result, FieldName.DATASET_KEY);
57      if (rawDatasetKey == null) {
58        throw new ValidationException("Fragment with key [" + key + "] has no datasetKey.");
59      }
60      UUID datasetKey = UUID.fromString(rawDatasetKey);
61  
62      Integer crawlId = OccurrenceResultReader.getInteger(result, FieldName.CRAWL_ID);
63      if (crawlId == null) {
64        throw new ValidationException("Fragment with key [" + key + "] has no crawlId.");
65      }
66      Long harvested = OccurrenceResultReader.getLong(result, FieldName.HARVESTED_DATE);
67      if (harvested == null) {
68        throw new ValidationException("Fragment with key [" + key + "] has no harvestedDate.");
69      }
70      Date harvestedDate = new Date(harvested);
71      String unitQualifier = OccurrenceResultReader.getString(result, FieldName.UNIT_QUALIFIER);
72      byte[] data = OccurrenceResultReader.getBytes(result, FieldName.FRAGMENT);
73      byte[] dataHash = OccurrenceResultReader.getBytes(result, FieldName.FRAGMENT_HASH);
74      Long created = OccurrenceResultReader.getLong(result, FieldName.CREATED);
75      String rawSchema = OccurrenceResultReader.getString(result, FieldName.XML_SCHEMA);
76      OccurrenceSchemaType schema;
77      if (rawSchema == null) {
78        // this is typically called just before updating the fragment, meaning schemaType will then be correctly set
79        LOG.debug("Fragment with key [{}] has no schema type - assuming DWCA.", key);
80        schema = OccurrenceSchemaType.DWCA;
81      } else {
82        schema = OccurrenceSchemaType.valueOf(rawSchema);
83      }
84      String rawProtocol = OccurrenceResultReader.getString(result, FieldName.PROTOCOL);
85      EndpointType protocol = rawProtocol == null ? null : EndpointType.valueOf(rawProtocol);
86  
87      Fragment frag;
88      if (schema == null || schema == OccurrenceSchemaType.DWCA) {
89        frag =
90          new Fragment(datasetKey, data, dataHash, Fragment.FragmentType.JSON, protocol, harvestedDate, crawlId, schema,
91            null, created);
92      } else {
93        frag =
94          new Fragment(datasetKey, data, dataHash, Fragment.FragmentType.XML, protocol, harvestedDate, crawlId, schema,
95            unitQualifier, created);
96      }
97      frag.setKey(key);
98  
99      return frag;
100   }

255   private static List<Identifier> extractIdentifiers(Integer key, Result result, String columnFamily) {
256     List<Identifier> records = Lists.newArrayList();
257     Integer maxCount = OccurrenceResultReader.getInteger(result, FieldName.IDENTIFIER_COUNT);
258     if (maxCount != null) {
259       for (int count = 0; count < maxCount; count++) {
260         String idCol = HBaseTableConstants.IDENTIFIER_COLUMN + count;
261         String idTypeCol = HBaseTableConstants.IDENTIFIER_TYPE_COLUMN + count;
262         String id = ResultReader.getString(result, columnFamily, idCol, null);
263         String rawType = ResultReader.getString(result, columnFamily, idTypeCol, null);
264         if (id != null && rawType != null) {
265           IdentifierType idType = null;
266           try {
267             idType = IdentifierType.valueOf(rawType);
268           } catch (IllegalArgumentException e) {
269             LOG.warn("Unrecognized value for IdentifierType from field [{}] - data is corrupt.", rawType);
270           }
271           if (idType != null) {
272             Identifier record = new Identifier();
273             record.setEntityKey(key);
274             record.setIdentifier(id);
275             record.setType(idType);
276             records.add(record);
277           }
278         }
279       }
280     }
281     return records;
282   }
