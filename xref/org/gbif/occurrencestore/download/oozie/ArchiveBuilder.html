175   /**
176    * Entry point for assembling the dwc archive.
177    * The thrown exception is the only way of telling Oozie that this job has failed.
178    *
179    * @throws IOException if any read/write operation failed
180    */
181   public static void main(String[] args) throws IOException {
182     final String nameNode = args[0];      // same as namenode, like hdfs://c1n2.gbif.org:8020
183     final String hdfsHivePath = args[1];  // path on hdfs to hive results
184     final String dataTable = args[2];     // hive occurrence results table
185     final String citationTable = args[3]; // hive citation results table
186     final String downloadDir = args[4];   // locally mounted download dir
187     // for example 0000020-130108132303336
188     final String downloadId = DownloadUtils.workflowToDownloadId(args[5]);
189     final String user = args[6];          // download user
190     final String query = args[7];         // download query filter
191     final String downloadLink = args[8];  // download link to the final zip archive
192     final String registryWs = args[9];    // registry ws url
193     final String isSmallDownload = args[10];    // isSmallDownload
194     // download link needs to be constructed
195     final String downloadLinkWithId = downloadLink.replace(DownloadUtils.DOWNLOAD_ID_PLACEHOLDER, downloadId);
196 
197     // create temporary, local, download specific directory
198     File archiveDir = new File(downloadDir, downloadId);
199     RegistryClientUtil registryClientUtil = new RegistryClientUtil("occurrence-download.properties");
200 
201     // create registry client
202     DatasetService datasetService = registryClientUtil.setupDatasetService(registryWs);
203 
204     DatasetOccurrenceDownloadUsageService datasetUsageService = registryClientUtil.setupDatasetUsageService(registryWs);
205 
206     // filesystem configs
207     Configuration conf = new Configuration();
208     conf.set(CommonConfigurationKeys.FS_DEFAULT_NAME_KEY, nameNode);
209     FileSystem hdfs = FileSystem.get(conf);
210     FileSystem localfs = FileSystem.getLocal(conf);
211 
212     // build archive
213     ArchiveBuilder generator =
214       new ArchiveBuilder(downloadId, user, query, datasetService, datasetUsageService, conf, hdfs, localfs, archiveDir,
215         dataTable, citationTable, hdfsHivePath, downloadLinkWithId, Boolean.parseBoolean(isSmallDownload));
216     log("ArchiveBuilder created");
217     generator.buildArchive(new File(downloadDir, downloadId + ".zip"));
218 
219     // SUCCESS!
220   }

235   /**
236    * Main method to assemble the dwc archive and do all the work until we have a final zip file.
237    *
238    * @param zipFile the final zip file holding the entire archive
239    */
240   public void buildArchive(File zipFile) throws IOException {
241     log("Start building the archive ...");
242 
243     // oozie might try several times to run this job, so make sure our filesystem is clean
244     cleanupFS();
245 
246     // create the temp archive dir
247     archiveDir.mkdirs();
248 
249     try {
250       // occurrence file
251       addOccurrenceDataFile();
252 
253       // metadata, citation and rights
254       addDatasetMetadata();
255 
256       // metadata about the entire archive data
257       addQueryMetadata();
258 
259       // meta.xml
260       addArchiveDescriptor();
261 
262       // zip up
263       log(String.format("Zipping archive %s", archiveDir.toString()));
264       CompressionUtil.zipDir(archiveDir, zipFile, true);
265 
266     } finally {
267       // always cleanUp temp dir
268       cleanupFS();
269     }
270 
271   }

273   public void createEmlFile(final UUID constituentId, final File emlDir) throws IOException {
274     Closer closer = Closer.create();
275     try {
276       // store dataset EML as constituent metadata
277       InputStream in = closer.register(datasetService.getMetadataDocument(constituentId));
278       if (in != null) {
279         // copy into archive, reading stream from registry services
280         OutputStream out = closer.register(new FileOutputStream(new File(emlDir, constituentId + ".xml")));
281         ByteStreams.copy(in, out);
282       } else {
283         logError("Found no EML for datasetId: " + constituentId);
284       }
285 
286     } catch (FileNotFoundException ex) {
287       logError("Error creating eml file", ex);
288     } catch (IOException ex) {
289       logError("Error creating eml file", ex);
290     } finally {
291       closer.close();
292     }
293   }

318   /**
319    * Adds an eml file per dataset involved into a subfolder "dataset" which is supported by our dwc archive reader.
320    * Create a rights.txt and citation.txt file targeted at humans to quickly yield an overview about rights and
321    * datasets involved.
322    *
323    * @throws IOException
324    */
325   private void addDatasetMetadata() throws IOException {
326     log("Add constituent dataset metadata to archive");
327     Path citationSrc = new Path(hdfsPath + Path.SEPARATOR + citationTable);
328     // now read the dataset citation table and create an EML file per datasetId
329     // first copy from HDFS to local file
330     if (!hdfs.exists(citationSrc)) {
331       logError("No citation file directory existing on HDFS, skip creating of dataset metadata: " + citationSrc);
332       return;
333     }
334 
335     final Map<UUID, Integer> srcDatasets = readDatasetCounts(citationSrc);
336 
337     File emlDir = new File(archiveDir, "dataset");
338     if (!srcDatasets.isEmpty()) {
339       emlDir.mkdir();
340     }
341     Closer closer = Closer.create();
342 
343     Writer rightsWriter = closer.register(FileUtils.startNewUtf8File(new File(archiveDir, RIGHTS_FILENAME)));
344     Writer citationWriter = closer.register(FileUtils.startNewUtf8File(new File(archiveDir, CITATIONS_FILENAME)));
345 
346     closer.register(citationWriter);
347     // write fixed citations header
348     citationWriter.write(CITATION_HEADER);
349     // now iterate over constituent UUIDs
350 
351     for (Entry<UUID, Integer> dsEntry : srcDatasets.entrySet()) {
352       final UUID constituentId = dsEntry.getKey();
353       log("Processing constituent dataset: " + constituentId);
354       // catch errors for each uuid to make sure one broken dataset does not bring down the entire process
355       try {
356         Dataset srcDataset = datasetService.get(constituentId);
357         // contacts are not eagerly loaded
358         srcDataset.setContacts(datasetService.listContacts(constituentId));
359 
360         // citation
361         String citationLink = writeCitation(citationWriter, srcDataset, constituentId);
362         // rights
363         writeRights(rightsWriter, srcDataset, citationLink);
364         // eml file
365         createEmlFile(constituentId, emlDir);
366 
367         // add as constituent for later
368         constituents.add(new Constituent(srcDataset.getTitle(), dsEntry.getValue()));
369 
370         // add original author as content provider to main dataset description
371         Contact provider = getContentProviderContact(srcDataset);
372         if (provider != null) {
373           dataset.getContacts().add(provider);
374         }
375       } catch (UniformInterfaceException e) {
376         logError("Registry client http exception: " + e.getResponse().getStatus() + '\n' + e.getResponse()
377           .getEntity(String.class));
378       } catch (Exception e) {
379         logError("Error creating download file", e);
380       }
381     }
382     closer.close();
383   }

385   /**
386    * Copies and merges the hive query results files into a single, local occurrence data file.
387    */
388   private void addOccurrenceDataFile() throws IOException {
389     log("Copy-merge occurrence data hdfs file to local filesystem");
390     Path dataSrc = new Path(hdfsPath + Path.SEPARATOR + dataTable);
391     if (!isSmallDownload) { // small downloads already include the headers
392       FileUtil.copy(new File(HEADERS_FILENAME), hdfs, dataSrc, false, conf);
393     }
394     File rawDataResult = new File(archiveDir, DATA_FILENAME);
395     Path dataDest = new Path(rawDataResult.toURI());
396     FileUtil.copyMerge(hdfs, dataSrc, localfs, dataDest, false, conf, null);
397     // remove the CRC file created by copyMerge method
398     removeDataCRCFile();
399   }

401   /**
402    * Creates a single EML metadata file for the entire archive.
403    * Make sure we execute this method AFTER building the constituents metadata which adds to our dataset instance.
404    *
405    * @throws IOException
406    */
407   private void addQueryMetadata() {
408     log("Add query dataset metadata to archive");
409     try {
410       // Random UUID use because the downloadId is not a string in UUID format
411       dataset.setKey(UUID.randomUUID());
412       dataset.setTitle(String.format(DATASET_TITLE_FMT, downloadId));
413       dataset.setDescription(getDatasetDescription());
414       dataset.setCreated(new Date());
415       Citation citation = new Citation(String.format(DATASET_TITLE_FMT, downloadId), downloadId);
416       dataset.setCitation(citation);
417       // can we derive a link from the query to set the dataset.homepage?
418       dataset.setPubDate(new Date());
419       dataset.setDataLanguage(Language.ENGLISH);
420       dataset.setType(DatasetType.OCCURRENCE);
421       dataset.getDataDescriptions().add(createDataDescription());
422 
423       dataset.getContacts().add(createContact(user, null, ContactType.AUTHOR, true));
424       dataset.getContacts().add(createContact(DOWNLOAD_CONTACT_SERVICE, null, ContactType.ORIGINATOR, true));
425       dataset.getContacts().add(createContact(DOWNLOAD_CONTACT_SERVICE, null, ContactType.METADATA_AUTHOR, true));
426 
427 
428       File eml = new File(archiveDir, METADATA_FILENAME);
429       Writer writer = FileUtils.startNewUtf8File(eml);
430       EMLWriter.write(dataset, writer);
431 
432     } catch (Exception e) {
433       System.err.println("Failed to write query result dataset EML file");
434       e.printStackTrace();
435     }
436   }

459   /**
460    * Create the date file, core of the DwC-A.
461    */
462   private ArchiveFile createCoreFile() {
463     ArchiveFile occ = new ArchiveFile();
464     occ.addLocation(DATA_FILENAME);
465     occ.setRowType(DwcTerm.Occurrence.qualifiedName());
466     occ.setEncoding(Charsets.UTF_8.displayName());
467     occ.setIgnoreHeaderLines(0);
468     occ.setFieldsEnclosedBy(null);
469     occ.setFieldsTerminatedBy("\t");
470     occ.setLinesTerminatedBy("\n");
471 
472     int index = 0;
473     for (FieldName fn : HiveFieldUtil.DOWNLOAD_COLUMNS) {
474       Term term = HiveFieldUtil.getTerm(fn);
475       ArchiveField field = new ArchiveField();
476       field.setIndex(index);
477       field.setTerm(term);
478 
479       occ.addField(field);
480       if (index == 0) {
481         occ.setId(field);
482       }
483       index++;
484     }
485     return occ;
486   }

501   /**
502    * Checks the contacts of a dataset and finds the preferred contact that should be used as the main author
503    * of a dataset.
504    *
505    * @return preferred author contact or null
506    */
507   private Contact getContentProviderContact(Dataset dataset) {
508     Contact author = null;
509     for (ContactType type : AUTHOR_TYPES) {
510       for (Contact c : dataset.getContacts()) {
511         if (type == c.getType()) {
512           if (author == null) {
513             author = c;
514           } else if (c.isPrimary()) {
515             author = c;
516           }
517         }
518       }
519       if (author != null) {
520         Contact provider = new Contact();
521         try {
522           PropertyUtils.copyProperties(provider, author);
523           provider.setKey(null);
524           provider.setType(ContactType.CONTENT_PROVIDER);
525           provider.setPrimary(false);
526           return provider;
527         } catch (IllegalAccessException e) {
528           // TODO: Handle exception
529         } catch (InvocationTargetException e) {
530           // TODO: Handle exception
531         } catch (NoSuchMethodException e) {
532           // TODO: Handle exception
533         }
534       }
535     }
536     return null;
537   }

568   /**
569    * Creates Map with dataset UUIDs and its record counts.
570    */
571   private Map<UUID, Integer> readDatasetCounts(Path citationSrc) throws IOException {
572     // the hive query result is a directory with one or more files - read them all into a uuid set
573     Map<UUID, Integer> srcDatasets = Maps.newHashMap(); // map of uuids to occurrence counts
574     FileStatus[] citFiles = hdfs.listStatus(citationSrc);
575     int invalidUuids = 0;
576     Closer closer = Closer.create();
577     for (FileStatus fs : citFiles) {
578       if (!fs.isDirectory()) {
579         BufferedReader citationReader =
580           new BufferedReader(new InputStreamReader(hdfs.open(fs.getPath()), Charsets.UTF_8));
581         closer.register(citationReader);
582         try {
583           String line = citationReader.readLine();
584           while (line != null) {
585             if (!Strings.isNullOrEmpty(line)) {
586               // we also catch errors for every dataset so we dont break the loop
587               try {
588                 Iterator<String> iter = TAB_SPLITTER.split(line).iterator();
589                 // play safe and make sure we got a uuid - even though our api doesnt require it
590                 UUID key = UUID.fromString(iter.next());
591                 Integer count = Integer.parseInt(iter.next());
592                 srcDatasets.put(key, count);
593                 // small downloads persist dataset usages while builds the citations file
594                 if (!this.isSmallDownload) {
595                   persistDatasetUsage(count, downloadId, key);
596                 }
597               } catch (IllegalArgumentException e) {
598                 // ignore invalid UUIDs
599                 logError("Found invalid UUID as datasetId >>>" + line + "<<<");
600                 invalidUuids++;
601               }
602             }
603             line = citationReader.readLine();
604           }
605         } finally {
606           closer.close();
607         }
608       }
609     }
610     if (invalidUuids > 0) {
611       log("Found " + invalidUuids + " invalid dataset UUIDs.");
612     } else {
613       log("All " + srcDatasets.size() + " dataset UUIDs are valid.");
614     }
615     return srcDatasets;
616   }

629   private String writeCitation(final Writer citationWriter, final Dataset dataset, final UUID constituentId)
630     throws IOException {
631     // citation
632     String citationLink = null;
633     if (dataset.getCitation() != null && !Strings.isNullOrEmpty(dataset.getCitation().getText())) {
634       citationWriter.write('\n' + dataset.getCitation().getText());
635       if (!Strings.isNullOrEmpty(dataset.getCitation().getIdentifier())) {
636         citationLink = ", " + dataset.getCitation().getIdentifier();
637         citationWriter.write(citationLink);
638       }
639     } else {
640       logError("Constituent dataset misses mandatory citation for id: " + constituentId);
641     }
