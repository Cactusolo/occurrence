129   /**
130    * Method that determines if the Solr Query produces a "small" download file.
131    */
132   public Boolean isSmallDownloadCount(String solrQuery, String workflowId) {
133     try {
134       QueryResponse response = solrServer.query(new SolrQuery(solrQuery));
135       long recordCount = response.getResults().getNumFound();
136       updateTotalRecordsCount(workflowId, recordCount);
137       return recordCount <= smallDownloadLimit;
138     } catch (Exception e) {
139       System.err.println("Error getting the records count");
140       e.printStackTrace();
141       return false;
142     }
143   }

145   /**
146    * Sets the oozie action parameter 'is_small_download'.
147    * 
148    * @param solrQuery to be executed
149    * @throws IOException in case of error reading or writing the 'oozie.action.output.properties' file
150    */
151   public void setDownloadLimitWorkflowFlag(String solrQuery, String workflowId) throws IOException {
152     String oozieProp = System.getProperty(OOZIE_ACTION_OUTPUT_PROPERTIES);
153     Closer closer = Closer.create();
154     if (oozieProp != null) {
155       OutputStream os = null;
156       File propFile = new File(oozieProp);
157       Properties props = new Properties();
158       try {
159         os = closer.register(new FileOutputStream(propFile));
160         props.setProperty(IS_SMALL_DOWNLOAD, isSmallDownloadCount(solrQuery, workflowId).toString());
161         props.store(os, "");
162       } catch (FileNotFoundException e) {
163         System.err.println("Error reading properties file");
164         e.printStackTrace();
165         closer.rethrow(e);
166       } finally {
167         closer.close();
168       }
169     } else {
170       throw new IllegalStateException(OOZIE_ACTION_OUTPUT_PROPERTIES + " System property not defined");
171     }
172   }

174   /**
175    * Updates the record count of the download entity.
176    */
177   private void updateTotalRecordsCount(String workflowId, long recordCount) {
178     try {
179       String downloadId = DownloadUtils.workflowToDownloadId(workflowId);
180       System.out.println(String.format("Updating record count(%,d) of download %s", recordCount, downloadId));
181       Download download = occurrenceDownloadService.get(downloadId);
182       if (download == null) {
183         System.err.println(String.format("Download %s was not found!", downloadId));
184       } else {
185         download.setTotalRecords(recordCount);
186         occurrenceDownloadService.update(download);
187       }
188     } catch (Exception ex) {
189       System.err.println(String.format("Error updating record count for download worflow %s, reported count is %,d",
190         workflowId, recordCount));
191       ex.printStackTrace();
192     }
193   }
