18  /**
19   * SAX Handler that can parse a single oozie download job definition and extract a download instance from it.
20   * Extracts download properties using the Hadoop configuration conventions.
21   * 
22   * <pre>
23   * {@code
24   * <configuration>
25   *   <property>
26   *     <name>property name</name>
27   *     <value>property value</value>
28   *   </property>
29   * </configuration>}
30   * </pre>
31   */
32  public class OozieJobSaxHandler extends DefaultHandler {
33  
34    private static final String CONFIGURATION_ELEMENT = "configuration";
35    private static final String PROPERTY_ELEMENT = "property";
36    private static final String NAME_ELEMENT = "name";
37    private static final String VALUE_ELEMENT = "value";
38    // ObjectMappers are thread safe if not reconfigured in code
39    private final ObjectMapper mapper = new ObjectMapper();
40    private static final Splitter EMAIL_SPLITTER = Splitter.on(';').omitEmptyStrings().trimResults();
41  
42    // download data
43    private String creator;
44    private Predicate predicate;
45    private Set<String> notificationAddresses;
46  
47    private boolean inConfiguration;
48    private boolean inProperty;
49    private StringBuffer chars;
50    private String propName;
51    private String propValue;
52  
53    public DownloadRequest buildDownload() {
54      return new DownloadRequest(predicate, creator, notificationAddresses);
55    }
56  
57    @Override
58    public void characters(char[] ch, int start, int length) {
59      chars.append(ch, start, length);
60    }
61  
62    @Override
63    public void endElement(String uri, String localName, String qName) throws SAXException {
64      if (localName.equalsIgnoreCase(CONFIGURATION_ELEMENT)) {
65        inConfiguration = false;
66      }
67  
68      if (inProperty) {
69        String content = Strings.emptyToNull(chars.toString().trim());
70        if (localName.equals(PROPERTY_ELEMENT)) {
71          inProperty = false;
72          // interpret name & value
73          if (propNameEquals(Constants.USER_PROPERTY)) {
74            creator = propValue.trim();
75  
76          } else if (propNameEquals(Constants.NOTIFICATION_PROPERTY)) {
77            notificationAddresses = Sets.newHashSet(EMAIL_SPLITTER.split(Strings.nullToEmpty(propValue)));
78  
79          } else if (propNameEquals(Constants.FILTER_PROPERTY) && !Strings.isNullOrEmpty(propValue)) {
80            try {
81              predicate = mapper.readValue(propValue, Predicate.class);
82            } catch (IOException e) {
83              // TODO: Handle exception
84            }
85          }
86  
87        } else if (localName.equals(NAME_ELEMENT)) {
88          propName = content;
89  
90        } else if (localName.equals(VALUE_ELEMENT)) {
91          propValue = content;
92        }
93  
94      }
95    }
96  
97    @Override
98    public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
99      chars = new StringBuffer();
100     if (localName.equalsIgnoreCase(CONFIGURATION_ELEMENT)) {
101       inConfiguration = true;
102     }
103     if (inConfiguration && localName.equalsIgnoreCase(PROPERTY_ELEMENT)) {
104       inProperty = true;
105       propName = null;
106       propValue = null;
107     }
108   }

110   /**
111    * Normalizes the current property name so that old and new property naming styles with . or _ are supported
112    * and then does a case insensitive comparison against the desired name.
113    */
114   private boolean propNameEquals(String wantedPropName) {
115     return propName.replace('.', '_').equalsIgnoreCase(wantedPropName);
116   }
