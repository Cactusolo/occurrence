48  @Path("occurrence/download/request")
49  @Produces({MediaType.APPLICATION_JSON, ExtraMediaTypes.APPLICATION_JAVASCRIPT})
50  @Consumes(MediaType.APPLICATION_JSON)
51  @Singleton
52  public class DownloadResource {
53  
54    private static final Logger LOG = LoggerFactory.getLogger(DownloadResource.class);
55  
56    // low quality of source to default to JSON
57    private static final String OCT_STREAM_QS = ";qs=0.5";
58  
59    private final DownloadRequestService requestService;
60  
61    private final OccurrenceDownloadService occurrenceDownloadService;
62  
63    private final CallbackService callbackService;
64  
65    @Inject
66    public DownloadResource(DownloadRequestService service, CallbackService callbackService,
67      OccurrenceDownloadService occurrenceDownloadService) {
68      this.requestService = service;
69      this.callbackService = callbackService;
70      this.occurrenceDownloadService = occurrenceDownloadService;
71    }
72  
73    @DELETE
74    @Path("{key}")
75    public void delDownload(@PathParam("key") String jobId, @Context SecurityContext security) {
76      // service.get returns a download or throws NotFoundException
77      assertLoginMatches(occurrenceDownloadService.get(jobId).getRequest(), security);
78      LOG.debug("Delete download: [{}]", jobId);
79      requestService.cancel(jobId);
80    }
81  
82    @GET
83    @Path("{key}")
84    @Produces(MediaType.APPLICATION_OCTET_STREAM + OCT_STREAM_QS)
85    public InputStream getResult(@PathParam("key") String downloadKey, @Context HttpServletResponse response) {
86      LOG.debug("Get download data: [{}]", downloadKey);
87      // suggest filename for download in http headers
88      response.setHeader("Content-Disposition", "attachment; filename=" + downloadKey + ".zip");
89      return requestService.getResult(downloadKey);
90    }
91  
92  
93    @GET
94    @Path("callback")
95    public Response oozieCallback(@QueryParam("job_id") String jobId, @QueryParam("status") String status) {
96      LOG.debug("Received callback from Oozie for Job [{}] with status [{}]", jobId, status);
97      callbackService.processCallback(jobId, status);
98      return Response.ok().build();
99    }
100 
101   @POST
102   @Validate
103   public String startDownload(@Valid DownloadRequest request, @Context SecurityContext security) {
104     LOG.debug("Download: [{}]", request);
105     // assert authenticated user is the same as in download
106     assertLoginMatches(request, security);
107     String downloadKey = requestService.create(request);
108     LOG.info("Created new download job with key [{}]", downloadKey);
109     return downloadKey;
110   }
111 
112   /**
113    * Checks that a user is authenticated and the same user is the creator of the download.
114    * 
115    * @throws AccessControlException if no or wrong user is authenticated
116    */
117   private void assertLoginMatches(DownloadRequest request, SecurityContext security) {
118     // assert authenticated user is the same as in download
119     final Principal principal = security.getUserPrincipal();
120     if (principal == null) {
121       throw new NotAuthenticatedException("No user authenticated for creating a download");
122     } else if (!principal.getName().equals(request.getCreator())) {
123       LOG.warn("Different user authenticated [{}] than download specifies [{}]", principal.getName(),
124         request.getCreator());
125       throw new NotAllowedException(principal.getName() + " not allowed to create download with creator "
126         + request.getCreator());
127     }
128   }
