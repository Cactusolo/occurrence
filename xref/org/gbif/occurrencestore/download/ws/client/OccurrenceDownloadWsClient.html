43    }

46    @Override
47    public String create(DownloadRequest download) {
48      String jobId;
49      try {
50        jobId = getResource().type(MediaType.APPLICATION_JSON).post(String.class, mapper.writeValueAsBytes(download));
51      } catch (IOException e) {
52        throw new IllegalStateException(e);
53      }
54      LOG.debug("Submitted download, got job id [{}]", jobId);
55      return jobId;
56    }
57
