41    /**
42     * Appends the occurrence headers line to the output file.
43     */
44    public static void appendHeaders(OutputStream fileWriter) throws IOException {
45      Closer resultCloser = Closer.create();
46      try {
47        InputStream headerInputStream =
48          resultCloser.register(new ByteArrayInputStream(OCCURRENCE_HEADER.getBytes(Charsets.UTF_8)));
49        ByteStreams.copy(headerInputStream, fileWriter);
50      } finally {
51        resultCloser.close();
52      }
53    }
54  
55    /**
56     * Creates the headers file.
57     * The output file name can be specified as argument. If the file name is not specified the file is generated in the
58     * current directory with the name "headers.txt".
59     */
60    public static void generateHeadersFile(String outFileName) throws IOException {
61      Closer closer = Closer.create();
62      String fileName = Strings.isNullOrEmpty(outFileName) ? DEFAULT_FILE_NAME : outFileName;
63      try {
64        FileWriter fileWriter = closer.register(new FileWriter(new File(fileName)));
65        fileWriter.write(OCCURRENCE_HEADER);
66      } finally {
67        closer.close();
68      }
69    }

71    /**
72     * Returns the headers names of download columns.
73     */
74    public static String[] getColumnHeaders() {
75      List<String> headers = Lists.newArrayList();
76      for (FieldName field : DOWNLOAD_COLUMNS) {
77        headers.add(getHiveField(field));
78      }
79      return headers.toArray(new String[headers.size()]);
80    }
81  
82    /**
83     * Returns the headers names of download columns.
84     */
85    public static String[] getDownloadHeaders() {
86      List<String> headers = Lists.newArrayList();
87      for (FieldName field : DOWNLOAD_COLUMNS) {
88        String col = getHiveField(field);
89        if (col.endsWith("_")) {
90          col = col.substring(0, col.length() - 1);
91        }
92        headers.add(col);
93      }
94      return headers.toArray(new String[headers.size()]);
95    }
