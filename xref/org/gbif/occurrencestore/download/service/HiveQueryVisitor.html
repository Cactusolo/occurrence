51  /**
52   * This class builds a WHERE clause for a Hive query from a {@link org.gbif.api.model.occurrence.predicate.Predicate}
53   * object.
54   * </p>
55   * This is not thread-safe but one instance can be reused. It is package-local and should usually be accessed through
56   * {@link DownloadRequestServiceImpl}. All {@code visit} methods have to be public for the
57   * {@link Class#getMethod(String, Class[])} call to work. This is the primary reason for this class being package-local.
58   * </p>
59   * The only entry point into this class is the {@code getHiveQuery} method!
60   */

118   // IS_GEOREFERENCED_CHECK, IS_NOT_GEOREFERENCED_CHECK, SPATIAL_ISSUES_CHECK and NO_SPATIAL_ISSUES_CHECK are

126     + IS_NULL_OPERATOR + ')';

136   private static String toHiveField(OccurrenceSearchParameter param) {
137     if (PARAM_TO_FIELD.containsKey(param)) {
138       return HiveFieldUtil.getHiveField(PARAM_TO_FIELD.get(param));
139     }
140     // QueryBuildingException requires an underlying exception
141     throw new IllegalArgumentException("Search parameter " + param + " is not mapped to Hive");
142   }

144   /**
145    * Translates a valid {@link Download} object and translates it into a
146    * strings that can be used as the <em>WHERE</em> clause for a Hive download.
147    * 
148    * @param download to translate
149    * @return WHERE clause
150    */
151   public String getHiveQuery(Predicate predicate) throws QueryBuildingException {
152     String hiveQuery = ALL_QUERY;
153     if (predicate != null) { // null predicate means a SELECT ALL
154       builder = new StringBuilder();
155       visit(predicate);
156       hiveQuery = builder.toString();
157     }
158 
159     // Set to null to prevent old StringBuilders hanging around in case this class is reused somewhere else
160     builder = null;
161     return hiveQuery;
162   }

172   /**
173    * Supports all parameters incl taxonKey expansion for higher taxa.

175    * @param predicate

177   public void visit(EqualsPredicate predicate) throws QueryBuildingException {
178     if (OccurrenceSearchParameter.TAXON_KEY == predicate.getKey()) {
179       appendTaxonKeyFilter(predicate.getValue());

194   public void visit(InPredicate predicate) throws QueryBuildingException {
195     builder.append('(');
196     Iterator<String> iterator = predicate.getValues().iterator();
197     while (iterator.hasNext()) {
198       String value = iterator.next();
199       builder.append('(');
200       builder.append(toHiveField(predicate.getKey()));
201       builder.append(EQUALS_OPERATOR);
202       builder.append(toHiveValue(predicate.getKey(), value));
203       builder.append(')');
204       if (iterator.hasNext()) {
205         builder.append(DISJUNCTION_OPERATOR);
206       }
207     }
208     builder.append(')');
209   }

229   public void visit(WithinPredicate within) {
230     // the geometry must be valid - it was validated in the predicates constructor
231     builder.append("contains(\"");
232     builder.append(within.getGeometry());
233     builder.append("\", ");
234     builder.append(HiveFieldUtil.getHiveField(FieldName.I_LATITUDE));
235     builder.append(", ");
236     builder.append(HiveFieldUtil.getHiveField(FieldName.I_LONGITUDE));
237     builder.append(')');
238   }

240   /**
241    * Builds a list of predicates joined by 'op' statements.
242    * The final statement will look like this:
243    * 
244    * <pre>
245    * ((predicate) op (predicate) ... op (predicate))
246    * </pre>
247    */
248   public void visitCompoundPredicate(CompoundPredicate predicate, String op) throws QueryBuildingException {
249     builder.append('(');
250     Iterator<Predicate> iterator = predicate.getPredicates().iterator();
251     while (iterator.hasNext()) {
252       Predicate subPredicate = iterator.next();
253       builder.append('(');
254       visit(subPredicate);
255       builder.append(')');
256       if (iterator.hasNext()) {
257         builder.append(op);
258       }
259     }
260     builder.append(')');
261   }

264   public void visitSimplePredicate(SimplePredicate predicate, String op) throws QueryBuildingException {
265     if (predicate.getKey() == OccurrenceSearchParameter.SPATIAL_ISSUES) {
266       appendSpatialIssuePredicate(predicate.getValue());
267     } else if (predicate.getKey() == OccurrenceSearchParameter.GEOREFERENCED) {
268       appendGeoreferencedPredicate(predicate.getValue());
269     } else {
270       builder.append(toHiveField(predicate.getKey()));
271       builder.append(op);
272       builder.append(toHiveValue(predicate.getKey(), predicate.getValue()));
273     }
274   }

276   /**
277    * OccurrenceSearchParameter.GEOREFERENCED is managed specially.
278    * The search parameter is a boolean value but in hive must be converted into NULL comparison of the latitude and
279    * longitude columns.
280    * Be aware that the operator is ignored and it's handled as expression that checks if the record has a spatial issue.
281    */
282   private void appendGeoreferencedPredicate(String value) {
283     if (Boolean.parseBoolean(value)) {
284       builder.append(IS_GEOREFERENCED_CHECK);
285     } else {
286       builder.append(IS_NOT_GEOREFERENCED_CHECK);
287     }
288   }
289 
290   /**
291    * OccurrenceSearchParameter.SPATIAL_ISSUES is managed specially. The search parameter is a boolean value but in hive
292    * is an integer.
293    * Be aware that the operator is ignored and it's handled as expression that checks if the record has a spatial issue.
294    */
295   private void appendSpatialIssuePredicate(String value) {
296     if (Boolean.parseBoolean(value)) {
297       builder.append(SPATIAL_ISSUES_CHECK);
298     } else {
299       builder.append(NO_SPATIAL_ISSUES_CHECK);
300     }
301   }

303   /**
304    * Searches any of the nub keys in hbase of any rank.
305    * 
306    * @param taxonKey
307    */
308   private void appendTaxonKeyFilter(String taxonKey) {
309     builder.append('(');
310     boolean first = true;
311     for (FieldName col : NUB_KEY_COLUMNS) {
312       if (!first) {
313         builder.append(DISJUNCTION_OPERATOR);
314       }
315       builder.append(HiveFieldUtil.getHiveField(col));
316       builder.append(EQUALS_OPERATOR);
317       builder.append(taxonKey);
318       first = false;
319     }
320     builder.append(')');
321   }

323   /**
324    * Converts a value to the form expected by Hive/Hbase based on the OccurrenceSearchParameter.
325    * Most values pass by unaltered. Quotes are added for values that need to be quoted, escaping any existing quotes.
326    * 
327    * @param param the type of parameter defining the expected type
328    * @param value the original query value
329    * @return the converted value expected by HBase
330    */
331   private String toHiveValue(OccurrenceSearchParameter param, String value) throws QueryBuildingException {
332     if (param == OccurrenceSearchParameter.BASIS_OF_RECORD) {
333       // special int
334       BasisOfRecord bor = (BasisOfRecord) VocabularyUtils.lookupEnum(value, BasisOfRecord.class);
335       return bor == null ? "" : BOR_CONVERTER.fromEnum(bor).toString();
336     } else if (param == OccurrenceSearchParameter.COUNTRY) {
337       // upper case 2 letter iso code
338       return '\'' + value.toUpperCase() + '\'';
339     }
340 
341     if (Date.class.isAssignableFrom(param.type())) {
342       // use longs for timestamps expressed as ISO dates
343       Date d = IsoDateParsingUtils.parseDate(value);
344       return String.valueOf(d.getTime());
345 
346     } else if (Number.class.isAssignableFrom(param.type())) {
347       // dont quote numbers
348       return value;
349 
350     } else {
351       // quote value, escape existing quotes
352       return '\'' + APOSTROPHE_MATCHER.replaceFrom(value, "\\\'") + '\'';
353     }
354   }

356   private void visit(Object object) throws QueryBuildingException {
357     Method method = null;
358     try {
359       method = getClass().getMethod("visit", new Class[] {object.getClass()});
360     } catch (NoSuchMethodException e) {
361       LOG
362         .warn(
363           "Visit method could not be found. That means a Predicate has been passed in that is unknown to this "
364             + "class",
365           e);
366       throw new IllegalArgumentException("Unknown Predicate", e);
367     }
368     try {
369       method.invoke(this, object);
370     } catch (IllegalAccessException e) {
371       LOG.error("This should never happen as all our methods are public and missing methods should have been caught "
372         + "before. Probably a programming error", e);
373       throw new RuntimeException("Programming error", e);
374     } catch (InvocationTargetException e) {
375       LOG.info("Exception thrown while building the Hive Download", e);
376       throw new QueryBuildingException(e);
377     }
378   }
