89    /**
90     * Converts the byte size into human-readable format.
91     * Support both SI and byte format.
92     */
93    private static String humanReadableByteCount(long bytes, boolean si) {
94      int unit = si ? 1000 : 1024;
95      if (bytes < unit) {
96        return bytes + " B";
97      }
98      int exp = (int) (Math.log(bytes) / Math.log(unit));
99      String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
100     return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
101   }

135   /**
136    * Gets a human readable version of the occurrence search filter used.
137    */
138   private String getHumanReadableFilter(Download download) {
139     // TODO: should escapeXml be false here?
140     HumanFilterBuilder filter = new HumanFilterBuilder(RESOURCES, datasetService, nameUsageService, false);
141     if (download.getRequest().getPredicate() != null) {
142       StringBuilder stringBuilder = new StringBuilder();
143       Map<OccurrenceSearchParameter, LinkedList<String>> params =
144         filter.humanFilter(download.getRequest().getPredicate());
145       for (Iterator<Entry<OccurrenceSearchParameter, LinkedList<String>>> paramEntryIt = params.entrySet().iterator(); paramEntryIt
146         .hasNext();) {
147         Entry<OccurrenceSearchParameter, LinkedList<String>> paramEntry = paramEntryIt.next();
148         stringBuilder.append('\t');
149         stringBuilder.append(paramEntry.getKey().name() + ": ");
150         stringBuilder.append(COMMA_JOINER.join(paramEntry.getValue()));
151         if (paramEntryIt.hasNext()) {
152           stringBuilder.append('\n');
153         }
154       }
155       return stringBuilder.toString();
156     }
157     return RESOURCES.getString("filter.allrecords");
158   }

160   /**
161    * Gets the list of notification addresses from the download object.
162    * If the list of addresses is empty, the email of the creator is used.
163    */
164   private List<Address> getNotificationAddresses(Download download) {
165     List<Address> emails = Lists.newArrayList();
166     if (download.getRequest().getNotificationAddresses() == null
167       || download.getRequest().getNotificationAddresses().isEmpty()) {
168       User user = userService.get(download.getRequest().getCreator());
169       if (user != null) {
170         try {
171           emails.add(new InternetAddress(user.getEmail()));
172         } catch (AddressException e) {
173           // bad address?
174           LOG.warn("Ignore corrupt email address {}", user.getEmail());
175         }
176       }
177     } else {
178       emails = toInternetAddresses(download.getRequest().getNotificationAddresses());
179     }
180     return emails;
181   }

184   /**
185    * Utility method that sends a notification email.
186    */
187   private void sendNotificationMail(Download d, String subject, String body) {
188     List<Address> emails = getNotificationAddresses(d);
189     if (emails.isEmpty() && bccAddresses.isEmpty()) {
190       LOG.warn("No valid notification addresses given for download {}", d.getKey());
191       return;
192     }
193     try {
194       // Send E-Mail
195       MimeMessage msg = new MimeMessage(session);
196       msg.setFrom();
197       msg.setRecipients(Message.RecipientType.TO, emails.toArray(new Address[emails.size()]));
198       msg.setRecipients(Message.RecipientType.BCC, bccAddresses.toArray(new Address[bccAddresses.size()]));
199       msg.setSubject(subject);
200       msg.setSentDate(new Date());
201       msg.setText(body + '\n' + getDownloadDetails(d));
202       Transport.send(msg);
203     } catch (MessagingException e) {
204       LOG.error(NOTIFY_ADMIN, "Sending of notification Mail for download [{}] failed", d.getKey(), e);
205     }
206   }

208   /**
209    * Transforms a iterable of string into a list of email addresses.
210    */
211   private List<Address> toInternetAddresses(Iterable<String> strEmails) {
212     List<Address> emails = Lists.newArrayList();
213     for (String address : strEmails) {
214       try {
215         emails.add(new InternetAddress(address));
216       } catch (AddressException e) {
217         // bad address?
218         LOG.warn("Ignore corrupt email address {}", address);
219       }
220     }
221     return emails;
222   }
