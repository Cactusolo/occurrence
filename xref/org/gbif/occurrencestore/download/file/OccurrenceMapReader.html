138   /**
139    * Cleans specials characters from a string value.
140    * Removes tabs, line breaks and new lines.
141    */
142   public static String getCleanString(Result row, FieldName column) {
143     String value = getString(row, column);
144     return value != null ? value.replaceAll(DownloadUtils.DELIMETERS_MATCH, " ") : value;

157   /**
158    * Reads an occurrence record from HBase into Map.
159    * The occurrence record
160    */
161   public Map<String, Object> get(@Nonnull Integer key) throws IOException {
162     Preconditions.checkNotNull(key, "Ocurrence key can't be null");
163     Map<String, Object> occurrence = null;
164     HTableInterface table = null;
165     Closer closer = Closer.create();
166     try {
167       table = tablePool.getTable(occurrenceTableName);
168       closer.register(table);
169       Get get = new Get(Bytes.toBytes(key));
170       Result result = table.get(get);
171       if (result == null || result.isEmpty()) {
172         LOG.debug("Couldn't find occurrence for key [{}], returning null", key);
173         return null;
174       }
175       occurrence = buildOccurrence(result);
176     } catch (IOException e) {
177       throw new ServiceUnavailableException("Could not read from HBase", e);
178     } finally {
179       closer.close();
180     }
181     return occurrence;
182   }
