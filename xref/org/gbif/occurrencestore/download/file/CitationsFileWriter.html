22  /**
23   * Utility class that creates the citations file using a faceted Solr response.
24   * The output file contains a list of datasets keys/uuids and its counts of occurrence records.
25   */
26  public class CitationsFileWriter {
27  
28    // Java fields of class solr.FacetField.Count that are used to create the citations file.
29    private static final String[] HEADER = new String[] {"name", "count"};
30  
31    // Processors for fields name and count of class solr.FacetField.Count
32    private static final CellProcessor[] PROCESSORS = new CellProcessor[] {new NotNull(), new ParseLong()};
33  
34    /**
35     * Private/default constructor.
36     */
37    private CitationsFileWriter() {
38      // private constructor
39    }
40  
41    /**
42     * Creates the dataset citation file using the the Solr query response.
43     * 
44     * @param datasetUsages record count per dataset
45     * @param citationFileName output file name
46     * @param datasetOccUsageService usage service
47     * @param downloadKey download key
48     * @throws IOException if an error occurs while the citation file is being created
49     */
50    public static void createCitationFile(final Map<UUID, Long> datasetUsages, String citationFileName,
51      DatasetOccurrenceDownloadUsageService datasetOccUsageService, String downloadKey) throws IOException {
52      if (datasetUsages != null && !datasetUsages.isEmpty()) {
53        ICsvBeanWriter beanWriter = null;
54        Closer closer = Closer.create();
55        try {
56          beanWriter =
57            new CsvBeanWriter(new FileWriterWithEncoding(citationFileName, Charsets.UTF_8), CsvPreference.TAB_PREFERENCE);
58          closer.register(beanWriter);
59          for (Entry<UUID, Long> entry : datasetUsages.entrySet()) {
60            if (entry.getKey() != null) {
61              beanWriter.write(new Facet.Count(entry.getKey().toString(), entry.getValue()), HEADER, PROCESSORS);
62              persistDatasetUsage(entry, downloadKey, datasetOccUsageService);
63            }
64          }
65        } catch (IOException e) {
66          closer.rethrow(e);
67        } finally {
68          if (beanWriter != null) {
69            beanWriter.flush();
70          }
71          closer.close();
72        }
73      }
74    }
75  
76    /**
77     * Persists the dataset usage information and swallows any exception to avoid an error during the file building.
78     */
79    private static void persistDatasetUsage(Entry<UUID, Long> usage, String downloadKey,
80      DatasetOccurrenceDownloadUsageService datasetOccUsageService) {
81      try {
82        DatasetOccurrenceDownloadUsage datasetUsage = new DatasetOccurrenceDownloadUsage();
83        datasetUsage.setDatasetKey(usage.getKey());
84        datasetUsage.setNumberRecords(usage.getValue());
85        datasetUsage.setDownloadKey(downloadKey);
86        datasetOccUsageService.create(datasetUsage);
87      } catch (Exception e) {
88        System.err.println("Error persisting dataset usage information");
89        e.printStackTrace();
90      }
91    }
