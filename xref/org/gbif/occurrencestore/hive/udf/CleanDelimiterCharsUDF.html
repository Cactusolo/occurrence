10   * A simple UDF for Hive that replaces specials characters with blanks.
11   * The characters replaced by this UDF can break a download format and those are: tabs, line breaks and new lines.
12   * If the input value is null or can't be parsed, and empty string is returned.
13   */
14  @Description(

16    value = "_FUNC_(field)")

18  
19    private final Text text = new Text();
20  
21    public Text evaluate(Text field) {
22      if (field == null) {
23        text.set("");
24      } else {

26      }
27      return text;
28    }
