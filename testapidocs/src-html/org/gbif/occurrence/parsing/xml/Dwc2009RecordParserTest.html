001package org.gbif.occurrence.parsing.xml;
002
003import org.gbif.occurrence.model.IdentifierRecord;
004import org.gbif.occurrence.model.RawOccurrenceRecord;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009
010public class Dwc2009RecordParserTest extends ParserTestCase {
011
012  @Test
013  public void testParseBasicFields() {
014    String fileName = getClass().getResource("/responses/dwc2009/dwc2009_simple_fields.gz").getFile();
015    RawOccurrenceRecord ror = setupRor(fileName).get(0);
016
017    // common to all schemas
018    assertEquals("BGBM", ror.getInstitutionCode());
019    assertEquals("Pontaurus", ror.getCollectionCode());
020    assertEquals("specimen", ror.getBasisOfRecord());
021    assertEquals("2360", ror.getMinAltitude());
022
023    // common to dwc
024    assertEquals("Acantholimon armenum var. armenum", ror.getScientificName());
025    assertEquals("Markus Döring", ror.getCollectorName());
026    assertEquals("Markus Döring", ror.getIdentifierName());
027    assertEquals("Plantae", ror.getKingdom());
028    assertEquals("Magnoliophyta", ror.getPhylum());
029    assertEquals("Magnoliopsida", ror.getKlass());
030    assertEquals("Plumbaginales", ror.getOrder());
031    assertEquals("Plumbaginaceae", ror.getFamily());
032    assertEquals("Acantholimon", ror.getGenus());
033
034    // dwc 1.4/2009 only
035    assertEquals("2004", ror.getCatalogueNumber());
036    assertEquals("Turkey", ror.getCountry());
037    assertEquals("29", ror.getDay());
038    assertEquals("7", ror.getMonth());
039    assertEquals("1999", ror.getYear());
040    assertEquals("1999-7-29", ror.getOccurrenceDate());
041    assertEquals("37.42", ror.getLatitude());
042    assertEquals("34.568", ror.getLongitude());
043
044    IdentifierRecord id = ror.getIdentifierRecords().get(0);
045    assertEquals("142316220", id.getIdentifier());
046
047    /** TODO: more complete sample */
048  }
