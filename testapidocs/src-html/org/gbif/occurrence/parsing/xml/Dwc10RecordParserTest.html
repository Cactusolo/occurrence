001package org.gbif.occurrence.parsing.xml;
002
003import org.gbif.occurrence.model.IdentifierRecord;
004import org.gbif.occurrence.model.RawOccurrenceRecord;
005import org.gbif.occurrence.model.TypificationRecord;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class Dwc10RecordParserTest extends ParserTestCase {
012
013  @Test
014  public void testParseBasicFields() {
015    String fileName = getClass().getResource("/responses/dwc10/dwc10_all_simple_fields.gz").getFile();
016    RawOccurrenceRecord ror = setupRor(fileName).get(0);
017
018    // System.out.println(ror.debugDump());
019    // common to all schemas
020    assertEquals("BR", ror.getInstitutionCode());
021    assertEquals("MYCO", ror.getCollectionCode());
022    assertEquals("S", ror.getBasisOfRecord());
023
024    // common to dwc
025    assertEquals("Amaurochaete", ror.getScientificName());
026    assertEquals("Protista", ror.getKingdom());
027    assertEquals("Myxomycota", ror.getPhylum());
028    assertEquals("Myxomycetes", ror.getKlass());
029    assertEquals("STEMONITALES", ror.getOrder());
030    assertEquals("STEMONITIDACEAE", ror.getFamily());
031    assertEquals("Amaurochaete", ror.getGenus());
032    assertEquals("Mannenga-Bremekamp N.", ror.getCollectorName());
033    assertEquals("Wetering; Beukenburg - Beukenlaan", ror.getLocality());
034    assertEquals("NL", ror.getCountry());
035
036    // dwc 1.0 only
037    assertEquals("1", ror.getCatalogueNumber());
038    assertEquals("52.755", ror.getLatitude());
039    assertEquals("5.9963", ror.getLongitude());
040    assertEquals("1958-8-15", ror.getOccurrenceDate());
041  }
042
043  @Test
044  public void testParseIdentifiers() {
045    String fileName = getClass().getResource("/responses/dwc10/dwc10_idtype.gz").getFile();
046    RawOccurrenceRecord ror = setupRor(fileName).get(0);
047
048    // showIdentifiers(ror);
049    assertEquals(1, ror.getIdentifierRecords().size());
050    IdentifierRecord idRec = ror.getIdentifierRecords().get(0);
051    assertEquals(3, idRec.getIdentifierType().intValue());
052    assertEquals("2873", idRec.getIdentifier());
053  }
054
055  @Test
056  public void testParseTypification() {
057    String fileName = getClass().getResource("/responses/dwc10/dwc10_typification.gz").getFile();
058    RawOccurrenceRecord ror = setupRor(fileName).get(0);
059
060    // showTypifications(ror);
061    assertEquals(1, ror.getTypificationRecords().size());
062    TypificationRecord typRecord = ror.getTypificationRecords().iterator().next();
063    assertEquals("Isotype", typRecord.getTypeStatus());
064  }
065}
