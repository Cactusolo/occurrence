045  @Ignore("too expensive for constant use")
046  @Test
047  public void testMultiThreadLoad() throws Exception {
048    String xml = Resources.toString(Resources.getResource("id_extraction/abcd1_umlaut.xml"), Charsets.UTF_8);
049
050    // 5 parsers in 5 threads parse 1000 records each
051    List<RawXmlOccurrence> raws = new ArrayList<RawXmlOccurrence>();
052    for (int i = 0; i < 5000; i++) {
053      raws.add(createFakeOcc(xml));
054    }
055
056    ExecutorService tp = Executors.newFixedThreadPool(5);
057    List<Future<List<RawOccurrenceRecord>>> futures = new ArrayList<Future<List<RawOccurrenceRecord>>>();
058    for (int i = 0; i < 5; i++) {
059      int start = i * 1000;
060      int end = start + 1000;
061      futures.add(tp.submit(new BatchRecordParser(raws.subList(start, end))));
062    }
063
064    List<RawOccurrenceRecord> rors = new ArrayList<RawOccurrenceRecord>();
065    for (Future<List<RawOccurrenceRecord>> future : futures) {
066      rors.addAll(future.get());
067    }
068
069    assertEquals(5000, rors.size());
070  }
071
072  private class BatchRecordParser implements Callable<List<RawOccurrenceRecord>> {
073
074    private List<RawXmlOccurrence> raws;
075
076    public BatchRecordParser(List<RawXmlOccurrence> raws) {
077      this.raws = raws;
078    }
079
080    public List<RawOccurrenceRecord> call() {
081      List<RawOccurrenceRecord> rors = new ArrayList<RawOccurrenceRecord>();
082      for (RawXmlOccurrence raw : raws) {
083        rors.addAll(XmlFragmentParser.parseRecord(raw));
084      }
085      return rors;
086    }
087  }

089  @Test
090  public void testIdExtractionSimple() throws IOException {
091    String xml = Resources.toString(Resources.getResource("id_extraction/abcd1_simple.xml"), Charsets.UTF_8);
092    UUID datasetKey = UUID.randomUUID();
093    HolyTriplet target =
094      new HolyTriplet(datasetKey, "TLMF", "Tiroler Landesmuseum Ferdinandeum", "82D45C93-B297-490E-B7B0-E0A9BEED1326",
095        null);
096    byte[] xmlBytes = xml.getBytes(Charset.forName("UTF8"));
097    Set<IdentifierExtractionResult> extractionResults =
098      XmlFragmentParser.extractIdentifiers(datasetKey, xmlBytes, OccurrenceSchemaType.ABCD_1_2, true,
099        true);
100    Set<UniqueIdentifier> ids = extractionResults.iterator().next().getUniqueIdentifiers();
101    assertEquals(1, ids.size());
102    UniqueIdentifier id = ids.iterator().next();
103    assertEquals(datasetKey, id.getDatasetKey());
104    assertEquals(OccurrenceKeyHelper.buildKey(target), id.getUniqueString());
105  }

107  @Test
108  public void testIdExtractionMultipleIdsUnitQualifier() throws IOException {
109    String xml = Resources.toString(Resources.getResource("id_extraction/abcd2_multi.xml"), Charsets.UTF_8);
110
111    UUID datasetKey = UUID.randomUUID();
112    byte[] xmlBytes = xml.getBytes(Charset.forName("UTF8"));
113    Set<IdentifierExtractionResult> extractionResults =
114      XmlFragmentParser.extractIdentifiers(datasetKey, xmlBytes, OccurrenceSchemaType.ABCD_2_0_6, true,
115        true);
116    HolyTriplet holyTriplet1 =
117      new HolyTriplet(datasetKey, "BGBM", "Bridel Herbar", "Bridel-1-428", "Grimmia alpicola Sw. ex Hedw.");
118    HolyTriplet holyTriplet2 = new HolyTriplet(datasetKey, "BGBM", "Bridel Herbar", "Bridel-1-428",
119      "Schistidium agassizii Sull. & Lesq. in Sull.");
120    assertEquals(2, extractionResults.size());
121    for (IdentifierExtractionResult result : extractionResults) {
122      String uniqueId = result.getUniqueIdentifiers().iterator().next().getUniqueString();
123      assertTrue(uniqueId.equals(OccurrenceKeyHelper.buildKey(holyTriplet1)) || uniqueId
124        .equals(OccurrenceKeyHelper.buildKey(holyTriplet2)));
125    }
126  }

143        .equals(OccurrenceKeyHelper.buildKey(pubProvided)));
