001package org.gbif.occurrence.parsing.xml;
002
003import org.gbif.occurrence.model.IdentifierRecord;
004import org.gbif.occurrence.model.RawOccurrenceRecord;
005import org.gbif.occurrence.model.TypificationRecord;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class DwcManisRecordParserTest extends ParserTestCase {
012
013  @Test
014  public void testParseBasicFields() {
015    String fileName = getClass().getResource("/responses/dwc_manis/dwc_manis_all_simple_fields.gz").getFile();
016    RawOccurrenceRecord ror = setupRor(fileName).get(0);
017
018    // System.out.println(ror.debugDump());
019    // common to all schemas
020    assertEquals("ANSP", ror.getInstitutionCode());
021    assertEquals("HRP", ror.getCollectionCode());
022    assertEquals("PreservedSpecimen", ror.getBasisOfRecord());
023
024    // common to dwc
025    assertEquals("Gyrinophilus porphyriticus porphyriticus", ror.getScientificName());
026    assertEquals("Amphibia", ror.getKlass());
027    assertEquals("Caudata", ror.getOrder());
028    assertEquals("Plethodontidae", ror.getFamily());
029    assertEquals("Gyrinophilus", ror.getGenus());
030    assertEquals("E. S. & W. I. Mattin", ror.getCollectorName());
031    assertEquals("New Jersey", ror.getStateOrProvince());
032    assertEquals("USA", ror.getCountry());
033    assertEquals("Sussex Co.", ror.getCounty());
034
035    // dwc manis only
036    assertEquals("513", ror.getCatalogueNumber());
037    assertEquals("17-21-10", ror.getOccurrenceDate());
038  }
039
040  @Test
041  public void testParseIdentifiers() {
042    String fileName = getClass().getResource("/responses/dwc_manis/dwc_manis_idtype.gz").getFile();
043    RawOccurrenceRecord ror = setupRor(fileName).get(0);
044
045    // System.out.println(ror.debugDump());
046    // showIdentifiers(ror);
047    assertEquals(1, ror.getIdentifierRecords().size());
048    IdentifierRecord idRec = ror.getIdentifierRecords().get(0);
049    assertEquals(3, idRec.getIdentifierType().intValue());
050    assertEquals("SZ 233", idRec.getIdentifier());
051  }
052
053  @Test
054  public void testParseTypification() {
055    String fileName = getClass().getResource("/responses/dwc_manis/dwc_manis_typification.gz").getFile();
056    RawOccurrenceRecord ror = setupRor(fileName).get(0);
057
058    /** TODO: is this a valid TypificationRecord? */
059    // showTypifications(ror);

061    TypificationRecord typRecord = ror.getTypificationRecords().iterator().next();
062    assertEquals("0", typRecord.getTypeStatus());
063  }
