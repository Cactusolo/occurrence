011  @Test
012  public void testParseBasicFields() {
013    String fileName = getClass().getResource("/responses/dwc14/dwc14_all_simple_fields.gz").getFile();
014    RawOccurrenceRecord ror = setupRor(fileName).get(0);
015
016    // System.out.println(ror.debugDump());
017    // common to all schemas
018    assertEquals("UGENT", ror.getInstitutionCode());
019    assertEquals("vertebrata", ror.getCollectionCode());
020    assertEquals("PreservedSpecimen", ror.getBasisOfRecord());
021    assertEquals("200", ror.getMinAltitude());
022    assertEquals("400", ror.getMaxAltitude());
023    assertEquals("25", ror.getLatLongPrecision());
024
025    // common to dwc
026    assertEquals("Alouatta villosa Gray, 1845", ror.getScientificName());
027    assertEquals("Gray, 1845", ror.getAuthor());
028    assertEquals("Animalia", ror.getKingdom());
029    assertEquals("Chordata", ror.getPhylum());
030    assertEquals("Mammalia", ror.getKlass());
031    assertEquals("Primates", ror.getOrder());
032    assertEquals("Atelidae", ror.getFamily());
033    assertEquals("Alouatta", ror.getGenus());
034    assertEquals("villosa", ror.getSpecies());
035
036    // dwc 1.4 only
037    assertEquals("50058", ror.getCatalogueNumber());
038    /** TODO: more complete sample */
039  }
