038    assertEquals("Tiroler Landesmuseum Ferdinandeum", ror.getCollectionCode());
039    assertEquals("AUT", ror.getCountry());
040    assertEquals("Osch√ºtz", ror.getCollectorName());
041    assertEquals("Seefeld", ror.getLocality());
042    assertEquals("82D45C93-B297-490E-B7B0-E0A9BEED1326", ror.getCatalogueNumber());
043    assertEquals("47.3303167192", ror.getLatitude());
044    assertEquals("11.2041081855", ror.getLongitude());
045    assertEquals("1999-12-31", ror.getOccurrenceDate());
046  }

065  @Test
066  public void testParseMultiIdentificationPreferredTrue() {
067    String fileName = getClass().getResource("/responses/abcd12/abcd12_multi_ids_preferred_true.gz").getFile();
068    List<RawOccurrenceRecord> rors = setupRor(fileName);
069    assertEquals(1, rors.size());

071    RawOccurrenceRecord ror = rors.get(0);
072    // System.out.println(ror.debugDump());

088  @Test
089  public void testParseTypification() {
090    String fileName = getClass().getResource("/responses/abcd12/abcd12_typification.gz").getFile();
091    RawOccurrenceRecord ror = setupRor(fileName).get(0);

093    assertEquals(1, ror.getTypificationRecords().size());
