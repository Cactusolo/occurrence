016  @Test
017  public void testParseBasicFields() {
018    String fileName = getClass().getResource("/responses/abcd206/abcd206_all_simple_fields.gz").getFile();
019    RawOccurrenceRecord ror = setupRor(fileName).get(0);
020
021    // System.out.println(ror.debugDump());
022    assertEquals("BGBM", ror.getInstitutionCode());
023    assertEquals("AlgaTerra", ror.getCollectionCode());
024    assertEquals("HumanObservation", ror.getBasisOfRecord());
025    assertEquals("DE", ror.getCountry());
026    assertEquals("Kusber, W.-H.", ror.getCollectorName());
027    assertEquals("Nikolassee, Berlin", ror.getLocality());
028    assertEquals("5834", ror.getCatalogueNumber());
029    assertEquals("52.423798", ror.getLatitude());
030    assertEquals("13.191434", ror.getLongitude());
031    assertEquals("1987-04-13T00:00:00", ror.getOccurrenceDate());
032    assertEquals("400", ror.getMinAltitude());
033    assertEquals("500", ror.getMaxAltitude());
034    assertEquals("50", ror.getLatLongPrecision());
035  }
