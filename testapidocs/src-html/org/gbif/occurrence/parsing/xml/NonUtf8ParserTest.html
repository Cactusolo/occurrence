001package org.gbif.occurrence.parsing.xml;
002
003import org.gbif.occurrence.parsing.RawXmlOccurrence;
004
005import java.io.File;
006import java.util.List;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011
012public class NonUtf8ParserTest extends ParserTestCase {
013
014  @Test
015  public void testParseIso8859_1() {
016    String fileName = getClass().getResource("/responses/problematic/dwc_manis_iso8859-1.gz").getFile();
017    List<RawXmlOccurrence> records = occurrenceParser.parseResponseFileToRawXml(new File(fileName));
018    assertEquals(900, records.size());
019  }
020
021  @Test
022  public void testParseNoXmlDeclaration() {
023    String fileName = getClass().getResource("/responses/problematic/dwc_manis_no_encoding.gz").getFile();
024
025    List<RawXmlOccurrence> records = occurrenceParser.parseResponseFileToRawXml(new File(fileName));
026    assertEquals(900, records.size());
027  }
028
029  @Test
030  public void testParseUtf8WrongEncodingGiven() {
031    // file is really iso8859-1 but claims to be utf8
032    String fileName = getClass().getResource("/responses/problematic/dwc_10_utf8_badcase.gz").getFile();
033    List<RawXmlOccurrence> records = occurrenceParser.parseResponseFileToRawXml(new File(fileName));
034    assertEquals(21, records.size());
035  }
036
037}
