018import org.gbif.occurrence.OccurrenceParser;

031
032  OccurrenceParser occParser;
033
034  @Before
035  public void setUp() {
036    occParser = new OccurrenceParser();
037  }
038
039  @Test
040  public void testArchiveUncompressed() {
041    String fileName = getClass().getResource("/dwca/plants_dwca").getFile();
042    List<RawOccurrenceRecord> records = new ArrayList<RawOccurrenceRecord>();
043    for (RawOccurrenceRecord record : occParser.parseDwca(new File(fileName))) {
044      records.add(record);
045    }
046    assertEquals(307, records.size());
047  }
048
049  @Test

057  }

088
089  @Test
090  public void testDwcaParserStartingAtDwcaParser() {
091    String fileName = getClass().getResource("/dwca/plants_dwca").getFile();
092    BufferedDwcaParser parser = new BufferedDwcaParser(new File(fileName), 100);
093    List<RawOccurrenceRecord> records = new ArrayList<RawOccurrenceRecord>();
094    for (RawOccurrenceRecord record : parser) {
095      records.add(record);
096    }
097    assertEquals(307, records.size());
098  }
