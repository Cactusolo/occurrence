016package org.gbif.occurrence.parsing.response_file;
017
018import org.gbif.api.vocabulary.OccurrenceSchemaType;
019
020import java.io.IOException;
021import javax.xml.parsers.DocumentBuilder;
022import javax.xml.parsers.DocumentBuilderFactory;
023import javax.xml.parsers.ParserConfigurationException;
024
025import com.google.common.io.Resources;
026import org.apache.commons.io.Charsets;
027import org.junit.Before;
028import org.junit.Test;
029import org.w3c.dom.Document;
030import org.w3c.dom.Element;
031
032import static org.junit.Assert.assertEquals;
033
034public class ResponseSchemaDetectorTest {
035
036  private ResponseSchemaDetector detector;
037  private DocumentBuilder docBuilder;
038  private Document doc;
039  private Element root;
040
041  @Before
042  public void setUp() throws ParserConfigurationException {
043    detector = new ResponseSchemaDetector();
044    docBuilder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
045    doc = docBuilder.newDocument();
046    root = doc.createElement("occurrence");
047    doc.appendChild(root);
048  }
049
050  @Test
051  public void testAbcd1() throws IOException {
052    String xml = Resources.toString(Resources.getResource("response_schema/abcd1.xml"), Charsets.UTF_8);
053    OccurrenceSchemaType result = detector.detectSchema(xml);
054    assertEquals(OccurrenceSchemaType.ABCD_1_2, result);
055  }
056
057  @Test
058  public void testAbcd2() throws IOException {
059    String xml = Resources.toString(Resources.getResource("response_schema/abcd2.xml"), Charsets.UTF_8);
060    OccurrenceSchemaType result = detector.detectSchema(xml);
061    assertEquals(OccurrenceSchemaType.ABCD_2_0_6, result);
062  }
063
064  @Test
065  public void testDwc1_0() throws IOException {
066    String xml = Resources.toString(Resources.getResource("response_schema/dwc_1_0.xml"), Charsets.UTF_8);

069  }
