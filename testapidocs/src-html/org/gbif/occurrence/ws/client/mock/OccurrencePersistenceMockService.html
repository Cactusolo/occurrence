001package org.gbif.occurrence.ws.client.mock;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.api.service.occurrence.OccurrenceService;
005import org.gbif.api.vocabulary.BasisOfRecord;
006import org.gbif.api.vocabulary.Kingdom;
007import org.gbif.dwc.terms.DwcTerm;
008import org.gbif.dwc.terms.Term;
009
010import java.util.List;
011import java.util.Map;
012import java.util.UUID;
013
014import com.google.common.collect.Lists;
015import org.apache.commons.lang3.StringUtils;
016
017
018public class OccurrencePersistenceMockService implements OccurrenceService {
019
020  protected static final List<UUID> DATASETS = Lists.newArrayList();
021  private static final String UUID_PREFIX = "111aaa11-0000-1111-2222-f5f5f5d8888";
022
023  static {
024    for (int i = 0; i < 10; i++) {
025      DATASETS.add(UUID.fromString(UUID_PREFIX + i));
026    }
027  }
028
029  @Override
030  public Occurrence get(Integer key) {
031    if (key == null || key < 1 || key >= 1000000) {
032      return null;
033    }
034
035    Occurrence occ = new Occurrence();
036    occ.setKey(key);
037    occ.setDatasetKey(DATASETS.get(key % 10));
038
039    int ord = key % BasisOfRecord.values().length;
040    occ.setBasisOfRecord(BasisOfRecord.values()[ord]);
041    occ.setYear(1800 + (key % 200));
042    occ.setAltitude(key % 2000);
043    Map<Term, String> fields = occ.getFields();
044    fields.put(DwcTerm.catalogNumber, "cat-" + key);
045    occ.setFields(fields);
046    occ.setMonth(1 + (key % 12));
047
048    ord = key % Kingdom.values().length;
049    Kingdom k = Kingdom.values()[ord];
050    occ.setKingdomKey(k.nubUsageID());
051    occ.setKingdom(StringUtils.capitalize(k.name().toLowerCase()));
052
053    return occ;
054  }
055
056  @Override
057  public String getFragment(int i) {
058    return "<record>mock parsing</record>";
059  }
060}
