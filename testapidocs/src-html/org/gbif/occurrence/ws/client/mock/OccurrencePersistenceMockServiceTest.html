001package org.gbif.occurrence.ws.client.mock;
002
003import org.gbif.api.model.occurrence.Occurrence;
004import org.gbif.api.vocabulary.BasisOfRecord;
005import org.gbif.dwc.terms.DwcTerm;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class OccurrencePersistenceMockServiceTest {
012
013  @Test
014  public void testGenerator() {
015    OccurrencePersistenceMockService srv = new OccurrencePersistenceMockService();
016    Occurrence occ = srv.get(112);
017    assertEquals((Integer) 112, occ.getKey());
018    assertEquals(OccurrencePersistenceMockService.DATASETS.get(2), occ.getDatasetKey());
019    assertEquals("cat-112", occ.getField(DwcTerm.catalogNumber));
020    assertEquals("Chromista", occ.getKingdom());
021    assertEquals((Integer) 4, occ.getKingdomKey());
022    assertEquals(BasisOfRecord.PRESERVED_SPECIMEN, occ.getBasisOfRecord());
023    assertEquals((Integer) 1912, occ.getYear());
024    assertEquals((Integer) 5, occ.getMonth());
025    assertEquals((Integer) 112, occ.getAltitude());
026  }
027}
