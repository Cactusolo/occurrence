001package org.gbif.occurrence.download.util;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.terms.GbifTerm;
005import org.gbif.occurrencestore.common.model.constants.FieldName;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class HiveFieldUtilTest {
012
013  @Test
014  public void testGetHiveField() throws Exception {
015    assertEquals("id", HiveFieldUtil.getHiveField(FieldName.ID));
016    assertEquals("verbatim_basis_of_record", HiveFieldUtil.getHiveField(FieldName.BASIS_OF_RECORD));
017    assertEquals("verbatim_class", HiveFieldUtil.getHiveField(FieldName.CLASS));
018    assertEquals("verbatim_order", HiveFieldUtil.getHiveField(FieldName.ORDER));
019    assertEquals("class_", HiveFieldUtil.getHiveField(FieldName.I_CLASS));
020    assertEquals("order_", HiveFieldUtil.getHiveField(FieldName.I_ORDER));
021    assertEquals("taxon_id", HiveFieldUtil.getHiveField(FieldName.I_NUB_ID));
022    assertEquals("mod360_cell_id", HiveFieldUtil.getHiveField(FieldName.I_MOD360_CELL_ID));
023    assertEquals("unit_qualifier", HiveFieldUtil.getHiveField(FieldName.UNIT_QUALIFIER));
024    assertEquals("created", HiveFieldUtil.getHiveField(FieldName.CREATED));
025    assertEquals("geospatial_issue", HiveFieldUtil.getHiveField(FieldName.I_GEOSPATIAL_ISSUE));
026  }
027
028  @Test
029  public void testGetSimpleTermName() throws Exception {
030    assertEquals("occurrenceID", HiveFieldUtil.getTerm(FieldName.ID).simpleName());
031    assertEquals("verbatimBasisOfRecord", HiveFieldUtil.getTerm(FieldName.BASIS_OF_RECORD).simpleName());
032    assertEquals("verbatimClass", HiveFieldUtil.getTerm(FieldName.CLASS).simpleName());
033    assertEquals("verbatimOrder", HiveFieldUtil.getTerm(FieldName.ORDER).simpleName());
034    assertEquals("class", HiveFieldUtil.getTerm(FieldName.I_CLASS).simpleName());
035    assertEquals("order", HiveFieldUtil.getTerm(FieldName.I_ORDER).simpleName());
036    assertEquals("taxonID", HiveFieldUtil.getTerm(FieldName.I_NUB_ID).simpleName());
037    assertEquals("unitQualifier", HiveFieldUtil.getTerm(FieldName.UNIT_QUALIFIER).simpleName());
038    assertEquals("created", HiveFieldUtil.getTerm(FieldName.CREATED).simpleName());
039  }
040
041  @Test
042  public void testGetTermName() throws Exception {
043    assertEquals(DwcTerm.occurrenceID, HiveFieldUtil.getTerm(FieldName.ID));
044    assertEquals(DwcTerm.basisOfRecord, HiveFieldUtil.getTerm(FieldName.I_BASIS_OF_RECORD));
045    assertEquals(GbifTerm.verbatimBasisOfRecord, HiveFieldUtil.getTerm(FieldName.BASIS_OF_RECORD));
046    assertEquals(GbifTerm.verbatimClass, HiveFieldUtil.getTerm(FieldName.CLASS));
047    assertEquals(GbifTerm.verbatimOrder, HiveFieldUtil.getTerm(FieldName.ORDER));
048    assertEquals(DwcTerm.class_, HiveFieldUtil.getTerm(FieldName.I_CLASS));
049    assertEquals(DwcTerm.order, HiveFieldUtil.getTerm(FieldName.I_ORDER));
050    assertEquals(DwcTerm.taxonID, HiveFieldUtil.getTerm(FieldName.I_NUB_ID));
051    assertEquals(GbifTerm.unitQualifier, HiveFieldUtil.getTerm(FieldName.UNIT_QUALIFIER));
052    assertEquals(GbifTerm.created, HiveFieldUtil.getTerm(FieldName.CREATED));
053  }
054}
