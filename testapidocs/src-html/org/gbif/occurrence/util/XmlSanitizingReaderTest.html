033  private String doSingleReads(String test) throws IOException {
034    StringReader reader = new StringReader(test);
035    XmlSanitizingReader xmlReader = new XmlSanitizingReader(reader);
036
037    StringBuilder sb = new StringBuilder();
038    while (xmlReader.ready()) {
039      int nextIntChar = xmlReader.read();
040      char nextChar = (char) nextIntChar;
041      if (nextIntChar != -1) sb.append(nextChar);
042    }
043    String result = sb.toString();
044
045    // System.out.println("Single reads result [" + result + "]");
046    return result;
047  }
048
049  private String doSimpleBufferRead(String test) throws IOException {
050    StringReader reader = new StringReader(test);
051    XmlSanitizingReader xmlReader = new XmlSanitizingReader(reader);
052
053    StringBuilder sb = new StringBuilder();
054    while (xmlReader.ready()) {
055      char[] buffer = new char[12];
056      xmlReader.read(buffer);
057      sb.append(new String(buffer));
058    }
059    String result = sb.toString().trim();
060
061    // System.out.println("Simple buffer read result [" + result + "]");
062    return result;
063  }
064
065  private String doOffsetBufferRead(String test) throws IOException {
066    StringReader reader = new StringReader(test);
067    XmlSanitizingReader xmlReader = new XmlSanitizingReader(reader);
068
069    StringBuilder sb = new StringBuilder();
070    int totalCharsRead = 0;
071    while (xmlReader.ready()) {
072      char[] buffer = new char[12];
073      int charsRead = xmlReader.read(buffer, 0, 12);
074      totalCharsRead += charsRead;
075      sb.append(new String(buffer));
076    }
077    String result = sb.toString().trim();
078
079    // System.out.println("Offset buffer read result [" + result + "]");
080    return result;
081  }

164  @Test
165  public void testBadXmlFileRead() throws Exception {
166    String fileName = getClass().getResource("/responses/problematic/spanish_bad_xml.gz").getFile();
167    File file = new File(fileName);
168    FileInputStream fis = new FileInputStream(file);
169    GZIPInputStream inputStream = new GZIPInputStream(fis);
170
171    StringBuilder sb = new StringBuilder();
172    XmlSanitizingReader xmlReader = new XmlSanitizingReader(new InputStreamReader(inputStream, "UTF-8"));
173    while (xmlReader.ready()) {
174      char[] buff = new char[8192];
175      xmlReader.read(buff, 0, 8192);
176      sb.append(buff);
177    }
178    assertEquals(6210, sb.toString().trim().length());
179  }
