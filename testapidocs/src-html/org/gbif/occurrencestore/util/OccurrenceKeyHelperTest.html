001package org.gbif.occurrencestore.util;
002
003import org.gbif.occurrencestore.common.model.HolyTriplet;
004import org.gbif.occurrencestore.common.model.PublisherProvidedUniqueIdentifier;
005
006import java.util.UUID;
007
008import org.junit.Assert;
009import org.junit.Test;
010
011public class OccurrenceKeyHelperTest {
012
013  private static final UUID DATASET_KEY = UUID.randomUUID();
014  private static final String IC = "BGBM";
015  private static final String CC = "Vascular Plants";
016  private static final String CN = "00234-asdfa-234as-asdf-cvb";
017  private static final String UQ = "Abies alba";
018  private static final String DWC = "98098098-234asd-asdfa-234df";
019
020  private static final HolyTriplet TRIPLET = new HolyTriplet(DATASET_KEY, IC, CC, CN, UQ);
021  private static final PublisherProvidedUniqueIdentifier PUB_PROVIDED = new PublisherProvidedUniqueIdentifier(DATASET_KEY, DWC);
022
023  @Test
024  public void testTripletKey() {
025    String testKey = OccurrenceKeyHelper.buildKey(TRIPLET);
026    Assert.assertTrue(testKey.equals(DATASET_KEY.toString() + "|" + IC + "|" + CC + "|" + CN + "|" + UQ));
027  }
028
029  @Test
030  public void testSingleDwcKey() {
031    String testKey = OccurrenceKeyHelper.buildKey(PUB_PROVIDED);
032    Assert.assertTrue(testKey.equals(DATASET_KEY.toString() + "|" + DWC));
033  }
034}
