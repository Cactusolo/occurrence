016  @Test
017  public void testLookupCountryMatch() {
018    Double lat = 43.65;
019    Double lng = -79.40;
020    String country = "CA";
021    CoordinateInterpretationResult result =
022      CoordinateInterpreter.interpretCoordinates(lat.toString(), lng.toString(), country);
023    assertEquals(lat, result.getLatitude().doubleValue(), 0.0001);
024    assertEquals(lng, result.getLongitude().doubleValue(), 0.0001);
025    assertEquals(country, result.getCountryCode());
026    assertEquals(0, result.getGeoSpatialIssue().intValue());
027  }

106  @Test
107  public void testRoundPrecision() {
108    Double lat = 43.6512345;
109    Double lng = -79.4056789;
110    // expect max 5 decimals
111    double roundedLat = Math.round(lat * Math.pow(10, 5)) / Math.pow(10, 5);
112    double roundedLng = Math.round(lng * Math.pow(10, 5)) / Math.pow(10, 5);
113    String country = "CA";
114    CoordinateInterpretationResult result =
115      CoordinateInterpreter.interpretCoordinates(lat.toString(), lng.toString(), country);
116    assertEquals(roundedLat, result.getLatitude(), 0.000001);
117    assertEquals(roundedLng, result.getLongitude(), 0.000001);
118    assertEquals(country, result.getCountryCode());
119    assertEquals(0, result.getGeoSpatialIssue().intValue());
120  }

128  @Test
129  public void testFuzzyCountry() {
130    // Belfast is GB not IE
131    Double lat = 54.597;
132    Double lng = -5.93;
133    String country = "IE";
134    CoordinateInterpretationResult result =
135      CoordinateInterpreter.interpretCoordinates(lat.toString(), lng.toString(), country);
136    assertEquals("GB", result.getCountryCode());
137    assertEquals(0, result.getGeoSpatialIssue().intValue());
138
139    // Isle of Man is IM not GB
140    lat = 54.25;
141    lng = -4.5;
142    country = "GB";
143    result = CoordinateInterpreter.interpretCoordinates(lat.toString(), lng.toString(), country);
144    assertEquals("IM", result.getCountryCode());
145    assertEquals(0, result.getGeoSpatialIssue().intValue());
146  }
