001package org.gbif.occurrencestore.interpreters;
002
003import org.gbif.api.model.registry.Organization;
004import org.gbif.api.vocabulary.Country;
005
006import java.util.UUID;
007
008import org.junit.Ignore;
009import org.junit.Test;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNull;
013
014@Ignore("requires live webservice")
015public class OrganizationLookupTest {
016
017  private static final String BGBM_KEY = "57254bd0-8256-11d8-b7ed-b8a03c50a862";
018  private static final String BOGART_DATASET_KEY = "85697f04-f762-11e1-a439-00145eb45e9a";
019
020  @Test
021  public void testOrgLookup(){
022    Organization org = OrganizationLookup.getOrgByDataset(UUID.fromString(BOGART_DATASET_KEY));
023    assertEquals(BGBM_KEY, org.getKey().toString());
024  }
025
026  @Test
027  public void testCountryLookup() {
028    Country result = OrganizationLookup.getOrgCountry(UUID.fromString(BGBM_KEY));
029    assertEquals(Country.GERMANY, result);
030  }
031
032  @Test
033  public void testBadCountryLookup() {
034    Country result = OrganizationLookup.getOrgCountry(UUID.randomUUID());
035    assertNull(result);
036  }
037}
