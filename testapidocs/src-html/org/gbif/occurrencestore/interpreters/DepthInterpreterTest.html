001package org.gbif.occurrencestore.interpreters;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertNotNull;
007import static org.junit.Assert.assertNull;
008
009public class DepthInterpreterTest {
010
011  @Test
012  public void testGoodDepthNoPrecision() {
013    String min = "10";
014    String max = "100";
015    String precision = "0";
016    Integer result = DepthInterpreter.interpretDepth(min, max, precision);
017    assertEquals(55l, result.longValue());
018  }
019
020  @Test
021  public void testGoodDepthWithPrecision() {
022    // in current implementation precision isn't used
023    String min = "10";
024    String max = "100";
025    String precision = "150";
026    Integer result = DepthInterpreter.interpretDepth(min, max, precision);
027    assertEquals(55l, result.longValue());
028  }
029
030  @Test
031  public void testRoundTo0Depth() {
032    String min = "0";
033    String max = "1";
034    String precision = "0";
035    Integer result = DepthInterpreter.interpretDepth(min, max, precision);
036    assertEquals(0l, result.longValue());
037  }
038
039  @Test
040  public void testReversedDepth() {
041    String min = "100";
042    String max = "10";
043    String precision = "0";
044    Integer result = DepthInterpreter.interpretDepth(min, max, precision);
045    assertEquals(55l, result.longValue());
046  }
047
048  @Test
049  public void testSomeNulls() {
050    Integer result = DepthInterpreter.interpretDepth("200", null, null);
051    assertNotNull(result);
052    assertEquals(200, result.intValue());
053
054    result = DepthInterpreter.interpretDepth(null, "300", null);
055    assertNotNull(result);
056    assertEquals(300, result.intValue());
057  }
058
059  @Test
060  public void testAllNulls() {
061    Integer result = DepthInterpreter.interpretDepth(null, null, null);
062    assertNull(result);
063  }
064}
