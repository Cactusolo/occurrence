083  @Before
084  public void setUp() throws Exception {
085    ConnectionParameters connectionParams = new ConnectionParameters("localhost", 5672, "guest", "guest", "/");
086    messagePublisher =
087      new DefaultMessagePublisher(connectionParams);
088    messageListener = new MessageListener(connectionParams);
089
090    zkServer = new TestingServer();
091    curator = CuratorFrameworkFactory.builder().connectString(zkServer.getConnectString()).namespace("crawlertest")
092      .retryPolicy(new RetryNTimes(1, 1000)).build();
093    curator.start();
094    zookeeperConnector = new ZookeeperConnector(curator);
095
096    long now = System.currentTimeMillis();
097    fragmentProcessor =
098      new FragmentProcessor(fragmentPersister, occurrenceKeyService, messagePublisher, zookeeperConnector);
099    occurrenceFragmentedListener = new OccurrenceFragmentedListener(fragmentProcessor);
100    messageListener.listen("occ_frag_test_" + now, 1, occurrenceFragmentedListener);
101    verbatimProcessor =
102      new VerbatimProcessor(fragmentPersister, verbatimPersister, messagePublisher, zookeeperConnector);
103    fragmentPersistedListener = new FragmentPersistedListener(verbatimProcessor);
104    messageListener.listen("frag_persisted_test_" + now, 1, fragmentPersistedListener);
105    verbatimInterpreter = new VerbatimOccurrenceInterpreter(occurrenceService, zookeeperConnector);
106    interpretedProcessor = new InterpretedProcessor(fragmentPersister, verbatimInterpreter, verbatimPersister, messagePublisher, zookeeperConnector);
107    verbatimPersistedListener = new VerbatimPersistedListener(interpretedProcessor);
108    messageListener.listen("verb_persisted_test_" + now, 1, verbatimPersistedListener);
109  }

162  @Test
163  public void testEndToEndZkCounts() throws IOException, InterruptedException {
164    UUID datasetKey = UUID.randomUUID();
165
166    // one abcd206
167    OccurrenceSchemaType schemaType = OccurrenceSchemaType.ABCD_2_0_6;
168    Integer crawlId = 1;
169    OccurrenceFragmentedMessage msg =
170      new OccurrenceFragmentedMessage(datasetKey, crawlId, abcd206Single.getBytes(), schemaType, EndpointType.BIOCASE,
171        null);
172    messagePublisher.send(msg);
173
174    // two in one abcd 2
175    schemaType = OccurrenceSchemaType.ABCD_2_0_6;
176    crawlId = 1;
177    msg =
178      new OccurrenceFragmentedMessage(datasetKey, crawlId, abcd206Multi.getBytes(), schemaType, EndpointType.BIOCASE,
179        null);
180    messagePublisher.send(msg);
181
182    // one dwc 1.4
183    schemaType = OccurrenceSchemaType.DWC_1_4;
184    crawlId = 1;
185    msg =
186      new OccurrenceFragmentedMessage(datasetKey, crawlId, dwc14.getBytes(), schemaType, EndpointType.BIOCASE, null);
187    messagePublisher.send(msg);
188
189    // dupe of dwc 1.4
190    messagePublisher.send(msg);
191
192    // update of dwc 1.4
193    schemaType = OccurrenceSchemaType.DWC_1_4;
194    crawlId = 1;
195    msg =
196      new OccurrenceFragmentedMessage(datasetKey, crawlId, dwc14_modified.getBytes(), schemaType, EndpointType.DIGIR,
197        null);
198    messagePublisher.send(msg);
199
200    TimeUnit.MILLISECONDS.sleep(6000);
201
202    assertEquals(5l,
203      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.FRAGMENT_RECEIVED).longValue());
204    assertEquals(5l,
205      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.FRAGMENT_PROCESSED).longValue());
206    assertEquals(0l,
207      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_ERROR)
208        .longValue());
209    assertEquals(1l,
210      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_UNCHANGED)
211        .longValue());
212    assertEquals(1l,
213      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_UPDATED)
214        .longValue());
215    assertEquals(4l,
216      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.RAW_OCCURRENCE_PERSISTED_NEW)
217        .longValue());
218    assertEquals(0l,
219      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_ERROR)
220        .longValue());
221    assertEquals(5l,
222      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.VERBATIM_OCCURRENCE_PERSISTED_SUCCESS)
223        .longValue());
224    assertEquals(0l,
225      zookeeperConnector.readCounter(datasetKey, ZookeeperConnector.CounterName.INTERPRETED_OCCURRENCE_PERSISTED_ERROR)
226        .longValue());
227    assertEquals(5l, zookeeperConnector
228      .readCounter(datasetKey, ZookeeperConnector.CounterName.INTERPRETED_OCCURRENCE_PERSISTED_SUCCESS).longValue());
229  }
