084  public void testNoContention(int threadCount) throws InterruptedException {
085    // test generating ids as fast as possible in the ideal case of no waiting for contention (all ids are globally
086    // unique)
087    int genPerThread = 10000;
088    List<Thread> threads = Lists.newArrayList();
089    for (int i = 0; i < threadCount; i++) {
090      Thread thread = new Thread(new FragmentPersister(fragmentProcessor, UUID.randomUUID(), genPerThread));
091      thread.start();
092      threads.add(thread);
093    }
094
095    Thread rateReporter = new Thread(new RateReporter(threadCount));
096    rateReporter.start();
097
098    for (Thread thread : threads) {
099      thread.join();
100    }
101
102    rateReporter.interrupt();
103    rateReporter.join();
104  }
105
106  private static class RateReporter implements Runnable {
107
108    private final int threadCount;
109
110    private RateReporter(int threadCount) {
111      this.threadCount = threadCount;
112    }
113
114    @Override
115    public void run() {
116      int periods = 0;
117      int runningAvg = 0;
118      int buildAverageAfter = 15;
119      int lastCount = 0;
120      boolean interrupted = false;
121      while (!interrupted) {
122        try {
123          TimeUnit.SECONDS.sleep(1);
124        } catch (InterruptedException e) {
125          interrupted = true;
126        }
127        int generated = fragsPersisted.intValue() - lastCount;
128        if (periods > buildAverageAfter) {
129          if (runningAvg == 0) {
130            runningAvg = generated;
131          } else {
132            int netPeriods = periods - buildAverageAfter;
133            runningAvg = (netPeriods * runningAvg + generated) / (netPeriods + 1);
134          }
135          System.out.println("Frags persisted at [" + generated + " frags/s] for running avg of [" + runningAvg
136                             + " frags/s] and per thread [" + (runningAvg / threadCount)
137                             + " frags/sec] with frag process time of [" + (threadCount * 1000 / runningAvg)
138                             + " ms/frag]");
139        } else {
140          System.out.println("Stats in [" + (buildAverageAfter - periods) + "] seconds.");
141        }
142        periods++;
143        lastCount = fragsPersisted.intValue();
144      }
145    }
146  }
147
148  private static class FragmentPersister implements Runnable {
149
150    private final FragmentProcessor fragProcessor;
151    private final UUID datasetKey;
152    private final OccurrenceSchemaType schema = OccurrenceSchemaType.DWC_MANIS;
153    private final int genCount;
154
155    private FragmentPersister(FragmentProcessor fragProcessor, UUID datasetKey, int genCount) {
156      this.fragProcessor = fragProcessor;
157      this.datasetKey = datasetKey;
158      this.genCount = genCount;
159    }
160
161    @Override
162    public void run() {
163      for (int i = 0; i < genCount; i++) {
164        String xml = xmlPrefix + i + xmlSuffix;
165        fragProcessor.buildFragments(datasetKey, xml.getBytes(Charsets.UTF_8), schema, EndpointType.BIOCASE, 1, null);
166        fragsPersisted.incrementAndGet();
167      }
168    }
169  }

171  public static void main(String[] args) throws InterruptedException, IOException {
172    // against c4 runs around 3-3.3k/sec - appears limited by persistence
173    int hbasePoolSize = 20;
174    int persistingThreads = 20;
175    if (args.length == 2) {
176      hbasePoolSize = Integer.valueOf(args[0]);
177      persistingThreads = Integer.valueOf(args[1]);
178    }
179    System.out
180      .println("Running test with hbasePool [" + hbasePoolSize + "] and persistingThreads [" + persistingThreads + "]");
181    FragmentProcessorThroughputTest instance = new FragmentProcessorThroughputTest(hbasePoolSize);
182    instance.testNoContention(persistingThreads);
183  }
