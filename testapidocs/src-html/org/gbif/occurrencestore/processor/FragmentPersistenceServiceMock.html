001package org.gbif.occurrencestore.processor;
002
003import org.gbif.occurrencestore.common.model.UniqueIdentifier;
004import org.gbif.occurrencestore.persistence.api.Fragment;
005import org.gbif.occurrencestore.persistence.api.FragmentCreationResult;
006import org.gbif.occurrencestore.persistence.api.OccurrenceKeyPersistenceService;
007import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
008
009import java.util.HashMap;
010import java.util.Map;
011import java.util.Set;
012
013public class FragmentPersistenceServiceMock implements FragmentPersistenceService {
014
015  private OccurrenceKeyPersistenceService occurrenceKeyService;
016  private Map<Integer, Fragment> cache = new HashMap<Integer, Fragment>();
017
018  public FragmentPersistenceServiceMock(OccurrenceKeyPersistenceService occurrenceKeyService) {
019    this.occurrenceKeyService = occurrenceKeyService;
020  }
021
022  @Override
023  public Fragment get(Integer key) {
024    return cache.get(key);
025  }
026
027  @Override
028  public void update(Fragment fragment) {
029    cache.put(fragment.getKey(), fragment);
030  }
031
032  @Override
033  public FragmentCreationResult insert(Fragment fragment, Set<UniqueIdentifier> uniqueIds) {
034    Integer key = occurrenceKeyService.generateKey(uniqueIds).getKey();
035    fragment.setKey(key);
036    cache.put(key, fragment);
037    return new FragmentCreationResult(fragment, true);
038  }
039}
