029  @Test
030  public void testUniqueIds() throws IOException {
031    UUID datasetKey = UUID.randomUUID();
032    String json = Resources.toString(Resources.getResource("uniqueids.json"), Charsets.UTF_8);
033    IdentifierExtractionResult result = JsonFragmentParser.extractIdentifiers(datasetKey, json.getBytes(), true, true);
034    Set<UniqueIdentifier> uniqueIds = result.getUniqueIdentifiers();
035    Assert.assertEquals(2, uniqueIds.size());
036    for (UniqueIdentifier uniqueId : uniqueIds) {
037      if (uniqueId instanceof HolyTriplet) {
038        HolyTriplet triplet = (HolyTriplet) uniqueId;
039        assertEquals("ic980897", triplet.getInstitutionCode());
040        assertEquals("cc1234", triplet.getCollectionCode());
041        assertEquals("cn90734", triplet.getCatalogNumber());
042      } else {
043        PublisherProvidedUniqueIdentifier pubProvided = (PublisherProvidedUniqueIdentifier) uniqueId;
044        assertEquals("occId3870", pubProvided.getPublisherProvidedIdentifier());
045      }
046    }
047  }

075  @Test
076  public void testParseUnpreferred() throws IOException {
077    UUID datasetKey = UUID.randomUUID();
078    String json = Resources.toString(Resources.getResource("fragment.json"), Charsets.UTF_8);
079    Fragment fragment = new Fragment(datasetKey, json.getBytes("UTF-8"), DigestUtils.md5(json.getBytes("UTF-8")),
080      Fragment.FragmentType.JSON, EndpointType.DWC_ARCHIVE, new Date(), 1,
081      OccurrenceSchemaType.DWCA, null, null);
082    VerbatimOccurrence verb = JsonFragmentParser.parseRecord(fragment);
083    assertNotNull(verb);
084    assertEquals("Plantae", verb.getKingdom());
085    assertNull(verb.getAuthor());
086    assertEquals("2400", verb.getMinAltitude());
087    assertEquals("Magnoliophyta", verb.getPhylum());
088    assertEquals("BGBM", verb.getInstitutionCode());
089    assertEquals("specimen", verb.getBasisOfRecord());
090    assertEquals("Verbascum cheiranthifolium var. cheiranthifolium", verb.getScientificName());
091    assertEquals("Pontaurus", verb.getCollectionCode());
092    assertEquals("Markus Döring", verb.getCollectorName());
093    assertNull(verb.getLocality());
094    assertEquals("7", verb.getMonth());
095    assertEquals("988", verb.getCatalogNumber());
096    assertEquals("37.42", verb.getLatitude());
097    assertEquals("1999", verb.getYear());
098    assertEquals("Verbascum", verb.getGenus());
099    assertEquals("Markus Döring", verb.getIdentifierName());
100    assertEquals("Scrophulariales", verb.getOrder());
101    assertEquals("30", verb.getDay());
102    assertEquals("Turkey", verb.getCountry());
103    assertEquals("34.568", verb.getLongitude());
104    assertEquals("Magnoliopsida", verb.getKlass());
105    assertEquals("Scrophulariaceae", verb.getFamily());
106    assertEquals("Bosphorus", verb.getContinentOrOcean());
107  }
