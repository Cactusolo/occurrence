001package org.gbif.occurrencestore.processor;
002
003import org.gbif.api.vocabulary.EndpointType;
004import org.gbif.api.vocabulary.OccurrenceSchemaType;
005import org.gbif.occurrencestore.persistence.api.Fragment;
006import org.gbif.occurrencestore.persistence.api.VerbatimOccurrence;
007import org.gbif.occurrencestore.processor.parsing.FragmentParser;
008
009import java.io.IOException;
010import java.util.Date;
011import java.util.UUID;
012
013import com.google.common.io.Resources;
014import org.apache.commons.codec.digest.DigestUtils;
015import org.apache.commons.io.Charsets;
016import org.junit.BeforeClass;
017import org.junit.Test;
018
019import static org.junit.Assert.assertEquals;
020import static org.junit.Assert.assertNotNull;
021import static org.junit.Assert.assertNull;
022
023public class FragmentParserTest {
024
025  private static String abcd206Single;
026  private static String dwc14;
027
028  @BeforeClass
029  public static void preClass() throws IOException {
030    abcd206Single = Resources.toString(Resources.getResource("abcd206_single.xml"), Charsets.UTF_8);
031    dwc14 = Resources.toString(Resources.getResource("dwc14.xml"), Charsets.UTF_8);
032  }
033
034  @Test
035  public void testAbcd206() {
036    OccurrenceSchemaType schema = OccurrenceSchemaType.ABCD_2_0_6;
037    UUID datasetKey = UUID.randomUUID();
038    Fragment frag = new Fragment(datasetKey, abcd206Single.getBytes(), DigestUtils.md5(abcd206Single.getBytes()),
039      Fragment.FragmentType.XML, EndpointType.BIOCASE, new Date(), 1, schema, null, null);
040    frag.setKey(1);
041
042    VerbatimOccurrence got = FragmentParser.parse(frag);
043    assertNotNull(got);
044    assertEquals("BGBM", got.getInstitutionCode());
045    assertEquals("AlgaTerra", got.getCollectionCode());
046    assertEquals("5834", got.getCatalogNumber());
047    assertEquals(datasetKey, got.getDatasetKey());
048    assertNull(got.getUnitQualifier());
049
050    assertEquals(1, got.getKey().intValue());
051    assertEquals("Tetraedron caudatum (Corda) Hansg.", got.getScientificName());
052    assertEquals("52.423798", got.getLatitude());
053    assertEquals("13.191434", got.getLongitude());
054    assertEquals("50", got.getLatLongPrecision());
055    assertEquals("400", got.getMinAltitude());
056    assertEquals("500", got.getMaxAltitude());
057    assertEquals("DE", got.getCountry());
058    assertEquals("Kusber, W.-H.", got.getCollectorName());
059    assertEquals("Nikolassee, Berlin", got.getLocality());
060    assertEquals("1987-04-13T00:00:00", got.getOccurrenceDate());
061    assertEquals("HumanObservation", got.getBasisOfRecord());
062    assertEquals("Kusber, W.-H.", got.getIdentifierName());
063  }
064
065  @Test
066  public void testDwc14() {
067    OccurrenceSchemaType schema = OccurrenceSchemaType.DWC_1_4;
068    UUID datasetKey = UUID.randomUUID();
069    Fragment frag =
070      new Fragment(datasetKey, dwc14.getBytes(), DigestUtils.md5(dwc14.getBytes()), Fragment.FragmentType.XML,
071        EndpointType.DIGIR, new Date(), 1, schema, null, null);
072
073    VerbatimOccurrence got = FragmentParser.parse(frag);
074    assertNotNull(got);
075    assertEquals("UGENT", got.getInstitutionCode());
076    assertEquals("vertebrata", got.getCollectionCode());
077    assertEquals("50058", got.getCatalogNumber());
078    assertEquals(datasetKey, got.getDatasetKey());
079    assertNull(got.getUnitQualifier());
080
081    assertEquals("Alouatta villosa Gray, 1845", got.getScientificName());
082    assertEquals("Gray, 1845", got.getAuthor());
083    assertEquals("Animalia", got.getKingdom());
084    assertEquals("Chordata", got.getPhylum());
085    assertEquals("Mammalia", got.getKlass());
086    assertEquals("Primates", got.getOrder());
087    assertEquals("Atelidae", got.getFamily());
088    assertEquals("Alouatta", got.getGenus());
089    assertEquals("villosa", got.getSpecies());
090    assertEquals("25", got.getLatLongPrecision());
091    assertEquals("200", got.getMinAltitude());
092    assertEquals("400", got.getMaxAltitude());
093    assertEquals("PreservedSpecimen", got.getBasisOfRecord());
094  }
