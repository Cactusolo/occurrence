121  @Test
122  @Ignore
123  public void testMultiThreadedAdd() throws InterruptedException {
124    UUID datasetKey = UUID.randomUUID();
125
126    List<Thread> threads = Lists.newArrayList();
127    for (int i = 0; i < 20; i++) {
128      Thread thread = new CountAddingThread(connector, ZookeeperConnector.CounterName.FRAGMENT_RECEIVED, datasetKey, 1);
129      thread.start();
130      threads.add(thread);
131    }
132    for (Thread thread : threads) {
133      thread.join();
134    }
135
136    Long test = connector.readCounter(datasetKey, ZookeeperConnector.CounterName.FRAGMENT_RECEIVED);
137    assertEquals(20l, test.longValue());
138  }

140  @Test
141  @Ignore("too expensive")
142  public void testPerformance() throws InterruptedException {
143    // add 1M counts on 100 threads for the same counter
144    UUID datasetKey = UUID.randomUUID();
145
146    Stopwatch stopwatch = new Stopwatch().start();
147    int totalCounts = 100000;
148    int threadCount = 100;
149    ExecutorService tp = Executors.newFixedThreadPool(threadCount);
150    for (int i = 0; i < threadCount; i++) {
151      tp.execute(new CountAddingThread(connector, ZookeeperConnector.CounterName.FRAGMENT_RECEIVED, datasetKey,
152        totalCounts / threadCount));
153    }
154    tp.shutdown();
155    tp.awaitTermination(10, TimeUnit.MINUTES);
156    stopwatch.stop();
157    System.out.println(
158      "zk counter adds took [" + stopwatch + "] for an average of [" + (stopwatch.elapsedMillis() / totalCounts)
159      + " ms/write]");
160    fail();
161  }
162
163  private static class CountAddingThread extends Thread {
164
165    private final ZookeeperConnector zkConnector;
166    private final ZookeeperConnector.CounterName counterName;
167    private final UUID uuid;
168    private final int count;
169
170    private CountAddingThread(ZookeeperConnector zkConnector, ZookeeperConnector.CounterName counterName, UUID uuid,
171      int count) {
172      this.zkConnector = zkConnector;
173      this.counterName = counterName;
174      this.uuid = uuid;
175      this.count = count;
176    }
177
178    @Override
179    public void run() {
180      for (int i = 0; i < count; i++) {
181        zkConnector.addCounter(uuid, counterName);
182      }
183    }
184  }
