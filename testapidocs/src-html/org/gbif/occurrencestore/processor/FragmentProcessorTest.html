094  @Test
095  public void testNewAbcd206Single() {
096    UUID datasetKey = UUID.randomUUID();
097    OccurrenceSchemaType schemaType = OccurrenceSchemaType.ABCD_2_0_6;
098    EndpointType protocol = EndpointType.BIOCASE;
099    Integer crawlId = 1;
100    fragmentProcessor.buildFragments(datasetKey, abcd206Single.getBytes(), schemaType, protocol, crawlId, null);
101
102    Fragment got = fragmentPersistenceService.get(1);
103    assertTrue(got.getKey() > 0);
104    Calendar resultCal = Calendar.getInstance();
105    resultCal.setTime(got.getHarvestedDate());
106    Calendar cal = Calendar.getInstance();
107    assertEquals(cal.get(Calendar.DAY_OF_YEAR), resultCal.get(Calendar.DAY_OF_YEAR));
108    assertEquals(Fragment.FragmentType.XML, got.getFragmentType());
109    assertEquals(schemaType, got.getXmlSchema());
110    assertEquals(datasetKey, got.getDatasetKey());
111    assertEquals(protocol, got.getProtocol());
112    assertEquals(crawlId, got.getCrawlId());
113    assertTrue(Arrays.equals(abcd206Single.getBytes(), got.getData()));
114    assertTrue(Arrays.equals(DigestUtils.md5(abcd206Single), got.getDataHash()));
115    assertNull(got.getUnitQualifier());
116  }

118  @Test
119  public void testNewAbcd206Multiple() {
120    UUID datasetKey = UUID.randomUUID();
121    OccurrenceSchemaType schemaType = OccurrenceSchemaType.ABCD_2_0_6;
122    Integer crawlId = 1;
123    fragmentProcessor
124      .buildFragments(datasetKey, abcd206Multi.getBytes(), schemaType, EndpointType.BIOCASE, crawlId, null);
125    Set<Fragment> fragments = Sets.newHashSet();
126    Fragment first = fragmentPersistenceService.get(1);
127    fragments.add(first);
128    Fragment second = fragmentPersistenceService.get(2);
129    fragments.add(second);
130
131    int count = 0;
132    for (Fragment got : fragments) {
133      assertEquals(datasetKey, got.getDatasetKey());
134      assertEquals(crawlId, got.getCrawlId());
135      assertTrue(Arrays.equals(abcd206Multi.getBytes(), got.getData()));
136      if ("Schistidium agassizii Sull. & Lesq. in Sull.".equals(got.getUnitQualifier())) {
137        count++;
138      } else if ("Grimmia alpicola Sw. ex Hedw.".equals(got.getUnitQualifier())) {
139        count++;
140      } else {
141        fail("Unexpected UnitQualifier");
142      }
143    }
144    assertEquals(2, count);
145  }

164    fragments.add(second);
165
166    int count = 0;

169      assertEquals(crawlId, got.getCrawlId());
170      assertTrue(Arrays.equals(abcd206MultiModified.getBytes(), got.getData()));
171      if ("Schistidium agassizii Sull. & Lesq. in Sull.".equals(got.getUnitQualifier())) {
172        count++;
173      } else if ("Grimmia alpicola Sw. ex Hedw.".equals(got.getUnitQualifier())) {
174        count++;
