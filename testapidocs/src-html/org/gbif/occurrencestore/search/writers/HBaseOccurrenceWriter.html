017/**
018 * Utility class for writing occurrence records into an HBase table.
019 */
020public class HBaseOccurrenceWriter implements Predicate<Occurrence> {
021
022
023  // HBase table
024  private final HTableInterface hTable;
025
026  // Column family
027  private final static byte[] CF = Bytes.toBytes(HBaseTableConstants.OCCURRENCE_COLUMN_FAMILY);
028
029  private static final BasisOfRecordConverter BOR_CONVERTER = new BasisOfRecordConverter();
030
031
032  /**
033   * Default constructor.
034   */
035  public HBaseOccurrenceWriter(HTableInterface hTable) {
036    this.hTable = hTable;
037  }
038
039  /**
040   * Reads and processes the occurrence object.
041   */
042  @Override
043  public boolean apply(Occurrence input) {
044    try {
045      write(input);
046      return true;
047    } catch (IOException e) {
048      return false;
049    }
050  }
051
052
053  /**
054   * Writes the occurrence record into the hbase table.
055   *
056   * @param occ occurrence object that will be written to hbase
057   * @throws IOException
058   */
059  private void write(Occurrence occ) throws IOException {
060    Put put = new Put(Bytes.toBytes(occ.getKey()));
061
062
063    if (occ.getAltitude() != null) {
064      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_ALTITUDE).getColumnName()),
065        Bytes.toBytes(occ.getAltitude()));
066    }
067
068    if (occ.getBasisOfRecord() != null) {
069      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_BASIS_OF_RECORD).getColumnName()),
070        Bytes.toBytes(BOR_CONVERTER.fromEnum(occ.getBasisOfRecord())));
071    }
072    if (occ.getField(DwcTerm.catalogNumber) != null) {
073      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.CATALOG_NUMBER).getColumnName()),
074        Bytes.toBytes(occ.getField(DwcTerm.catalogNumber)));
075    }
076
077    if (occ.getClassKey() != null) {
078      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_CLASS_ID).getColumnName()),
079        Bytes.toBytes(occ.getClassKey()));
080    }
081    if (occ.getClazz() != null) {
082      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_CLASS).getColumnName()),
083        Bytes.toBytes(occ.getClazz()));
084    }
085
086    if (occ.getField(DwcTerm.collectionCode) != null) {
087      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.COLLECTION_CODE).getColumnName()),
088        Bytes.toBytes(occ.getField(DwcTerm.collectionCode)));
089    }
090
091    // deprecated - remove
092//    if (occ.getDataProviderId() != null) {
093//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATA_PROVIDER_ID).getColumnName()),
094//        Bytes.toBytes(occ.getDataProviderId()));
095//    }
096//
097//    if (occ.getDataResourceId() != null) {
098//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATA_RESOURCE_ID).getColumnName()),
099//        Bytes.toBytes(occ.getDataResourceId()));
100//    }
101
102    if (occ.getDatasetKey() != null) {
103      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATASET_KEY).getColumnName()),
104        Bytes.toBytes(occ.getDatasetKey().toString()));
105    }
106
107    if (occ.getDepth() != null) {
108      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_DEPTH).getColumnName()),
109        Bytes.toBytes(occ.getDepth()));
110    }
111
112    if (occ.getFamily() != null) {
113      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_FAMILY).getColumnName()),
114        Bytes.toBytes(occ.getFamily()));
115    }
116
117    if (occ.getFamilyKey() != null) {
118      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_FAMILY_ID).getColumnName()),
119        Bytes.toBytes(occ.getFamilyKey()));
120    }
121
122    if (occ.getGenus() != null) {
123      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_GENUS).getColumnName()),
124        Bytes.toBytes(occ.getGenus()));
125    }
126
127    if (occ.getGenusKey() != null) {
128      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_GENUS_ID).getColumnName()),
129        Bytes.toBytes(occ.getGenusKey()));
130    }
131
132    // TODO geospatial issue has changed a lot
133//    if (occ.getGeospatialIssue() != null) {
134//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_GEOSPATIAL_ISSUE).getColumnName()),
135//        Bytes.toBytes(occ.getGeospatialIssue()));
136//    }
137
138    if (occ.getField(DwcTerm.institutionCode) != null) {
139      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.INSTITUTION_CODE).getColumnName()),
140        Bytes.toBytes(occ.getField(DwcTerm.institutionCode)));
141    }
142
143    if (occ.getKingdom() != null) {
144      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_KINGDOM).getColumnName()),
145        Bytes.toBytes(occ.getKingdom()));
146    }
147
148    if (occ.getKingdomKey() != null) {
149      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_KINGDOM_ID).getColumnName()),
150        Bytes.toBytes(occ.getKingdomKey()));
151    }
152
153    if (occ.getLatitude() != null) {
154      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_LATITUDE).getColumnName()),
155        Bytes.toBytes(occ.getLatitude()));
156    }
157
158    if (occ.getLongitude() != null) {
159      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_LONGITUDE).getColumnName()),
160        Bytes.toBytes(occ.getLongitude()));

162
163    if (occ.getField(DwcTerm.locality) != null) {

165        Bytes.toBytes(occ.getField(DwcTerm.locality)));
166    }
167
168    if (occ.getCountry() != null) {
169      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_ISO_COUNTRY_CODE).getColumnName()),
170        Bytes.toBytes(occ.getCountry().getIso2LetterCode()));
171    }
172

174      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.COUNTY).getColumnName()),
175        Bytes.toBytes(occ.getField(DwcTerm.county)));
176    }
177

179      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.STATE_PROVINCE).getColumnName()),
180        Bytes.toBytes(occ.getStateProvince()));

192    }
193
194    if (occ.getField(DwcTerm.identifiedBy) != null) {
195      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.IDENTIFIER_NAME).getColumnName()),
196        Bytes.toBytes(occ.getField(DwcTerm.identifiedBy)));
197    }
198
199    if (occ.getIdentificationDate() != null) {
200      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.IDENTIFICATION_DATE).getColumnName()),
201        Bytes.toBytes(occ.getIdentificationDate().getTime()));
202    }
203
204    if (occ.getModified() != null) {
205      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_MODIFIED).getColumnName()),
206        Bytes.toBytes(occ.getModified().getTime()));
207    }
208
209    if (occ.getMonth() != null) {
210      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_MONTH).getColumnName()),
211        Bytes.toBytes(occ.getMonth()));
212    }
213
214    if (occ.getTaxonKey() != null) {

216        Bytes.toBytes(occ.getTaxonKey()));
217    }
218

220      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_OCCURRENCE_DATE).getColumnName()),
221        Bytes.toBytes(occ.getEventDate().getTime()));
222    }
223
224    if (occ.getOrder() != null) {
225      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_ORDER).getColumnName()),
226        Bytes.toBytes(occ.getOrder()));
227    }
228
229    if (occ.getOrderKey() != null) {
230      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_ORDER_ID).getColumnName()),
231        Bytes.toBytes(occ.getOrderKey()));
232    }
233
234    // TODO: other issue now deprecated
235//    if (occ.getOtherIssue() != null) {
236//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_OTHER_ISSUE).getColumnName()),
237//        Bytes.toBytes(occ.getOtherIssue()));
238//    }
239
240    if (occ.getPublishingOrgKey() != null) {
241      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.OWNING_ORG_KEY).getColumnName()),
242        Bytes.toBytes(occ.getPublishingOrgKey().toString()));
243    }
244
245    if (occ.getPhylum() != null) {
246      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_PHYLUM).getColumnName()),
247        Bytes.toBytes(occ.getPhylum()));
248    }
249
250    if (occ.getPhylumKey() != null) {
251      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_PHYLUM_ID).getColumnName()),
252        Bytes.toBytes(occ.getPhylumKey()));
253    }
254
255    // deprecated
256//    if (occ.getResourceAccessPointId() != null) {
257//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.RESOURCE_ACCESS_POINT_ID).getColumnName()),
258//        Bytes.toBytes(occ.getResourceAccessPointId()));
259//    }
260
261    if (occ.getScientificName() != null) {
262      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_SCIENTIFIC_NAME).getColumnName()),
263        Bytes.toBytes(occ.getScientificName()));
264    }
265
266    if (occ.getSpecies() != null) {
267      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_SPECIES).getColumnName()),
268        Bytes.toBytes(occ.getSpecies()));
269    }
270
271    if (occ.getSpeciesKey() != null) {
272      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_SPECIES_ID).getColumnName()),
273        Bytes.toBytes(occ.getSpeciesKey()));
274    }
275
276    // TODO: deprecated
277//    if (occ.getTaxonomicIssue() != null) {
278//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_TAXONOMIC_ISSUE).getColumnName()),
279//        Bytes.toBytes(occ.getTaxonomicIssue()));
280//    }
281
282    // TODO: deprecated
283//    if (occ.getUnitQualifier() != null) {
284//      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.UNIT_QUALIFIER).getColumnName()),
285//        Bytes.toBytes(occ.getUnitQualifier()));
286//    }
287
288    if (occ.getYear() != null) {
289      put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_YEAR).getColumnName()),
290        Bytes.toBytes(occ.getYear()));
291    }
292
293    hTable.put(put);
294    hTable.flushCommits();
295  }
