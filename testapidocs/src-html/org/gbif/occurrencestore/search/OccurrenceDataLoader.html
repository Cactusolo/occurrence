027/**
028 * Utility class that loads and processes occurrence records read from a CSV file.
029 * The expected columns in that file are:
030 * "key","altitude","basisOfRecord","catalogNumber","classKey","clazz","collectionCode","dataProviderId",
031 * "dataResourceId","datasetKey","depth","occurrenceId","family","familyKey","genus",
032 * "genusKey","geospatialIssue","institutionCode","country","kingdom","kingdomKey","latitude","longitude","modified",
033 * "occurrenceMonth","nubKey","occurrenceDate","order","orderKey","otherIssue",
034 * "owningOrgKey","phylum","phylumKey","resourceAccessPointId","scientificName","species","speciesKey","taxonomicIssue",
035 * "unitQualifier","occurrenceYear","locality","county","stateProvince","continent",
036 * "collectorName","identifierName", "identificationDate".
037 * Each cvs line is interpreted into an Occurrence object; to process each object a predicate or list of predicates are
038 * passed as parameters to the function loadOccurrences.
039 */
040public class OccurrenceDataLoader {
041
042  /**
043   * Produces a Basis of Record instance.
044   */
045  private static class BasisOfRecordProcessor implements CellProcessor {
046
047    @Override
048    public BasisOfRecord execute(Object value, CsvContext context) {
049      return BOR_CONVERTER.toEnum(Integer.parseInt((String) value));
050    }
051
052  }
053
054  /**
055   * Produces a Country instance.
056   */
057  private static class CountryProcessor implements CellProcessor {
058
059    @Override
060    public Country execute(Object value, CsvContext context) {
061      return Country.fromIsoCode((String) value);
062    }
063
064  }
065
066
067  /**
068   * Produces an UUID instance from a string object.
069   */
070  private static class UUIDProcessor implements CellProcessor {
071
072    @Override
073    public UUID execute(Object value, CsvContext context) {
074      return UUID.fromString((String) value);
075    }
076
077  }
078

193  /**
194   * Reads a CSV file and produces occurrence records for each line.
195   * Each occurrence object is processed by the list of processors.
196   * 
197   * @param fileName CSV file
198   * @param processors list of processors(predicates) that consume occurrence objects
199   */
200  public static void processOccurrences(String fileName, Predicate<Occurrence>... processors) {
201    ICsvBeanReader reader = null;
202    int line = 1;
203    try {
204      reader =
205        new CsvBeanReader(new FileReader(new File(Resources.getResource(fileName).toURI())),
206          CsvPreference.STANDARD_PREFERENCE);
207      reader.getHeader(true);
208      Occurrence occurrence;
209      while ((occurrence = reader.read(Occurrence.class, HEADER, CELL_PROCESSORS)) != null) {
210        for (Predicate<Occurrence> predicate : processors) {
211          predicate.apply(occurrence);
212        }
213        line++;
214      }
215    } catch (Exception e) {
216      LOG.error(String.format("Error parsing occurrence object from file %d", line), e);
217    } finally {
218      Closeables.closeQuietly(reader);
219    }
220  }
