044 * Internally loads data from a csv file into a HBase minicluster and an embedded Solr server; both server instances are

054  public static class OccurrenceSearchTestModule extends PrivateServiceModule {
055
056    private static final String PREFIX = "occurrencestore.search.";
057
058
059    /**
060     * Default constructor.
061     */
062    public OccurrenceSearchTestModule(Properties properties) {
063      super(PREFIX, properties);
064    }

066    /**
067     * Provides instances of OccurrenceService, this is required by the OccurrenceSearchService.
068     */
069    @Provides
070    @Singleton
071    public OccurrenceService provideOccurrenceService() {
072      OccurrenceService occurrenceService = new OccurrencePersistenceServiceImpl(OCCURRENCE_TABLE_NAME, hTablePool);
073      return occurrenceService;
074    }
075
076    @Override
077    protected void configureService() {
078      install(new SolrModule());
079      bind(NameUsageMatchingService.class).toInstance(Mockito.mock(NameUsageMatchingService.class));
080      bind(OccurrenceSearchService.class).to(OccurrenceSearchImpl.class);
081      expose(OccurrenceSearchService.class);
082      // Exposes the SolrServer because it is required to create the index.
083      expose(SolrServer.class);
084    }

122  @BeforeClass
123  public static void beforeClass() throws Exception {
124    TEST_UTIL.startMiniCluster(1);
125    TEST_UTIL.createTable(OCCURRENCE_TABLE, CF);
126    hTablePool = new HTablePool(TEST_UTIL.getConfiguration(), 1);
127    occurrenceSearchService = injector.getInstance(OccurrenceSearchService.class);
128    solrServer = injector.getInstance(SolrServer.class);
129    // deletes all previous data
130    solrServer.deleteByQuery("*:*");
131    loadOccurrences();
132  }

135  /**
136   * Commits changes to the Solr server.
137   * Exceptions {@link SolrServerException} and {@link IOException} are swallowed.
138   */
139  private static void commitToSolrQuietly() {
140    try {
141      solrServer.commit();
142    } catch (SolrServerException e) {
143      LOG.error("Solr error commiting on Solr server", e);
144    } catch (IOException e) {
145      LOG.error("I/O error commiting on Solr server", e);
146    }
147  }
148
149  /**
150   * Creates an instance of a guice injector using the OccurrenceSearchTestModule.
151   *
152   * @return
153   */
154  private static Injector getInjector() {
155    try {
156      Properties properties = new Properties();
157      properties.load(Resources.newInputStreamSupplier(Resources.getResource("occurrence.properties")).getInput());
158      return Guice.createInjector(new OccurrenceSearchTestModule(properties));
159    } catch (IllegalArgumentException e) {
160      throw e;
161    } catch (IOException e) {
162      throw new IllegalStateException(e);
163    }
164  }
165
166
167  /**
168   * Loads test data from the CSV file and store it into Solr and HBase.
169   */
170  private static void loadOccurrences() throws IOException {
171    final HTableInterface hTable = hTablePool.getTable(OCCURRENCE_TABLE);
172    try {
173      OccurrenceDataLoader.processOccurrences(CSV_TEST_FILE, new HBaseOccurrenceWriter(hTable),
174        new SolrOccurrenceWriter(
175          solrServer));
176    } catch (Exception e) {
177      LOG.error("Error processing occurrence objects from file", e);
178    } finally {
179      commitToSolrQuietly();
180      hTable.close();
181    }
182  }
