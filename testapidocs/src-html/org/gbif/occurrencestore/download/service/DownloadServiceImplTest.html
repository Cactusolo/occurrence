089  @Test
090  public void testList() {
091    List<Download> peterDownloads = Lists.newArrayList();
092    List<Download> karlDownloads = Lists.newArrayList();
093    List<Download> allDownloads = Lists.newArrayList();
094    List<Download> emptyDownloads = Lists.newArrayList();
095
096    Download job1 = mockDownload("1-oozie-oozi-W", "peter");
097    peterDownloads.add(job1);
098
099    Download job2 = mockDownload("2-oozie-oozi-W", "karl");
100    karlDownloads.add(job2);
101    allDownloads.add(job1);
102    allDownloads.add(job2);
103    // always get 3 job infos until we hit an offset of 100
104    when(downloadService.listByUser(any(String.class), any(Pageable.class))).thenReturn(
105      new PagingResponse<Download>(0L, 0, 0L, emptyDownloads));
106    when(downloadService.listByUser(eq("peter"), any(Pageable.class))).thenReturn(
107      new PagingResponse<Download>(0L, peterDownloads.size(), new Long(peterDownloads.size()), peterDownloads));
108    when(downloadService.listByUser(eq("karl"), any(Pageable.class))).thenReturn(
109      new PagingResponse<Download>(0L, peterDownloads.size(), new Long(peterDownloads.size()), karlDownloads));
110    when(downloadService.list(any(Pageable.class))).thenReturn(
111      new PagingResponse<Download>(0L, allDownloads.size(), new Long(allDownloads.size()), allDownloads));
112    // mock get details
113    when(downloadService.get(eq("1-oozie-oozi-W"))).thenReturn(job1);
114    when(downloadService.get(eq("2-oozie-oozi-W"))).thenReturn(job2);
115
116    // test
117    PagingRequest req = new PagingRequest(0, 2);
118    PagingResponse<Download> x = downloadService.list(req);
119    assertEquals(2, x.getResults().size());
120
121    x = downloadService.listByUser("harald", req);
122    assertEquals(0, x.getResults().size());
123
124    x = downloadService.listByUser("karl", req);
125    assertEquals(1, x.getResults().size());
126
127    x = downloadService.listByUser("peter", req);
128    assertEquals(1, x.getResults().size());
129  }

131  @Test
132  public void testNotification() throws OozieClientException {
133    when(oozieClient.run(any(Properties.class))).thenReturn(JOB_ID);
134    DownloadRequest dl =
135      new DownloadRequest(DEFAULT_TEST_PREDICATE, "markus", Lists.newArrayList(TEST_EMAIL));
136
137    String downloadKey = requestService.create(dl);
138    assertThat(downloadKey, equalTo(DOWNLOAD_ID));
139
140    ArgumentCaptor<Properties> argument = ArgumentCaptor.forClass(Properties.class);
141    verify(oozieClient).run(argument.capture());
142    assertThat(argument.getValue().getProperty(Constants.NOTIFICATION_PROPERTY), equalTo(TEST_EMAIL));
143  }
144
145  private Download mockDownload(String downloadKey, String creator) {
146    DownloadRequest downloadRequest = new DownloadRequest(DEFAULT_TEST_PREDICATE, creator, null);
147    Download download = new Download();
148    download.setRequest(downloadRequest);
149    download.setKey(downloadKey);
150    download.setCreated(new Date());
151    download.setModified(new Date());
152    download.setStatus(Status.SUCCEEDED);
153    return download;
154  }
