016import com.google.common.collect.Lists;

019import static org.hamcrest.CoreMatchers.equalTo;
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertThat;
022
023public class SolrQueryVisitorTest {
024
025  private static final OccurrenceSearchParameter PARAM = OccurrenceSearchParameter.CATALOG_NUMBER;
026  private static final OccurrenceSearchParameter PARAM2 = OccurrenceSearchParameter.INSTITUTION_CODE;
027  private final SolrQueryVisitor visitor = new SolrQueryVisitor();
028
029  @Test
030  public void testComplexQuery() throws QueryBuildingException {
031    Predicate aves = new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212");
032    Predicate UK = new EqualsPredicate(OccurrenceSearchParameter.COUNTRY, "GB");
033    Predicate before1989 = new LessThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "1989");
034
035    ConjunctionPredicate p = new ConjunctionPredicate(Lists.newArrayList(aves, UK, before1989));
036    String where = visitor.getQuery(p);
037    assertEquals("((taxon_key:212) AND (country:GB) AND (year:[* TO 1989]))", where);
038  }
039
040  @Test
041  public void testConjunctionPredicate() throws QueryBuildingException {
042    Predicate p1 = new EqualsPredicate(PARAM, "value_1");
043    Predicate p2 = new EqualsPredicate(PARAM2, "value_2");
044
045    ConjunctionPredicate p = new ConjunctionPredicate(Lists.newArrayList(p1, p2));
046    String query = visitor.getQuery(p);
047    assertThat(query, equalTo("((catalog_number:value_1) AND (institution_code:value_2))"));
048  }
049
050  @Test
051  public void testDisjunctionPredicate() throws QueryBuildingException {
052    Predicate p1 = new EqualsPredicate(PARAM, "value_1");
053    Predicate p2 = new EqualsPredicate(PARAM2, "value_2");
054
055    DisjunctionPredicate p = new DisjunctionPredicate(Lists.newArrayList(p1, p2));
056    String query = visitor.getQuery(p);
057    assertThat(query, equalTo("((catalog_number:value_1) OR (institution_code:value_2))"));
058  }
059
060  @Test
061  public void testEqualsPredicate() throws QueryBuildingException {
062    Predicate p = new EqualsPredicate(PARAM, "value");
063    String query = visitor.getQuery(p);
064    assertThat(query, equalTo("catalog_number:value"));
065  }
066
067  @Test
068  public void testGreaterThanOrEqualPredicate() throws QueryBuildingException {
069    Predicate p = new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.ALTITUDE, "222");
070    String query = visitor.getQuery(p);
071    assertThat(query, equalTo("altitude:[222 TO *]"));
072  }
073
074  @Test
075  public void testGreaterThanPredicate() throws QueryBuildingException {
076    Predicate p = new GreaterThanPredicate(OccurrenceSearchParameter.ALTITUDE, "1000");
077    String query = visitor.getQuery(p);
078    assertThat(query, equalTo("altitude:{1000 TO *]"));
079  }
080
081  @Test
082  public void testInPredicate() throws QueryBuildingException {
083    Predicate p = new InPredicate(PARAM, Lists.newArrayList("value_1", "value_2", "value_3"));
084    String query = visitor.getQuery(p);
085    assertThat(query,
086      equalTo("((catalog_number:value_1) OR (catalog_number:value_2) OR (catalog_number:value_3))"));
087  }
088
089  @Test
090  public void testLessThanOrEqualPredicate() throws QueryBuildingException {
091    Predicate p = new LessThanOrEqualsPredicate(OccurrenceSearchParameter.ALTITUDE, "1000");
092    String query = visitor.getQuery(p);
093    assertThat(query, equalTo("altitude:[* TO 1000]"));
094  }
095
096  @Test
097  public void testLessThanPredicate() throws QueryBuildingException {
098    Predicate p = new LessThanPredicate(OccurrenceSearchParameter.ALTITUDE, "1000");
099    String query = visitor.getQuery(p);
100    assertThat(query, equalTo("altitude:[* TO 1000}"));
101  }
102
103  @Test
104  public void testNotPredicate() throws QueryBuildingException {
105    Predicate p = new NotPredicate(new EqualsPredicate(PARAM, "value"));
106    String query = visitor.getQuery(p);
107    assertThat(query, equalTo("-catalog_number:value"));
108  }
109
110  @Test
111  public void testNotPredicateComplex() throws QueryBuildingException {
112    Predicate p1 = new EqualsPredicate(PARAM, "value_1");
113    Predicate p2 = new EqualsPredicate(PARAM2, "value_2");
114
115    ConjunctionPredicate cp = new ConjunctionPredicate(Lists.newArrayList(p1, p2));
116
117    Predicate p = new NotPredicate(cp);
118    String query = visitor.getQuery(p);
119    assertThat(query, equalTo("-((catalog_number:value_1) AND (institution_code:value_2))"));
120
121
122  }
123
124  @Test
125  public void testQuotes() throws QueryBuildingException {
126    Predicate p = new EqualsPredicate(PARAM, "my 'pleasure'");
127    String query = visitor.getQuery(p);
128    assertThat(query, equalTo("catalog_number:\"my\\ 'pleasure'\""));
129
130    p = new LessThanOrEqualsPredicate(OccurrenceSearchParameter.ALTITUDE, "101");
131    query = visitor.getQuery(p);
132    assertThat(query, equalTo("altitude:[* TO 101]"));
133
134    p = new GreaterThanPredicate(OccurrenceSearchParameter.YEAR, "1998");
135    query = visitor.getQuery(p);
136    assertThat(query, equalTo("year:{1998 TO *]"));
137  }
