001package org.gbif.occurrencestore.download.service;
002
003import org.gbif.api.model.occurrence.Download;
004import org.gbif.api.model.occurrence.Download.Status;
005import org.gbif.api.model.occurrence.DownloadRequest;
006import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
007import org.gbif.api.model.occurrence.predicate.Predicate;
008import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
009import org.gbif.api.service.registry.OccurrenceDownloadService;
010
011import java.util.Date;
012import java.util.List;
013
014import javax.mail.MessagingException;
015
016import com.google.common.collect.Lists;
017import com.google.common.collect.Maps;
018import org.apache.oozie.client.OozieClient;
019import org.apache.oozie.client.OozieClientException;
020import org.apache.oozie.client.WorkflowJob;
021import org.junit.Before;
022import org.junit.Test;
023import org.junit.runner.RunWith;
024import org.powermock.api.mockito.mockpolicies.Slf4jMockPolicy;
025import org.powermock.core.classloader.annotations.MockPolicy;
026import org.powermock.core.classloader.annotations.PowerMockIgnore;
027import org.powermock.modules.junit4.PowerMockRunner;
028import org.slf4j.Logger;
029import org.slf4j.LoggerFactory;
030
031import static org.mockito.Matchers.anyString;
032import static org.mockito.Matchers.eq;
033import static org.mockito.Mockito.mock;
034import static org.mockito.Mockito.reset;
035import static org.mockito.Mockito.verify;
036import static org.mockito.Mockito.verifyNoMoreInteractions;
037import static org.mockito.Mockito.when;
038
039
040@MockPolicy(Slf4jMockPolicy.class)
041@PowerMockIgnore({"javax.management.*", "javax.xml.*", "org.xml.*", "com.sun.org.apache.xerces.*", "org.w3c.dom.*"})
042@RunWith(PowerMockRunner.class)
043public class CallbackServiceTest {
044
045  private static final String DOWNLOAD_ID = "0000092-120518122602221";
046  private static final String JOB_ID = DOWNLOAD_ID + "-oozie-oozi-W";
047  private static final String KILLED = "KILLED";
048  private static final String FAILED = "FAILED";
049  private static final String SUCCEEDED = "SUCCEEDED";
050  private static final String RUNNING = "RUNNING";
051  private static final Predicate DEFAULT_TEST_PREDICATE = new EqualsPredicate(OccurrenceSearchParameter.CATALOG_NUMBER,
052    "bar");
053  private static final String TEST_USER = "admin";
054  private static final List<String> EMAILS = Lists.newArrayList("tests@gbif.org");
055
056  private OozieClient oozieClient;
057  private CallbackService service;
058  private OccurrenceDownloadService occurrenceDownloadService;
059  private DownloadEmailUtils downloadEmailUtils;
060
061  /**
062   * Creates a mock download object.
063   */
064  private static Download mockDownload() {
065    DownloadRequest downloadRequest = new DownloadRequest(DEFAULT_TEST_PREDICATE, TEST_USER, EMAILS);
066    Download download = new Download();
067    download.setRequest(downloadRequest);
068    download.setKey(DOWNLOAD_ID);
069    download.setDownloadLink(JOB_ID + ".zip");
070    download.setCreated(new Date());
071    download.setModified(new Date());
072    download.setStatus(Status.PREPARING);
073    return download;
074  }
075
076  @Before
077  public void setup() {
078    downloadEmailUtils = mock(DownloadEmailUtils.class);
079    occurrenceDownloadService = mock(OccurrenceDownloadService.class);
080    when(occurrenceDownloadService.get(anyString())).thenReturn(mockDownload());
081    oozieClient = mock(OozieClient.class);
082    service =
083      new DownloadRequestServiceImpl(oozieClient, Maps.<String, String>newHashMap(), "http://localhost:8080/",
084        "", occurrenceDownloadService, downloadEmailUtils);
085  }
086
087
088  @Test
089  public void testIgnoreRunningJobs() {
090    service.processCallback(JOB_ID, RUNNING);
091    verifyNoMoreInteractions(oozieClient);
092  }
093
094
095  @Test(expected = IllegalArgumentException.class)
096  public void testIgnoreWrongStatuses() {
097    service.processCallback(JOB_ID, "INVALID");
098    verifyNoMoreInteractions(downloadEmailUtils);
099  }
100
101  @Test
102  public void testNotificationSent() throws OozieClientException, MessagingException {
103    WorkflowJob job = mock(WorkflowJob.class);
104    when(oozieClient.getJobInfo(JOB_ID)).thenReturn(job);
105    when(job.getId()).thenReturn(JOB_ID);
106    when(job.getCreatedTime()).thenReturn(new Date());
107    when(job.getConf())
108      .thenReturn(
109        "<configuration>"
110          + "<property><name>"
111          + Constants.USER_PROPERTY
112          + "</name>"
113          + "<value>test</value></property>"
114
115          + "<property><name>"
116          + Constants.NOTIFICATION_PROPERTY
117          + "</name>"
118          + "<value>test@gbif.org</value></property>"
119
120          + "<property><name>"
121          + Constants.FILTER_PROPERTY
122          + "</name>"
123          + "<value>{\"type\":\"equals\",\"key\":\"DATASET_KEY\",\"value\":\"8575f23e-f762-11e1-a439-00145eb45e9a\"}</value></property>"
124          + "</configuration>");
125
126    service.processCallback(JOB_ID, SUCCEEDED);
127  }
128
129  @Test
130  public void testNotifyAdminForFailedJobs() {
131    Logger logger = LoggerFactory.getLogger(CallbackService.class);
132    service.processCallback(JOB_ID, KILLED);
133    verify(logger).error(eq(Constants.NOTIFY_ADMIN), anyString(), eq(JOB_ID), eq(KILLED));
134    reset(logger);
135
136    service.processCallback(JOB_ID, FAILED);
137    verify(logger).error(eq(Constants.NOTIFY_ADMIN), anyString(), eq(JOB_ID), eq(FAILED));
138    reset(logger);
139  }
140
141}
