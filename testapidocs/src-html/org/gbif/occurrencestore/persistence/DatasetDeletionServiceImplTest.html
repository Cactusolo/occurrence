138  @Test
139  public void testDatasetKeyExists() {
140    Iterator<Integer> iterator =
141      occurrenceService.getKeysByColumn(Bytes.toBytes(GOOD_DATASET_KEY.toString()), FieldName.DATASET_KEY);
142    int count = 0;
143    while (iterator.hasNext()) {
144      iterator.next();
145      count++;
146    }
147    assertEquals(3, count);
148    PublisherProvidedUniqueIdentifier uniqueId = new PublisherProvidedUniqueIdentifier(GOOD_DATASET_KEY, "abc3");
149    Set<UniqueIdentifier> uniqueIdentifiers = Sets.newHashSet();
150    uniqueIdentifiers.add(uniqueId);
151    KeyLookupResult result = occurrenceKeyService.findKey(uniqueIdentifiers);
152    assertEquals(3, result.getKey());
153    deletionService.deleteDataset(GOOD_DATASET_KEY);
154
155    iterator = occurrenceService.getKeysByColumn(Bytes.toBytes(GOOD_DATASET_KEY.toString()), FieldName.DATASET_KEY);
156    assertFalse(iterator.hasNext());
157
158    result = occurrenceKeyService.findKey(uniqueIdentifiers);
159    assertNull(result);
160  }
161
162  @Test
163  public void testDatasetKeyDoesntExist() {
164    UUID datasetKey = UUID.randomUUID();
165    Iterator<Integer> iterator =
166      occurrenceService.getKeysByColumn(Bytes.toBytes(datasetKey.toString()), FieldName.DATASET_KEY);
167    assertFalse(iterator.hasNext());
168    deletionService.deleteDataset(datasetKey);
169
170    iterator = occurrenceService.getKeysByColumn(Bytes.toBytes(datasetKey.toString()), FieldName.DATASET_KEY);
171    assertFalse(iterator.hasNext());
172  }
