125  @Test
126  public void testLockWriteDie() throws IOException {
127    // setup: 2 rows, each one gets as far as writing the new id but "dies" before releasing lock
128    HTableInterface lookupTable = tablePool.getTable(LOOKUP_TABLE);
129    String datasetKey = UUID.randomUUID().toString();
130
131    byte[] lock1 = Bytes.toBytes(UUID.randomUUID().toString());
132    byte[] lookupKey1 = Bytes.toBytes(datasetKey + "|ABCD");
133    Put put = new Put(lookupKey1);
134    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
135      Bytes.toBytes(HBaseTableConstants.LOOKUP_LOCK_COLUMN), 0, lock1);
136    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
137      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN), Bytes.toBytes(2));
138    lookupTable.put(put);
139
140    byte[] lock2 = Bytes.toBytes(UUID.randomUUID().toString());
141    byte[] lookupKey2 = Bytes.toBytes(datasetKey + "|EFGH");
142    put = new Put(lookupKey2);
143    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
144      Bytes.toBytes(HBaseTableConstants.LOOKUP_LOCK_COLUMN), 0, lock2);
145    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
146      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN), Bytes.toBytes(3));
147    lookupTable.put(put);
148    lookupTable.flushCommits();
149    lookupTable.close();
150
151    // test: 3rd keygen attempt uses both previous unique ids, expects a new key to be generated
152    KeyLookupResult result = keyService.generateKey(ImmutableSet.of("ABCD", "EFGH"), datasetKey);
153    assertEquals(1, result.getKey());
154  }

156  @Test
157  public void testConflictingIds() throws IOException {
158    // setup: 2 rows with different lookupkeys and assigned ids
159    HTableInterface lookupTable = tablePool.getTable(LOOKUP_TABLE);
160    String datasetKey = "fakeuuid";
161
162    byte[] lookupKey1 = Bytes.toBytes(datasetKey + "|ABCD");
163    Put put = new Put(lookupKey1);
164    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
165      Bytes.toBytes(HBaseTableConstants.LOOKUP_STATUS_COLUMN), Bytes.toBytes("ALLOCATED"));
166    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
167      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN), Bytes.toBytes(1));
168    lookupTable.put(put);
169
170    byte[] lookupKey2 = Bytes.toBytes(datasetKey + "|EFGH");
171    put = new Put(lookupKey2);
172    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
173      Bytes.toBytes(HBaseTableConstants.LOOKUP_STATUS_COLUMN), Bytes.toBytes("ALLOCATED"));
174    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
175      Bytes.toBytes(HBaseTableConstants.LOOKUP_KEY_COLUMN), Bytes.toBytes(2));
176    lookupTable.put(put);
177    lookupTable.flushCommits();
178    lookupTable.close();
179
180    // test: gen id for one occ with both lookupkeys
181    exception.expect(RuntimeException.class);
182    exception.expectMessage(
183      "Found inconsistent occurrence keys in looking up unique identifiers:[" + datasetKey + "|ABCD]=[1][" + datasetKey
184      + "|EFGH]=[2]");
185    keyService.generateKey(ImmutableSet.of("ABCD", "EFGH"), datasetKey);
186  }

188  @Test
189  public void testStaleLock() throws IOException {
190    String datasetKey = UUID.randomUUID().toString();
191    // setup: lookupkey | null for status | uuid with old ts for lock | null for occurrence key
192
193    byte[] lookupKey = Bytes.toBytes(datasetKey + "|ABCD");
194    byte[] lock = Bytes.toBytes(UUID.randomUUID().toString());
195    Put put = new Put(lookupKey);
196    put.add(Bytes.toBytes(HBaseTableConstants.LOOKUP_COLUMN_FAMILY),
197      Bytes.toBytes(HBaseTableConstants.LOOKUP_LOCK_COLUMN), 0, lock);
198    HTableInterface lookupTable = tablePool.getTable(LOOKUP_TABLE);
199    lookupTable.put(put);
200    lookupTable.flushCommits();
201    lookupTable.close();
202
203    KeyLookupResult result = keyService.generateKey(ImmutableSet.of("ABCD"), datasetKey);
204    assertEquals(1, result.getKey());
205  }

230  @Test
231  public void testThreadedIdContig() throws InterruptedException {
232    // 5 threads concurrently allocated 1000 ids each, expect the next call to produce id 5001
233    List<Thread> threads = Lists.newArrayList();
234    for (int i = 0; i < 5; i++) {
235      Thread thread = new Thread(new KeyRequester(1000, keyService, String.valueOf(i)));
236      thread.start();
237      threads.add(thread);
238    }
239    for (Thread thread : threads) {
240      thread.join();
241    }
242    KeyLookupResult result = keyService.generateKey(ImmutableSet.of("asdf"), "wqer");
243    assertEquals(5001, result.getKey());
244  }
245
246  private static class KeyRequester implements Runnable {
247
248    private final int keyCount;
249    private final HBaseLockingKeyService keyService;
250    private final String name;
251
252    private KeyRequester(int keyCount, HBaseLockingKeyService keyService, String name) {
253      this.keyCount = keyCount;
254      this.keyService = keyService;
255      this.name = name;
256    }
257
258    @Override
259    public void run() {
260      for (int i = 0; i < keyCount; i++) {
261        Set<String> uniqueIds = ImmutableSet.of(name + i);
262        keyService.generateKey(uniqueIds, "boo");
263      }
264    }
265  }
