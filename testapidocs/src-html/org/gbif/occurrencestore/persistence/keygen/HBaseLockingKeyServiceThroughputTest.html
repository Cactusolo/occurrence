001package org.gbif.occurrencestore.persistence.keygen;
002
003import java.util.List;
004import java.util.UUID;
005import java.util.concurrent.TimeUnit;
006import java.util.concurrent.atomic.AtomicInteger;
007
008import com.google.common.collect.ImmutableSet;
009import com.google.common.collect.Lists;
010import org.apache.hadoop.hbase.HBaseConfiguration;
011import org.apache.hadoop.hbase.client.HTablePool;
012
013/**
014 * Note not a real JUnit test, but an extremely expensive performance test that should use the real cluster.
015 */
016public class HBaseLockingKeyServiceThroughputTest {
017
018  private static final String LOOKUP_TABLE_NAME = "keygen_test_occurrence_lookup";
019  private static final String COUNTER_TABLE_NAME = "keygen_test_occurrence_counter";
020  private static final String OCCURRENCE_TABLE_NAME = "keygen_test_occurrence";
021
022  private HTablePool tablePool = null;
023  private final HBaseLockingKeyService keyService;
024
025  private static final AtomicInteger keysGenerated = new AtomicInteger(0);
026
027  public HBaseLockingKeyServiceThroughputTest(int hbasePoolSize) {
028    tablePool = new HTablePool(HBaseConfiguration.create(), hbasePoolSize);
029    keyService = new HBaseLockingKeyService(LOOKUP_TABLE_NAME, COUNTER_TABLE_NAME, OCCURRENCE_TABLE_NAME, tablePool);
030  }
031
032  public void testNoContention(int threadCount) throws InterruptedException {
033    // test generating ids as fast as possible in the ideal case of no waiting for contention (all ids are globally
034    // unique)
035    int genPerThread = 100000;
036    List<Thread> threads = Lists.newArrayList();
037    for (int i = 0; i < threadCount; i++) {
038      Thread thread = new Thread(new KeyGenerator(keyService, UUID.randomUUID(), genPerThread));
039      thread.start();
040      threads.add(thread);
041    }
042
043    Thread rateReporter = new Thread(new RateReporter(threadCount));
044    rateReporter.start();
045
046    for (Thread thread : threads) {
047      thread.join();
048    }
049
050    rateReporter.interrupt();
051    rateReporter.join();
052  }
053
054  private static class RateReporter implements Runnable {
055
056    private final int threadCount;
057
058    private RateReporter(int threadCount) {
059      this.threadCount = threadCount;
060    }
061
062    @Override
063    public void run() {
064      int periods = 0;
065      int runningAvg = 0;
066      int buildAverageAfter = 15;
067      int lastCount = 0;
068      boolean interrupted = false;
069      while (!interrupted) {
070        try {
071          TimeUnit.SECONDS.sleep(1);
072        } catch (InterruptedException e) {
073          interrupted = true;
074        }
075        int generated = keysGenerated.intValue() - lastCount;
076        if (periods > buildAverageAfter) {
077          if (runningAvg == 0) {
078            runningAvg = generated;
079          } else {
080            int netPeriods = periods - buildAverageAfter;
081            runningAvg = (netPeriods * runningAvg + generated) / (netPeriods + 1);
082          }
083          System.out.println("Key generation at [" + generated + " keys/s] for running avg of [" + runningAvg
084                             + " keys/s] and per thread [" + (runningAvg / threadCount)
085                             + " keys/sec] with id generation time of [" + (threadCount * 1000 / runningAvg)
086                             + " ms/id]");
087        } else {
088          System.out.println("Stats in [" + (buildAverageAfter - periods) + "] seconds.");
089        }
090        periods++;
091        lastCount = keysGenerated.intValue();
092      }
093    }
094  }
095
096  private static class KeyGenerator implements Runnable {
097
098    private final HBaseLockingKeyService keyService;
099    private final UUID datasetKey;
100    private final int genCount;
101
102    private KeyGenerator(HBaseLockingKeyService keyService, UUID datasetKey, int genCount) {
103      this.keyService = keyService;
104      this.datasetKey = datasetKey;
105      this.genCount = genCount;
106    }
107
108    @Override
109    public void run() {
110      for (int i = 0; i < genCount; i++) {
111        keyService.generateKey(ImmutableSet.of(String.valueOf(i)), datasetKey.toString());
112        keysGenerated.incrementAndGet();
113      }
114    }
115  }
116
117  public static void main(String[] args) throws InterruptedException {
118    int hbasePoolSize = 100;
119    int persistingThreads = 100;
120    if (args.length == 2) {
121      hbasePoolSize = Integer.valueOf(args[0]);
122      persistingThreads = Integer.valueOf(args[1]);
123    }
124    System.out
125      .println("Running test with hbasePool [" + hbasePoolSize + "] and persistingThreads [" + persistingThreads + "]");
126    HBaseLockingKeyServiceThroughputTest instance = new HBaseLockingKeyServiceThroughputTest(hbasePoolSize);
127    instance.testNoContention(persistingThreads);
128  }
129}
