009
010import org.apache.hadoop.hbase.HBaseTestingUtility;
011import org.apache.hadoop.hbase.client.HTableInterface;

013import org.apache.hadoop.hbase.client.Put;
014import org.apache.hadoop.hbase.client.Scan;
015import org.apache.hadoop.hbase.filter.CompareFilter;
016import org.apache.hadoop.hbase.filter.SingleColumnValueFilter;
017import org.apache.hadoop.hbase.util.Bytes;
018import org.junit.AfterClass;
019import org.junit.Before;
020import org.junit.BeforeClass;

022import org.junit.Test;
023
024import static org.junit.Assert.assertEquals;
025
026@Ignore("As per http://dev.gbif.org/issues/browse/OCC-109")
027public class OccurrenceKeyIteratorTest {
028
029  private static final String TABLE_NAME = "occurrence_test";
030  private static final byte[] TABLE = Bytes.toBytes(TABLE_NAME);
031  private static final String CF_NAME = "o";
032  private static final byte[] CF = Bytes.toBytes(CF_NAME);
033  private static final UUID DATASET_1 = UUID.randomUUID();
034  private static final int ID_1_COUNT = 200;
035  private static final UUID DATASET_2 = UUID.randomUUID();
036  private static final int ID_2_COUNT = 750;
037  private static final UUID DATASET_3 = UUID.randomUUID();
038  private static final int ID_3_COUNT = 1200;
039  private static final HBaseTestingUtility TEST_UTIL = new HBaseTestingUtility();
040
041  private HTablePool tablePool = null;
042
043  @AfterClass
044  public static void afterClass() throws Exception {
045    TEST_UTIL.shutdownMiniCluster();
046  }
047
048  @BeforeClass
049  public static void beforeClass() throws Exception {
050    TEST_UTIL.startMiniCluster(1);
051    TEST_UTIL.createTable(TABLE, CF);
052  }
053
054  @Before
055  public void setUp() throws Exception {
056    TEST_UTIL.truncateTable(TABLE);
057
058    tablePool = new HTablePool(TEST_UTIL.getConfiguration(), 1);
059
060    HTableInterface table = tablePool.getTable(TABLE_NAME);
061    int keyCount = 0;
062    for (int i = 0; i < ID_1_COUNT; i++) {
063      keyCount++;
064      addId(keyCount, DATASET_1.toString(), table);
065    }
066    for (int i = 0; i < ID_2_COUNT; i++) {
067      keyCount++;
068      addId(keyCount, DATASET_2.toString(), table);
069    }
070    for (int i = 0; i < ID_3_COUNT; i++) {
071      keyCount++;
072      addId(keyCount, DATASET_3.toString(), table);
073    }
074    table.flushCommits();
075    table.close();
076  }
077
078  private void addId(int key, String value, HTableInterface table) throws IOException {
079    Put put = new Put(Bytes.toBytes(key));
080    put.add(CF, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATASET_KEY).getColumnName()),
081      Bytes.toBytes(value));
082    table.put(put);
083  }
084
085  @Test
086  public void testOneRecord() throws IOException {
087    TEST_UTIL.truncateTable(TABLE);
088    HTableInterface table = tablePool.getTable(TABLE);
089    addId(1, DATASET_1.toString(), table);
090    table.flushCommits();
091    table.close();
092
093    byte[] col = Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATASET_KEY).getColumnName());
094    Scan scan = new Scan();
095    scan.addColumn(CF, col);
096    scan.setBatch(500);
097    scan.setCaching(500);
098    SingleColumnValueFilter filter =
099      new SingleColumnValueFilter(CF, col, CompareFilter.CompareOp.EQUAL, Bytes.toBytes(DATASET_1.toString()));
100    scan.setFilter(filter);
101
102    Iterator<Integer> iterator = new OccurrenceKeyIterator(tablePool, TABLE_NAME, scan);
103    int count = 0;
104    while (iterator.hasNext()) {
105      iterator.next();
106      count++;
107    }
108    assertEquals(1, count);
109  }
110
111
112  @Test
113  public void testOnePage() {
114    byte[] col = Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATASET_KEY).getColumnName());
115    Scan scan = new Scan();
116    scan.addColumn(CF, col);
117    scan.setBatch(500);
118    scan.setCaching(500);
119    SingleColumnValueFilter filter =
120      new SingleColumnValueFilter(CF, col, CompareFilter.CompareOp.EQUAL, Bytes.toBytes(DATASET_1.toString()));
121    scan.setFilter(filter);
122
123    Iterator<Integer> iterator = new OccurrenceKeyIterator(tablePool, TABLE_NAME, scan);
124    int count = 0;
125    while (iterator.hasNext()) {
126      iterator.next();
127      count++;
128    }
129    assertEquals(ID_1_COUNT, count);
130  }
131

133  public void testMultiPage() {

135    Scan scan = new Scan();
136    scan.addColumn(CF, col);
137    scan.setBatch(500);
138    scan.setCaching(500);
139    SingleColumnValueFilter filter =
140      new SingleColumnValueFilter(CF, col, CompareFilter.CompareOp.EQUAL, Bytes.toBytes(DATASET_3.toString()));
141    scan.setFilter(filter);
142
143    Iterator<Integer> iterator = new OccurrenceKeyIterator(tablePool, TABLE_NAME, scan);
144    int count = 0;
145    while (iterator.hasNext()) {
146      iterator.next();

149    assertEquals(ID_3_COUNT, count);
150  }
