309  @Test
310  public void testRaceForOneKey() throws InterruptedException {
311    UUID datasetKey = UUID.randomUUID();
312    HolyTriplet triplet = new HolyTriplet(datasetKey, "ic", "cc", "cn", null);
313    PublisherProvidedUniqueIdentifier pubProvided1 = new PublisherProvidedUniqueIdentifier(datasetKey, "abc:123");
314    PublisherProvidedUniqueIdentifier pubProvided2 = new PublisherProvidedUniqueIdentifier(datasetKey, "xcb08725");
315    Set<UniqueIdentifier> ids = Sets.newHashSet();
316    ids.add(triplet);
317    ids.add(pubProvided1);
318    ids.add(pubProvided2);
319
320    // generate 100 threads, all attempt to generate new key for same unique ids - ensure all return same value
321    List<KeyGeneratorThread> threads = Lists.newArrayList();
322    for (int i = 0; i < 100; i++) {
323      KeyGeneratorThread thread = new KeyGeneratorThread(ids);
324      thread.start();
325      threads.add(thread);
326    }
327    for (KeyGeneratorThread thread : threads) {
328      thread.join();
329    }
330
331    // next available key in the test counter table is 3
332    int expectedKey = 3;
333    assertEquals(expectedKey, occurrenceKeyService.findKey(ids).getKey());
334  }

336  @Test
337  public void testRaceForManyKeys() throws InterruptedException {
338    UUID datasetKey = UUID.randomUUID();
339
340    // generate 100 threads, all attempt to generate new key for same unique ids - ensure all return same value
341    List<KeyGeneratorThread> threads = Lists.newArrayList();
342    int threadCount = 100;
343    for (int i = 0; i < threadCount; i++) {
344      PublisherProvidedUniqueIdentifier pubProvided =
345        new PublisherProvidedUniqueIdentifier(datasetKey, UUID.randomUUID().toString());
346      Set<UniqueIdentifier> ids = Sets.newHashSet();
347      ids.add(pubProvided);
348      KeyGeneratorThread thread = new KeyGeneratorThread(ids);
349      thread.start();
350      threads.add(thread);
351      KeyGeneratorThread thread2 = new KeyGeneratorThread(ids);
352      thread2.start();
353      threads.add(thread2);
354    }
355    for (KeyGeneratorThread thread : threads) {
356      thread.join();
357    }
358
359    // test counter table is at 2, then 100 + 1 for this test
360    int expectedKey = 2 + threadCount + 1;
361    PublisherProvidedUniqueIdentifier pubProvided =
362      new PublisherProvidedUniqueIdentifier(datasetKey, UUID.randomUUID().toString());
363    Set<UniqueIdentifier> ids = Sets.newHashSet();
364    ids.add(pubProvided);
365    assertEquals(expectedKey, occurrenceKeyService.generateKey(ids).getKey());
366  }

368  private class KeyGeneratorThread extends Thread {
369
370    private final Set<UniqueIdentifier> ids;
371    private final OccurrenceKeyPersistenceService occKeyService;
372
373    private KeyGeneratorThread(Set<UniqueIdentifier> ids) {
374      this.ids = ids;
375      ThreadLocalLockProvider threadLocalLockProvider = new ThreadLocalLockProvider(CURATOR);
376      KeyPersistenceService keyPersistenceService = new ZkLockingKeyService(LOOKUP_TABLE_NAME, COUNTER_TABLE_NAME,
377        OCCURRENCE_TABLE_NAME, tablePool, threadLocalLockProvider);
378      this.occKeyService =
379        new OccurrenceKeyPersistenceServiceImpl(keyPersistenceService);
380    }
381
382    public void run() {
383      occKeyService.generateKey(ids);
384    }
385  }
