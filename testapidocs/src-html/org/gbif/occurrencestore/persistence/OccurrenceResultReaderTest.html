001package org.gbif.occurrencestore.persistence;
002
003import org.gbif.occurrencestore.common.model.constants.FieldName;
004import org.gbif.occurrencestore.persistence.hbase.HBaseFieldUtil;
005
006import java.util.ArrayList;
007import java.util.List;
008
009import org.apache.hadoop.hbase.KeyValue;
010import org.apache.hadoop.hbase.client.Result;
011import org.apache.hadoop.hbase.util.Bytes;
012import org.junit.Before;
013import org.junit.Rule;
014import org.junit.Test;
015import org.junit.rules.ExpectedException;
016
017import static org.junit.Assert.assertEquals;
018
019public class OccurrenceResultReaderTest {
020
021  private static final String CF1_NAME = "o";
022  private static final byte[] CF1 = Bytes.toBytes(CF1_NAME);
023  private static final int KEY_NAME = 12345;
024  private static final byte[] KEY = Bytes.toBytes(KEY_NAME);
025
026  private static final int INT_VAL_1 = 1111;
027  private static final double DOUBLE_VAL_1 = 2.2222222222222222222d;
028  private static final long LONG_VAL_1 = 33333333333333l;
029  private static final String STRING_VAL_1 = "not numbers";
030
031  private Result result = null;
032
033  @Rule
034  public ExpectedException exception = ExpectedException.none();
035
036  @Before
037  public void setup() {
038    List<KeyValue> kvs = new ArrayList<KeyValue>();
039
040    // mimic four fields of an occurrence row
041    KeyValue kv = new KeyValue(KEY, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATA_PROVIDER_ID).getColumnFamilyName()),
042      Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.DATA_PROVIDER_ID).getColumnName()), Bytes.toBytes(INT_VAL_1));
043    kvs.add(kv);
044    kv = new KeyValue(KEY, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_LATITUDE).getColumnFamilyName()),
045      Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.I_LATITUDE).getColumnName()), Bytes.toBytes(DOUBLE_VAL_1));
046    kvs.add(kv);
047    kv = new KeyValue(KEY, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.MODIFIED).getColumnFamilyName()),
048      Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.MODIFIED).getColumnName()), Bytes.toBytes(LONG_VAL_1));
049    kvs.add(kv);
050    kv = new KeyValue(KEY, Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.SCIENTIFIC_NAME).getColumnFamilyName()),
051      Bytes.toBytes(HBaseFieldUtil.getHBaseColumn(FieldName.SCIENTIFIC_NAME).getColumnName()), Bytes.toBytes(STRING_VAL_1));
052    kvs.add(kv);
053
054    result = new Result(kvs);
055  }
056
057  @Test
058  public void testKey() {
059    Integer test = OccurrenceResultReader.getKey(result);
060    assertEquals(KEY_NAME, test.intValue());
061  }
062
063  @Test
064  public void testString() {
065    String test = OccurrenceResultReader.getString(result, FieldName.SCIENTIFIC_NAME, null);
066    assertEquals(STRING_VAL_1, test);
067  }
068
069  @Test
070  public void testStringFail() {
071    exception.expect(IllegalArgumentException.class);
072    exception.expectMessage("FieldName [MODIFIED] is not of type String");
073    OccurrenceResultReader.getString(result, FieldName.MODIFIED, null);
074  }
075
076  @Test
077  public void testInteger() {
078    Integer test = OccurrenceResultReader.getInteger(result, FieldName.DATA_PROVIDER_ID, null);
079    assertEquals(INT_VAL_1, test.intValue());
080  }
081
082  @Test
083  public void testIntegerFail() {
084    exception.expect(IllegalArgumentException.class);
085    exception.expectMessage("FieldName [MODIFIED] is not of type Integer");
086    OccurrenceResultReader.getInteger(result, FieldName.MODIFIED, null);
087  }
088
089  @Test
090  public void testLong() {
091    Long test = OccurrenceResultReader.getLong(result, FieldName.MODIFIED, null);
092    assertEquals(LONG_VAL_1, test.longValue());
093  }
094
095  @Test
096  public void testLongFail() {
097    exception.expect(IllegalArgumentException.class);
098    exception.expectMessage("FieldName [SCIENTIFIC_NAME] is not of type Long");
099    OccurrenceResultReader.getLong(result, FieldName.SCIENTIFIC_NAME, null);
100  }
101
102  @Test
103  public void testDouble() {
104    Double test = OccurrenceResultReader.getDouble(result, FieldName.I_LATITUDE, null);
105    assertEquals(DOUBLE_VAL_1, test, 0.00001);
106  }
107
108  @Test
109  public void testDoubleFail() {
110    exception.expect(IllegalArgumentException.class);
111    exception.expectMessage("FieldName [SCIENTIFIC_NAME] is not of type Double");
112    OccurrenceResultReader.getDouble(result, FieldName.SCIENTIFIC_NAME, null);
113  }
114
115  @Test
116  public void testObject() {
117    Object test = OccurrenceResultReader.get(result, FieldName.I_LATITUDE);
118    assertEquals(DOUBLE_VAL_1, (Double) test, 0.00001);
119  }
120}
