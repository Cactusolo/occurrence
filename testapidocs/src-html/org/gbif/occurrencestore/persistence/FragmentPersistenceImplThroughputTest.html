025 * Note not a real JUnit test, but an extremely expensive performance test that should use the real cluster.

027public class FragmentPersistenceImplThroughputTest {
028
029  private static final String LOOKUP_TABLE_NAME = "keygen_test_occurrence_lookup";
030  private static final String COUNTER_TABLE_NAME = "keygen_test_occurrence_counter";
031  private static final String OCCURRENCE_TABLE_NAME = "keygen_test_occurrence";
032
033  private final FragmentPersistenceServiceImpl fragService;
034
035  private static final AtomicInteger fragsPersisted = new AtomicInteger(0);
036
037  public FragmentPersistenceImplThroughputTest(int hbasePoolSize) {
038    HTablePool tablePool = new HTablePool(HBaseConfiguration.create(), hbasePoolSize);
039    HBaseLockingKeyService keyService =
040      new HBaseLockingKeyService(LOOKUP_TABLE_NAME, COUNTER_TABLE_NAME, OCCURRENCE_TABLE_NAME, tablePool);
041    fragService = new FragmentPersistenceServiceImpl(OCCURRENCE_TABLE_NAME, tablePool,
042      new OccurrenceKeyPersistenceServiceImpl(keyService));
043  }
044
045  public void testNoContention(int threadCount) throws InterruptedException {
046    // test generating ids as fast as possible in the ideal case of no waiting for contention (all ids are globally
047    // unique)
048    int genPerThread = 10000;
049    List<Thread> threads = Lists.newArrayList();
050    for (int i = 0; i < threadCount; i++) {
051      Thread thread = new Thread(new FragmentPersister(fragService, UUID.randomUUID(), genPerThread));
052      thread.start();
053      threads.add(thread);
054    }
055
056    Thread rateReporter = new Thread(new RateReporter(threadCount));
057    rateReporter.start();
058
059    for (Thread thread : threads) {
060      thread.join();
061    }
062
063    rateReporter.interrupt();
064    rateReporter.join();
065  }
066
067  private static class RateReporter implements Runnable {
068
069    private final int threadCount;
070
071    private RateReporter(int threadCount) {
072      this.threadCount = threadCount;
073    }
074
075    @Override
076    public void run() {
077      int periods = 0;
078      int runningAvg = 0;
079      int buildAverageAfter = 15;
080      int lastCount = 0;
081      boolean interrupted = false;
082      while (!interrupted) {
083        try {
084          TimeUnit.SECONDS.sleep(1);
085        } catch (InterruptedException e) {
086          interrupted = true;
087        }
088        int generated = fragsPersisted.intValue() - lastCount;
089        if (periods > buildAverageAfter) {
090          if (runningAvg == 0) {
091            runningAvg = generated;
092          } else {
093            int netPeriods = periods - buildAverageAfter;
094            runningAvg = (netPeriods * runningAvg + generated) / (netPeriods + 1);
095          }
096          System.out.println("Frags persisted at [" + generated + " frags/s] for running avg of [" + runningAvg
097                             + " frags/s] and per thread [" + (runningAvg / threadCount)
098                             + " frags/sec] with frag persist time of [" + (threadCount * 1000 / runningAvg)
099                             + " ms/frag]");
100        } else {
101          System.out.println("Stats in [" + (buildAverageAfter - periods) + "] seconds.");
102        }
103        periods++;
104        lastCount = fragsPersisted.intValue();
105      }
106    }
107  }
108
109  private static class FragmentPersister implements Runnable {
110
111    private final FragmentPersistenceService fragService;
112    private final UUID datasetKey;
113    private final int genCount;
114    private final Fragment fragment;
115
116    private FragmentPersister(FragmentPersistenceService fragService, UUID datasetKey, int genCount) {
117      this.fragService = fragService;
118      this.datasetKey = datasetKey;
119      this.genCount = genCount;
120      String xml = "<DarwinRecord>\n"
121                   + "  <GlobalUniqueIdentifier>ZMA:Entomology:Diptera_Tipulidae_NL_TEMP_09183</GlobalUniqueIdentifier>\n"
122                   + "  <DateLastModified>2007-05-02</DateLastModified>\n"
123                   + "  <BasisOfRecord>Museum specimen</BasisOfRecord>\n" + "  <InstitutionCode>ZMA</InstitutionCode>\n"
124                   + "  <CollectionCode>Tipulidae</CollectionCode>\n" + "  <CatalogNumber>TEMP_09183</CatalogNumber>\n"
125                   + "  <ScientificName>Ctenophora (Cnemoncosis) festiva Meigen, 1804</ScientificName>\n"
126                   + "  <Kingdom>Animalia</Kingdom>\n" + "  <Phylum nil=\"true\"/>\n" + "  <Class>Insecta</Class>\n"
127                   + "  <Order nil=\"true\"/>\n" + "  <Family>Tipulidae</Family>\n" + "  <Genus>Ctenophora</Genus>\n"
128                   + "  <SpecificEpithet>festiva</SpecificEpithet>\n" + "  <InfraspecificEpithet nil=\"true\"/>\n"
129                   + "  <Continent nil=\"true\"/>\n" + "  <WaterBody nil=\"true\"/>\n"
130                   + "  <Country>Netherlands</Country>\n" + "  <StateProvince>Utrecht</StateProvince>\n"
131                   + "  <Locality>Leusden</Locality>\n" + "  <MinimumElevationInMeters nil=\"true\"/>\n"
132                   + "  <MaximumElevationInMeters nil=\"true\"/>\n" + "  <MinimumDepthInMeters nil=\"true\"/>\n"
133                   + "  <MaximumDepthInMeters nil=\"true\"/>\n" + "  <CollectingMethod>unrecorded</CollectingMethod>\n"
134                   + "  <EarliestDateCollected>1985-06-08 00:00:00</EarliestDateCollected>\n"
135                   + "  <LatestDateCollected>1985-06-08 00:00:00</LatestDateCollected>\n"
136                   + "  <DayOfYear nil=\"true\"/>\n" + "  <Collector>Zeegers, T.</Collector>\n"
137                   + "  <DecimalLatitude nil=\"true\"/>\n" + "  <DecimalLongitude nil=\"true\"/>\n"
138                   + "  <CoordinateUncertaintyInMeters nil=\"true\"/>\n" + "  <Preparations>dry pinned</Preparations>\n"
139                   + "  <TypeStatus nil=\"true\"/>\n" + "</DarwinRecord>\n";
140      byte[] hash = DigestUtils.md5(xml);
141      fragment =
142        new Fragment(datasetKey, xml.getBytes(Charsets.UTF_8), hash, Fragment.FragmentType.XML, EndpointType.DIGIR,
143          new Date(), 1, OccurrenceSchemaType.DWC_MANIS, null, System.currentTimeMillis());
144    }
145
146    @Override
147    public void run() {
148      for (int i = 0; i < genCount; i++) {
149        UniqueIdentifier unique = new PublisherProvidedUniqueIdentifier(datasetKey, String.valueOf(i));
150        fragService.insert(fragment, ImmutableSet.of(unique));
151        //        fragment.setKey(i);
152        //        fragService.update(fragment);
153        fragsPersisted.incrementAndGet();
154      }
155    }
156  }
157
158  public static void main(String[] args) throws InterruptedException {
159    // following stats from single regionserver, single region
160    // on pure insert, raw rate is ~3k/sec
161    // on pure update where some fields null, raw rate is ~8k/sec
162    // on pure update where no fields null, raw rate is ~15k/sec
163
164    int hbasePoolSize = 100;
165    int persistingThreads = 100;
166    if (args.length == 2) {
167      hbasePoolSize = Integer.valueOf(args[0]);
168      persistingThreads = Integer.valueOf(args[1]);
169    }
170    System.out
171      .println("Running test with hbasePool [" + hbasePoolSize + "] and persistingThreads [" + persistingThreads + "]");
172    FragmentPersistenceImplThroughputTest instance = new FragmentPersistenceImplThroughputTest(hbasePoolSize);
173    instance.testNoContention(persistingThreads);
174  }
