001package org.gbif.occurrencestore.persistence.guice;
002
003import org.gbif.api.service.occurrence.OccurrenceService;
004import org.gbif.occurrencestore.persistence.api.DatasetDeletionService;
005import org.gbif.occurrencestore.persistence.api.FragmentPersistenceService;
006import org.gbif.occurrencestore.persistence.api.VerbatimOccurrencePersistenceService;
007import org.gbif.occurrencestore.persistence.zookeeper.ZookeeperLockManager;
008
009import java.util.Properties;
010
011import com.google.inject.Guice;
012import com.google.inject.Injector;
013import org.junit.Test;
014
015import static org.junit.Assert.assertEquals;
016import static org.junit.Assert.assertNotNull;
017
018public class OccurrencePersistenceModuleTest {
019
020  // ensure that the guice module is currrent - if you change this, change the README to match!
021  @Test
022  public void testModule() {
023    Properties props = new Properties();
024    props.setProperty("occurrencestore.db.table_name", "occurrence");
025    props.setProperty("occurrencestore.db.counter_table_name", "occurrence_counter");
026    props.setProperty("occurrencestore.db.id_lookup_table_name", "occurrence_id");
027    props.setProperty("occurrencestore.db.max_connection_pool", "1");
028    props.setProperty("occurrencestore.db.zookeeper.connection_string", "localhost:2181");
029
030    Injector injector = Guice.createInjector(new OccurrencePersistenceModule(props));
031    OccurrenceService occService = injector.getInstance(OccurrenceService.class);
032    assertNotNull(occService);
033    VerbatimOccurrencePersistenceService verbService1 = injector.getInstance(VerbatimOccurrencePersistenceService.class);
034    assertNotNull(verbService1);
035    VerbatimOccurrencePersistenceService verbService2 = injector.getInstance(VerbatimOccurrencePersistenceService.class);
036    assertEquals(verbService1, verbService2);
037    FragmentPersistenceService fragService = injector.getInstance(FragmentPersistenceService.class);
038    assertNotNull(fragService);
039    ZookeeperLockManager lockManager1 = injector.getInstance(ZookeeperLockManager.class);
040    assertNotNull(lockManager1);
041    ZookeeperLockManager lockManager2 = injector.getInstance(ZookeeperLockManager.class);
042    assertEquals(lockManager1, lockManager2);
043    DatasetDeletionService ddService = injector.getInstance(DatasetDeletionService.class);
044    assertNotNull(ddService);
045  }
046}
